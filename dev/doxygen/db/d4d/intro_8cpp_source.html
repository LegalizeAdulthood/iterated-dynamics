33static std::vector<int> get_authors(const char *credits)
34{
35 std::vector<int> authors;
36 authors.push_back(0); // find the start of each credit-line
37 int i;
38 for (i = 0; credits[i] != 0; i++)
39 {
40 if (credits[i] == '\n')
41 {
42 authors.push_back(i + 1);
43 }
44 }
45 authors.push_back(i);
46 return authors;
47}

110 driver_get_key();

115 i++;

117 {
118 i = 0;
119 }
120 old_char = credits[authors.at(i+1)];
121 credits[authors.at(i+1)] = 0;
122 driver_put_string(bot_row, 0, C_CONTRIB, &credits[authors.at(i)]);
123 driver_set_attr(bot_row, 0, C_CONTRIB, 80);
124 credits[authors.at(i+1)] = old_char;
125 driver_hide_text_cursor(); // turn it off
126 }
127}
