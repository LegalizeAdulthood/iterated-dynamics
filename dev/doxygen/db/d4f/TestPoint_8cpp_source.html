4
5Write your fractal program here. initreal and initimag are the values in
6the complex plane; parm1, and parm2 are paramaters to be entered with the
7"params=" option (if needed). The function should return the color associated
8with initreal and initimag. Id will repeatedly call your function with
9the values of initreal and initimag ranging over the rectangle defined by the
10"corners=" option. Assuming your formula is iterative, "maxit" is the maximum
11iteration. If "maxit" is hit, color "inside" should be returned.

13Note that this routine could be sped up using external variables/arrays

66void TestPoint::iterate()
67{
68 if (g_row <= g_i_stop_pt.y)
69 {
70 if (g_col <= g_i_stop_pt.x) // look at each point on screen
71 {
72 g_init.x = dx_pixel();
73 g_init.y = dy_pixel();
74 int color =
75 per_pixel(g_init.x, g_init.y, g_param_z1.x, g_param_z1.y, g_max_iterations, g_inside_color);
76 if (color >= g_colors)
77 {
78 // avoid trouble if color is 0
79 if (g_colors < 16)
80 {
81 color &= g_and_color;
82 }
83 else
84 {
85 color = (color - 1) % g_and_color + 1; // skip color zero
86 }
87 }
88 g_plot(g_col, g_row, color);
89 if (num_passes && passes == 0)
90 {
91 g_plot(g_col, g_row + 1, color);
92 }
93 ++g_col;
94 }
95 else
96 {
97 g_col = 0;
98 g_row += 1 + num_passes;
99 }
100 }
101 else
102 {
103 start_row = passes + 1;
104 g_row = start_row;
105 passes++;
106 }
107}

110// this routine is called once for every pixel
111// (note: possibly using the dual-pass / solid-guessing options
112int TestPoint::per_pixel(const double init_real, const double init_imag, //
113 const double param1, const double param2, //
114 const long max_iter, const int inside)
115{
116 double old_real = param1;
117 double old_imag = param2;
118 double magnitude = 0.0;
119 long iter = 0;
120 while (magnitude < 4.0 && iter < max_iter)
121 {
122 const double new_real = old_real * old_real - old_imag * old_imag + init_real;
123 const double new_imag = 2 * old_real * old_imag + init_imag;
124 iter++;
125 old_real = new_real;
126 old_imag = new_imag;
127 magnitude = new_real * new_real + new_imag * new_imag;
128 }
129 if (iter >= max_iter)
130 {
131 iter = inside;
132 }
133 return static_cast<int>(iter);
134}
