61int halley_orbit()
62{
63 // X(X^a - 1) = 0, Halley Map
64 // a = parm.x = degree, relaxation coeff. = parm.y, epsilon = parm2.x
65 DComplex x_to_a_less_one;
66 DComplex x_to_a;
67 DComplex x_to_a_plus_one; // a-1, a, a+1
68 DComplex fx;
69 DComplex f1_prime;
70 DComplex f2_prime;
71 DComplex numer1;
72 DComplex numer2;
73 DComplex denom;
74 DComplex relax;
75
76 x_to_a_less_one = g_old_z;
77 for (int deg = 2; deg < g_degree; deg++)
78 {
79 fpu_cmplx_mul(g_old_z, x_to_a_less_one, x_to_a_less_one);
80 }
81 fpu_cmplx_mul(g_old_z, x_to_a_less_one, x_to_a);
82 fpu_cmplx_mul(g_old_z, x_to_a, x_to_a_plus_one);
83
84 fx = x_to_a_plus_one - g_old_z; // FX = X^(a+1) - X = F
85 f2_prime.x = s_halley_a_plus_one_times_degree * x_to_a_less_one.x; // g_halley_a_plus_one_times_degree in setup
86 f2_prime.y = s_halley_a_plus_one_times_degree * x_to_a_less_one.y; // F"
87
88 f1_prime.x = s_halley_a_plus_one * x_to_a.x - 1.0;
89 f1_prime.y = s_halley_a_plus_one * x_to_a.y; // F'
90
91 fpu_cmplx_mul(f2_prime, fx, numer1); // F * F"
92 denom.x = f1_prime.x + f1_prime.x;
93 denom.y = f1_prime.y + f1_prime.y; // 2 * F'
94
95 fpu_cmplx_div(numer1, denom, numer1); // F"F/2F'
96 numer2 = f1_prime - numer1; // F' - F"F/2F'
97 fpu_cmplx_div(fx, numer2, numer2);
98 // parm.y is relaxation coef.
99 relax.x = g_param_z1.y;
100 relax.y = g_params[3];
101 fpu_cmplx_mul(relax, numer2, numer2);
102 g_new_z.x = g_old_z.x - numer2.x;
103 g_new_z.y = g_old_z.y - numer2.y;
104 return halley_bailout();
105}
