14std::string trim_filename(const std::string &filename, const int length)
15{
16 if (static_cast<int>(filename.length()) <= length)
17 {
18 return filename;
19 }
20
21 fs::path path{filename, fs::path::format::generic_format};
22 std::vector<fs::path> parts;
23 while (path.has_parent_path() && path.parent_path() != path)
24 {
25 parts.push_back(path.filename());
26 path = path.parent_path();
27 }
28 const fs::path start{path / parts.back() / "..."};
29 parts.pop_back();
30 const auto path_length = [](const int len, const fs::path &item)
31 { return static_cast<int>(item.filename().string().size()) + len + 1; };
32 const int remaining{length - static_cast<int>(start.string().size())};
33 while (std::accumulate(parts.begin(), parts.end(), 0, path_length) > remaining)
34 {
35 parts.pop_back();
36 }
37 const auto join_paths = [](const fs::path &result, const fs::path &item) { return result / item; };
38 return std::accumulate(parts.rbegin(), parts.rend(), start, join_paths).make_preferred().string();
39}
