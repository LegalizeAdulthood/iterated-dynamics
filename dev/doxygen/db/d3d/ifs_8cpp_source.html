68int ifs_load() // read in IFS parameters
69{
70 // release prior params
71 g_ifs_definition.clear();
72 g_ifs_dim = IFSDimension::TWO;
73 std::FILE *ifs_file;
74 if (find_file_item(g_ifs_filename, g_ifs_name, &ifs_file, ItemType::IFS))
75 {
76 return -1;
77 }
78
79 char buf[201];
80 file_gets(buf, 200, ifs_file);
81 char *buf_ptr = std::strchr(buf, ';');
82 if (buf_ptr != nullptr) // use ';' as comment to eol
83 {
84 *buf_ptr = 0;
85 }
86
87 string_lower(buf);
88 const int row_size = std::strstr(buf, "(3d)") != nullptr ? NUM_IFS_3D_PARAMS : NUM_IFS_2D_PARAMS;
89 g_ifs_dim = row_size == NUM_IFS_3D_PARAMS ? IFSDimension::THREE : IFSDimension::TWO;
90
91 int ret = 0;
92 int i = ret;
93 buf_ptr = get_ifs_token(buf, ifs_file);
94 while (buf_ptr != nullptr)
95 {
96 float value = 0.0f;
97 if (std::sscanf(buf_ptr, " %f ", &value) != 1)
98 {
99 break;
100 }
101 g_ifs_definition.push_back(value);
102 if (++i >= NUM_IFS_FUNCTIONS*row_size)
103 {
104 stop_msg("IFS definition has too many lines");
105 ret = -1;
106 break;
107 }
108 buf_ptr = get_next_ifs_token(buf, ifs_file);
109 if (buf_ptr == nullptr)
110 {
111 ret = -1;
112 break;
113 }
114 if (*buf_ptr == '}')
115 {
116 break;
117 }
118 }
119
120 if (i % row_size != 0 || (buf_ptr && *buf_ptr != '}'))
121 {
122 stop_msg("invalid IFS definition");
123 ret = -1;
124 }
125 if (i == 0 && ret == 0)
126 {
127 stop_msg("Empty IFS definition");
128 ret = -1;
129 }
130 std::fclose(ifs_file);
131
132 if (ret == 0)
133 {
134 g_num_affine_transforms = i/row_size;
135 }
136 return ret;
137}
