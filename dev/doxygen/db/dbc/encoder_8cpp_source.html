130// Save-To-Disk Routines (GIF)
131//
132// The following routines perform the GIF encoding when the 's' key is pressed.
133//
134// The compression logic in this file has been replaced by the classic
135// UNIX compress code. We have extensively modified the sources to fit
136// our needs, but have left the original credits where they
137// appear. Thanks to the original authors for making available these
138// classic and reliable sources. Of course, they are not responsible for
139// all the changes we have made to integrate their sources into this program.
140//
141// MEMORY ALLOCATION
142//
143// There are two large arrays:
144//
145// long htab[HSIZE] (5003*4 = 20012 bytes)
146// unsigned short codetab[HSIZE] (5003*2 = 10006 bytes)
147//
148// At the moment these arrays reuse extraseg and strlocn, respectively.
149//
150//
151
152// save-to-disk routine
153int save_image(std::filesystem::path &filename)
154{
155 std::filesystem::path open_file;
156 std::string open_file_ext;
157 std::filesystem::path tmp_file;
158 bool new_file;
159 int interrupted;
160
161restart:
162 s_save_16bit = g_disk_16_bit;
163 open_file = filename; // decode and open the filename
164 open_file_ext = DEFAULT_FRACTAL_TYPE; // determine the file extension
165 if (s_save_16bit)
166 {
167 open_file_ext = ".pot";
168 }
169
170 if (open_file.has_extension())
171 {
172 open_file_ext = open_file.extension().string();
173 open_file.replace_extension();
174 }
175 if (g_resave_flag != TimedSave::STARTED)
176 {
177 update_save_name(filename); // for next time
178 }
179
180 open_file.replace_extension(open_file_ext);
181
182 open_file = get_save_path(WriteFile::IMAGE, open_file.string());
183 assert(!open_file.empty());
184 tmp_file = open_file;
185 if (!std::filesystem::exists(open_file)) // file doesn't exist
186 {
187 new_file = true;
188 }
189 else
190 {
191 // file already exists
192 if (!g_overwrite_file)
193 {
194 if (g_resave_flag == TimedSave::NONE)
195 {
196 goto restart;
197 }
198 if (!g_started_resaves)
199 {
200 // first save of a savetime set
201 update_save_name(filename);
202 goto restart;
203 }
204 }
205 if (!is_writeable(open_file))
206 {
207 stop_msg(std::string{"Can't write "} + open_file.string());
208 return -1;
209 }
210 new_file = false;
211 tmp_file.replace_filename("id.tmp");
212 }
213
214 g_started_resaves = g_resave_flag == TimedSave::STARTED;
215 if (g_resave_flag == TimedSave::FINAL) // final save of savetime set?
216 {
217 g_resave_flag = TimedSave::NONE;
218 }
219
220 s_outfile = std::fopen(tmp_file.string().c_str(), "wb");
221 if (s_outfile == nullptr)
222 {
223 stop_msg("Can't create " + tmp_file.string());
224 return -1;
225 }
226
227 if (driver_is_disk())
228 {
229 // disk-video
230 dvid_status(1, "Saving " + open_file.filename().string());
231 }
232
233 g_busy = true;
234
235 if (g_debug_flag != DebugFlags::BENCHMARK_ENCODER)
236 {
237 interrupted = encoder() ? 1 : 0;
238 }
239 else
240 {
241 interrupted = encoder_timer(); // invoke encoder() via timer
242 }
243
244 g_busy = false;
245
246 std::fclose(s_outfile);
247
248 if (interrupted)
249 {
250 std::string buf{"Save of "};
251 buf += open_file.string();
252 buf += " interrupted.\nCancel to ";
253 if (new_file)
254 {
255 buf += "delete the file,\ncontinue to keep the partial image.";
256 }
257 else
258 {
259 buf += "retain the original file,\ncontinue to replace original with new partial image.";
260 }
261 interrupted = 1;
262 if (stop_msg(StopMsgFlags::CANCEL, buf))
263 {
264 interrupted = -1;
265 std::filesystem::remove(tmp_file);
266 }
267 }
268
269 if (!new_file && interrupted >= 0)
270 {
271 // replace the real file
272 std::filesystem::remove(open_file); // success assumed since we checked
273 std::filesystem::rename(tmp_file, open_file); // earlier with access
274 }
275
276 if (!driver_is_disk())
277 {
278 // supress this on disk-video
279 int out_color1 = s_out_color_1s;
280 int out_color2 = s_out_color_2s;
281 for (int j = 0; j <= s_last_color_bar; j++)
282 {
283 if ((j & 4) == 0)
284 {
285 if (++out_color1 >= g_colors)
286 {
287 out_color1 = 0;
288 }
289 if (++out_color2 >= g_colors)
290 {
291 out_color2 = 0;
292 }
293 }
294 for (int i = 0; 250*i < g_logical_screen.x_dots; i++)
295 {
296 // clear vert status bars
297 g_put_color(i, j, get_color(i, j) ^ out_color1);
298 g_put_color(g_logical_screen.x_dots - 1 - i, j,
299 get_color(g_logical_screen.x_dots - 1 - i, j) ^ out_color2);
300 }
301 }
302 }
303 else // disk-video
304 {
305 dvid_status(1, "");
306 }
307
308 if (interrupted)
309 {
310 text_temp_msg(" *interrupted* save ");
311 if (g_init_batch >= BatchMode::NORMAL)
312 {
313 g_init_batch = BatchMode::BAILOUT_ERROR_NO_SAVE; // if batch mode, set error level
314 }
315 return -1;
316 }
317 if (g_timed_save == TimedSave::NONE)
318 {
319 driver_buzzer(Buzzer::COMPLETE);
320 if (g_init_batch == BatchMode::NONE)
321 {
322 text_temp_msg((" File saved as " + open_file.filename().string() + ' ').c_str());
323 }
324 }
325 if (g_save_time_interval < 0)
326 {
327 goodbye();
328 }
329 return 0;
330}

332bool encoder()
333{
334 bool interrupted;
335 int width;
336 int row_limit;
337 Byte bits_per_pixel;
338 Byte x;
339 FractalInfo save_info;
340
341 setup_save_info(&save_info);
342
343 bits_per_pixel = 0; // calculate bits / pixel
344 for (int i = g_colors; i >= 2; i /= 2)
345 {
346 bits_per_pixel++;
347 }
348
349 s_start_bits = bits_per_pixel + 1;// start coding with this many bits
350 if (g_colors == 2)
351 {
352 s_start_bits++; // B&W Klooge
353 }
354
355 int i = 0;
356 if (std::fputs("GIF89a", s_outfile) < 0)
357 {
358 goto oops; // new GIF Signature
359 }
360
361 width = g_logical_screen.x_dots;
362 row_limit = g_logical_screen.y_dots;
363 if (s_save_16bit)
364 {
365 // pot16bit info is stored as: file: double width rows, right side
366 // of row is low 8 bits diskvid: ydots rows of colors followed by ydots
367 // rows of low 8 bits decoder: returns (row of color info then row of
368 // low 8 bits) * ydots
369 row_limit <<= 1;
370 width <<= 1;
371 }
372 if (std::fwrite(&width, 2, 1, s_outfile) != 1)
373 {
374 goto oops; // screen descriptor
375 }
376 if (std::fwrite(&g_logical_screen.y_dots, 2, 1, s_outfile) != 1)
377 {
378 goto oops;
379 }
380 static constexpr int COLOR_RESOLUTION{8};
381 x = static_cast<Byte>(128 + ((COLOR_RESOLUTION - 1) << 4) + (bits_per_pixel - 1));
382 if (std::fwrite(&x, 1, 1, s_outfile) != 1)
383
384 {
385 goto oops;
386 }
387 if (std::fputc(0, s_outfile) != 0)
388 {
389 goto oops; // background color
390 }
391
392 // TODO: pixel aspect ratio should be 1:1?
393 if (g_viewport.enabled // less than full screen?
394 && (g_viewport.x_dots == 0 || g_viewport.y_dots == 0)) // and we picked the dots?
395 {
396 i = static_cast<int>(static_cast<double>(g_screen_y_dots) / static_cast<double>(g_screen_x_dots) *
397 64.0 / g_screen_aspect -
398 14.5);
399 }
400 else // must risk loss of precision if numbers low
401 {
402 i = static_cast<int>(static_cast<double>(g_logical_screen.y_dots) /
403 static_cast<double>(g_logical_screen.x_dots) / g_viewport.final_aspect_ratio * 64 -
404 14.5);
405 }
406 i = std::max(i, 1);
407 i = std::min(i, 255);
408 if (std::fputc(i, s_outfile) != i)
409 {
410 goto oops; // pixel aspect ratio
411 }
412
413 if (g_colors == 256)
414 {
415 // write out the 256-color palette
416 if (g_got_real_dac)
417 {
418 // got a DAC - must be a VGA
419 if (!shift_write(reinterpret_cast<Byte *>(g_dac_box), g_colors))
420 {
421 goto oops;
422 }
423 }
424 else
425 {
426 // uh oh - better fake it
427 for (int j = 0; j < 256; j += 16)
428 {
429 if (!shift_write(s_palette_ega, 16))
430 {
431 goto oops;
432 }
433 }
434 }
435 }
436 if (g_colors == 2)
437 {
438 // write out the B&W palette
439 if (!shift_write(s_palette_bw, g_colors))
440 {
441 goto oops;
442 }
443 }
444 if (g_colors == 4)
445 {
446 // write out the CGA palette
447 if (!shift_write(s_palette_cga, g_colors))
448 {
449 goto oops;
450 }
451 }
452 if (g_colors == 16)
453 {
454 // Either EGA or VGA
455 if (g_got_real_dac)
456 {
457 if (!shift_write(reinterpret_cast<Byte *>(g_dac_box), g_colors))
458 {
459 goto oops;
460 }
461 }
462 else
463 {
464 // no DAC - must be an EGA
465 if (!shift_write(s_palette_ega, g_colors))
466 {
467 goto oops;
468 }
469 }
470 }
471
472 if (std::fputs(",", s_outfile) < 0)
473 {
474 goto oops; // Image Descriptor
475 }
476 i = 0;
477 if (std::fwrite(&i, 2, 1, s_outfile) != 1)
478 {
479 goto oops;
480 }
481 if (std::fwrite(&i, 2, 1, s_outfile) != 1)
482 {
483 goto oops;
484 }
485 if (std::fwrite(&width, 2, 1, s_outfile) != 1)
486 {
487 goto oops;
488 }
489 if (std::fwrite(&g_logical_screen.y_dots, 2, 1, s_outfile) != 1)
490 {
491 goto oops;
492 }
493 if (std::fwrite(&i, 1, 1, s_outfile) != 1)
494 {
495 goto oops;
496 }
497
498 bits_per_pixel = static_cast<Byte>(s_start_bits - 1);
499
500 if (std::fwrite(&bits_per_pixel, 1, 1, s_outfile) != 1)
501 {
502 goto oops;
503 }
504
505 interrupted = compress(row_limit);
506
507 if (std::ferror(s_outfile))
508 {
509 goto oops;
510 }
511
512 if (std::fputc(0, s_outfile) != 0)
513 {
514 goto oops;
515 }
516
517 // store non-standard fractal info
518 // loadfile has notes about extension block structure
519 if (interrupted)
520 {
521 save_info.calc_status = static_cast<short>(CalcStatus::PARAMS_CHANGED); // partial save is not resumable
522 }
523 save_info.tot_extend_len = 0;
524 if (!g_resume_data.empty() && save_info.calc_status == static_cast<short>(CalcStatus::RESUMABLE))
525 {
526 // resume info block, 002
527 save_info.tot_extend_len += extend_blk_len(g_resume_len);
528 std::copy(g_resume_data.data(), &g_resume_data[g_resume_len], &g_block[0]);
529 if (!put_extend_blk(2, g_resume_len, reinterpret_cast<char *>(g_block)))
530 {
531 goto oops;
532 }
533 }
534 // save_info.fractal_type gets modified in setup_save_info() in float only version, so we need to use fractype.
535 if (g_fractal_type == FractalType::FORMULA)
536 {
537 save_info.tot_extend_len += store_item_name(g_formula_name.c_str());
538 }
539 // if (save_info.fractal_type == LSYSTEM)
540 if (g_fractal_type == FractalType::L_SYSTEM)
541 {
542 save_info.tot_extend_len += store_item_name(g_l_system_name.c_str());
543 }
544 // if (save_info.fractal_type == IFS || save_info.fractal_type == IFS3D)
545 if (g_fractal_type == FractalType::IFS || g_fractal_type == FractalType::IFS_3D)
546 {
547 save_info.tot_extend_len += store_item_name(g_ifs_name.c_str());
548 }
549 if (g_display_3d <= Display3DMode::NONE && !g_iteration_ranges.empty())
550 {
551 // ranges block, 004
552 const int num_bytes = g_iteration_ranges.size() * 2;
553 save_info.tot_extend_len += extend_blk_len(num_bytes);
554 std::vector<char> buffer;
555 for (int range : g_iteration_ranges)
556 {
557 // ranges are stored as 16-bit ints in little-endian byte order
558 buffer.push_back(range & 0xFF);
559 buffer.push_back(static_cast<unsigned>(range & 0xFFFF) >> 8);
560 }
561 if (!put_extend_blk(4, num_bytes, buffer.data()))
562 {
563 goto oops;
564 }
565 }
566 // Extended parameters block 005
567 if (g_bf_math != BFMathType::NONE)
568 {
569 save_info.tot_extend_len += extend_blk_len(22 * (g_bf_length + 2));
570 // note: this assumes variables allocated in order starting with
571 // g_bf_x_min in init_bf_2() in BIGNUM.C
572 if (!put_extend_blk(5, 22 * (g_bf_length + 2), reinterpret_cast<char *>(g_bf_x_min)))
573 {
574 goto oops;
575 }
576 }
577
578 // Extended parameters block 006
579 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
580 {
581 EvolutionInfo evolution_info;
582 if (!g_have_evolve_info || g_calc_status == CalcStatus::COMPLETED)
583 {
584 evolution_info.x_parameter_range = g_evolve_x_parameter_range;
585 evolution_info.y_parameter_range = g_evolve_y_parameter_range;
586 evolution_info.x_parameter_offset = g_evolve_x_parameter_offset;
587 evolution_info.y_parameter_offset = g_evolve_y_parameter_offset;
588 evolution_info.discrete_x_parameter_offset =
589 static_cast<std::int16_t>(g_evolve_discrete_x_parameter_offset);
590 evolution_info.discrete_y_parameter_offset = static_cast<std::int16_t>(g_evolve_discrete_y_parameter_offset);
591 evolution_info.px = static_cast<std::int16_t>(g_evolve_param_grid_x);
592 evolution_info.py = static_cast<std::int16_t>(g_evolve_param_grid_y);
593 evolution_info.screen_x_offset = static_cast<std::int16_t>(g_logical_screen.x_offset);
594 evolution_info.screen_y_offset = static_cast<std::int16_t>(g_logical_screen.y_offset);
595 evolution_info.x_dots = static_cast<std::int16_t>(g_logical_screen.x_dots);
596 evolution_info.y_dots = static_cast<std::int16_t>(g_logical_screen.y_dots);
597 evolution_info.image_grid_size = static_cast<std::int16_t>(g_evolve_image_grid_size);
598 evolution_info.evolving = static_cast<std::int16_t>(+g_evolving);
599 evolution_info.this_generation_random_seed =
600 static_cast<std::uint16_t>(g_evolve_this_generation_random_seed);
601 evolution_info.max_random_mutation = g_evolve_max_random_mutation;
602 evolution_info.count =
603 static_cast<std::int16_t>(g_evolve_image_grid_size * g_evolve_image_grid_size); // flag for done
604 }
605 else
606 {
607 // we will need the resuming information
608 evolution_info.x_parameter_range = g_evolve_info.x_parameter_range;
609 evolution_info.y_parameter_range = g_evolve_info.y_parameter_range;
610 evolution_info.x_parameter_offset = g_evolve_info.x_parameter_offset;
611 evolution_info.y_parameter_offset = g_evolve_info.y_parameter_offset;
612 evolution_info.discrete_x_parameter_offset = g_evolve_info.discrete_x_parameter_offset;
613 evolution_info.discrete_y_parameter_offset = g_evolve_info.discrete_y_parameter_offset;
614 evolution_info.px = g_evolve_info.px;
615 evolution_info.py = g_evolve_info.py;
616 evolution_info.screen_x_offset = g_evolve_info.screen_x_offset;
617 evolution_info.screen_y_offset = g_evolve_info.screen_y_offset;
618 evolution_info.x_dots = g_evolve_info.x_dots;
619 evolution_info.y_dots = g_evolve_info.y_dots;
620 evolution_info.image_grid_size = g_evolve_info.image_grid_size;
621 evolution_info.evolving = g_evolve_info.evolving;
622 evolution_info.this_generation_random_seed = g_evolve_info.this_generation_random_seed;
623 evolution_info.max_random_mutation = g_evolve_info.max_random_mutation;
624 evolution_info.count = g_evolve_info.count;
625 }
626 for (int j = 0; j < NUM_GENES; j++)
627 {
628 evolution_info.mutate[j] = static_cast<std::int16_t>(g_gene_bank[j].mutate);
629 }
630
631 for (int j = 0; j < sizeof(evolution_info.future) / sizeof(std::int16_t); j++) // NOLINT(modernize-loop-convert)
632 {
633 evolution_info.future[j] = 0;
634 }
635
636 decode_evolver_info(&evolution_info, 0);
637 // evolution info block, 006
638 save_info.tot_extend_len += extend_blk_len(sizeof(evolution_info));
639 if (!put_extend_blk(6, sizeof(evolution_info), reinterpret_cast<char *>(&evolution_info)))
640 {
641 goto oops;
642 }
643 }
644
645 // Extended parameters block 007
646 if (g_std_calc_mode == CalcMode::ORBIT)
647 {
648 OrbitsInfo orbits_info{};
649 orbits_info.orbit_corner_min_x = g_orbit_corner.m_min.x;
650 orbits_info.orbit_corner_max_x = g_orbit_corner.m_max.x;
651 orbits_info.orbit_corner_min_y = g_orbit_corner.m_min.y;
652 orbits_info.orbit_corner_max_y = g_orbit_corner.m_max.y;
653 orbits_info.orbit_corner_3rd_x = g_orbit_corner.m_3rd.x;
654 orbits_info.orbit_corner_3rd_y = g_orbit_corner.m_3rd.y;
655 orbits_info.keep_screen_coords = static_cast<std::int16_t>(g_keep_screen_coords);
656 orbits_info.draw_mode = static_cast<char>(g_draw_mode);
657
658 // some big-endian logic for the doubles needed here
659 decode_orbits_info(&orbits_info, 0);
660 // orbits info block, 007
661 save_info.tot_extend_len += extend_blk_len(sizeof(orbits_info));
662 if (!put_extend_blk(7, sizeof(orbits_info), reinterpret_cast<char *>(&orbits_info)))
663 {
664 goto oops;
665 }
666 }
667
668 // main and last block, 001
669 save_info.tot_extend_len += extend_blk_len(sizeof(FractalInfo));
670 decode_fractal_info(&save_info, 0);
671 if (!put_extend_blk(1, sizeof(FractalInfo), reinterpret_cast<char *>(&save_info)))
672 {
673 goto oops;
674 }
675
676 if (std::fputs(";", s_outfile) < 0)
677 {
678 goto oops; // GIF Terminator
679 }
680
681 return interrupted;
682
683oops:
684 {
685 fflush(s_outfile);
686 stop_msg("Error Writing to disk (Disk full?)");
687 return true;
688 }

965 H_SIZE = 5003 // 80% occupancy

968// GIF Image compression - modified 'compress'
969//
970// Based on: compress.c - File compression ala IEEE Computer, June 1984.
971//
972// By Authors: Spencer W. Thomas
973// Jim McKie
974// Steve Davies
975// Ken Turkowski

987static int s_n_bits{}; // number of bits/code
988static int s_max_bits{BITS_F}; // user settable max # bits/code
989static int s_max_code{}; // maximum code, given n_bits
990static int s_max_max_cde{1 << BITS_F}; // should NEVER generate this code
991
992constexpr static int max_code(const int n_bits)
993{
994 return (1 << n_bits) - 1;
995}

997Byte g_block[4096]{};
998
999static long s_h_tab[H_SIZE]{};
1000static unsigned short s_code_tab[10240]{};
1001
1002static int s_free_ent{}; // first unused entry
1003
1004// block compression parameters -- after all codes are used up,
1005// and compression rate changes, start over.
1006static bool s_clear_flag{};
1007
1008//
1009// compress stdin to stdout
1010//
1011// Algorithm: use open addressing double hashing (no chaining) on the
1012// prefix code / next character combination. We do a variant of Knuth's
1013// algorithm D (vol. 3, sec. 6.4) along with G. Knott's relatively-prime
1014// secondary probe. Here, the modular division first probe is gives way
1015// to a faster exclusive-or manipulation. Also do block compression with
1016// an adaptive reset, whereby the code table is cleared when the compression
1017// ratio decreases, but after the table fills. The variable-length output
1018// codes are re-sized at this point, and a special CLEAR code is generated
1019// for the decompressor. Late addition: construct the table according to
1020// file size for noticeable speed improvement on small files. Please direct
1021// questions about this implementation to ames!jaw.
1022//
1023
1024static int s_clear_code{};
1025static int s_eof_code{};
1026static int s_a_count{}; // Number of characters so far in this 'packet'
1027static unsigned long s_cur_accum{};
1028static int s_cur_bits{};
1029
1030//
1031// Define the storage for the packet accumulator
1032//
1033static char s_accum[256]{};
1034
1035static bool compress(const int row_limit)
1036{
1037 int color;
1038 int in_count = 0;
1039 // TODO: why does this remain if it's never changed?
1040 const bool interrupted = false;
1041
1042 int out_color1 = 0; // use these colors to show progress
1043 int out_color2 = 1; // (this has nothing to do with GIF)
1044
1045 if (g_colors > 2)
1046 {
1047 out_color1 = 2;
1048 out_color2 = 3;
1049 }
1050 if ((++s_num_saves & 1) == 0)
1051 {
1052 // reverse the colors on alt saves
1053 int i = out_color1;
1054 out_color1 = out_color2;
1055 out_color2 = i;
1056 }
1057 s_out_color_1s = out_color1;
1058 s_out_color_2s = out_color2;
1059
1060 // Set up the necessary values
1061 s_cur_accum = 0;
1062 s_cur_bits = 0;
1063 s_clear_flag = false;
1064 int ent = 0;
1065 s_n_bits = s_start_bits;
1066 s_max_code = max_code(s_n_bits);
1067
1068 s_clear_code = 1 << (s_start_bits - 1);
1069 s_eof_code = s_clear_code + 1;
1070 s_free_ent = s_clear_code + 2;
1071
1072 s_a_count = 0;
1073 int h_shift = 0;
1074 for (long f_code = H_SIZE; f_code < 65536L; f_code *= 2L)
1075 {
1076 h_shift++;
1077 }
1078 h_shift = 8 - h_shift; // set hash code range bound
1079
1080 std::memset(s_h_tab, 0xff, static_cast<unsigned>(H_SIZE) *sizeof(long));
1081
1082 output(s_clear_code);
1083
1084 for (int row_num = 0; row_num < g_logical_screen.y_dots; row_num++)
1085 {
1086 // scan through the dots
1087 for (int y_dot = row_num; y_dot < row_limit; y_dot += g_logical_screen.y_dots)
1088 {
1089 for (int x_dot = 0; x_dot < g_logical_screen.x_dots; x_dot++)
1090 {
1091 if (s_save_16bit == 0 || y_dot < g_logical_screen.y_dots)
1092 {
1093 color = get_color(x_dot, y_dot);
1094 }
1095 else
1096 {
1097 color = disk_read_pixel(x_dot + g_logical_screen.x_offset, y_dot + g_logical_screen.y_offset);
1098 }
1099 if (in_count == 0)
1100 {
1101 in_count = 1;
1102 ent = color;
1103 continue;
1104 }
1105 const long f_code = (static_cast<long>(color) << s_max_bits) + ent;
1106 int i = color << h_shift ^ ent; // xor hashing
1107 int disp{};
1108
1109 if (s_h_tab[i] == f_code)
1110 {
1111 ent = s_code_tab[i];
1112 continue;
1113 }
1114 if (s_h_tab[i] < 0) // empty slot
1115 {
1116 goto no_match;
1117 }
1118 disp = H_SIZE - i; // secondary hash (after G. Knott)
1119 if (i == 0)
1120 {
1121 disp = 1;
1122 }
1123probe:
1124 if ((i -= disp) < 0)
1125 {
1126 i += H_SIZE;
1127 }
1128
1129 if (s_h_tab[i] == f_code)
1130 {
1131 ent = s_code_tab[i];
1132 continue;
1133 }
1134 if (s_h_tab[i] > 0)
1135 {
1136 goto probe;
1137 }
1138no_match:
1139 output(ent);
1140 ent = color;
1141 if (s_free_ent < s_max_max_cde)
1142 {
1143 // code -> hashtable
1144 s_code_tab[i] = static_cast<unsigned short>(s_free_ent++);
1145 s_h_tab[i] = f_code;
1146 }
1147 else
1148 {
1149 cl_block();
1150 }
1151 } // end for xdot
1152 if (! driver_is_disk() // supress this on disk-video
1153 && y_dot == row_num)
1154 {
1155 if ((y_dot & 4) == 0)
1156 {
1157 if (++out_color1 >= g_colors)
1158 {
1159 out_color1 = 0;
1160 }
1161 if (++out_color2 >= g_colors)
1162 {
1163 out_color2 = 0;
1164 }
1165 }
1166 for (int i = 0; 250*i < g_logical_screen.x_dots; i++)
1167 {
1168 // display vert status bars

1186// TAG(output)
1187//
1188// Output the given code.
1189// Inputs:
1190// code: A n_bits-bit integer. If == -1, then EOF. This assumes
1191// that n_bits =< (long)wordsize - 1.
1192// Outputs:
1193// Outputs code to the file.
1194// Assumptions:
1195// Chars are 8 bits long.
1196// Algorithm:
1197// Maintain a BITSF character long buffer (so that 8 codes will
1198// fit in it exactly). Use the VAX insv instruction to insert each
1199// code in turn. When the buffer fills up empty it and start over.
1200//
1201
1202static void output(const int code)
1203{
1204 static unsigned long masks[] =
1205 {
1206 0x0000, 0x0001, 0x0003, 0x0007, 0x000F,
1207 0x001F, 0x003F, 0x007F, 0x00FF,
1208 0x01FF, 0x03FF, 0x07FF, 0x0FFF,
1209 0x1FFF, 0x3FFF, 0x7FFF, 0xFFFF
1210 };
1211
1212 s_cur_accum &= masks[ s_cur_bits ];
1213
1214 if (s_cur_bits > 0)
1215 {
1216 s_cur_accum |= static_cast<long>(code) << s_cur_bits;
1217 }
1218 else
1219 {
1220 s_cur_accum = code;
1221 }
1222
1223 s_cur_bits += s_n_bits;
1224
1225 while (s_cur_bits >= 8)
1226 {
1227 char_out(static_cast<unsigned int>(s_cur_accum & 0xff));
1228 s_cur_accum >>= 8;
1229 s_cur_bits -= 8;
1230 }
1231
1232 // If the next entry is going to be too big for the code size,
1233 // then increase it, if possible.
1234 if (s_free_ent > s_max_code || s_clear_flag)
1235 {
1236 if (s_clear_flag)
1237 {
1238 s_n_bits = s_start_bits;
1239 s_max_code = max_code(s_n_bits);
1240 s_clear_flag = false;
1241 }
1242 else
1243 {
1244 s_n_bits++;
1245 if (s_n_bits == s_max_bits)
1246 {
1247 s_max_code = s_max_max_cde;
1248 }
1249 else
1250 {
1251 s_max_code = max_code(s_n_bits);
1252 }
1253 }
1254 }
1255
1256 if (code == s_eof_code)
1257 {
1258 // At EOF, write the rest of the buffer.
1259 while (s_cur_bits > 0)
1260 {
1261 char_out(static_cast<unsigned int>(s_cur_accum & 0xff));
1262 s_cur_accum >>= 8;
1263 s_cur_bits -= 8;
1264 }
1265
1266 flush_char();
1267
1268 fflush(s_outfile);
1269 }
1270}
