128// Save-To-Disk Routines (GIF)
129//
130// The following routines perform the GIF encoding when the 's' key is pressed.
131//
132// The compression logic in this file has been replaced by the classic
133// UNIX compress code. We have extensively modified the sources to fit
134// our needs, but have left the original credits where they
135// appear. Thanks to the original authors for making available these
136// classic and reliable sources. Of course, they are not responsible for
137// all the changes we have made to integrate their sources into this program.
138//
139// MEMORY ALLOCATION
140//
141// There are two large arrays:
142//
143// long htab[HSIZE] (5003*4 = 20012 bytes)
144// unsigned short codetab[HSIZE] (5003*2 = 10006 bytes)
145//
146// At the moment these arrays reuse extraseg and strlocn, respectively.
147//
148//
149
150// save-to-disk routine
151int save_image(std::filesystem::path &filename)
152{
153 std::filesystem::path open_file;
154 std::string open_file_ext;
155 std::filesystem::path tmp_file;
156 bool new_file;
157 int interrupted;
158
159restart:
160 s_save_16bit = g_disk_16_bit;
161 open_file = filename; // decode and open the filename
162 open_file_ext = DEFAULT_FRACTAL_TYPE; // determine the file extension
163 if (s_save_16bit)
164 {
165 open_file_ext = ".pot";
166 }
167
168 if (open_file.has_extension())
169 {
170 open_file_ext = open_file.extension().string();
171 open_file.replace_extension();
172 }
173 if (g_resave_flag != TimedSave::STARTED)
174 {
175 update_save_name(filename); // for next time
176 }
177
178 open_file.replace_extension(open_file_ext);
179
180 open_file = get_save_path(WriteFile::IMAGE, open_file.string());
181 assert(!open_file.empty());
182 tmp_file = open_file;
183 if (!std::filesystem::exists(open_file)) // file doesn't exist
184 {
185 new_file = true;
186 }
187 else
188 {
189 // file already exists
190 if (!g_overwrite_file)
191 {
192 if (g_resave_flag == TimedSave::NONE)
193 {
194 goto restart;
195 }
196 if (!g_started_resaves)
197 {
198 // first save of a savetime set
199 update_save_name(filename);
200 goto restart;
201 }
202 }
203 if (!is_writeable(open_file))
204 {
205 stop_msg(std::string{"Can't write "} + open_file.string());
206 return -1;
207 }
208 new_file = false;
209 tmp_file.replace_filename("id.tmp");
210 }
211
212 g_started_resaves = g_resave_flag == TimedSave::STARTED;
213 if (g_resave_flag == TimedSave::FINAL) // final save of savetime set?
214 {
215 g_resave_flag = TimedSave::NONE;
216 }
217
218 s_outfile = std::fopen(tmp_file.string().c_str(), "wb");
219 if (s_outfile == nullptr)
220 {
221 stop_msg("Can't create " + tmp_file.string());
222 return -1;
223 }
224
225 if (driver_is_disk())
226 {
227 // disk-video
228 dvid_status(1, "Saving " + open_file.filename().string());
229 }
230
231 g_busy = true;
232
233 if (g_debug_flag != DebugFlags::BENCHMARK_ENCODER)
234 {
235 interrupted = encoder() ? 1 : 0;
236 }
237 else
238 {
239 interrupted = encoder_timer(); // invoke encoder() via timer
240 }
241
242 g_busy = false;
243
244 std::fclose(s_outfile);
245
246 if (interrupted)
247 {
248 std::string buf{"Save of "};
249 buf += open_file.string();
250 buf += " interrupted.\nCancel to ";
251 if (new_file)
252 {
253 buf += "delete the file,\ncontinue to keep the partial image.";
254 }
255 else
256 {
257 buf += "retain the original file,\ncontinue to replace original with new partial image.";
258 }
259 interrupted = 1;
260 if (stop_msg(StopMsgFlags::CANCEL, buf))
261 {
262 interrupted = -1;
263 std::filesystem::remove(tmp_file);
264 }
265 }
266
267 if (!new_file && interrupted >= 0)
268 {
269 // replace the real file
270 std::filesystem::remove(open_file); // success assumed since we checked
271 std::filesystem::rename(tmp_file, open_file); // earlier with access
272 }
273
274 if (!driver_is_disk())
275 {
276 // supress this on disk-video
277 int out_color1 = s_out_color_1s;
278 int out_color2 = s_out_color_2s;
279 for (int j = 0; j <= s_last_color_bar; j++)
280 {
281 if ((j & 4) == 0)
282 {
283 if (++out_color1 >= g_colors)
284 {
285 out_color1 = 0;
286 }
287 if (++out_color2 >= g_colors)
288 {
289 out_color2 = 0;
290 }
291 }
292 for (int i = 0; 250*i < g_logical_screen.x_dots; i++)
293 {
294 // clear vert status bars
295 g_put_color(i, j, get_color(i, j) ^ out_color1);
296 g_put_color(g_logical_screen.x_dots - 1 - i, j,
297 get_color(g_logical_screen.x_dots - 1 - i, j) ^ out_color2);
298 }
299 }
300 }
301 else // disk-video
302 {
303 dvid_status(1, "");
304 }
305
306 if (interrupted)
307 {
308 text_temp_msg(" *interrupted* save ");
309 if (g_init_batch >= BatchMode::NORMAL)
310 {
311 g_init_batch = BatchMode::BAILOUT_ERROR_NO_SAVE; // if batch mode, set error level
312 }
313 return -1;
314 }
315 if (g_timed_save == TimedSave::NONE)
316 {
317 driver_buzzer(Buzzer::COMPLETE);
318 if (g_init_batch == BatchMode::NONE)
319 {
320 text_temp_msg((" File saved as " + open_file.filename().string() + ' ').c_str());
321 }
322 }
323 if (g_save_time_interval < 0)
324 {
325 goodbye();
326 }
327 return 0;
328}

330bool encoder()
331{
332 bool interrupted;
333 int width;
334 int row_limit;
335 Byte bits_per_pixel;
336 Byte x;
337 FractalInfo save_info;
338
339 setup_save_info(&save_info);
340
341 bits_per_pixel = 0; // calculate bits / pixel
342 for (int i = g_colors; i >= 2; i /= 2)
343 {
344 bits_per_pixel++;
345 }
346
347 s_start_bits = bits_per_pixel + 1;// start coding with this many bits
348 if (g_colors == 2)
349 {
350 s_start_bits++; // B&W Klooge
351 }
352
353 int i = 0;
354 if (std::fputs("GIF89a", s_outfile) < 0)
355 {
356 goto oops; // new GIF Signature
357 }
358
359 width = g_logical_screen.x_dots;
360 row_limit = g_logical_screen.y_dots;
361 if (s_save_16bit)
362 {
363 // pot16bit info is stored as: file: double width rows, right side
364 // of row is low 8 bits diskvid: ydots rows of colors followed by ydots
365 // rows of low 8 bits decoder: returns (row of color info then row of
366 // low 8 bits) * ydots
367 row_limit <<= 1;
368 width <<= 1;
369 }
370 if (std::fwrite(&width, 2, 1, s_outfile) != 1)
371 {
372 goto oops; // screen descriptor
373 }
374 if (std::fwrite(&g_logical_screen.y_dots, 2, 1, s_outfile) != 1)
375 {
376 goto oops;
377 }
378 static constexpr int COLOR_RESOLUTION{8};
379 x = static_cast<Byte>(128 + ((COLOR_RESOLUTION - 1) << 4) + (bits_per_pixel - 1));
380 if (std::fwrite(&x, 1, 1, s_outfile) != 1)
381
382 {
383 goto oops;
384 }
385 if (std::fputc(0, s_outfile) != 0)
386 {
387 goto oops; // background color
388 }
389
390 // TODO: pixel aspect ratio should be 1:1?
391 if (g_viewport.enabled // less than full screen?
392 && (g_viewport.x_dots == 0 || g_viewport.y_dots == 0)) // and we picked the dots?
393 {
394 i = static_cast<int>(static_cast<double>(g_screen_y_dots) / static_cast<double>(g_screen_x_dots) *
395 64.0 / g_screen_aspect -
396 14.5);
397 }
398 else // must risk loss of precision if numbers low
399 {
400 i = static_cast<int>(static_cast<double>(g_logical_screen.y_dots) /
401 static_cast<double>(g_logical_screen.x_dots) / g_viewport.final_aspect_ratio * 64 -
402 14.5);
403 }
404 i = std::max(i, 1);
405 i = std::min(i, 255);
406 if (std::fputc(i, s_outfile) != i)
407 {
408 goto oops; // pixel aspect ratio
409 }
410
411 if (g_colors == 256)
412 {
413 // write out the 256-color palette
414 if (g_got_real_dac)
415 {
416 // got a DAC - must be a VGA
417 if (!shift_write(reinterpret_cast<Byte *>(g_dac_box), g_colors))
418 {
419 goto oops;
420 }
421 }
422 else
423 {
424 // uh oh - better fake it
425 for (int j = 0; j < 256; j += 16)
426 {
427 if (!shift_write(s_palette_ega, 16))
428 {
429 goto oops;
430 }
431 }
432 }
433 }
434 if (g_colors == 2)
435 {
436 // write out the B&W palette
437 if (!shift_write(s_palette_bw, g_colors))
438 {
439 goto oops;
440 }
441 }
442 if (g_colors == 4)
443 {
444 // write out the CGA palette
445 if (!shift_write(s_palette_cga, g_colors))
446 {
447 goto oops;
448 }
449 }
450 if (g_colors == 16)
451 {
452 // Either EGA or VGA
453 if (g_got_real_dac)
454 {
455 if (!shift_write(reinterpret_cast<Byte *>(g_dac_box), g_colors))
456 {
457 goto oops;
458 }
459 }
460 else
461 {
462 // no DAC - must be an EGA
463 if (!shift_write(s_palette_ega, g_colors))
464 {
465 goto oops;
466 }
467 }
468 }
469
470 if (std::fputs(",", s_outfile) < 0)
471 {
472 goto oops; // Image Descriptor
473 }
474 i = 0;
475 if (std::fwrite(&i, 2, 1, s_outfile) != 1)
476 {
477 goto oops;
478 }
479 if (std::fwrite(&i, 2, 1, s_outfile) != 1)
480 {
481 goto oops;
482 }
483 if (std::fwrite(&width, 2, 1, s_outfile) != 1)
484 {
485 goto oops;
486 }
487 if (std::fwrite(&g_logical_screen.y_dots, 2, 1, s_outfile) != 1)
488 {
489 goto oops;
490 }
491 if (std::fwrite(&i, 1, 1, s_outfile) != 1)
492 {
493 goto oops;
494 }
495
496 bits_per_pixel = static_cast<Byte>(s_start_bits - 1);
497
498 if (std::fwrite(&bits_per_pixel, 1, 1, s_outfile) != 1)
499 {
500 goto oops;
501 }
502
503 interrupted = compress(row_limit);
504
505 if (std::ferror(s_outfile))
506 {
507 goto oops;
508 }
509
510 if (std::fputc(0, s_outfile) != 0)
511 {
512 goto oops;
513 }
514
515 // store non-standard fractal info
516 // loadfile has notes about extension block structure
517 if (interrupted)
518 {
519 save_info.calc_status = static_cast<short>(CalcStatus::PARAMS_CHANGED); // partial save is not resumable
520 }
521 save_info.tot_extend_len = 0;
522 if (!g_resume_data.empty() && save_info.calc_status == static_cast<short>(CalcStatus::RESUMABLE))
523 {
524 // resume info block, 002
525 save_info.tot_extend_len += extend_blk_len(g_resume_len);
526 std::copy(g_resume_data.data(), &g_resume_data[g_resume_len], &g_block[0]);
527 if (!put_extend_blk(2, g_resume_len, reinterpret_cast<char *>(g_block)))
528 {
529 goto oops;
530 }
531 }
532 // save_info.fractal_type gets modified in setup_save_info() in float only version, so we need to use fractype.
533 if (g_fractal_type == FractalType::FORMULA)
534 {
535 save_info.tot_extend_len += store_item_name(g_formula_name.c_str());
536 }
537 // if (save_info.fractal_type == LSYSTEM)
538 if (g_fractal_type == FractalType::L_SYSTEM)
539 {
540 save_info.tot_extend_len += store_item_name(g_l_system_name.c_str());
541 }
542 // if (save_info.fractal_type == IFS || save_info.fractal_type == IFS3D)
543 if (g_fractal_type == FractalType::IFS || g_fractal_type == FractalType::IFS_3D)
544 {
545 save_info.tot_extend_len += store_item_name(g_ifs_name.c_str());
546 }
547 if (g_display_3d <= Display3DMode::NONE && !g_iteration_ranges.empty())
548 {
549 // ranges block, 004
550 const int num_bytes = g_iteration_ranges.size() * 2;
551 save_info.tot_extend_len += extend_blk_len(num_bytes);
552 std::vector<char> buffer;
553 for (int range : g_iteration_ranges)
554 {
555 // ranges are stored as 16-bit ints in little-endian byte order
556 buffer.push_back(range & 0xFF);
557 buffer.push_back(static_cast<unsigned>(range & 0xFFFF) >> 8);
558 }
559 if (!put_extend_blk(4, num_bytes, buffer.data()))
560 {
561 goto oops;
562 }
563 }
564 // Extended parameters block 005
565 if (g_bf_math != BFMathType::NONE)
566 {
567 save_info.tot_extend_len += extend_blk_len(22 * (g_bf_length + 2));
568 // note: this assumes variables allocated in order starting with
569 // g_bf_x_min in init_bf_2() in BIGNUM.C
570 if (!put_extend_blk(5, 22 * (g_bf_length + 2), reinterpret_cast<char *>(g_bf_x_min)))
571 {
572 goto oops;
573 }
574 }
575
576 // Extended parameters block 006
577 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
578 {
579 EvolutionInfo evolution_info;
580 if (!g_have_evolve_info || g_calc_status == CalcStatus::COMPLETED)
581 {
582 evolution_info.x_parameter_range = g_evolve_x_parameter_range;
583 evolution_info.y_parameter_range = g_evolve_y_parameter_range;
584 evolution_info.x_parameter_offset = g_evolve_x_parameter_offset;
585 evolution_info.y_parameter_offset = g_evolve_y_parameter_offset;
586 evolution_info.discrete_x_parameter_offset =
587 static_cast<std::int16_t>(g_evolve_discrete_x_parameter_offset);
588 evolution_info.discrete_y_parameter_offset = static_cast<std::int16_t>(g_evolve_discrete_y_parameter_offset);
589 evolution_info.px = static_cast<std::int16_t>(g_evolve_param_grid_x);
590 evolution_info.py = static_cast<std::int16_t>(g_evolve_param_grid_y);
591 evolution_info.screen_x_offset = static_cast<std::int16_t>(g_logical_screen.x_offset);
592 evolution_info.screen_y_offset = static_cast<std::int16_t>(g_logical_screen.y_offset);
593 evolution_info.x_dots = static_cast<std::int16_t>(g_logical_screen.x_dots);
594 evolution_info.y_dots = static_cast<std::int16_t>(g_logical_screen.y_dots);
595 evolution_info.image_grid_size = static_cast<std::int16_t>(g_evolve_image_grid_size);
596 evolution_info.evolving = static_cast<std::int16_t>(+g_evolving);
597 evolution_info.this_generation_random_seed =
598 static_cast<std::uint16_t>(g_evolve_this_generation_random_seed);
599 evolution_info.max_random_mutation = g_evolve_max_random_mutation;
600 evolution_info.count =
601 static_cast<std::int16_t>(g_evolve_image_grid_size * g_evolve_image_grid_size); // flag for done
602 }
603 else
604 {
605 // we will need the resuming information
606 evolution_info.x_parameter_range = g_evolve_info.x_parameter_range;
607 evolution_info.y_parameter_range = g_evolve_info.y_parameter_range;
608 evolution_info.x_parameter_offset = g_evolve_info.x_parameter_offset;
609 evolution_info.y_parameter_offset = g_evolve_info.y_parameter_offset;
610 evolution_info.discrete_x_parameter_offset = g_evolve_info.discrete_x_parameter_offset;
611 evolution_info.discrete_y_parameter_offset = g_evolve_info.discrete_y_parameter_offset;
612 evolution_info.px = g_evolve_info.px;
613 evolution_info.py = g_evolve_info.py;
614 evolution_info.screen_x_offset = g_evolve_info.screen_x_offset;
615 evolution_info.screen_y_offset = g_evolve_info.screen_y_offset;
616 evolution_info.x_dots = g_evolve_info.x_dots;
617 evolution_info.y_dots = g_evolve_info.y_dots;
618 evolution_info.image_grid_size = g_evolve_info.image_grid_size;
619 evolution_info.evolving = g_evolve_info.evolving;
620 evolution_info.this_generation_random_seed = g_evolve_info.this_generation_random_seed;
621 evolution_info.max_random_mutation = g_evolve_info.max_random_mutation;
622 evolution_info.count = g_evolve_info.count;
623 }
624 for (int j = 0; j < NUM_GENES; j++)
625 {
626 evolution_info.mutate[j] = static_cast<std::int16_t>(g_gene_bank[j].mutate);
627 }
628
629 for (int j = 0; j < sizeof(evolution_info.future) / sizeof(std::int16_t); j++) // NOLINT(modernize-loop-convert)
630 {
631 evolution_info.future[j] = 0;
632 }
633
634 decode_evolver_info(&evolution_info, 0);
635 // evolution info block, 006
636 save_info.tot_extend_len += extend_blk_len(sizeof(evolution_info));
637 if (!put_extend_blk(6, sizeof(evolution_info), reinterpret_cast<char *>(&evolution_info)))
638 {
639 goto oops;
640 }
641 }
642
643 // Extended parameters block 007
644 if (g_std_calc_mode == CalcMode::ORBIT)
645 {
646 OrbitsInfo orbits_info{};
647 orbits_info.orbit_corner_min_x = g_orbit_corner.m_min.x;
648 orbits_info.orbit_corner_max_x = g_orbit_corner.m_max.x;
649 orbits_info.orbit_corner_min_y = g_orbit_corner.m_min.y;
650 orbits_info.orbit_corner_max_y = g_orbit_corner.m_max.y;
651 orbits_info.orbit_corner_3rd_x = g_orbit_corner.m_3rd.x;
652 orbits_info.orbit_corner_3rd_y = g_orbit_corner.m_3rd.y;
653 orbits_info.keep_screen_coords = static_cast<std::int16_t>(g_keep_screen_coords);
654 orbits_info.draw_mode = static_cast<char>(g_draw_mode);
655
656 // some big-endian logic for the doubles needed here
657 decode_orbits_info(&orbits_info, 0);
658 // orbits info block, 007
659 save_info.tot_extend_len += extend_blk_len(sizeof(orbits_info));
660 if (!put_extend_blk(7, sizeof(orbits_info), reinterpret_cast<char *>(&orbits_info)))
661 {
662 goto oops;
663 }
664 }
665
666 // main and last block, 001
667 save_info.tot_extend_len += extend_blk_len(sizeof(FractalInfo));
668 decode_fractal_info(&save_info, 0);
669 if (!put_extend_blk(1, sizeof(FractalInfo), reinterpret_cast<char *>(&save_info)))
670 {
671 goto oops;
672 }
673
674 if (std::fputs(";", s_outfile) < 0)
675 {
676 goto oops; // GIF Terminator
677 }
678
679 return interrupted;
680
681oops:
682 {
683 fflush(s_outfile);
684 stop_msg("Error Writing to disk (Disk full?)");
685 return true;
686 }
687}

689// shift IBM colors to GIF
690static int shift_write(const Byte *color, const int num_colors)
691{
692 for (int i = 0; i < num_colors; i++)
693 {
694 for (int j = 0; j < 3; j++)
695 {
696 Byte this_color = color[3 * i + j];
697 if (std::fputc(this_color, s_outfile) != static_cast<int>(this_color))
698 {
699 return 0;
700 }
701 }
702 }
703 return 1;
704}

712static int put_extend_blk(const int block_id, int block_len, const char *block_data)
713{
714 int j;
715 char header[15];
716 std::strcpy(header, "!\377\013fractint");
717 std::sprintf(&header[11], "%03d", block_id);
718 if (std::fwrite(header, 14, 1, s_outfile) != 1)
719 {
720 return 0;
721 }
722 int i = (block_len + 254) / 255;
723 while (--i >= 0)
724 {
725 block_len -= j = std::min(block_len, 255);
726 if (std::fputc(j, s_outfile) != j)
727 {
728 return 0;
729 }
730 while (--j >= 0)
731 {
732 std::fputc(*block_data++, s_outfile);
733 }
734 }
735 if (std::fputc(0, s_outfile) != 0)
736 {
737 return 0;
738 }
739 return 1;
740}

937//
938// GIFENCOD.C - GIF Image compression routines
939//
940// Lempel-Ziv compression based on 'compress'. GIF modifications by
941// David Rowley (mgardi@watdcsu.waterloo.edu).
942// Thoroughly massaged by the Stone Soup team for our purposes.
943//
944//
945
946enum
947{
948 BITS_F = 12,
949 H_SIZE = 5003 // 80% occupancy
950};
951
952// GIF Image compression - modified 'compress'
953//
954// Based on: compress.c - File compression ala IEEE Computer, June 1984.
955//
956// By Authors: Spencer W. Thomas
957// Jim McKie
958// Steve Davies
959// Ken Turkowski
960// James A. Woods
961// Joe Orost
962//
963
964// prototypes
965
966static void output(int code);
967static void char_out(int c);
968static void flush_char();
969static void cl_block();
970
971static int s_n_bits{}; // number of bits/code
972static int s_max_bits{BITS_F}; // user settable max # bits/code
973static int s_max_code{}; // maximum code, given n_bits
974static int s_max_max_cde{1 << BITS_F}; // should NEVER generate this code
975
976constexpr static int max_code(const int n_bits)
977{
978 return (1 << n_bits) - 1;
979}
980
981Byte g_block[4096]{};
982
983static long s_h_tab[H_SIZE]{};
984static unsigned short s_code_tab[10240]{};
985
986static int s_free_ent{}; // first unused entry
987
988// block compression parameters -- after all codes are used up,
989// and compression rate changes, start over.
990static bool s_clear_flag{};
991
992//
993// compress stdin to stdout
994//
995// Algorithm: use open addressing double hashing (no chaining) on the
996// prefix code / next character combination. We do a variant of Knuth's
997// algorithm D (vol. 3, sec. 6.4) along with G. Knott's relatively-prime
998// secondary probe. Here, the modular division first probe is gives way
999// to a faster exclusive-or manipulation. Also do block compression with
1000// an adaptive reset, whereby the code table is cleared when the compression
1001// ratio decreases, but after the table fills. The variable-length output
1002// codes are re-sized at this point, and a special CLEAR code is generated
1003// for the decompressor. Late addition: construct the table according to
1004// file size for noticeable speed improvement on small files. Please direct
1005// questions about this implementation to ames!jaw.
1006//
1007
1008static int s_clear_code{};
1009static int s_eof_code{};
1010static int s_a_count{}; // Number of characters so far in this 'packet'
1011static unsigned long s_cur_accum{};
1012static int s_cur_bits{};
1013
1014//
1015// Define the storage for the packet accumulator
1016//
1017static char s_accum[256]{};
1018
1019static bool compress(const int row_limit)
1020{
1021 int color;
1022 int in_count = 0;
1023 // TODO: why does this remain if it's never changed?
1024 const bool interrupted = false;
1025
1026 int out_color1 = 0; // use these colors to show progress
1027 int out_color2 = 1; // (this has nothing to do with GIF)
1028
1029 if (g_colors > 2)
1030 {
1031 out_color1 = 2;
1032 out_color2 = 3;
1033 }
1034 if ((++s_num_saves & 1) == 0)
1035 {
1036 // reverse the colors on alt saves
1037 int i = out_color1;
1038 out_color1 = out_color2;
1039 out_color2 = i;
1040 }
1041 s_out_color_1s = out_color1;
1042 s_out_color_2s = out_color2;
1043
1044 // Set up the necessary values
1045 s_cur_accum = 0;
1046 s_cur_bits = 0;
1047 s_clear_flag = false;
1048 int ent = 0;
1049 s_n_bits = s_start_bits;
1050 s_max_code = max_code(s_n_bits);
1051
1052 s_clear_code = 1 << (s_start_bits - 1);
1053 s_eof_code = s_clear_code + 1;
1054 s_free_ent = s_clear_code + 2;
1055
1056 s_a_count = 0;
1057 int h_shift = 0;
1058 for (long f_code = H_SIZE; f_code < 65536L; f_code *= 2L)
1059 {
1060 h_shift++;
1061 }
1062 h_shift = 8 - h_shift; // set hash code range bound
1063
1064 std::memset(s_h_tab, 0xff, static_cast<unsigned>(H_SIZE) *sizeof(long));
1065
1066 output(s_clear_code);
1067
1068 for (int row_num = 0; row_num < g_logical_screen.y_dots; row_num++)
1069 {
1070 // scan through the dots
1071 for (int y_dot = row_num; y_dot < row_limit; y_dot += g_logical_screen.y_dots)
1072 {
1073 for (int x_dot = 0; x_dot < g_logical_screen.x_dots; x_dot++)
1074 {
1075 if (s_save_16bit == 0 || y_dot < g_logical_screen.y_dots)
1076 {
1077 color = get_color(x_dot, y_dot);
1078 }
1079 else
1080 {
1081 color = disk_read_pixel(x_dot + g_logical_screen.x_offset, y_dot + g_logical_screen.y_offset);
1082 }
1083 if (in_count == 0)
1084 {
1085 in_count = 1;
1086 ent = color;
1087 continue;
1088 }
1089 const long f_code = (static_cast<long>(color) << s_max_bits) + ent;
1090 int i = color << h_shift ^ ent; // xor hashing
1091 int disp{};
1092
1093 if (s_h_tab[i] == f_code)
1094 {
1095 ent = s_code_tab[i];
1096 continue;
1097 }
1098 if (s_h_tab[i] < 0) // empty slot
1099 {
1100 goto no_match;
1101 }
1102 disp = H_SIZE - i; // secondary hash (after G. Knott)
1103 if (i == 0)
1104 {
1105 disp = 1;
1106 }
1107probe:
1108 if ((i -= disp) < 0)
1109 {
1110 i += H_SIZE;
1111 }
1112
1113 if (s_h_tab[i] == f_code)
1114 {
1115 ent = s_code_tab[i];
1116 continue;
1117 }
1118 if (s_h_tab[i] > 0)
1119 {
1120 goto probe;
1121 }
1122no_match:
1123 output(ent);
1124 ent = color;
1125 if (s_free_ent < s_max_max_cde)
1126 {
1127 // code -> hashtable
1128 s_code_tab[i] = static_cast<unsigned short>(s_free_ent++);
1129 s_h_tab[i] = f_code;
1130 }
1131 else
1132 {
1133 cl_block();
1134 }
1135 } // end for xdot
1136 if (! driver_is_disk() // supress this on disk-video
1137 && y_dot == row_num)
1138 {
1139 if ((y_dot & 4) == 0)
1140 {
1141 if (++out_color1 >= g_colors)
1142 {
1143 out_color1 = 0;
1144 }
1145 if (++out_color2 >= g_colors)
1146 {
1147 out_color2 = 0;
1148 }
1149 }
1150 for (int i = 0; 250*i < g_logical_screen.x_dots; i++)
1151 {
1152 // display vert status bars
1153 // (this is NOT GIF-related)
1154 g_put_color(i, y_dot, get_color(i, y_dot) ^ out_color1);
1155 g_put_color(g_logical_screen.x_dots - 1 - i, y_dot,
1156 get_color(g_logical_screen.x_dots - 1 - i, y_dot) ^ out_color2);
1157 }
1158 s_last_color_bar = y_dot;
1159 } // end if !driver_diskp()
1160 } // end for ydot
1161 } // end for rownum
1162
1163 // Put out the final code.
1164 output(ent);
1165 output(s_eof_code);
1166 return interrupted;
1167}

1172// Output the given code.
1173// Inputs:
1174// code: A n_bits-bit integer. If == -1, then EOF. This assumes
1175// that n_bits =< (long)wordsize - 1.
1176// Outputs:
1177// Outputs code to the file.
1178// Assumptions:
1179// Chars are 8 bits long.
1180// Algorithm:
1181// Maintain a BITSF character long buffer (so that 8 codes will
1182// fit in it exactly). Use the VAX insv instruction to insert each
1183// code in turn. When the buffer fills up empty it and start over.
1184//
1185
1186static void output(const int code)
1187{
1188 static unsigned long masks[] =
1189 {
1190 0x0000, 0x0001, 0x0003, 0x0007, 0x000F,
1191 0x001F, 0x003F, 0x007F, 0x00FF,
1192 0x01FF, 0x03FF, 0x07FF, 0x0FFF,
1193 0x1FFF, 0x3FFF, 0x7FFF, 0xFFFF
1194 };
1195
1196 s_cur_accum &= masks[ s_cur_bits ];
1197
1198 if (s_cur_bits > 0)
1199 {
1200 s_cur_accum |= static_cast<long>(code) << s_cur_bits;
1201 }
1202 else
1203 {
1204 s_cur_accum = code;
1205 }
1206
1207 s_cur_bits += s_n_bits;
1208
1209 while (s_cur_bits >= 8)
1210 {
1211 char_out(static_cast<unsigned int>(s_cur_accum & 0xff));
1212 s_cur_accum >>= 8;
1213 s_cur_bits -= 8;
1214 }
1215
1216 // If the next entry is going to be too big for the code size,
1217 // then increase it, if possible.
1218 if (s_free_ent > s_max_code || s_clear_flag)
1219 {
1220 if (s_clear_flag)
1221 {
1222 s_n_bits = s_start_bits;
1223 s_max_code = max_code(s_n_bits);
1224 s_clear_flag = false;
1225 }
1226 else
1227 {
1228 s_n_bits++;
1229 if (s_n_bits == s_max_bits)
1230 {
1231 s_max_code = s_max_max_cde;
1232 }
1233 else
1234 {
1235 s_max_code = max_code(s_n_bits);
1236 }
1237 }
1238 }
1239
1240 if (code == s_eof_code)
1241 {
1242 // At EOF, write the rest of the buffer.
1243 while (s_cur_bits > 0)
1244 {
1245 char_out(static_cast<unsigned int>(s_cur_accum & 0xff));
1246 s_cur_accum >>= 8;
1247 s_cur_bits -= 8;
1248 }
1249
1250 flush_char();
1251
1252 fflush(s_outfile);
1253 }
1254}
