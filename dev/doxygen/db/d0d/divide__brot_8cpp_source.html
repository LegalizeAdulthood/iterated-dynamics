25int divide_brot5_bn_per_pixel()
26{
27 /* parm.x = xxmin + g_col*delx + g_row*delx2 */
28 mult_bn_int(g_param_z_bn.x, g_delta_x_bn, static_cast<U16>(g_col));
29 mult_bn_int(g_bn_tmp, g_delta2_x_bn, static_cast<U16>(g_row));
30
31 add_a_bn(g_param_z_bn.x, g_bn_tmp);
32 add_a_bn(g_param_z_bn.x, g_x_min_bn);
33
34 /* parm.y = yymax - g_row*dely - g_col*dely2; */
35 /* note: in next four lines, g_old_z_bn is just used as a temporary variable */
36 mult_bn_int(g_old_z_bn.x, g_delta_y_bn, static_cast<U16>(g_row));
37 mult_bn_int(g_old_z_bn.y, g_delta2_y_bn, static_cast<U16>(g_col));
38 add_a_bn(g_old_z_bn.x, g_old_z_bn.y);
39 sub_bn(g_param_z_bn.y, g_y_max_bn, g_old_z_bn.x);
40
41 clear_bn(g_tmp_sqr_x_bn);
42 clear_bn(g_tmp_sqr_y_bn);
43 clear_bn(g_old_z_bn.x);
44 clear_bn(g_old_z_bn.y);
45
46 return 0; /* 1st iteration has NOT been done */
47}

73int divide_brot5_bn_fractal()
74{
75 BigStackSaver saved;
76 BNComplex bn_tmp_new;
77 BNComplex bn_numer;
78 BNComplex bn_c_exp;
79
80 bn_tmp_new.x = alloc_stack(g_r_length);
81 bn_tmp_new.y = alloc_stack(g_r_length);
82 bn_numer.x = alloc_stack(g_r_length);
83 bn_numer.y = alloc_stack(g_r_length);
84 bn_c_exp.x = alloc_stack(g_bn_length);
85 bn_c_exp.y = alloc_stack(g_bn_length);
86 BigNum tmp1 = alloc_stack(g_bn_length);
87 BigNum tmp2 = alloc_stack(g_r_length);
88
89 /* g_tmp_sqr_x_bn and g_tmp_sqr_y_bn were previously squared before getting to */
90 /* this function, so they must be shifted. */
91
92 /* sqr(z) */
93 /* bnnumer.x = g_tmp_sqr_x_bn - g_tmp_sqr_y_bn; */
94 sub_bn(bn_numer.x, g_tmp_sqr_x_bn + g_shift_factor, g_tmp_sqr_y_bn + g_shift_factor);
95
96 /* bnnumer.y = 2 * g_old_z_bn.x * g_old_z_bn.y; */
97 mult_bn(tmp2, g_old_z_bn.x, g_old_z_bn.y);
98 double_a_bn(tmp2 + g_shift_factor);
99 copy_bn(bn_numer.y, tmp2 + g_shift_factor);
100
101 /* z^(a) */
102 int_to_bn(bn_c_exp.x, g_c_exponent);
103 clear_bn(bn_c_exp.y);
104 cmplx_pow_bn(&bn_tmp_new, &g_old_z_bn, &bn_c_exp);
105 /* then add b */
106 float_to_bn(tmp1, g_b_const);
107 add_bn(bn_tmp_new.x, tmp1, bn_tmp_new.x + g_shift_factor);
108 /* need to g_shift_factor bntmpnew.y */
109 copy_bn(tmp2, bn_tmp_new.y + g_shift_factor);
110 copy_bn(bn_tmp_new.y, tmp2);
111
112 /* sqr(z)/(z^(a)+b) */
113 cmplx_div_bn(&g_new_z_bn, &bn_numer, &bn_tmp_new);
114
115 add_a_bn(g_new_z_bn.x, g_param_z_bn.x);
116 add_a_bn(g_new_z_bn.y, g_param_z_bn.y);
117
118 return g_bailout_bignum();
119}

121int divide_brot5_orbit_bf()
122{
123 BigStackSaver saved;
124 BFComplex bf_tmp_new;
125 BFComplex bf_numer;
126 BFComplex bf_c_exp;
127
128 bf_tmp_new.x = alloc_stack(g_r_bf_length + 2);
129 bf_tmp_new.y = alloc_stack(g_r_bf_length + 2);
130 bf_numer.x = alloc_stack(g_r_bf_length + 2);
131 bf_numer.y = alloc_stack(g_r_bf_length + 2);
132 bf_c_exp.x = alloc_stack(g_bf_length + 2);
133 bf_c_exp.y = alloc_stack(g_bf_length + 2);
134 BigFloat tmp1 = alloc_stack(g_bf_length + 2);
135
136 /* sqr(z) */
137 /* bfnumer.x = g_tmp_sqr_x_bf - g_tmp_sqr_y_bf; */
138 sub_bf(bf_numer.x, g_tmp_sqr_x_bf, g_tmp_sqr_y_bf);
139
140 /* bfnumer.y = 2 * g_old_z_bf.x * g_old_z_bf.y; */
141 mult_bf(bf_numer.y, g_old_z_bf.x, g_old_z_bf.y);
142 double_a_bf(bf_numer.y);
143
144 /* z^(a) */
145 int_to_bf(bf_c_exp.x, g_c_exponent);
146 clear_bf(bf_c_exp.y);
147 cmplx_pow_bf(&bf_tmp_new, &g_old_z_bf, &bf_c_exp);
148 /* then add b */
149 float_to_bf(tmp1, g_b_const);
150 add_a_bf(bf_tmp_new.x, tmp1);
151
152 /* sqr(z)/(z^(a)+b) */
153 cmplx_div_bf(&g_new_z_bf, &bf_numer, &bf_tmp_new);
154
155 add_a_bf(g_new_z_bf.x, g_param_z_bf.x);
156 add_a_bf(g_new_z_bf.y, g_param_z_bf.y);

187int divide_brot5_orbit() /* from formula by Jim Muth */
188{
189 /* z=sqr(z)/(z^(-a)+b)+c */
190 /* we'll set a to -a in setup, so don't need it here */
191 /* z=sqr(z)/(z^(a)+b)+c */
192 DComplex tmp_sqr;
193 DComplex tmp1;
194 DComplex tmp2;
195
196 /* sqr(z) */
197 tmp_sqr.x = g_temp_sqr_x - g_temp_sqr_y;
198 tmp_sqr.y = g_old_z.x * g_old_z.y * 2.0;
199
200 /* z^(a) = e^(a * log(z))*/
201 cmplx_log(g_old_z, tmp1);
202 tmp1.x *= g_c_exponent;
203 tmp1.y *= g_c_exponent;
204 cmplx_exp(tmp1, tmp2);
205 /* then add b */
206 tmp2.x += g_b_const;
207 /* sqr(z)/(z^(a)+b) */
208 fpu_cmplx_div(tmp_sqr, tmp2, g_new_z);
209 /* then add c = g_init = pixel */
210 g_new_z.x += g_init.x;
211 g_new_z.y += g_init.y;
212
213 return g_bailout_float();
214}
