143bool operator==(const EvolutionInfo &lhs, const EvolutionInfo &rhs)
144{
145 return lhs.evolving == rhs.evolving //
146 && lhs.image_grid_size == rhs.image_grid_size //
147 && lhs.this_generation_random_seed == rhs.this_generation_random_seed //
148 && lhs.max_random_mutation == rhs.max_random_mutation //
149 && within_eps(lhs.x_parameter_range, rhs.x_parameter_range) //
150 && within_eps(lhs.y_parameter_range, rhs.y_parameter_range) //
151 && within_eps(lhs.x_parameter_offset, rhs.x_parameter_offset) //
152 && within_eps(lhs.y_parameter_offset, rhs.y_parameter_offset) //
153 && lhs.discrete_x_parameter_offset == rhs.discrete_x_parameter_offset //
154 && lhs.discrete_y_parameter_offset == rhs.discrete_y_parameter_offset //
155 && lhs.px == rhs.px //
156 && lhs.py == rhs.py //
157 && lhs.screen_x_offset == rhs.screen_x_offset //
158 && lhs.screen_y_offset == rhs.screen_y_offset //
159 && lhs.x_dots == rhs.x_dots //
160 && lhs.y_dots == rhs.y_dots //
161 && equal(lhs.mutate, rhs.mutate) //
162 && lhs.count == rhs.count; //
163}

238void vary_dbl(GeneBase gene[], const int rand_val, const int i)
239{
240 const int delta_y = g_evolve_image_grid_size - g_evolve_param_grid_y - 1;
241 switch (gene[i].mutate)
242 {
243 default:
244 case Variations::NONE:
245 break;
246 case Variations::X:
247 *static_cast<double *>(gene[i].addr) = g_evolve_param_grid_x * g_evolve_dist_per_x + g_evolve_x_parameter_offset; //paramspace x coord * per view delta px + offset
248 break;
249 case Variations::Y:
250 *static_cast<double *>(gene[i].addr) = delta_y * g_evolve_dist_per_y + g_evolve_y_parameter_offset; //same for y
251 break;
252 case Variations::X_PLUS_Y:
253 *static_cast<double *>(gene[i].addr) = g_evolve_param_grid_x*g_evolve_dist_per_x+ g_evolve_x_parameter_offset +
254 delta_y * g_evolve_dist_per_y + g_evolve_y_parameter_offset; //and x+y
255 break;
256 case Variations::X_MINUS_Y:
257 *static_cast<double *>(gene[i].addr) = g_evolve_param_grid_x * g_evolve_dist_per_x + g_evolve_x_parameter_offset -(delta_y*g_evolve_dist_per_y+ g_evolve_y_parameter_offset); //and x-y
258 break;
259 case Variations::RANDOM:
260 *static_cast<double *>(gene[i].addr) += static_cast<double>(rand_val) / RAND_MAX * 2 * g_evolve_max_random_mutation - g_evolve_max_random_mutation;
261 break;
262 case Variations::WEIGHTED_RANDOM:
263 {
264 const int mid = g_evolve_image_grid_size / 2;
265 const double radius = std::sqrt(static_cast<double>(sqr(g_evolve_param_grid_x - mid) + sqr(delta_y - mid)));
266 *static_cast<double *>(gene[i].addr) += (static_cast<double>(rand_val) / RAND_MAX * 2 * g_evolve_max_random_mutation - g_evolve_max_random_mutation) * radius;
267 }
268 break;
269 }
270}

272static int vary_int(const int rand_value, const int limit, const Variations mode)
273{
274 int ret = 0;
275 const int delta_y = g_evolve_image_grid_size - g_evolve_param_grid_y - 1;
276 switch (mode)
277 {
278 default:
279 case Variations::NONE:
280 break;
281 case Variations::X:
282 ret = (g_evolve_discrete_x_parameter_offset +g_evolve_param_grid_x)%limit;
283 break;
284 case Variations::Y:
285 ret = (g_evolve_discrete_y_parameter_offset +delta_y)%limit;
286 break;
287 case Variations::X_PLUS_Y:
288 ret = (g_evolve_discrete_x_parameter_offset +g_evolve_param_grid_x+ g_evolve_discrete_y_parameter_offset +delta_y)%limit;
289 break;
290 case Variations::X_MINUS_Y:
291 ret = g_evolve_discrete_x_parameter_offset + g_evolve_param_grid_x -(g_evolve_discrete_y_parameter_offset +delta_y)%limit;
292 break;
293 case Variations::RANDOM:
294 ret = rand_value % limit;
295 break;
296 case Variations::WEIGHTED_RANDOM:
297 {
298 const int mid = g_evolve_image_grid_size / 2;
299 const double radius = std::sqrt(static_cast<double>(sqr(g_evolve_param_grid_x - mid) + sqr(delta_y - mid)));
300 ret = static_cast<int>(
301 (rand_value / RAND_MAX * 2 * g_evolve_max_random_mutation - g_evolve_max_random_mutation) *
302 radius);
303 ret %= limit;
304 break;
305 }
306 }
307 return ret;
308}

1008void spiral_map(const int count)
1009{
1010 // maps out a clockwise spiral for a prettier and possibly
1011 // more intuitively useful order of drawing the sub images.
1012 // All the malarky with count is to allow resuming
1013 int i = 0;
1014 const int mid = g_evolve_image_grid_size / 2;
1015 if (count == 0)
1016 {
1017 // start in the middle
1018 g_evolve_param_grid_y = mid;
1019 g_evolve_param_grid_x = g_evolve_param_grid_y;
1020 return;
1021 }
1022 for (int offset = 1; offset <= mid; offset ++)
1023 {
1024 // first do the top row
1025 g_evolve_param_grid_y = mid - offset;
1026 for (g_evolve_param_grid_x = mid - offset +1; g_evolve_param_grid_x < mid+offset; g_evolve_param_grid_x++)
1027 {
1028 i++;
1029 if (i == count)
1030 {
1031 return;
1032 }
1033 }
1034 // then do the right hand column
1035 for (; g_evolve_param_grid_y < mid + offset; g_evolve_param_grid_y++)
1036 {
1037 i++;
1038 if (i == count)
1039 {
1040 return;
1041 }
1042 }
1043 // then reverse along the bottom row
1044 for (; g_evolve_param_grid_x > mid - offset; g_evolve_param_grid_x--)
1045 {
1046 i++;
1047 if (i == count)
1048 {
1049 return;
1050 }
1051 }
1052 // then up the left to finish
1053 for (; g_evolve_param_grid_y >= mid - offset; g_evolve_param_grid_y--)
1054 {
1055 i++;
1056 if (i == count)
1057 {
1058 return;
1059 }
1060 }
1061 }
