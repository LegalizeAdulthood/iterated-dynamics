21void get_julia_attractor(const double real, const double imag)
22{
23 if (g_attractor.count == 0 && !g_attractor.enabled) // not magnet & not requested
24 {
25 return;
26 }
27
28 if (g_attractor.count >= MAX_NUM_ATTRACTORS) // space for more attractors ?
29 {
30 return; // Bad luck - no room left !
31 }
32
33 int save_periodicity_check = g_periodicity_check;
34 long save_max_iterations = g_max_iterations;
35 g_periodicity_check = 0;
36 g_old_z.x = real; // prepare for f.p orbit calc
37 g_old_z.y = imag;
38 g_temp_sqr_x = sqr(g_old_z.x);
39 g_temp_sqr_y = sqr(g_old_z.y);
40
41 // we're going to try at least this hard
42 g_max_iterations = std::max(g_max_iterations, 500L);
43 g_color_iter = 0;
44 g_overflow = false;
45 while (++g_color_iter < g_max_iterations)
46 {
47 if (g_cur_fractal_specific->orbit_calc() || g_overflow)
48 {
49 break;
50 }
51 }
52 if (g_color_iter >= g_max_iterations) // if orbit stays in the lake
53 {
54 const DComplex prev_z = g_new_z;
55 for (int i = 0; i < 10; i++)
56 {
57 g_overflow = false;
58 if (!g_cur_fractal_specific->orbit_calc() && !g_overflow) // if it stays in the lake
59 {
60 // and doesn't move far, probably found a finite attractor
61 if (std::abs(prev_z.x - g_new_z.x) < g_close_enough &&
62 std::abs(prev_z.y - g_new_z.y) < g_close_enough)
63 {
64 g_attractor.z[g_attractor.count] = g_new_z;
65 g_attractor.period[g_attractor.count] = i + 1;
66 g_attractor.count++; // another attractor - coloured lakes !
67 break;
68 }
69 }
70 else
71 {
72 break;
73 }
74 }
75 }
76 if (g_attractor.count == 0)
77 {
78 g_periodicity_check = save_periodicity_check;
79 }
80 g_max_iterations = save_max_iterations;
81}
