143// TODO: move this to the UI folder
144int file_get_window()
145{
146 Affine stack_cvt;
147 std::time_t this_time;
148 std::time_t last_time;
149 int c;
150 FileWindowStatus status{};
151 int win_count;
152 bool toggle{};
153 int color_of_box;
154 FileWindow window;
155 char drive[ID_FILE_MAX_DRIVE];
156 char dir[ID_FILE_MAX_DIR];
157 char fname[ID_FILE_MAX_FNAME];
158 char ext[ID_FILE_MAX_EXT];
159 char tmp_mask[ID_FILE_MAX_PATH];
160 bool vid_too_big{};
161 int saved;
162
163 s_old_bf_math = g_bf_math;
164 g_bf_math = BFMathType::BIG_FLT;
165 if (s_old_bf_math == BFMathType::NONE)
166 {
167 CalcStatus old_calc_status = g_calc_status; // kludge because next sets it = 0
168 fractal_float_to_bf();
169 g_calc_status = old_calc_status;
170 }
171 saved = save_stack();
172 s_bt_a = alloc_stack(g_r_bf_length+2);
173 s_bt_b = alloc_stack(g_r_bf_length+2);
174 s_bt_c = alloc_stack(g_r_bf_length+2);
175 s_bt_d = alloc_stack(g_r_bf_length+2);
176 s_bt_e = alloc_stack(g_r_bf_length+2);
177 s_bt_f = alloc_stack(g_r_bf_length + 2);
178
179 const int num_dots = g_screen_x_dots + g_screen_y_dots;
180 vid_too_big = num_dots > 4096;
181 s_browse_box_x.resize(num_dots*MAX_WINDOWS_OPEN);
182 s_browse_box_y.resize(num_dots*MAX_WINDOWS_OPEN);
183 s_browse_box_values.resize(num_dots*MAX_WINDOWS_OPEN);
184
185 // set up complex-plane-to-screen transformation
186 if (s_old_bf_math != BFMathType::NONE)
187 {
188 bf_setup_convert_to_screen();
189 }
190 else
191 {
192 s_cvt = &stack_cvt; // use stack
193 setup_convert_to_screen(s_cvt);
194 // put in bf variables
195 float_to_bf(s_bt_a, s_cvt->a);
196 float_to_bf(s_bt_b, s_cvt->b);
197 float_to_bf(s_bt_c, s_cvt->c);
198 float_to_bf(s_bt_d, s_cvt->d);
199 float_to_bf(s_bt_e, s_cvt->e);
200 float_to_bf(s_bt_f, s_cvt->f);
201 }
202 find_special_colors();
203 color_of_box = g_color_medium;
204rescan: // entry for changed browse parms
205 std::time(&last_time);
206 toggle = false;
207 win_count = 0;
208 g_browse.sub_images = true;
209 split_drive_dir(g_read_filename, drive, dir);
210 split_fname_ext(g_browse.mask, fname, ext);
211 make_path(tmp_mask, drive, dir, fname, ext);
212 std::filesystem::path path{find_wildcard_first(ReadFile::IMAGE, tmp_mask)};
213 status = vid_too_big || path.empty() ? FileWindowStatus::EXIT : FileWindowStatus::CONTINUE;
214 // draw all visible windows
215 while (status == FileWindowStatus::CONTINUE)
216 {
217 if (driver_key_pressed())
218 {
219 driver_get_key();
220 break;
221 }
222 FractalInfo read_info;
223 ExtBlock2 blk_2_info;
224 ExtBlock3 blk_3_info;
225 ExtBlock4 blk_4_info;
226 ExtBlock5 blk_5_info;
227 ExtBlock6 blk_6_info;
228 ExtBlock7 blk_7_info;
229 if (!find_fractal_info(path.string(), &read_info, //
230 &blk_2_info, &blk_3_info, &blk_4_info, &blk_5_info, &blk_6_info, &blk_7_info) //
231 && (type_ok(&read_info, &blk_3_info) || !g_browse.check_fractal_type) //
232 && (params_ok(&read_info) || !g_browse.check_fractal_params) //
233 && g_browse.name != path.string() //
234 && !blk_6_info.got_data //
235 && window.is_visible(&read_info, &blk_5_info)) //
236 {
237 window.path = path;
238 window.filename = path.filename().string();
239 window.draw(color_of_box);
240 window.box_count = g_box_count;
241 s_browse_windows.push_back(window);
242 save_box(num_dots, win_count);
243 win_count++;
244 assert(static_cast<size_t>(win_count) == s_browse_windows.size());
245 }
246 path = find_wildcard_next();
247 status = path.empty() || win_count >= MAX_WINDOWS_OPEN ? FileWindowStatus::EXIT : FileWindowStatus::CONTINUE;
248 }
249
250 if (win_count >= MAX_WINDOWS_OPEN)
251 {
252 // hard code message at MAX_WINDOWS_OPEN = 450
253 text_temp_msg("Sorry...no more space, 450 displayed.");
254 }
255 if (vid_too_big)
256 {
257 text_temp_msg("Xdots + Ydots > 4096.");
258 }
259 c = 0;
260 if (win_count)
261 {
262 driver_buzzer(Buzzer::COMPLETE); //let user know we've finished
263 int index = 0;
264 status = FileWindowStatus::CONTINUE;
265 window = s_browse_windows[index];
266 restore_box(num_dots, index);
267 show_temp_msg(window.filename);
268 while (status == FileWindowStatus::CONTINUE)
269 {
270 char msg[40];
271 while (!driver_key_pressed())
272 {
273 std::time(&this_time);
274 if (static_cast<double>(this_time - last_time) > 0.2)
275 {
276 last_time = this_time;
277 toggle = !toggle;
278 }
279 if (toggle)
280 {
281 window.draw(g_color_bright); // flash current window
282 }
283 else
284 {
285 window.draw(g_color_dark);
286 }
287 }
288
289 c = driver_get_key();
290 switch (c)
291 {
292 case ID_KEY_RIGHT_ARROW:
293 case ID_KEY_LEFT_ARROW:
294 case ID_KEY_DOWN_ARROW:
295 case ID_KEY_UP_ARROW:
296 clear_temp_msg();
297 window.draw(color_of_box);// dim last window
298 if (c == ID_KEY_RIGHT_ARROW || c == ID_KEY_UP_ARROW)
299 {
300 index++; // shift attention to next window
301 if (index >= win_count)
302 {
303 index = 0;
304 }
305 }
306 else
307 {
308 index -- ;
309 if (index < 0)
310 {
311 index = win_count -1 ;
312 }
313 }
314 window = s_browse_windows[index];
315 restore_box(num_dots, index);
316 show_temp_msg(window.filename);
317 break;
318 case ID_KEY_CTL_INSERT:
319 color_of_box += key_count(ID_KEY_CTL_INSERT);
320 for (FileWindow &s_browse_window : s_browse_windows)
321 {
322 s_browse_window.draw(color_of_box);
323 }
324 window = s_browse_windows[index];
325 window.draw(color_of_box);
326 break;
327
328 case ID_KEY_CTL_DEL:
329 color_of_box -= key_count(ID_KEY_CTL_DEL);
330 for (int i = 0; i < win_count ; i++)
331 {
332 s_browse_windows[i].draw(color_of_box);
333 }
334 window = s_browse_windows[index];
335 window.draw(color_of_box);
336 break;
337 case ID_KEY_ENTER:
338 case ID_KEY_ENTER_2: // this file please
339 g_browse.name = window.filename;
340 status = FileWindowStatus::EXIT;
341 break;
342
343 case ID_KEY_ESC:
344 case 'l':
345 case 'L':
346 g_browse.auto_browse = false;
347 status = FileWindowStatus::ERASE_BOXES_EXIT;
348 break;
349
350 case 'D': // delete file
351 clear_temp_msg();
352 show_temp_msg(fmt::format("Delete {:s}? (Y/N)", window.filename));
353 driver_wait_key_pressed(false);
354 clear_temp_msg();
355 c = driver_get_key();
356 if (c == 'Y' && g_browse.confirm_delete)
357 {
358 text_temp_msg("ARE YOU SURE???? (Y/N)");
359 if (driver_get_key() != 'Y')
360 {
361 c = 'N';
362 }
363 }
364 if (c == 'Y')
365 {
366 std::error_code ec;
367 if (std::filesystem::remove(window.path, ec))
368 {
369 // do a rescan
370 status = FileWindowStatus::RESCAN;
371 check_history(window.filename.c_str(), "");
372 break;
373 }
374 if (ec.value() == EACCES)
375 {
376 text_temp_msg("Sorry...it's a read only file, can't del");
377 show_temp_msg(window.filename);
378 break;
379 }
380 }
381 {
382 text_temp_msg("file not deleted (phew!)");
383 }
384 show_temp_msg(window.filename);
385 break;
386
387 case 'R':
388 {
389 clear_temp_msg();
390 driver_stack_screen();
391 char new_name[60];
392 new_name[0] = 0;
393 std::strcpy(msg, "Enter the new filename for ");
394 std::strcpy(new_name, window.filename.c_str());
395 int i = field_prompt(msg, nullptr, new_name, 60, nullptr);
396 driver_unstack_screen();
397 if (i != -1)
398 {
399 std::error_code ec;
400 std::string new_filename{std::filesystem::path{new_name}.filename().string()};
401 std::filesystem::path new_path{window.path.parent_path() / new_filename};
402 std::filesystem::rename(window.path, new_path, ec);
403 if (ec)
404 {
405 if (ec.value() == EACCES)
406 {
407 text_temp_msg("Sorry....can't rename");
408 }
409 }
410 else
411 {
412 check_history(window.filename.c_str(), new_filename.c_str());
413 window.path = new_path;
414 window.filename = new_filename;
415 }
416 }
417 s_browse_windows[index] = window;
418 show_temp_msg(window.filename);
419 break;
420 }
421
422 case ID_KEY_CTL_B:
423 clear_temp_msg();
424 driver_stack_screen();
425 status = static_cast<FileWindowStatus>(std::abs(get_browse_params()));
426 driver_unstack_screen();
427 show_temp_msg(window.filename);
428 break;
429
430 case 's': // save image with boxes
431 g_browse.auto_browse = false;
432 window.draw(color_of_box); // current window white
433 status = FileWindowStatus::SAVE_BOXES_EXIT;
434 break;
435
436 case '\\': //back out to last image
437 status = FileWindowStatus::ERASE_BOXES_EXIT;
438 break;
439
440 default:
441 break;
442 } //switch
443 } //while
444
445 // now clean up memory (and the screen if necessary)
446 clear_temp_msg();
447 if (status >= FileWindowStatus::EXIT && status < FileWindowStatus::SAVE_BOXES_EXIT)
448 {
449 for (int i = win_count-1; i >= 0; i--)
450 {
451 window = s_browse_windows[i];
452 g_box_count = window.box_count;
453 restore_box(num_dots, i);
454 if (g_box_count > 0)
455 {
456 clear_box();
457 }
458 }
459 }
460 if (status == FileWindowStatus::RESCAN)
461 {
462 goto rescan; // hey everybody I just used the g word!
463 }
464 }//if
465 else
466 {
467 driver_buzzer(Buzzer::INTERRUPT); //no suitable files in directory!
468 text_temp_msg("Sorry... I can't find anything");
469 g_browse.sub_images = false;
470 }
471
472 s_browse_windows.clear();
473 s_browse_box_x.clear();
474 s_browse_box_y.clear();
475 s_browse_box_values.clear();
476 restore_stack(saved);
477 if (s_old_bf_math == BFMathType::NONE)
478 {
479 free_bf_vars();
480 }
481 g_bf_math = s_old_bf_math;
482
483 return c;
484}

486void FileWindow::draw(const int color)
487{
488 g_box_color = color;
489 g_box_count = 0;
490 if (win_size >= g_browse.smallest_box)
491 {
492 // big enough on screen to show up as a box so draw it
493 // corner pixels
494 add_box(top_left);
495 add_box(top_right);
496 add_box(bot_left);
497 add_box(bot_right);
498 draw_lines(
499 top_left, top_right, bot_left.x - top_left.x, bot_left.y - top_left.y); // top & bottom lines
500 draw_lines(
501 top_left, bot_left, top_right.x - top_left.x, top_right.y - top_left.y); // left & right lines
502 display_box();
503 }
504 else
505 {
506 Coord ibl;
507 Coord itr;
508 // draw crosshairs
509 int cross_size = g_logical_screen.y_dots / 45;
510 cross_size = std::max(cross_size, 2);
511 itr.x = top_left.x - cross_size;
512 itr.y = top_left.y;
513 ibl.y = top_left.y - cross_size;
514 ibl.x = top_left.x;
515 draw_lines(top_left, itr, ibl.x - itr.x, 0); // top & bottom lines
516 draw_lines(top_left, ibl, 0, itr.y - ibl.y); // left & right lines
517 display_box();
518 }

521bool FileWindow::is_visible(const FractalInfo *info, const ExtBlock5 *blk_5_info)
522{
523 DblCoords tl;
524 DblCoords tr;
525 DblCoords bl;
526 DblCoords br;
527 const double too_big = std::sqrt(sqr(static_cast<double>(g_screen_x_dots)) + sqr(static_cast<double>(g_screen_y_dots))) * 1.5;
528 // arbitrary value... stops browser zooming out too far
529 int corner_count = 0;
530 bool cant_see = false;
531
532 BigStackSaver saved;
533 // Save original values.
534 int orig_bf_length = g_bf_length;
535 int orig_bn_length = g_bn_length;
536 int orig_padding = g_padding;
537 int orig_r_length = g_r_length;
538 int orig_shift_factor = g_shift_factor;
539 int orig_r_bf_length = g_r_bf_length;
540 /*
541 if (oldbf_math && info->bf_math && (g_bn_length+4 < info->g_bf_length)) {
542 g_bn_length = info->g_bf_length;
543 calc_lengths();
544 }
545 */
546 const int two_len = g_bf_length + 2;
547 BigFloat bt_x = alloc_stack(two_len);
548 BigFloat bt_y = alloc_stack(two_len);
549 BigFloat bt_x_min = alloc_stack(two_len);
550 BigFloat bt_x_max = alloc_stack(two_len);
551 BigFloat bt_y_min = alloc_stack(two_len);
552 BigFloat bt_y_max = alloc_stack(two_len);
553 BigFloat bt_x_3rd = alloc_stack(two_len);
554 BigFloat bt_y_3rd = alloc_stack(two_len);
555
556 if (info->bf_math)
557 {
558 const int di_bf_length = info->bf_length + g_bn_step;
559 const int two_di_len = di_bf_length + 2;
560 const int two_r_bf = g_r_bf_length + 2;
561
562 s_n_a = alloc_stack(two_r_bf);
563 s_n_b = alloc_stack(two_r_bf);
564 s_n_c = alloc_stack(two_r_bf);
565 s_n_d = alloc_stack(two_r_bf);
566 s_n_e = alloc_stack(two_r_bf);
567 s_n_f = alloc_stack(two_r_bf);
568
569 convert_bf(s_n_a, s_bt_a, g_r_bf_length, orig_r_bf_length);
570 convert_bf(s_n_b, s_bt_b, g_r_bf_length, orig_r_bf_length);
571 convert_bf(s_n_c, s_bt_c, g_r_bf_length, orig_r_bf_length);
572 convert_bf(s_n_d, s_bt_d, g_r_bf_length, orig_r_bf_length);
573 convert_bf(s_n_e, s_bt_e, g_r_bf_length, orig_r_bf_length);
574 convert_bf(s_n_f, s_bt_f, g_r_bf_length, orig_r_bf_length);
575
576 BigFloat bt_t1 = alloc_stack(two_di_len);
577 BigFloat bt_t2 = alloc_stack(two_di_len);
578 BigFloat bt_t3 = alloc_stack(two_di_len);
579 BigFloat bt_t4 = alloc_stack(two_di_len);
580 BigFloat bt_t5 = alloc_stack(two_di_len);
581 BigFloat bt_t6 = alloc_stack(two_di_len);
582
583 std::memcpy(bt_t1, blk_5_info->apm_data.data(), two_di_len);
584 std::memcpy(bt_t2, &blk_5_info->apm_data[two_di_len], two_di_len);
585 std::memcpy(bt_t3, &blk_5_info->apm_data[2*two_di_len], two_di_len);
586 std::memcpy(bt_t4, &blk_5_info->apm_data[3*two_di_len], two_di_len);
587 std::memcpy(bt_t5, &blk_5_info->apm_data[4*two_di_len], two_di_len);
588 std::memcpy(bt_t6, &blk_5_info->apm_data[5*two_di_len], two_di_len);
589
590 convert_bf(bt_x_min, bt_t1, two_len, two_di_len);
591 convert_bf(bt_x_max, bt_t2, two_len, two_di_len);
592 convert_bf(bt_y_min, bt_t3, two_len, two_di_len);
593 convert_bf(bt_y_max, bt_t4, two_len, two_di_len);
594 convert_bf(bt_x_3rd, bt_t5, two_len, two_di_len);
595 convert_bf(bt_y_3rd, bt_t6, two_len, two_di_len);
596 }
597
598 /* transform maps real plane co-ords onto the current screen view see above */
599 if (s_old_bf_math != BFMathType::NONE || info->bf_math != 0)
600 {
601 if (!info->bf_math)
602 {
603 float_to_bf(bt_x, info->x_min);
604 float_to_bf(bt_y, info->y_max);
605 }
606 else
607 {
608 copy_bf(bt_x, bt_x_min);
609 copy_bf(bt_y, bt_y_max);
610 }
611 bf_transform(bt_x, bt_y, &tl);
612 }
613 else
614 {
615 tl.x = info->x_min;
616 tl.y = info->y_max;
617 transform(&tl);
618 }
619 top_left.x = static_cast<int>(std::lround(tl.x));
620 top_left.y = static_cast<int>(std::lround(tl.y));
621 if (s_old_bf_math != BFMathType::NONE || info->bf_math)
622 {
623 if (!info->bf_math)
624 {
625 float_to_bf(bt_x, info->x_max -(info->x3rd-info->x_min));
626 float_to_bf(bt_y, info->y_max +(info->y_min-info->y3rd));
627 }
628 else
629 {
630 neg_a_bf(sub_bf(bt_x, bt_x_3rd, bt_x_min));
631 add_a_bf(bt_x, bt_x_max);
632 sub_bf(bt_y, bt_y_min, bt_y_3rd);
633 add_a_bf(bt_y, bt_y_max);
634 }
635 bf_transform(bt_x, bt_y, &tr);
636 }
637 else
638 {
639 tr.x = info->x_max -(info->x3rd-info->x_min);
640 tr.y = info->y_max +(info->y_min-info->y3rd);
641 transform(&tr);
642 }
643 top_right.x = static_cast<int>(std::lround(tr.x));
644 top_right.y = static_cast<int>(std::lround(tr.y));

718 && bl.y >= 0 - g_logical_screen.y_offset && bl.y <= g_screen_y_dots - g_logical_screen.y_offset)

852static void bf_setup_convert_to_screen()
853{
854 BigStackSaver saved;
855
856 // setup_convert_to_screen() in LORENZ.C, converted to g_bf_math
857 // Call only from within fgetwindow()
858
859 BigFloat bt_inter1 = alloc_stack(g_r_bf_length + 2);
860 BigFloat bt_inter2 = alloc_stack(g_r_bf_length + 2);
861 BigFloat bt_det = alloc_stack(g_r_bf_length + 2);
862 BigFloat bt_xd = alloc_stack(g_r_bf_length + 2);
863 BigFloat bt_yd = alloc_stack(g_r_bf_length + 2);
864 BigFloat bt_tmp1 = alloc_stack(g_r_bf_length + 2);
865 BigFloat bt_tmp2 = alloc_stack(g_r_bf_length + 2);
866
867 // xx3rd-xxmin
868 sub_bf(bt_inter1, g_bf_x_3rd, g_bf_x_min);
869 // yymin-yymax
870 sub_bf(bt_inter2, g_bf_y_min, g_bf_y_max);
871 // (xx3rd-xxmin)*(yymin-yymax)
872 mult_bf(bt_tmp1, bt_inter1, bt_inter2);
873
874 // yymax-yy3rd
875 sub_bf(bt_inter1, g_bf_y_max, g_bf_y_3rd);
876 // xxmax-xxmin
877 sub_bf(bt_inter2, g_bf_x_max, g_bf_x_min);
878 // (yymax-yy3rd)*(xxmax-xxmin)
879 mult_bf(bt_tmp2, bt_inter1, bt_inter2);
880
881 // det = (xx3rd-xxmin)*(yymin-yymax) + (yymax-yy3rd)*(xxmax-xxmin)
882 add_bf(bt_det, bt_tmp1, bt_tmp2);
883
884 // xd = x_size_d/det
885 float_to_bf(bt_tmp1, g_logical_screen.x_size_dots);
886 div_bf(bt_xd, bt_tmp1, bt_det);
887
888 // a = xd*(yymax-yy3rd)
889 sub_bf(bt_inter1, g_bf_y_max, g_bf_y_3rd);
890 mult_bf(s_bt_a, bt_xd, bt_inter1);
891
892 // b = xd*(xx3rd-xxmin)
893 sub_bf(bt_inter1, g_bf_x_3rd, g_bf_x_min);
894 mult_bf(s_bt_b, bt_xd, bt_inter1);
895
896 // e = -(a*xxmin + b*yymax)
897 mult_bf(bt_tmp1, s_bt_a, g_bf_x_min);
898 mult_bf(bt_tmp2, s_bt_b, g_bf_y_max);
899 neg_a_bf(add_bf(s_bt_e, bt_tmp1, bt_tmp2));
900
901 // xx3rd-xxmax
902 sub_bf(bt_inter1, g_bf_x_3rd, g_bf_x_max);
903 // yymin-yymax
904 sub_bf(bt_inter2, g_bf_y_min, g_bf_y_max);
905 // (xx3rd-xxmax)*(yymin-yymax)
906 mult_bf(bt_tmp1, bt_inter1, bt_inter2);
907
908 // yymin-yy3rd
909 sub_bf(bt_inter1, g_bf_y_min, g_bf_y_3rd);
910 // xxmax-xxmin
911 sub_bf(bt_inter2, g_bf_x_max, g_bf_x_min);
912 // (yymin-yy3rd)*(xxmax-xxmin)
913 mult_bf(bt_tmp2, bt_inter1, bt_inter2);
914
915 // det = (xx3rd-xxmax)*(yymin-yymax) + (yymin-yy3rd)*(xxmax-xxmin)
916 add_bf(bt_det, bt_tmp1, bt_tmp2);
917
918 // yd = y_size_d/det
919 float_to_bf(bt_tmp2, g_logical_screen.y_size_dots);
920 div_bf(bt_yd, bt_tmp2, bt_det);
921
922 // c = yd*(yymin-yy3rd)
923 sub_bf(bt_inter1, g_bf_y_min, g_bf_y_3rd);
924 mult_bf(s_bt_c, bt_yd, bt_inter1);
925
926 // d = yd*(xx3rd-xxmax)
927 sub_bf(bt_inter1, g_bf_x_3rd, g_bf_x_max);
928 mult_bf(s_bt_d, bt_yd, bt_inter1);
929
930 // f = -(c*xxmin + d*yymax)
931 mult_bf(bt_tmp1, s_bt_c, g_bf_x_min);
932 mult_bf(bt_tmp2, s_bt_d, g_bf_y_max);
933 neg_a_bf(add_bf(s_bt_f, bt_tmp1, bt_tmp2));
934}
