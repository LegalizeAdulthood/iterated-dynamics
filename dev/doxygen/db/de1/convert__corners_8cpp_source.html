23// convert center/mag to corners
24void cvt_corners(const double ctr_x, const double ctr_y, const LDouble mag, double x_mag_factor, double rot,
25 const double skew)
26{
27 if (x_mag_factor == 0.0)
28 {
29 x_mag_factor = 1.0;
30 }
31
32 // half height, width
33 const double h = static_cast<double>(1.0 / mag);
34 const double w = h / (DEFAULT_ASPECT * x_mag_factor);
35
36 if (rot == 0.0 && skew == 0.0)
37 {
38 // simple, faster case
39 g_image_region.m_min.x = ctr_x - w;
40 g_image_region.m_3rd.x = ctr_x - w;
41 g_image_region.m_max.x = ctr_x + w;
42 g_image_region.m_min.y = ctr_y - h;
43 g_image_region.m_3rd.y = ctr_y - h;
44 g_image_region.m_max.y = ctr_y + h;
45 return;
46 }
47
48 // in unrotated, untranslated coordinate system
49 const double tan_skew = std::tan(deg_to_rad(skew));
50 g_image_region.m_min.x = -w + h*tan_skew;
51 g_image_region.m_max.x = w - h*tan_skew;
52 g_image_region.m_3rd.x = -w - h*tan_skew;
53 g_image_region.m_max.y = h;
54 g_image_region.m_min.y = -h;
55 g_image_region.m_3rd.y = g_image_region.m_min.y;
56
57 // rotate coord system and then translate it
58 rot = deg_to_rad(rot);
59 const double sin_rot = std::sin(rot);
60 const double cos_rot = std::cos(rot);
61
62 // top left
63 double x = g_image_region.m_min.x * cos_rot + g_image_region.m_max.y * sin_rot;
64 double y = -g_image_region.m_min.x * sin_rot + g_image_region.m_max.y * cos_rot;
65 g_image_region.m_min.x = x + ctr_x;
66 g_image_region.m_max.y = y + ctr_y;
67
68 // bottom right
69 x = g_image_region.m_max.x * cos_rot + g_image_region.m_min.y * sin_rot;
70 y = -g_image_region.m_max.x * sin_rot + g_image_region.m_min.y * cos_rot;
71 g_image_region.m_max.x = x + ctr_x;
72 g_image_region.m_min.y = y + ctr_y;
73
74 // bottom left
75 x = g_image_region.m_3rd.x * cos_rot + g_image_region.m_3rd.y * sin_rot;
76 y = -g_image_region.m_3rd.x * sin_rot + g_image_region.m_3rd.y * cos_rot;
77 g_image_region.m_3rd.x = x + ctr_x;
78 g_image_region.m_3rd.y = y + ctr_y;
79}

81// convert center/mag to corners using bf
82void cvt_corners_bf(
83 BigFloat ctr_x, BigFloat ctr_y, const LDouble mag, double x_mag_factor, double rot, const double skew)
84{
85 BigStackSaver saved;
86 BigFloat bfh = alloc_stack(g_bf_length + 2);
87 BigFloat bfw = alloc_stack(g_bf_length + 2);
88
89 if (x_mag_factor == 0.0)
90 {
91 x_mag_factor = 1.0;
92 }
93
94 // half height, width
95 const LDouble h = 1 / mag;
96 float_to_bf(bfh, h);
97 const LDouble w = h / (DEFAULT_ASPECT * x_mag_factor);
98 float_to_bf(bfw, w);
99
100 if (rot == 0.0 && skew == 0.0)
101 {
102 // simple, faster case
103 // xx3rd = xxmin = x_ctr - w;
104 sub_bf(g_bf_x_min, ctr_x, bfw);
105 copy_bf(g_bf_x_3rd, g_bf_x_min);
106 // xxmax = x_ctr + w;
107 add_bf(g_bf_x_max, ctr_x, bfw);
108 // yy3rd = yymin = y_ctr - h;
109 sub_bf(g_bf_y_min, ctr_y, bfh);
110 copy_bf(g_bf_y_3rd, g_bf_y_min);
111 // yymax = y_ctr + h;
112 add_bf(g_bf_y_max, ctr_y, bfh);
113 return;
114 }
115
116 BigFloat bf_tmp = alloc_stack(g_bf_length + 2);
117 // in unrotated, untranslated coordinate system
118 const double tan_skew = std::tan(deg_to_rad(skew));
119 const LDouble x_min = -w + h * tan_skew;
120 const LDouble x_max = w - h * tan_skew;
121 const LDouble x_3rd = -w - h * tan_skew;
122 const LDouble y_max = h;
123 const LDouble y_min = -h;
124 const LDouble y_3rd = y_min;
125
126 // rotate coord system and then translate it
127 rot = deg_to_rad(rot);
128 const double sin_rot = std::sin(rot);
129 const double cos_rot = std::cos(rot);
130
131 // top left
132 LDouble x = x_min * cos_rot + y_max * sin_rot;
133 LDouble y = -x_min * sin_rot + y_max * cos_rot;
134 // xxmin = x + x_ctr;
135 float_to_bf(bf_tmp, x);
136 add_bf(g_bf_x_min, bf_tmp, ctr_x);
137 // yymax = y + y_ctr;
138 float_to_bf(bf_tmp, y);
139 add_bf(g_bf_y_max, bf_tmp, ctr_y);
140
141 // bottom right
142 x = x_max * cos_rot + y_min * sin_rot;
143 y = -x_max * sin_rot + y_min * cos_rot;
144 // xxmax = x + x_ctr;
145 float_to_bf(bf_tmp, x);
146 add_bf(g_bf_x_max, bf_tmp, ctr_x);
147 // yymin = y + y_ctr;
148 float_to_bf(bf_tmp, y);
149 add_bf(g_bf_y_min, bf_tmp, ctr_y);
150
151 // bottom left
152 x = x_3rd * cos_rot + y_3rd * sin_rot;
153 y = -x_3rd * sin_rot + y_3rd * cos_rot;
154 // xx3rd = x + x_ctr;
155 float_to_bf(bf_tmp, x);
156 add_bf(g_bf_x_3rd, bf_tmp, ctr_x);
157 // yy3rd = y + y_ctr;
158 float_to_bf(bf_tmp, y);
159 add_bf(g_bf_y_3rd, bf_tmp, ctr_y);
160}
