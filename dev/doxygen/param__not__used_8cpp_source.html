13/*
14 * Returns 1 if the formula parameter is not used in the current
15 * formula. If the parameter is used, or not a formula fractal,
16 * a 0 is returned. Note: this routine only works for formula types.
17 */
18bool param_not_used(const int param)
19{
20 bool ret = false;
21
22 // sanity check
23 if (g_fractal_type != FractalType::FORMULA)
24 {
25 return false;
26 }
27
28 switch (param/2)
29 {
30 case 0:
31 if (!g_frm_uses_p1)
32 {
33 ret = true;
34 }
35 break;
36 case 1:
37 if (!g_frm_uses_p2)
38 {
39 ret = true;
40 }
41 break;
42 case 2:
43 if (!g_frm_uses_p3)
44 {
45 ret = true;
46 }
47 break;
48 case 3:
49 if (!g_frm_uses_p4)
50 {
51 ret = true;
52 }
53 break;
54 case 4:
55 if (!g_frm_uses_p5)
56 {
57 ret = true;
58 }
59 break;
60 default:
61 ret = false;
62 break;
63 }
64 return ret;
65}
