56static void format_param_file_line(const int choice, char *buf)
57{
58 int c;
59 char line[80];
60 std::fseek(s_gfe.file, s_gfe.choices[choice]->point, SEEK_SET);
61 while (std::getc(s_gfe.file) != '{')
62 {
63 }
64 do
65 {
66 c = std::getc(s_gfe.file);
67 }
68 while (c == ' ' || c == '\t' || c == ';');
69 int i = 0;
70 while (i < 56 && !is_newline(c) && c != EOF)
71 {
72 line[i++] = c == '\t' ? ' ' : static_cast<char>(c);
73 c = std::getc(s_gfe.file);
74 }
75 line[i] = 0;
76 *fmt::format_to(buf, "{:<20s}{:<56s}", s_gfe.choices[choice]->name, line) = '\0';
77}

79static int check_gfe_key(const int key, const int choice)
80{
81 char blanks[79]; // used to clear the entry portion of screen
82 std::memset(blanks, ' ', 78);
83 blanks[78] = static_cast<char>(0);
84
85 if (key == ID_KEY_F6)
86 {
87 return 0-ID_KEY_F6;
88 }
89 if (key == ID_KEY_F4)
90 {
91 return 0-ID_KEY_F4;
92 }
93 if (key == ID_KEY_F2)
94 {
95 char inf_buf[25 * 80];
96 int widest_entry_line = 0;
97 int lines_in_entry = 0;
98 bool comment = false;
99 int c = 0;
100 int width_ct = 0;
101 std::fseek(s_gfe.file, s_gfe.choices[choice]->point, SEEK_SET);
102 while ((c = std::fgetc(s_gfe.file)) != EOF)
103 {
104 if (c == ';')
105 {
106 comment = true;
107 }
108 else if (c == '\n')
109 {
110 comment = false;
111 lines_in_entry++;
112 width_ct = -1;
113 }
114 else if (c == '\t')
115 {
116 width_ct += 7 - width_ct % 8;
117 }
118 else if (c == '\r')
119 {
120 continue;
121 }
122 if (++width_ct > widest_entry_line)
123 {
124 widest_entry_line = width_ct;
125 }
126 if (c == '}' && !comment)
127 {
128 lines_in_entry++;
129 break;
130 }
131 }
132 bool in_scrolling_mode = false; // true if entry doesn't fit available space
133 if (c == EOF)
134 {
135 // should never happen
136 std::fseek(s_gfe.file, s_gfe.choices[choice]->point, SEEK_SET);
137 in_scrolling_mode = false;
138 }
139 std::fseek(s_gfe.file, s_gfe.choices[choice]->point, SEEK_SET);
140 load_entry_text(s_gfe.file, inf_buf, 17, 0, 0);
141 if (lines_in_entry > 17 || widest_entry_line > 74)
142 {
143 in_scrolling_mode = true;
144 }
145 // ... instead, call help with buffer? heading added
146 driver_stack_screen();
147 help_title();
148 driver_set_attr(1, 0, C_GENERAL_MED, 24*80);
149
150 g_text_col_base = 0;
151 driver_put_string(2, 1, C_GENERAL_HI, s_gfe.title + std::string{" file entry:\n\n"});
152 g_text_col_base = 2; // left margin is 2
153 driver_put_string(4, 0, C_GENERAL_MED, inf_buf);
154 driver_put_string(-1, 0, C_GENERAL_LO,
155 "\n"
156 "\n"
157 " Use arrow keys, <PageUp>, <PageDown>, <Home>, and <End> to scroll text\n"
158 "Any other key to return to selection list");
159
160 int top_line = 0;
161 int left_column = 0;
162 bool done = false;
163 bool rewrite_inf_buf = false; // if true: rewrite the entry portion of screen
164 while (!done)
165 {
166 if (rewrite_inf_buf)
167 {
168 rewrite_inf_buf = false;
169 std::fseek(s_gfe.file, s_gfe.choices[choice]->point, SEEK_SET);
170 load_entry_text(s_gfe.file, inf_buf, 17, top_line, left_column);
171 for (int i = 4; i < (lines_in_entry < 17 ? lines_in_entry + 4 : 21); i++)
172 {
173 driver_put_string(i, 0, C_GENERAL_MED, blanks);
174 }
175 driver_put_string(4, 0, C_GENERAL_MED, inf_buf);
176 }
177 int i = get_a_key_no_help();
178 if (i == ID_KEY_DOWN_ARROW || i == ID_KEY_CTL_DOWN_ARROW
179 || i == ID_KEY_UP_ARROW || i == ID_KEY_CTL_UP_ARROW
180 || i == ID_KEY_LEFT_ARROW || i == ID_KEY_CTL_LEFT_ARROW
181 || i == ID_KEY_RIGHT_ARROW || i == ID_KEY_CTL_RIGHT_ARROW
182 || i == ID_KEY_HOME || i == ID_KEY_CTL_HOME
183 || i == ID_KEY_END || i == ID_KEY_CTL_END
184 || i == ID_KEY_PAGE_UP || i == ID_KEY_CTL_PAGE_UP
185 || i == ID_KEY_PAGE_DOWN || i == ID_KEY_CTL_PAGE_DOWN)
186 {
187 switch (i)
188 {
189 case ID_KEY_DOWN_ARROW:
190 case ID_KEY_CTL_DOWN_ARROW: // down one line
191 if (in_scrolling_mode && top_line < lines_in_entry - 17)
192 {
193 top_line++;
194 rewrite_inf_buf = true;
195 }
196 break;
197 case ID_KEY_UP_ARROW:
198 case ID_KEY_CTL_UP_ARROW: // up one line
199 if (in_scrolling_mode && top_line > 0)
200 {
201 top_line--;
202 rewrite_inf_buf = true;
203 }
204 break;
205 case ID_KEY_LEFT_ARROW:
206 case ID_KEY_CTL_LEFT_ARROW: // left one column
207 if (in_scrolling_mode && left_column > 0)
208 {
209 left_column--;
210 rewrite_inf_buf = true;
211 }
212 break;
213 case ID_KEY_RIGHT_ARROW:
214 case ID_KEY_CTL_RIGHT_ARROW: // right one column
215 if (in_scrolling_mode && std::strchr(inf_buf, SCROLL_MARKER) != nullptr)
216 {
217 left_column++;
218 rewrite_inf_buf = true;
219 }
220 break;
221 case ID_KEY_PAGE_DOWN:
222 case ID_KEY_CTL_PAGE_DOWN: // down 17 lines
223 if (in_scrolling_mode && top_line < lines_in_entry - 17)
224 {
225 top_line += 17;
226 top_line = std::min(top_line, lines_in_entry - 17);
227 rewrite_inf_buf = true;
228 }
229 break;
230 case ID_KEY_PAGE_UP:
231 case ID_KEY_CTL_PAGE_UP: // up 17 lines
232 if (in_scrolling_mode && top_line > 0)
233 {
234 top_line -= 17;
235 top_line = std::max(top_line, 0);
236 rewrite_inf_buf = true;
237 }
238 break;
239 case ID_KEY_END:
240 case ID_KEY_CTL_END: // to end of entry
241 if (in_scrolling_mode)
242 {
243 top_line = lines_in_entry - 17;
244 left_column = 0;
245 rewrite_inf_buf = true;
246 }
247 break;
248 case ID_KEY_HOME:
249 case ID_KEY_CTL_HOME: // to beginning of entry
250 if (in_scrolling_mode)
251 {
252 left_column = 0;
253 top_line = 0;
254 rewrite_inf_buf = true;
255 }
256 break;
257 default:
258 break;
259 }
260 }
261 else
262 {
263 done = true; // a key other than scrolling key was pressed
264 }
265 }
266 g_text_col_base = 0;
267 driver_hide_text_cursor();
268 driver_unstack_screen();
269 }
270 return 0;
271}

273static long gfe_choose_entry(
274 const ItemType type, const char *title, const std::filesystem::path &path, std::string &entry_name)
275{
276 const char *o_instr = "Press F6 to select different file, F2 for details, F4 to toggle sort ";
277 char buf[101];
278 FileEntry storage[MAX_ENTRIES + 1]{};
279 FileEntry *choices[MAX_ENTRIES + 1] = { nullptr };
280 int attributes[MAX_ENTRIES + 1]{};
281 char instr[80];
282
283 static bool do_sort = true;
284
285 s_gfe.choices = &choices[0];
286 s_gfe.title = title;
287
288retry:
289 for (int i = 0; i < MAX_ENTRIES+1; i++)
290 {
291 choices[i] = &storage[i];
292 attributes[i] = 1;
293 }
294
295 help_title(); // to display a clue when file big and next is slow
296
297 const int num_entries = scan_entries(s_gfe.file, &storage[0]);
298 if (num_entries == 0)
299 {
300 stop_msg("File doesn't contain any valid entries");
301 std::fclose(s_gfe.file);
302 return -2; // back to file list
303 }
304 std::strcpy(instr, o_instr);
305 if (do_sort)
306 {
307 std::strcat(instr, "off");
308 shell_sort(&choices, num_entries, sizeof(FileEntry *));
309 }
310 else
311 {
312 std::strcat(instr, "on");
313 }
314
315 std::strcpy(buf, entry_name.c_str()); // preset to last choice made
316 const std::string heading{std::string{title} + " Selection\n"
317 + "File: " + trim_filename(path, 68)};
318 FormatItem *format_item{};
319 int box_depth = 0;
320 int col_width = box_depth;
321 int box_width = col_width;
322 if (type == ItemType::PAR_SET)
323 {
324 format_item = format_param_file_line;
325 box_width = 1;
326 box_depth = 16;
327 col_width = 76;
328 }
329
330 const int i =
331 full_screen_choice(ChoiceFlags::INSTRUCTIONS | (do_sort ? ChoiceFlags::NONE : ChoiceFlags::NOT_SORTED),
332 heading.c_str(), nullptr, instr, num_entries, (const char **) choices, attributes, box_width,
333 box_depth, col_width, 0, format_item, buf, nullptr, check_gfe_key);
334 if (i == -ID_KEY_F4)
335 {
336 std::fseek(s_gfe.file, 0, SEEK_SET);
337 do_sort = !do_sort;
338 goto retry;
339 }
340 std::fclose(s_gfe.file);
341 if (i < 0)
342 {
343 // go back to file list or cancel
344 return i == -ID_KEY_F6 ? -2 : -1;
345 }
346 entry_name = choices[i]->name;
347 return choices[i]->point;
348}

350// Formula, LSystem, etc. type structure, select from file
351// containing definitions in the form name { ... }
352static long get_file_entry(const ItemType type, const char *type_desc, const char *type_wildcard,
353 std::filesystem::path &path, std::string &entry_name)
354{
355 const std::string hdg{fmt::format("Select {:s} File", type_desc)};
356 while (true)
357 {
358 // binary mode used here - it is more work, but much faster,
359 // especially when ftell or fgetpos is used
360 s_gfe.file = std::fopen(path.string().c_str(), "rb");
361 while (s_gfe.file == nullptr)
362 {
363 stop_msg("Couldn't open " + path.string());
364 if (driver_get_filename(hdg.c_str(), type_desc, type_wildcard, path))
365 {
366 return -1;
367 }
368 s_gfe.file = std::fopen(path.string().c_str(), "rb");
369 }
370 const long entry_pointer = gfe_choose_entry(type, type_desc, path, entry_name);
371 if (entry_pointer == -2)
372 {
373 std::fclose(s_gfe.file);
374 s_gfe.file = nullptr;
375 // either they cancel browsing for a new file, in which case
376 // filename is unchanged, or they browsed for a new file that's
377 // been stored in filename. Either way, we need to open the
378 // file and show the entries again.
379 driver_get_filename(hdg.c_str(), type_desc, type_wildcard, path);
380 continue;
381 }
382 return entry_pointer;
383 }
384}

386long get_file_entry(ItemType type, std::filesystem::path &path, std::string &entry_name)
387{
388 switch (type)
389 {
390 case ItemType::PAR_SET:
391 return get_file_entry(type, "Parameter Set", "*.par", path, entry_name);
392
393 case ItemType::FORMULA:
394 {
395 const long entry_pointer{get_file_entry(type, "Formula", "*.frm", path, entry_name)};
396 return entry_pointer >= 0 ? (parse_formula(g_formula_filename, entry_name, true) ? entry_pointer : 0) : entry_pointer;
397 }
398
399 case ItemType::L_SYSTEM:
400 {
401 const long entry_pointer{get_file_entry(type, "L-System", "*.l", path, entry_name)};
402 return entry_pointer >= 0 ? (lsystem_load() == 0 ? 0 : entry_pointer) : entry_pointer;
403 }
404
405 case ItemType::IFS:
406 {
407 const long entry_pointer{get_file_entry(type, "IFS", "*.ifs", path, entry_name)};
408 if (entry_pointer >= 0 && ifs_load() == 0)
409 {
410 set_fractal_type(g_ifs_dim == IFSDimension::TWO ? FractalType::IFS : FractalType::IFS_3D);
411 set_default_params(); // to correct them if 3d
412 return 0;
413 }
414 return entry_pointer;
415 }
416 }
417 throw std::runtime_error("Unknown item type " + std::to_string(static_cast<int>(type)));
418}
