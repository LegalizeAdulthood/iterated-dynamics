35//*********** standalone engine for "bifurcation" types **************
36
37//*************************************************************
38// The following code now forms a generalised Fractal Engine
39// for Bifurcation fractal typeS. By rights, it now belongs in
40// calcfract.cpp, but it's easier for me to leave it here !
41
42// Besides generalisation, enhancements include Periodicity
43// Checking during the plotting phase (AND halfway through the
44// filter cycle, if possible, to halve calc times), quicker
45// floating-point calculations for the standard Verhulst type,
46// and new bifurcation types (integer bifurcation, f.p & int
47// biflambda - the real equivalent of complex Lambda sets -
48// and f.p renditions of bifurcations of r*sin(Pi*p), which
49// spurred Mitchel Feigenbaum on to discover his Number).
50
51// To add further types, extend the FractalSpecific[] array in
52// usual way, with Bifurcation as the engine, and the name of
53// the routine that calculates the next bifurcation generation
54// as the "orbitcalc" routine in the FractalSpecific[] entry.
55
56// Bifurcation "orbitcalc" routines get called once per screen
57// pixel column. They should calculate the next generation
58// from the doubles Rate & Population (or the longs lRate &
59// lPopulation if they use integer math), placing the result
60// back in Population (or lPopulation). They should return 0
61// if all is ok, or any non-zero value if calculation bailout
62// is desirable (e.g. in case of errors, or the series tending
63// to infinity). Have fun !
64//*************************************************************

72Bifurcation::Bifurcation()
73{
74 m_verhulst.resize(g_i_stop_pt.y + 1);
75
76 for (int y = 0; y <= g_i_stop_pt.y; y++) // should be iystop
77 {
78 m_verhulst[y] = 0;
79 }
80
81 m_mono = false;
82 if (g_colors == 2)
83 {
84 m_mono = true;
85 }
86 if (m_mono)
87 {
88 if (g_inside_color != +ColorMethod::COLOR_BLACK)
89 {
90 m_outside_x = 0;
91 g_inside_color = 1;
92 }
93 else
94 {
95 m_outside_x = 1;
96 }
97 }
98
99 m_filter_cycles = g_param_z1.x <= 0 ? DEFAULT_FILTER : static_cast<long>(g_param_z1.x);
100 m_half_time_check = false;
101 if (g_periodicity_check && static_cast<unsigned long>(g_max_iterations) < m_filter_cycles)
102 {
103 m_filter_cycles = (m_filter_cycles - g_max_iterations + 1) / 2;
104 m_half_time_check = true;
105 }
106
107 g_init.y = static_cast<double>(g_image_region.m_max.y - g_i_stop_pt.y * g_delta_y); // bottom pixels
108}

117bool Bifurcation::iterate()
118{
119 if (m_x <= g_i_stop_pt.x)
120 {
121 g_rate = static_cast<double>(g_image_region.m_min.x + m_x * g_delta_x);
122 verhulst(); // calculate array once per column
123
124 for (int y = g_i_stop_pt.y; y >= 0; y--) // should be iystop & >=0
125 {
126 int color = m_verhulst[y];
127 if (color && m_mono)
128 {
129 color = g_inside_color;
130 }
131 else if (!color && m_mono)
132 {
133 color = m_outside_x;
134 }
135 else if (color>=g_colors)
136 {
137 color = g_colors-1;
138 }
139 m_verhulst[y] = 0;
140 g_plot(m_x, y, color); // was row-1, but that's not right?
141 }
142 ++m_x;
143 return true;
144 }
145 return false;
146}

148void Bifurcation::verhulst() // P. F. Verhulst (1845)
149{
150 g_population = g_param_z1.y == 0 ? SEED : g_param_z1.y;
151
152 g_overflow = false;
153
154 for (unsigned long counter = 0UL; counter < m_filter_cycles; counter++)
155 {
156 if (g_cur_fractal_specific->orbit_calc())
157 {
158 return;
159 }
160 }
161 if (m_half_time_check) // check for periodicity at half-time
162 {
163 s_period.init();
164 unsigned long counter;
165 for (counter = 0; counter < static_cast<unsigned long>(g_max_iterations) ; counter++)
166 {
167 if (g_cur_fractal_specific->orbit_calc())
168 {
169 return;
170 }
171 if (g_periodicity_check && s_period.periodic(counter))
172 {
173 break;
174 }
175 }
176 if (counter >= static_cast<unsigned long>(g_max_iterations)) // if not periodic, go the distance
177 {
178 for (counter = 0; counter < m_filter_cycles ; counter++)
179 {
180 if (g_cur_fractal_specific->orbit_calc())
181 {
182 return;
183 }
184 }
185 }
186 }
187
188 if (g_periodicity_check)
189 {
190 s_period.init();
191 }
192 for (unsigned long counter = 0UL; counter < static_cast<unsigned long>(g_max_iterations) ; counter++)
193 {
194 if (g_cur_fractal_specific->orbit_calc())
195 {
196 return;
197 }
198
199 // assign population value to Y coordinate in pixels
200 const unsigned int pixel_row = g_i_stop_pt.y - static_cast<int>((g_population - g_init.y) / g_delta_y);
201
202 // if it's visible on the screen, save it in the column array
203 if (pixel_row <= static_cast<unsigned int>(g_i_stop_pt.y))
204 {
205 m_verhulst[ pixel_row ] ++;
206 }
207 if (g_periodicity_check && s_period.periodic(counter))
208 {
209 if (pixel_row <= static_cast<unsigned int>(g_i_stop_pt.y))
210 {
211 m_verhulst[ pixel_row ] --;
212 }
213 break;
214 }
215 }

226// Bifurcation Population Periodicity Check
227// Returns : true if periodicity found, else false
228bool BifurcationPeriod::periodic(const long time)
229{
230 if ((time & saved_and) == 0) // time to save a new value
231 {
232 saved_pop = g_population;
233 if (--saved_inc == 0)
234 {
235 saved_and = (saved_and << 1) + 1;
236 saved_inc = 4;
237 }
238 }
239 else // check against an old save
240 {
241 if (std::abs(saved_pop - g_population) <= close_enough)
242 {
243 return true;
244 }
245 }
246 return false;
247}

249int bifurc_verhulst_trig_orbit()
250{
251 // Population = Pop + Rate * fn(Pop) * (1 - fn(Pop))
252 g_tmp_z.x = g_population;
253 g_tmp_z.y = 0;
254 cmplx_trig0(g_tmp_z, g_tmp_z);
255 g_population += g_rate * g_tmp_z.x * (1 - g_tmp_z.x);
256 return population_orbit();
257}
