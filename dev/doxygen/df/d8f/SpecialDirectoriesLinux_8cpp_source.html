42std::string_view trim(std::string_view text)
43{
44 while (!text.empty() && is_space(text.front()))

46 text.remove_prefix(1);

55std::string_view unquote(std::string_view text)
56{
57 text = trim(text);
58 if (text.size() >= 2 &&
59 ((text.front() == '"' && text.back() == '"') || (text.front() == '\'' && text.back() == '\'')))
60 {
61 text.remove_prefix(1);
62 text.remove_suffix(1);
63 }
64 return text;

78fs::path expand_user_dir(std::string_view value)
79{
80 value = unquote(value);
81 const fs::path home{home_dir()};
82 if (value == "$HOME" || value == "${HOME}")
83 {
84 return home;
85 }
86 if (starts_with(value, "$HOME/"))
87 {
88 return home / std::string{value.substr(6)};
89 }
90 if (starts_with(value, "${HOME}/"))
91 {
92 return home / std::string{value.substr(8)};
93 }
94
95 const fs::path path{std::string{value}};
96 if (path.is_absolute() || home.empty())
97 {
98 return path;
99 }
100 return home / path;
101}

114std::optional<fs::path> read_user_documents_dir()
115{
116 const fs::path config_home{xdg_config_home()};
117 if (config_home.empty())
118 {
119 return {};
120 }
121
122 std::ifstream user_dirs{config_home / USER_DIRS_FILE};
123 std::string line;
124 while (std::getline(user_dirs, line))
125 {
126 const std::string_view text{trim(line)};
127 if (text.empty() || text.front() == '#')
128 {
129 continue;
130 }
131 constexpr std::string_view SEPARATOR{"="};
132 const std::size_t separator{text.find(SEPARATOR)};
133 if (separator == std::string_view::npos || trim(text.substr(0, separator)) != DOCUMENTS_KEY)
134 {
135 continue;
136 }
137 const fs::path path{expand_user_dir(text.substr(separator + SEPARATOR.size()))};
138 if (!path.empty())
139 {
140 return path;
141 }
142 }
143 return {};
144}

146fs::path LinuxSpecialDirectories::program_dir() const
147{
148 std::error_code err;
149 const fs::path executable{fs::read_symlink(PROC_SELF_EXE, err)};
150 if (err)

157fs::path LinuxSpecialDirectories::documents_dir() const
158{
159 if (const char *docs = std::getenv("XDG_DOCUMENTS_DIR"); docs != nullptr && docs[0] != '\0')
160 {
161 return expand_user_dir(docs);
162 }
163 if (const std::optional<fs::path> docs{read_user_documents_dir()}; docs.has_value())
164 {
165 return *docs;
166 }
167
168 const fs::path home{home_dir()};
169 return home.empty() ? fs::current_path() : home / "Documents";
170}
