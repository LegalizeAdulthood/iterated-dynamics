58/* defines for flags; done this way instead of bit union to ensure ordering;
59 these bits represent the sort sequence for video mode list */
60enum
61{
62 VI_EXACT = 0x8000, // unless the one and only exact match
63 VI_DISK = 1024, // if video mode is disk video
64 VI_NO_KEY = 512, // if no function key assigned
65 VI_SCREEN_SMALLER = 128, // screen smaller than file's screen
66 VI_SCREEN_BIGGER = 64, // screen bigger than file's screen
67 VI_VIEW_SMALLER = 32, // screen smaller than file's view
68 VI_VIEW_BIGGER = 16, // screen bigger than file's view
69 VI_COLORS_SMALLER = 8, // mode has too few colors
70 VI_COLORS_BIGGER = 4, // mode has excess colors
71 VI_BAD_ASPECT = 1 // aspect ratio bad
72};

92static bool video_choice_less(const VideoModeChoice &lhs, const VideoModeChoice &rhs)
93{
94 if (lhs.flags < rhs.flags)
95 {
96 return true;
97 }
98 if (lhs.flags > rhs.flags)
99 {
100 return false;
101 }
102 if (g_video_table[lhs.entry_num].key < g_video_table[rhs.entry_num].key)
103 {
104 return true;
105 }
106 if (g_video_table[lhs.entry_num].key > g_video_table[rhs.entry_num].key)
107 {
108 return false;
109 }
110 if (lhs.entry_num < rhs.entry_num)
111 {
112 return true;
113 }
114 return false;
115}

128static double video_aspect(const int try_x_dots, const int try_y_dots)
129{
130 // calc resulting aspect ratio for specified dots in current mode
131 return static_cast<double>(try_y_dots) / static_cast<double>(try_x_dots)
132 * static_cast<double>(g_video_entry.x_dots) / static_cast<double>(g_video_entry.y_dots)
133 * g_screen_aspect;
134}

161int get_video_mode(FractalInfo *info, ExtBlock3 *blk_3_info)
162{
163 double f_temp;
164
165 g_init_mode = -1;
166
167 // try to find exact match for vid mode: first look for non-disk video
168 const VideoInfo *begin = g_video_table;
169 const VideoInfo *end = g_video_table + g_video_table_len;
170 auto it = std::find_if(begin, end,
171 [=](const VideoInfo &mode)
172 {
173 return info->x_dots == mode.x_dots //
174 && info->y_dots == mode.y_dots //
175 && g_file_colors == mode.colors //
176 && mode.driver != nullptr //
177 && !mode.driver->is_disk();
178 });
179 if (it == end)
180 {
181 it = std::find_if(begin, end,
182 [=](const VideoInfo &mode)
183 {
184 return info->x_dots == mode.x_dots //
185 && info->y_dots == mode.y_dots //
186 && g_file_colors == mode.colors;
187 });
188 }
189 if (it != end)
190 {
191 g_init_mode = static_cast<int>(it - begin);
192 }
193
194 // exit in makepar mode if no exact match of video mode in file
195 if (g_make_parameter_file && g_init_mode == -1)
196 {
197 return 0;
198 }
199
200 // setup table entry for each vid mode, flagged for how well it matches
201 s_video_choices.resize(g_video_table_len);
202 for (int i = 0; i < g_video_table_len; ++i)
203 {
204 g_video_entry = g_video_table[i];
205 unsigned tmp_flags = VI_EXACT;
206 if (g_video_entry.driver != nullptr && g_video_entry.driver->is_disk())
207 {
208 tmp_flags |= VI_DISK;
209 }
210 if (g_video_entry.key == 0)
211 {
212 tmp_flags |= VI_NO_KEY;
213 }
214 if (info->x_dots > g_video_entry.x_dots || info->y_dots > g_video_entry.y_dots)
215 {
216 tmp_flags |= VI_SCREEN_SMALLER;
217 }
218 else if (info->x_dots < g_video_entry.x_dots || info->y_dots < g_video_entry.y_dots)
219 {
220 tmp_flags |= VI_SCREEN_BIGGER;
221 }
222 if (g_file_x_dots > g_video_entry.x_dots || g_file_y_dots > g_video_entry.y_dots)
223 {
224 tmp_flags |= VI_VIEW_SMALLER;
225 }
226 else if (g_file_x_dots < g_video_entry.x_dots || g_file_y_dots < g_video_entry.y_dots)
227 {
228 tmp_flags |= VI_VIEW_BIGGER;
229 }
230 if (g_file_colors > g_video_entry.colors)
231 {
232 tmp_flags |= VI_COLORS_SMALLER;
233 }
234 if (g_file_colors < g_video_entry.colors)
235 {
236 tmp_flags |= VI_COLORS_BIGGER;
237 }
238 if (i == g_init_mode)
239 {
240 tmp_flags -= VI_EXACT;
241 }
242 if (g_file_aspect_ratio != 0.0f && (tmp_flags & VI_VIEW_SMALLER) == 0)
243 {
244 f_temp = video_aspect(g_file_x_dots, g_file_y_dots);
245 if (f_temp < g_file_aspect_ratio * 0.98
246 || f_temp > g_file_aspect_ratio * 1.02)
247 {
248 tmp_flags |= VI_BAD_ASPECT;
249 }
250 }
251 s_video_choices[i].entry_num = i;
252 // cppcheck-suppress unreadVariable
253 s_video_choices[i].flags = tmp_flags;
254 }
255
256 if (g_fast_restore && !g_ask_video)
257 {
258 g_init_mode = g_adapter;
259 }
260
261 bool got_real_mode = false;
262 if ((g_init_mode < 0 || (g_ask_video && g_init_batch == BatchMode::NONE)) && !g_make_parameter_file)
263 {
264 // no exact match or (askvideo=yes and batch=no), and not in makepar mode, talk to user
265 std::sort(s_video_choices.begin(), s_video_choices.end(), video_choice_less);
266
267 std::vector attributes(g_video_table_len, 1);
268
269 // format heading
270 std::string heading{fmt::format("File: {:<44s} {:d} x {:d} x {:d}\n"
271 "{:<52s}",
272 trim_filename(g_read_filename, 44), //
273 g_file_x_dots, g_file_y_dots, g_file_colors, //
274 heading_detail(info, blk_3_info))};
275 if (info->info_id[0] != 'G')
276 {
277 if (info->system)
278 {
279 heading += "Id ";
280 }
281 heading += to_display_string(g_file_version);
282 }
283 heading += '\n';
284 if (info->info_id[0] != 'G' && info->system == +SaveSystem::DOS)
285 {
286 if (g_init_mode < 0)
287 {
288 heading += "Saved in unknown video mode.";
289 }
290 else
291 {
292 char buff[80];
293 format_video_choice(g_init_mode, "", buff);
294 heading += buff;
295 }
296 }
297 if (g_file_aspect_ratio != 0 && g_file_aspect_ratio != g_screen_aspect)
298 {
299 heading += "\n"
300 "WARNING: non-standard aspect ratio; loading will change your <v>iew settings";
301 }
302 heading += '\n';
303 // set up instructions
304 std::string instructions{"Select a video mode. Use the cursor keypad to move the pointer.\n"
305 "Press ENTER for selected mode, or use a video mode function key.\n"
306 "Press F1 for help, "};
307 if (info->info_id[0] != 'G')
308 {
309 instructions += "TAB for fractal information, ";
310 }
311 instructions += "ESCAPE to back out.";
312
313 int i;
314 {
315 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_LOAD_FILE};
316 i = full_screen_choice(ChoiceFlags::NONE, heading.c_str(),
317 "key...name......................err...xdot..ydot.clr.comment..................",
318 instructions.c_str(), g_video_table_len, nullptr, attributes.data(), 1, 13, 78, 0, format_item,
319 nullptr, nullptr, check_mode_key);
320 }
321 if (i == -1)
322 {
323 return -1;
324 }
325 if (i < 0) // returned -100 - g_video_table entry number
326 {
327 g_init_mode = -100 - i;
328 got_real_mode = true;
329 }
330 else
331 {
332 g_init_mode = s_video_choices[i].entry_num;
333 }
334 }
335
336 if (!got_real_mode) // translate from temp table to permanent
337 {
338 int i = g_init_mode;
339 int j = g_video_table[i].key;
340 if (j != 0)
341 {
342 for (g_init_mode = 0; g_init_mode < MAX_VIDEO_MODES-1; ++g_init_mode)
343 {
344 if (g_video_table[g_init_mode].key == j)
345 {
346 break;
347 }
348 }
349 if (g_init_mode >= MAX_VIDEO_MODES-1)
350 {
351 j = 0;
352 }
353 }
354 if (j == 0) // mode has no key, add to reserved slot at end
355 {
356 std::memcpy(
357 &g_video_table[g_init_mode = MAX_VIDEO_MODES - 1],
358 &g_video_table[i], sizeof(*g_video_table));
359 }
360 }
361
362 // ok, we're going to return with a video mode
363 g_video_entry = g_video_table[g_init_mode];
364
365 if (g_viewport.enabled //
366 && g_file_x_dots == g_video_entry.x_dots //
367 && g_file_y_dots == g_video_entry.y_dots)
368 {
369 // pull image into a view window
370 if (g_calc_status != CalcStatus::COMPLETED) // if not complete
371 {
372 g_calc_status = CalcStatus::PARAMS_CHANGED; // can't resume anyway
373 }
374 if (g_viewport.x_dots)
375 {
376 g_viewport.reduction = static_cast<float>(g_video_entry.x_dots / g_viewport.x_dots);
377 g_viewport.y_dots = 0;
378 g_viewport.x_dots = 0; // easier to use auto reduction
379 }
380 g_viewport.reduction = std::round(g_viewport.reduction); // need integer value
381 g_skip_y_dots = static_cast<short>(g_viewport.reduction - 1);
382 g_skip_x_dots = static_cast<short>(g_viewport.reduction - 1);
383 return 0;
384 }
385
386 g_skip_y_dots = 0;
387 g_skip_x_dots = 0; // set for no reduction
388 if (g_video_entry.x_dots < g_file_x_dots || g_video_entry.y_dots < g_file_y_dots)
389 {
390 // set up to load only every nth pixel to make image fit
391 if (g_calc_status != CalcStatus::COMPLETED) // if not complete
392 {
393 g_calc_status = CalcStatus::PARAMS_CHANGED; // can't resume anyway
394 }
395 g_skip_y_dots = 1;
396 g_skip_x_dots = 1;
397 while (g_skip_x_dots * g_video_entry.x_dots < g_file_x_dots)
398 {
399 ++g_skip_x_dots;
400 }
401 while (g_skip_y_dots * g_video_entry.y_dots < g_file_y_dots)
402 {
403 ++g_skip_y_dots;
404 }
405 int i = 0;
406 int j = 0;
407 int tmp_x_dots;
408 int tmp_y_dots;
409 while (true)
410 {
411 tmp_x_dots = (g_file_x_dots + g_skip_x_dots - 1) / g_skip_x_dots;
412 tmp_y_dots = (g_file_y_dots + g_skip_y_dots - 1) / g_skip_y_dots;
413 // reduce further if that improves aspect
414 f_temp = video_aspect(tmp_x_dots, tmp_y_dots);
415 if (f_temp > g_file_aspect_ratio)
416 {
417 if (j)
418 {
419 break; // already reduced x, don't reduce y
420 }
421 const double f_temp2 = video_aspect(tmp_x_dots, (g_file_y_dots+g_skip_y_dots)/(g_skip_y_dots+1));
422 if (f_temp2 < g_file_aspect_ratio
423 && f_temp/g_file_aspect_ratio *0.9 <= g_file_aspect_ratio/f_temp2)
424 {
425 break; // further y reduction is worse
426 }
427 ++g_skip_y_dots;
428 ++i;
429 }
430 else
431 {
432 if (i)
433 {
434 break; // already reduced y, don't reduce x
435 }
436 const double f_temp2 = video_aspect((g_file_x_dots+g_skip_x_dots)/(g_skip_x_dots+1), tmp_y_dots);
437 if (f_temp2 > g_file_aspect_ratio
438 && g_file_aspect_ratio/f_temp *0.9 <= f_temp2/g_file_aspect_ratio)
439 {
440 break; // further x reduction is worse
441 }
442 ++g_skip_x_dots;
443 ++j;
444 }
445 }
446 g_file_x_dots = tmp_x_dots;
447 g_file_y_dots = tmp_y_dots;
448 --g_skip_x_dots;
449 --g_skip_y_dots;
450 }
451
452 g_viewport.final_aspect_ratio = g_file_aspect_ratio;
453 if (g_viewport.final_aspect_ratio == 0) // assume display correct
454 {
455 g_viewport.final_aspect_ratio = static_cast<float>(video_aspect(g_file_x_dots, g_file_y_dots));
456 }
457 if (g_viewport.final_aspect_ratio >= g_screen_aspect-0.02
458 && g_viewport.final_aspect_ratio <= g_screen_aspect+0.02)
459 {
460 g_viewport.final_aspect_ratio = g_screen_aspect;
461 }
462 {
463 int i = static_cast<int>(g_viewport.final_aspect_ratio * 1000.0 + 0.5);
464 g_viewport.final_aspect_ratio = static_cast<float>(i / 1000.0); // chop precision to 3 decimals
465 }
466
467 // setup view window stuff
468 g_viewport.enabled = false;
469 g_viewport.x_dots = 0;
470 g_viewport.y_dots = 0;
471 if (g_file_x_dots != g_video_entry.x_dots || g_file_y_dots != g_video_entry.y_dots)
472 {
473 // image not exactly same size as screen
474 g_viewport.enabled = true;
475 f_temp = g_viewport.final_aspect_ratio
476 * static_cast<double>(g_video_entry.y_dots) / static_cast<double>(g_video_entry.x_dots)
477 / g_screen_aspect;
478 float tmp_reduce;
479 int i;
480 int j;
481 if (g_viewport.final_aspect_ratio <= g_screen_aspect)
482 {
483 i = static_cast<int>(std::lround(
484 static_cast<double>(g_video_entry.x_dots) / static_cast<double>(g_file_x_dots) * 20.0));
485 tmp_reduce = static_cast<float>(i / 20.0); // chop precision to nearest .05
486 i = static_cast<int>(std::lround(static_cast<double>(g_video_entry.x_dots) / tmp_reduce));
487 j = static_cast<int>(std::lround(static_cast<double>(i) * f_temp));
488 }
489 else
490 {
491 i = static_cast<int>(std::lround(
492 static_cast<double>(g_video_entry.y_dots) / static_cast<double>(g_file_y_dots) * 20.0));
493 tmp_reduce = static_cast<float>(i / 20.0); // chop precision to nearest .05
494 j = static_cast<int>(std::lround(static_cast<double>(g_video_entry.y_dots) / tmp_reduce));
495 i = static_cast<int>(std::lround(static_cast<double>(j) / f_temp));
496 }
497 if (i != g_file_x_dots || j != g_file_y_dots) // too bad, must be explicit
498 {
499 g_viewport.x_dots = g_file_x_dots;
500 g_viewport.y_dots = g_file_y_dots;
501 }
502 else
503 {
504 g_viewport.reduction = tmp_reduce; // ok, this works
505 }
506 }
507 if (!g_make_parameter_file
508 && !g_fast_restore
509 && g_init_batch == BatchMode::NONE
510 && (std::abs(g_viewport.final_aspect_ratio - g_screen_aspect) > .00001 || g_viewport.x_dots != 0))
511 {
512 stop_msg(StopMsgFlags::NO_BUZZER,
513 "Warning: <V>iew parameters are being set to non-standard values.\n"
514 "Remember to reset them when finished with this image!");
515 }
516 return 0;
517}

519static void format_item(const int choice, char *buf)
520{
521 char err_buf[10];
522 err_buf[0] = 0;
523 const unsigned tmp_flags = s_video_choices[choice].flags;
524 if (tmp_flags & VI_VIEW_SMALLER + VI_COLORS_SMALLER + VI_BAD_ASPECT)
525 {
526 std::strcat(err_buf, "*");
527 }
528 if (tmp_flags & VI_DISK)
529 {
530 std::strcat(err_buf, "D");
531 }
532 if (tmp_flags & VI_VIEW_SMALLER)
533 {
534 std::strcat(err_buf, "R");
535 }
536 if (tmp_flags & VI_COLORS_SMALLER)
537 {
538 std::strcat(err_buf, "C");
539 }
540 if (tmp_flags & VI_BAD_ASPECT)
541 {
542 std::strcat(err_buf, "A");
543 }
544 if (tmp_flags & VI_VIEW_BIGGER)
545 {
546 std::strcat(err_buf, "v");
547 }
548 if (tmp_flags & VI_COLORS_BIGGER)
549 {
550 std::strcat(err_buf, "c");
551 }
552 format_video_choice(s_video_choices[choice].entry_num, err_buf, buf);
553}
