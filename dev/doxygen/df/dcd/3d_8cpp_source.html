3// A word about the 3D library. Even though this library supports
4// three dimensions, the matrices are 4x4 for the following reason.
5// With normal 3-dimensional vectors, translation is an ADDITION,
6// and rotation is a MULTIPLICATION. A vector {x,y,z} is represented
7// as a 4-tuple {x,y,z,1}. It is then possible to define a 4x4
8// matrix such that multiplying the vector by the matrix translates
9// the vector. This allows combinations of translation and rotation
10// to be obtained in a single matrix by multiplying a translation
11// matrix and a rotation matrix together. Note that in the code,
12// vectors have three components; since the fourth component is
13// always 1, that value is not included in the vector variable to
14// save space, but the routines make use of the fourth component
15// (see vec_mult()). Similarly, the fourth column of EVERY matrix is
16// always
17// 0
18// 0
19// 0
20// 1
21// but currently the C version of a matrix includes this even though
22// it could be left out of the data structure and assumed in the
23// routines. Vectors are ROW vectors, and are always multiplied with
24// matrices FROM THE LEFT (e.g. vector*matrix). Also note the order
25// of indices of a matrix is matrix[row][column], and in usual C
26// fashion, numbering starts with 0.
27//
28// TRANSLATION MATRIX = 1 0 0 0
29// 0 1 0 0
30// 0 0 1 0
31// Tx Ty Tz 1
32//
33// SCALE MATRIX = Sx 0 0 0
34// 0 Sy 0 0
35// 0 0 Sz 0
36// 0 0 0 1
37//
38// Rotation about x axis i degrees:
39// ROTX(i) = 1 0 0 0
40// 0 cosi sini 0
41// 0 -sini cosi 0
42// 0 0 0 1
43//
44// Rotation about y axis i degrees:
45// ROTY(i) = cosi 0 -sini 0
46// 0 1 0 0
47// sini 0 cosi 0
48// 0 0 0 1
49//
50// Rotation about z axis i degrees:
51// ROTZ(i) = cosi sini 0 0
52// -sini cosi 0 0
53// 0 0 1 0
54// 0 0 0 1
55//

96// initialize a matrix and set to identity matrix (all 0's, 1's on diagonal)
97void identity(Matrix m)
98{
99 for (int i = 0 ; i < COL_MAX; i++)
100 {
101 for (int j = 0; j < ROW_MAX; j++)
102 {
103 if (i == j)
104 {
105 m[j][i] = 1.0;
106 }
107 else
108 {
109 m[j][i] = 0.0;
110 }
111 }
112 }
113}

115// Multiply two matrices
116void mat_mul(Matrix lhs, Matrix rhs, Matrix result)
117{
118 // result stored in product to avoid problems
119 // in case parameter mat3 == mat2 or mat 1
120 Matrix product;
121 for (int i = 0; i < 4; i++)
122 {
123 for (int j = 0; j < 4; j++)
124 {
125 product[j][i] = //
126 lhs[j][0] * rhs[0][i] + //
127 lhs[j][1] * rhs[1][i] + //
128 lhs[j][2] * rhs[2][i] + //
129 lhs[j][3] * rhs[3][i];
130 }
131 }
132 std::memcpy(result, product, sizeof(product));
133}

199// cross product - useful because cross is perpendicular to v and w
200int cross_product(Vector v, Vector w, Vector cross)
201{
202 Vector tmp;
203 tmp[0] = v[1]*w[2] - w[1]*v[2];
204 tmp[1] = w[0]*v[2] - v[0]*w[2];
205 tmp[2] = v[0]*w[1] - w[0]*v[1];
206 cross[0] = tmp[0];
207 cross[1] = tmp[1];
208 cross[2] = tmp[2];
209 return 0;
210}

212// normalize a vector to length 1
213bool normalize_vector(Vector v)
214{
215 double length = dot_product(v, v);
216
217 // bailout if zero length
218 if (length < FLT_MIN || length > FLT_MAX)
219 {
220 return true;
221 }
222 length = std::sqrt(length);
223 if (length < FLT_MIN)
224 {
225 return true;
226 }
227
228 v[0] /= length;
229 v[1] /= length;
230 v[2] /= length;
231 return false;
232}

234// multiply source vector s by matrix m, result in target t
235// used to apply transformations to a vector
236int vec_mat_mul(Vector s, Matrix m, Vector t)
237{
238 Vector tmp;
239 for (int j = 0; j < COL_MAX-1; j++)
240 {
241 tmp[j] = 0.0;
242 for (int i = 0; i < ROW_MAX-1; i++)
243 {
244 tmp[j] += s[i]*m[i][j];
245 }
246 // vector is really four dimensional with last component always 1
247 tmp[j] += m[3][j];
248 }
249 // set target = tmp. Necessary to use tmp in case source = target
250 std::memcpy(t, tmp, sizeof(tmp));
251 return 0;
252}

275// perspective projection of vector v with respect to viewpoint vector view
276int perspective(Vector v)
277{
278 const double denom = g_view[2] - v[2];
279
280 if (denom >= 0.0)
281 {
282 v[0] = BAD_VALUE; // clipping will catch these values
283 v[1] = BAD_VALUE; // so they won't plot values BEHIND viewer
284 v[2] = BAD_VALUE;
285 return -1;
286 }
287 v[0] = (v[0]*g_view[2] - g_view[0]*v[2])/denom;
288 v[1] = (v[1]*g_view[2] - g_view[1]*v[2])/denom;
289
290 // calculation of z if needed later
291 // v[2] = v[2]/denom;
292 return 0;
293}

295// long version of vmult and perspective combined for speed
296int long_vec_mat_mul_persp(VectorL s, MatrixL m, VectorL t0, VectorL t, VectorL view, const int bit_shift)
297{
298 // s: source vector
299 // m: transformation matrix
300 // t0: after transformation, before persp
301 // t: target vector
302 // lview: perspective viewer coordinates
303 // bitshift: fixed point conversion bitshift
304 VectorL tmp;
305 g_overflow = false;
306 int k = COL_MAX - 1; // shorten the math if non-perspective and non-illum
307 if (view[2] == 0 && t0[0] == 0)
308 {
309 k--;
310 }
311
312 for (int j = 0; j < k; j++)
313 {
314 tmp[j] = 0;
315 for (int i = 0; i < ROW_MAX-1; i++)
316 {
317 tmp[j] += multiply(s[i], m[i][j], bit_shift);
318 }
319 // vector is really four dimensional with last component always 1
320 tmp[j] += m[3][j];
321 }
322 if (t0[0]) // first component of t0 used as flag
323 {
324 // faster than for loop, if less general
325 t0[0] = tmp[0];
326 t0[1] = tmp[1];
327 t0[2] = tmp[2];
328 }
329 if (view[2] != 0) // perspective 3D
330 {
331 const long denom = view[2] - tmp[2];
332 if (denom >= 0) // bail out if point is "behind" us
333 {
334 t[0] = BAD_VALUE;
335 t[0] = t[0] << bit_shift;
336 t[1] = t[0];
337 t[2] = t[0];
338 return -1;
339 }
340
341 // doing math in this order helps prevent overflow
342 VectorL tmp_view;
343 tmp_view[0] = divide(view[0], denom, bit_shift);
344 tmp_view[1] = divide(view[1], denom, bit_shift);
345 tmp_view[2] = divide(view[2], denom, bit_shift);
346
347 tmp[0] = multiply(tmp[0], tmp_view[2], bit_shift) -
348 multiply(tmp_view[0], tmp[2], bit_shift);
349
350 tmp[1] = multiply(tmp[1], tmp_view[2], bit_shift) -
351 multiply(tmp_view[1], tmp[2], bit_shift);
352 }
353
354 // set target = tmp. Necessary to use tmp in case source = target
355 // faster than for loop, if less general
356 t[0] = tmp[0];
357 t[1] = tmp[1];
358 t[2] = tmp[2];
359 return g_overflow ? 1 : 0;
360}
