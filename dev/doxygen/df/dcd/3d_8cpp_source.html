3// A word about the 3D library. Even though this library supports
4// three dimensions, the matrices are 4x4 for the following reason.
5// With normal 3-dimensional vectors, translation is an ADDITION,
6// and rotation is a MULTIPLICATION. A vector {x,y,z} is represented
7// as a 4-tuple {x,y,z,1}. It is then possible to define a 4x4
8// matrix such that multiplying the vector by the matrix translates
9// the vector. This allows combinations of translation and rotation
10// to be obtained in a single matrix by multiplying a translation
11// matrix and a rotation matrix together. Note that in the code,
12// vectors have three components; since the fourth component is
13// always 1, that value is not included in the vector variable to
14// save space, but the routines make use of the fourth component
15// (see vec_mult()). Similarly, the fourth column of EVERY matrix is
16// always
17// 0
18// 0
19// 0
20// 1
21// but currently the C version of a matrix includes this even though
22// it could be left out of the data structure and assumed in the
23// routines. Vectors are ROW vectors, and are always multiplied with
24// matrices FROM THE LEFT (e.g. vector*matrix). Also note the order
25// of indices of a matrix is matrix[row][column], and in usual C
26// fashion, numbering starts with 0.
27//
28// TRANSLATION MATRIX = 1 0 0 0
29// 0 1 0 0
30// 0 0 1 0
31// Tx Ty Tz 1
32//
33// SCALE MATRIX = Sx 0 0 0
34// 0 Sy 0 0
35// 0 0 Sz 0
36// 0 0 0 1
37//
38// Rotation about x axis i degrees:
39// ROTX(i) = 1 0 0 0
40// 0 cosi sini 0
41// 0 -sini cosi 0
42// 0 0 0 1
43//
44// Rotation about y axis i degrees:
45// ROTY(i) = cosi 0 -sini 0
46// 0 1 0 0
47// sini 0 cosi 0
48// 0 0 0 1
49//
50// Rotation about z axis i degrees:
51// ROTZ(i) = cosi sini 0 0
52// -sini cosi 0 0
53// 0 0 1 0
54// 0 0 0 1
55//

97// initialize a matrix and set to identity matrix (all 0's, 1's on diagonal)
98void identity(Matrix m)
99{
100 for (int i = 0 ; i < COL_MAX; i++)
101 {
102 for (int j = 0; j < ROW_MAX; j++)
103 {
104 if (i == j)
105 {
106 m[j][i] = 1.0;
107 }
108 else
109 {
110 m[j][i] = 0.0;
111 }
112 }
113 }
114}

116// Multiply two matrices
117void mat_mul(Matrix lhs, Matrix rhs, Matrix result)
118{
119 // result stored in product to avoid problems
120 // in case parameter mat3 == mat2 or mat 1
121 Matrix product;
122 for (int i = 0; i < 4; i++)
123 {
124 for (int j = 0; j < 4; j++)
125 {
126 product[j][i] = //
127 lhs[j][0] * rhs[0][i] + //
128 lhs[j][1] * rhs[1][i] + //
129 lhs[j][2] * rhs[2][i] + //
130 lhs[j][3] * rhs[3][i];
131 }
132 }
133 std::memcpy(result, product, sizeof(product));
134}

200// cross product - useful because cross is perpendicular to v and w
201int cross_product(Vector v, Vector w, Vector cross)
202{
203 Vector tmp;
204 tmp[0] = v[1]*w[2] - w[1]*v[2];
205 tmp[1] = w[0]*v[2] - v[0]*w[2];
206 tmp[2] = v[0]*w[1] - w[0]*v[1];
207 cross[0] = tmp[0];
208 cross[1] = tmp[1];
209 cross[2] = tmp[2];
210 return 0;
211}

213// normalize a vector to length 1
214bool normalize_vector(Vector v)
215{
216 double length = dot_product(v, v);
217
218 // bailout if zero length
219 if (length < std::numeric_limits<float>::min() || length > std::numeric_limits<float>::max())
220 {
221 return true;
222 }
223 length = std::sqrt(length);
224 if (length < std::numeric_limits<float>::min())
225 {
226 return true;
227 }
228
229 v[0] /= length;
230 v[1] /= length;
231 v[2] /= length;
232 return false;
233}

235// multiply source vector s by matrix m, result in target t
236// used to apply transformations to a vector
237int vec_mat_mul(Vector s, Matrix m, Vector t)
238{
239 Vector tmp;
240 for (int j = 0; j < COL_MAX-1; j++)
241 {
242 tmp[j] = 0.0;
243 for (int i = 0; i < ROW_MAX-1; i++)
244 {
245 tmp[j] += s[i]*m[i][j];
246 }
247 // vector is really four dimensional with last component always 1
248 tmp[j] += m[3][j];
249 }
250 // set target = tmp. Necessary to use tmp in case source = target
251 std::memcpy(t, tmp, sizeof(tmp));
252 return 0;
253}

276// perspective projection of vector v with respect to viewpoint vector view
277int perspective(Vector v)
278{
279 const double denom = g_view[2] - v[2];
280
281 if (denom >= 0.0)
282 {
283 v[0] = BAD_VALUE; // clipping will catch these values
284 v[1] = BAD_VALUE; // so they won't plot values BEHIND viewer
285 v[2] = BAD_VALUE;
286 return -1;
287 }
288 v[0] = (v[0]*g_view[2] - g_view[0]*v[2])/denom;
289 v[1] = (v[1]*g_view[2] - g_view[1]*v[2])/denom;
290
291 // calculation of z if needed later
292 // v[2] = v[2]/denom;
293 return 0;
294}

296// long version of vmult and perspective combined for speed
297int long_vec_mat_mul_persp(VectorL s, MatrixL m, VectorL t0, VectorL t, VectorL view, const int bit_shift)
298{
299 // s: source vector
300 // m: transformation matrix
301 // t0: after transformation, before persp
302 // t: target vector
303 // lview: perspective viewer coordinates
304 // bitshift: fixed point conversion bitshift
305 VectorL tmp;
306 g_overflow = false;
307 int k = COL_MAX - 1; // shorten the math if non-perspective and non-illum
308 if (view[2] == 0 && t0[0] == 0)
309 {
310 k--;
311 }
312
313 for (int j = 0; j < k; j++)
314 {
315 tmp[j] = 0;
316 for (int i = 0; i < ROW_MAX-1; i++)
317 {
318 tmp[j] += multiply(s[i], m[i][j], bit_shift);
319 }
320 // vector is really four dimensional with last component always 1
321 tmp[j] += m[3][j];
322 }
323 if (t0[0]) // first component of t0 used as flag
324 {
325 // faster than for loop, if less general
326 t0[0] = tmp[0];
327 t0[1] = tmp[1];
328 t0[2] = tmp[2];
329 }
330 if (view[2] != 0) // perspective 3D
331 {
332 const long denom = view[2] - tmp[2];
333 if (denom >= 0) // bail out if point is "behind" us
334 {
335 t[0] = BAD_VALUE;
336 t[0] = t[0] << bit_shift;
337 t[1] = t[0];
338 t[2] = t[0];
339 return -1;
340 }
341
342 // doing math in this order helps prevent overflow
343 VectorL tmp_view;
344 tmp_view[0] = divide(view[0], denom, bit_shift);
345 tmp_view[1] = divide(view[1], denom, bit_shift);
346 tmp_view[2] = divide(view[2], denom, bit_shift);
347
348 tmp[0] = multiply(tmp[0], tmp_view[2], bit_shift) -
349 multiply(tmp_view[0], tmp[2], bit_shift);
350
351 tmp[1] = multiply(tmp[1], tmp_view[2], bit_shift) -
352 multiply(tmp_view[1], tmp[2], bit_shift);
353 }
354
355 // set target = tmp. Necessary to use tmp in case source = target
356 // faster than for loop, if less general
357 t[0] = tmp[0];
358 t[1] = tmp[1];
359 t[2] = tmp[2];
360 return g_overflow ? 1 : 0;
361}
