37constexpr auto ANT_DELAY_SLEEP_CHUNK = std::chrono::milliseconds{1};
38
39void wait_for_ant_delay(const long delay_100us)
40{
41 const auto deadline = Clock::now() + std::chrono::microseconds{delay_100us * 100};
42 driver_flush();
43 auto now = Clock::now();
44 while (now < deadline)
45 {
46 const auto remaining = deadline - now;
47 if (remaining > ANT_DELAY_SLEEP_CHUNK)
48 {
49 driver_delay(static_cast<int>(ANT_DELAY_SLEEP_CHUNK.count()));
50 }
51 else
52 {
53 driver_delay(0);
54 }
55 now = Clock::now();
56 }
57}

112static long change_wait(long wait)
113{
114 while (true)
115 {
116 std::ostringstream msg;
117 msg << "Delay " << std::setw(4) << wait << " ";
118 show_temp_msg(msg.str());
119 switch (driver_get_key())
120 {
121 case ID_KEY_CTL_RIGHT_ARROW:
122 case ID_KEY_CTL_UP_ARROW:
123 wait += 100;
124 break;
125 case ID_KEY_RIGHT_ARROW:
126 case ID_KEY_UP_ARROW:
127 wait += 10;
128 break;
129 case ID_KEY_CTL_DOWN_ARROW:
130 case ID_KEY_CTL_LEFT_ARROW:
131 wait -= 100;
132 break;
133 case ID_KEY_LEFT_ARROW:
134 case ID_KEY_DOWN_ARROW:
135 wait -= 10;
136 break;
137 default:
138 clear_temp_msg();
139 return wait;
140 }
141 wait = std::max(wait, 0L);
142 }

145static bool ant_key(bool &step, long &wait)
146{
147 // check for a key only every inner_loop times
148 int key = driver_key_pressed();
149 const bool key_peeked{key != 0};
150 if (key || step)
151 {
152 if (key == 0)
153 {
154 key = driver_get_key();
155 }
156 switch (key)
157 {
158 case ID_KEY_SPACE:
159 step = !step;
160 break;
161 case ID_KEY_ESC:
162 return stop_ant(key, key_peeked);
163
164 case ID_KEY_RIGHT_ARROW:
165 case ID_KEY_UP_ARROW:
166 case ID_KEY_DOWN_ARROW:
167 case ID_KEY_LEFT_ARROW:
168 case ID_KEY_CTL_RIGHT_ARROW:
169 case ID_KEY_CTL_UP_ARROW:
170 case ID_KEY_CTL_DOWN_ARROW:
171 case ID_KEY_CTL_LEFT_ARROW:
172 wait = change_wait(wait);
173 break;
174
175 case ID_KEY_ENTER:
176 case ID_KEY_ENTER_2:
177 break;
178
179 default:
180 if (key_opens_options_menu(key))
181 {
182 if (key_peeked)
183 {
184 driver_get_key();
185 }
186 return run_options_menu(key);
187 }
188 return stop_ant(key, key_peeked);
189 }
190 if (key_peeked)
191 {
192 driver_get_key();
193 }
194 }
195 return false;
196}

198int ant_type()
199{
200 ValueSaver saved_help_mode(g_help_mode, HelpLabels::HELP_ANT_COMMANDS);
201 fractals::Ant ant;
202 long wait = std::abs(g_orbit_delay);
203
204 bool step = wait == 1;
205 if (step)
206 {
207 wait = 0;
208 }
209 while (!ant.done())
210 {
211 ant.iterate();
212 const bool batch_complete{ant.consume_batch_complete()};
213 if (!ant.done() && batch_complete && wait > 0 && !step)
214 {
215 wait_for_ant_delay(wait);
216 }
217 if (!ant.done() && batch_complete && step)
218 {
219 driver_flush();
220 driver_delay(0);
221 }
222 if (!ant.done() && batch_complete && ant_key(step, wait))
223 {
224 break;
225 }
226 }
227 return 0;
228}
