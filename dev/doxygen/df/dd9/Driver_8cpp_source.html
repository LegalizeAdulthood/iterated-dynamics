18// Array of drivers that are supported by source code in Id.
19// The default driver is first one in the list that initializes.
20static std::vector<Driver *> s_available;
21
22Driver *g_driver{};
23
24void load_driver(Driver *drv, int *argc, char **argv)
25{
26 if (drv == nullptr || !drv->init(argc, argv))
27 {
28 return;
29 }
30
31 if (!g_driver)
32 {
33 g_driver = drv;
34 }
35 s_available.push_back(drv);
36}

38//------------------------------------------------------------
39// init_drivers
40//
41// Go through the static list of drivers defined and try to initialize
42// them one at a time. Returns the number of drivers initialized.
43//
44int init_drivers(int *argc, char **argv)
45{
46#if ID_HAVE_WIN32_DISK_DRIVER

50#if ID_HAVE_GDI_DRIVER
51 load_driver(get_gdi_driver(), argc, argv);
52#endif
53
54#if ID_HAVE_WX_DRIVER
55 load_driver(get_wx_driver(), argc, argv);
56#endif
57#if ID_HAVE_WX_DISK_DRIVER
58 load_driver(get_wx_disk_driver(), argc, argv);
59#endif
60
61 return static_cast<int>(s_available.size()); // number of drivers supported at runtime
62}

64// add_video_mode
65//
66// a driver uses this to inform the system of an available video mode
67//
68void add_video_mode(Driver *drv, VideoInfo *mode)
69{
70#if defined(_WIN32)
71 _ASSERTE(g_video_table_len < MAX_VIDEO_MODES);
72#endif
73 // stash away driver pointer so we can init driver for selected mode
74 mode->driver = drv;
75 std::memcpy(&g_video_table[g_video_table_len], mode, sizeof(g_video_table[0]));
76 g_video_table_len++;
77}

79void close_drivers()
80{
81 // terminate drivers in reverse order of initialization
82 while (!s_available.empty())
83 {
84 Driver *drv = s_available.back();
85 s_available.pop_back();
86 drv->terminate();
87 }
88 g_driver = nullptr;
89}

91Driver *driver_find_by_name(const char *name)
92{
93 if (const auto it = std::find_if(
94 s_available.begin(), s_available.end(), [name](Driver *drv) { return drv->get_name() == name; });
95 it != s_available.end())

97 return *it;
98 }
99 return nullptr;
