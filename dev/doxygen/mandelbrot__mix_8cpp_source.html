44bool MandelbrotMix::setup()
45{
46 int sign_array = 0;
47 a.x = g_params[0];
48 a.y = 0.0; // a=real(p1),
49 b.x = g_params[1];
50 b.y = 0.0; // b=imag(p1),
51 d.x = g_params[2];
52 d.y = 0.0; // d=real(p2),
53 f.x = g_params[3];
54 f.y = 0.0; // f=imag(p2),
55 k.x = g_params[4] + 1.0;
56 k.y = 0.0; // k=real(p3)+1,
57 l.x = g_params[5] + 100.0;
58 l.y = 0.0; // l=imag(p3)+100,
59 cmplx_recip(f, g); // g=1/f,
60 cmplx_recip(d, h); // h=1/d,
61 g_tmp_z = f - b; // tmp = f-b
62 cmplx_recip(g_tmp_z, j); // j = 1/(f-b)
63 g_tmp_z = -a;
64 cmplx_mult(g_tmp_z, b, g_tmp_z); // z=(-a*b*g*h)^j,
65 cmplx_mult(g_tmp_z, g, g_tmp_z);
66 cmplx_mult(g_tmp_z, h, g_tmp_z);
67
68 /*
69 This code kludge attempts to duplicate the behavior
70 of the parser in determining the sign of zero of the
71 imaginary part of the argument of the power function. The
72 reason this is important is that the complex arctangent
73 returns PI in one case and -PI in the other, depending
74 on the sign bit of zero, and we wish the results to be
75 compatible with Jim Muth's mix4 formula using the parser.
76
77 First create a number encoding the signs of a, b, g , h. Our
78 kludge assumes that those signs determine the behavior.
79 */
80 if (a.x < 0.0)
81 {
82 sign_array += 8;
83 }
84 if (b.x < 0.0)
85 {
86 sign_array += 4;
87 }
88 if (g.x < 0.0)
89 {
90 sign_array += 2;
91 }
92 if (h.x < 0.0)
93 {
94 sign_array += 1;
95 }
96 if (g_tmp_z.y == 0.0) // we know tmp.y IS zero but ...
97 {
98 switch (sign_array)
99 {
100 /*
101 Add to this list the magic numbers of any cases
102 in which the fractal does not match the formula version
103 */
104 case 15: // 1111
105 case 10: // 1010
106 case 6: // 0110
107 case 5: // 0101
108 case 3: // 0011
109 case 0: // 0000
110 g_tmp_z.y = -g_tmp_z.y; // swap sign bit
111 default: // do nothing - remaining cases already OK
112 ;
113 }
114 // in case our kludge failed, let the user fix it
115 if (g_debug_flag == DebugFlags::MANDELBROT_MIX4_FLIP_SIGN)
116 {
117 g_tmp_z.y = -g_tmp_z.y;
118 }
119 }
120
121 cmplx_pwr(g_tmp_z, j, g_tmp_z); // note: z is old
122 // in case our kludge failed, let the user fix it
123 if (g_params[6] < 0.0)
124 {
125 g_tmp_z.y = -g_tmp_z.y;
126 }
127
128 if (g_user.bailout_value == 0)
129 {
130 g_magnitude_limit = l.x;
131 g_magnitude_limit2 = g_magnitude_limit * g_magnitude_limit;
132 }
133 return true;
134}

136int MandelbrotMix::per_pixel()
137{
138 if (g_inversion.invert != 0)
139 {
140 invertz2(&g_init);
141 }
142 else
143 {
144 g_init.x = dx_pixel();
145 g_init.y = dy_pixel();
146 }
147 g_old_z = g_tmp_z;
148 cmplx_trig0(g_init, c); // c=fn1(pixel):
149 return 0; // 1st iteration has been NOT been done
150}

152int MandelbrotMix::iterate()
153{
154 // z=k*((a*(z^b))+(d*(z^f)))+c,
155 DComplex z_b;
156 DComplex z_f;
157 cmplx_pwr(g_old_z, b, z_b); // (z^b)
158 cmplx_pwr(g_old_z, f, z_f); // (z^f)
159 g_new_z.x = k.x * a.x * z_b.x + k.x * d.x * z_f.x + c.x;
160 g_new_z.y = k.x * a.x * z_b.y + k.x * d.x * z_f.y + c.y;
161 return g_bailout_float();
162}
