27 BigStackSaver saved;

53 else
54 {
55 BigFloat tmp_bf{alloc_stack(g_bf_length + 2)};
56 sub_bf(tmp_bf, g_bf_y_max, g_bf_y_min);
57 mandel_width = bf_to_float(tmp_bf);
58 }
59
60 s_pert_engine.initialize_frame(center_bf, {center.x, center.y}, mandel_width / 2.0);
61 if (const int result = s_pert_engine.calculate_one_frame(); result < 0)
62 {
63 throw std::runtime_error("Failed to initialize perturbation engine (" + std::to_string(result) + ")");
64 }
65 g_calc_status = CalcStatus::COMPLETED;
66 return false;
67}
