45Ant::Ant()
46{
47 for (int i = 0; i < DIRS; i++)
48 {
49 inc_x[i].resize(g_logical_screen.x_dots + 2);
50 inc_y[i].resize(g_logical_screen.y_dots + 2);
51 }
52
53 // In this vectors put all the possible point that the ants can visit.
54 // Wrap them from a side to the other instead of simply end calculation
55 for (int i = 0; i < g_logical_screen.x_dots; i++)
56 {
57 inc_x[0][i] = i;
58 inc_x[2][i] = i;
59 }
60
61 for (int i = 0; i < g_logical_screen.x_dots; i++)
62 {
63 inc_x[3][i] = i + 1;
64 }
65 inc_x[3][g_logical_screen.x_dots - 1] = 0; // wrap from right of the screen to left
66
67 for (int i = 1; i < g_logical_screen.x_dots; i++)
68 {
69 inc_x[1][i] = i - 1;
70 }
71 inc_x[1][0] = g_logical_screen.x_dots - 1; // wrap from left of the screen to right
72
73 for (int i = 0; i < g_logical_screen.y_dots; i++)
74 {
75 inc_y[1][i] = i;
76 inc_y[3][i] = i;
77 }
78 for (int i = 0; i < g_logical_screen.y_dots; i++)
79 {
80 inc_y[0][i] = i + 1;
81 }
82 inc_y[0][g_logical_screen.y_dots - 1] = 0; // wrap from the top of the screen to the bottom
83 for (int i = 1; i < g_logical_screen.y_dots; i++)
84 {
85 inc_y[2][i] = i - 1;
86 }
87 inc_y[2][0] = g_logical_screen.y_dots - 1; // wrap from the bottom of the screen to the top
88
89 max_pts = std::abs(static_cast<long>(g_params[1]));
90 max_ants = static_cast<int>(g_params[2]);
91 if (max_ants < 1) // if maxants == 0 maxants random
92 {
93 max_ants = 2 + random(256 - 2);
94 }
95 else if (max_ants > MAX_ANTS)
96 {
97 max_ants = MAX_ANTS;
98 g_params[2] = MAX_ANTS;
99 }
100 count_end = max_pts / static_cast<long>(INNER_LOOP);
101
102 type = static_cast<AntType>(g_params[3]);
103 if (type < AntType::ONE || type > AntType::TWO)
104 {
105 type = static_cast<AntType>(random(2)); // if type == 0 choose a random type
106 }
107
108 wrap = g_params[4] != 0.0;
109
110 const auto get_rule = [](const double param)
111 {
112 std::ostringstream buff;
113 buff << std::setprecision(17) << std::fixed << param;
114 return buff.str();
115 };
116 rule_text = get_rule(g_params[0]);
117 rule_len = static_cast<int>(rule_text.length());
118 if (rule_len > 1)
119 {
120 // if rule_len == 0 random rule
121 for (size_t i = 0; i < rule_len; i++)
122 {
123 if (rule_text[i] != '1')
124 {
125 rule_text[i] = '\000';
126 }
127 else
128 {
129 rule_text[i] = '\001';
130 }
131 }
132 }
133 else
134 {
135 rule_len = 0;
136 }
137
138 // set random seed for reproducibility
139 if (!g_random_seed_flag && g_params[5] == 1.0)
140 {
141 --g_random_seed;
142 }
143 if (g_params[5] != 0.0 && g_params[5] != 1.0)
144 {
145 g_random_seed = static_cast<int>(g_params[5]);
146 }
147
148 set_random_seed();
149
150 switch (type)
151 {
152 case AntType::ONE:
153 init_mite1();
154 break;
155 case AntType::TWO:
156 init_mite2();
157 break;
158 }

161void Ant::init_mite1()
162{
163 if (rule_len == 0)
164 {
165 // random rule
166 for (g_color = 0; g_color < MAX_ANTS; g_color++)
167 {
168 // init the rules and colors for the
169 // turkmites: 1 turn left, -1 turn right
170 rule[g_color] = 1 - random(2) * 2;
171 next_col[g_color] = g_color + 1;
172 }
173 // close the cycle
174 next_col[g_color] = 0;
175 }
176 else
177 {
178 // user defined rule
179 for (g_color = 0; g_color < static_cast<int>(rule_len); g_color++)
180 {
181 // init the rules and colors for the
182 // turkmites: 1 turn left, -1 turn right
183 rule[g_color] = rule_text[g_color] * 2 - 1;
184 next_col[g_color] = g_color + 1;
185 }
186 // repeats to last color
187 for (g_color = static_cast<int>(rule_len); g_color < MAX_ANTS; g_color++)
188 {
189 // init the rules and colors for the
190 // turkmites: 1 turn left, -1 turn right
191 rule[g_color] = rule[g_color % rule_len];
192 next_col[g_color] = g_color + 1;
193 }
194 // close the cycle
195 next_col[g_color] = 0;
196 }
197 for (g_color = max_ants; g_color; g_color--)
198 {
199 // init the various turmites N.B. non usa
200 // x[0], y[0], dir[0]
201 if (rule_len)
202 {
203 dir[g_color] = 1;
204 x[g_color] = XO;
205 y[g_color] = YO;
206 }
207 else
208 {
209 dir[g_color] = random(DIRS);
210 x[g_color] = random(g_logical_screen.x_dots);
211 y[g_color] = random(g_logical_screen.y_dots);
212 }
213 }

216// turkmite from Scientific American July 1994 page 91
217// Tweaked by Luciano Genero & Fulvio Cappelli
218//
219void Ant::turk_mite1(const bool step, const long wait)
220{
221 for (int i = INNER_LOOP; i; i--)
222 {
223 for (int color = max_ants; color; color--)
224 {
225 // move the various turmites
226 const int x = this->x[color]; // temp vars
227 const int y = this->y[color];
228 int dir = this->dir[color];
229 const int pixel = get_color(x, y);
230 if (wait > 0 && !step)
231 {
232 g_put_color(x, y, 15);
233 sleep_ms(wait);
234 }
235 g_put_color(x, y, next_col[pixel]);
236 dir += rule[pixel];
237 dir &= 3;
238 if (!wrap)
239 {
240 if ((dir == 0 && y == g_logical_screen.y_dots - 1) ||
241 (dir == 1 && x == g_logical_screen.x_dots - 1) || (dir == 2 && y == 0) ||
242 (dir == 3 && x == 0))
243 {
244 return;
245 }
246 }
247 this->x[color] = inc_x[dir][x];
248 this->y[color] = inc_y[dir][y];
249 this->dir[color] = dir;
250 }
251 }
252}

254static unsigned rotate_left_one(const unsigned value)
255{
256 constexpr unsigned high_bit{~(~0U >> 1)};
257 const unsigned result{value << 1};
258 return value & high_bit ? result | 1U : result;
259}

261void Ant::init_mite2()
262{
263 if (rule_len == 0)
264 {
265 // random rule
266 for (int color = MAX_ANTS - 1; color; color--)
267 {
268 // init the various turmites N.B. don't use
269 // x[0], y[0], dir[0]
270 dir[color] = random(DIRS);
271 rule[color] = std::rand() << random(2) | random(2);
272 x[color] = random(g_logical_screen.x_dots);
273 y[color] = random(g_logical_screen.y_dots);
274 }
275 }
276 else
277 {
278 // the same rule the user wants for every
279 // turkmite (max rule_len = 16 bit)
280 rule_len = static_cast<int>(std::min(rule_len, 8 * sizeof(int) - 1));
281 rule[0] = 0;
282 for (size_t i = 0; i < rule_len; i++)
283 {
284 rule[0] = rule[0] << 1 | rule_text[i];
285 }
286 for (int color = MAX_ANTS - 1; color; color--)
287 {
288 // init the various turmites N.B. non usa
289 // x[0], y[0], dir[0]
290 dir[color] = 0;
291 rule[color] = rule[0];
292 x[color] = XO;
293 y[color] = YO;
294 }
295 }
296 // use this rule when a black pixel is found
297 rule[0] = 0;
298}

300// this one ignore the color of the current cell is more like a white ant
301void Ant::turk_mite2(const bool step, const long wait)
302{
303 for (int i = INNER_LOOP; i; i--)
304 {
305 for (int color = max_ants; color; color--)
306 {
307 // move the various turmites
308 const int x = this->x[color]; // temp vars
309 const int y = this->y[color];
310 int dir = this->dir[color];
311 const int pixel = get_color(x, y);
312 g_put_color(x, y, 15);
313
314 if (wait > 0 && !step)
315 {
316 sleep_ms(wait);
317 }
318
319 if (rule[pixel] & rule_mask)
320 {
321 // turn right
322 dir--;
323 g_put_color(x, y, 0);
324 }
325 else
326 {
327 // turn left
328 dir++;
329 g_put_color(x, y, color);
330 }
331 dir &= 3;
332 if (!wrap)
333 {
334 if ((dir == 0 && y == g_logical_screen.y_dots - 1) ||
335 (dir == 1 && x == g_logical_screen.x_dots - 1) || (dir == 2 && y == 0) ||
336 (dir == 3 && x == 0))
337 {
338 return;
339 }
340 }
341 this->x[color] = inc_x[dir][x];
342 this->y[color] = inc_y[dir][y];
343 this->dir[color] = dir;
344 }
345 rule_mask = rotate_left_one(rule_mask);
346 }
347}

349bool Ant::iterate(const bool step, const long wait)
350{
351 if (count < count_end)
352 {
353 switch (type)
354 {
355 case AntType::ONE:
356 turk_mite1(step, wait);
357 break;
358 case AntType::TWO:
359 turk_mite2(step, wait);
360 break;
361 }
362 ++count;
363 }
364 return count < count_end;
365}
