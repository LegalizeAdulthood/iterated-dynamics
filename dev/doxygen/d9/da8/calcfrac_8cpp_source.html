209static double fmod_test_bailout_or()
210{
211 const double tmp_x = sqr(g_new_z.x);
212 const double tmp_y = sqr(g_new_z.y);
213 if (tmp_x > tmp_y)
214 {
215 return tmp_x;

221// Makes the test condition for the FMOD coloring type
222// that of the current bailout method. 'or' and 'and'
223// methods are not used - in these cases a normal
224// modulus test is used
225//
226static double fmod_test()
227{
228 double result;
229
230 switch (g_bailout_test)
231 {
232 case Bailout::MOD:
233 if (g_magnitude == 0.0 || g_magnitude_calc)
234 {
235 result = sqr(g_new_z.x)+sqr(g_new_z.y);
236 }
237 else
238 {
239 result = g_magnitude; // don't recalculate
240 }
241 break;
242
243 case Bailout::REAL:
244 result = sqr(g_new_z.x);
245 break;
246
247 case Bailout::IMAG:
248 result = sqr(g_new_z.y);
249 break;
250
251 case Bailout::OR:
252 result = fmod_test_bailout_or();
253 break;
254
255 case Bailout::MANH:
256 result = sqr(std::abs(g_new_z.x) + std::abs(g_new_z.y));
257 break;
258
259 case Bailout::MANR:
260 result = sqr(g_new_z.x+g_new_z.y);
261 break;
262
263 default:
264 result = sqr(g_new_z.x)+sqr(g_new_z.y);
265 break;
266 }
267
268 return result;
269}

271// The sym_fill_line() routine was pulled out of the boundary tracing
272// code for re-use with show dot. Its purpose is to fill a line with a
273// solid color. This assumes that Byte *str is already filled
274// with the color. The routine does write the line using symmetry
275// in all cases, however the symmetry logic assumes that the line
276// is one color; it is not general enough to handle a row of
277// pixels of different colors.
278void sym_fill_line(const int row, const int left, const int right, const Byte *str)
279{
280 const int length = right - left + 1;
281 write_span(row, left, right, str);
282 // here's where all the symmetry goes
283 if (g_plot == g_put_color)
284 {
285 g_keyboard_check_interval -= length >> 4; // seems like a reasonable value
286 }
287 else if (g_plot == sym_plot2) // X-axis symmetry
288 {
289 if (int i = g_stop_pt.y - (row - g_start_pt.y); i > g_i_stop_pt.y && i < g_logical_screen.y_dots)
290 {
291 write_span(i, left, right, str);
292 g_keyboard_check_interval -= length >> 3;
293 }
294 }
295 else if (g_plot == sym_plot2y) // Y-axis symmetry
296 {
297 write_span(row, g_stop_pt.x-(right-g_start_pt.x), g_stop_pt.x-(left-g_start_pt.x), str);
298 g_keyboard_check_interval -= length >> 3;
299 }
300 else if (g_plot == sym_plot2j) // Origin symmetry
301 {
302 const int i = g_stop_pt.y - (row - g_start_pt.y);
303 const int j = std::min(g_stop_pt.x-(right-g_start_pt.x), g_logical_screen.x_dots-1);
304 const int k = std::min(g_stop_pt.x-(left -g_start_pt.x), g_logical_screen.x_dots-1);
305 if (i > g_i_stop_pt.y && i < g_logical_screen.y_dots && j <= k)
306 {
307 write_span(i, j, k, str);
308 }
309 g_keyboard_check_interval -= length >> 3;
310 }
311 else if (g_plot == sym_plot4) // X-axis and Y-axis symmetry
312 {
313 const int i = g_stop_pt.y - (row - g_start_pt.y);
314 const int j = std::min(g_stop_pt.x - (right - g_start_pt.x), g_logical_screen.x_dots - 1);
315 const int k = std::min(g_stop_pt.x-(left -g_start_pt.x), g_logical_screen.x_dots-1);
316 if (i > g_i_stop_pt.y && i < g_logical_screen.y_dots)
317 {
318 write_span(i, left, right, str);
319 if (j <= k)
320 {
321 write_span(i, j, k, str);
322 }
323 }
324 if (j <= k)
325 {
326 write_span(row, j, k, str);
327 }
328 g_keyboard_check_interval -= length >> 2;
329 }
330 else // cheap and easy way out
331 {
332 for (int i = left; i <= right; i++)
333 {
334 g_plot(i, row, str[i-left]);
335 }
336 g_keyboard_check_interval -= length >> 1;
337 }
338}

340// The sym_put_line() routine is the symmetry-aware version of put_line().
341// It only works efficiently in the no symmetry or X_AXIS symmetry case,
342// otherwise it just writes the pixels one-by-one.
343static void sym_put_line(const int row, const int left, const int right, const Byte *str)
344{
345 const int length = right-left+1;
346 write_span(row, left, right, str);
347 if (g_plot == g_put_color)
348 {
349 g_keyboard_check_interval -= length >> 4; // seems like a reasonable value
350 }
351 else if (g_plot == sym_plot2) // X-axis symmetry
352 {
353 if (const int i = g_stop_pt.y - (row - g_start_pt.y);
354 i > g_i_stop_pt.y && i < g_logical_screen.y_dots)
355 {
356 write_span(i, left, right, str);
357 }
358 g_keyboard_check_interval -= length >> 3;
359 }
360 else
361 {
362 for (int i = left; i <= right; i++)
363 {
364 g_plot(i, row, str[i-left]);
365 }
366 g_keyboard_check_interval -= length >> 1;
367 }
368}

370static void show_dot_save_restore(
371 int start_x, int stop_x, const int start_y, const int stop_y,
372 const ShowDotDirection direction, const ShowDotAction action)
373{
374 int ct = 0;
375 if (direction != ShowDotDirection::JUST_A_POINT)
376 {
377 if (s_save_dots.empty())
378 {
379 stop_msg("savedots empty");
380 std::exit(0);
381 }
382 if (s_fill_buff == nullptr)
383 {
384 stop_msg("fillbuff NULL");
385 std::exit(0);
386 }
387 }
388 switch (direction)
389 {
390 case ShowDotDirection::LOWER_RIGHT:
391 for (int j = start_y; j <= stop_y; start_x++, j++)
392 {
393 if (action == ShowDotAction::SAVE)
394 {
395 read_span(j, start_x, stop_x, s_save_dots.data() + ct);
396 sym_fill_line(j, start_x, stop_x, s_fill_buff);
397 }
398 else
399 {
400 sym_put_line(j, start_x, stop_x, s_save_dots.data() + ct);
401 }
402 ct += stop_x-start_x+1;
403 }
404 break;
405 case ShowDotDirection::UPPER_RIGHT:
406 for (int j = start_y; j >= stop_y; start_x++, j--)
407 {
408 if (action == ShowDotAction::SAVE)
409 {
410 read_span(j, start_x, stop_x, s_save_dots.data() + ct);
411 sym_fill_line(j, start_x, stop_x, s_fill_buff);
412 }
413 else
414 {
415 sym_put_line(j, start_x, stop_x, s_save_dots.data() + ct);
416 }
417 ct += stop_x-start_x+1;
418 }
419 break;
420 case ShowDotDirection::LOWER_LEFT:
421 for (int j = start_y; j <= stop_y; stop_x--, j++)
422 {
423 if (action == ShowDotAction::SAVE)
424 {
425 read_span(j, start_x, stop_x, s_save_dots.data() + ct);
426 sym_fill_line(j, start_x, stop_x, s_fill_buff);
427 }
428 else
429 {
430 sym_put_line(j, start_x, stop_x, s_save_dots.data() + ct);
431 }
432 ct += stop_x-start_x+1;
433 }
434 break;
435 case ShowDotDirection::UPPER_LEFT:
436 for (int j = start_y; j >= stop_y; stop_x--, j--)
437 {
438 if (action == ShowDotAction::SAVE)
439 {
440 read_span(j, start_x, stop_x, s_save_dots.data() + ct);
441 sym_fill_line(j, start_x, stop_x, s_fill_buff);
442 }
443 else
444 {
445 sym_put_line(j, start_x, stop_x, s_save_dots.data() + ct);
446 }
447 ct += stop_x-start_x+1;
448 }
449 break;
450 case ShowDotDirection::JUST_A_POINT:
451 break;
452 }
453 if (action == ShowDotAction::SAVE)
454 {
455 g_plot(g_col, g_row, s_show_dot_color);
456 }
457}

459static int calc_type_show_dot()
460{
461 ShowDotDirection direction = ShowDotDirection::JUST_A_POINT;
462 int stop_x = g_col;
463 int start_x = g_col;
464 int stop_y = g_row;
465 int start_y = g_row;
466 if (const int width = s_show_dot_width + 1; width > 0)
467 {
468 if (g_col+width <= g_i_stop_pt.x && g_row+width <= g_i_stop_pt.y)
469 {
470 // preferred show dot shape
471 direction = ShowDotDirection::UPPER_LEFT;
472 start_x = g_col;
473 stop_x = g_col+width;
474 start_y = g_row+width;
475 stop_y = g_row+1;
476 }
477 else if (g_col-width >= g_i_start_pt.x && g_row+width <= g_i_stop_pt.y)
478 {
479 // second choice
480 direction = ShowDotDirection::UPPER_RIGHT;
481 start_x = g_col-width;
482 stop_x = g_col;
483 start_y = g_row+width;
484 stop_y = g_row+1;
485 }
486 else if (g_col-width >= g_i_start_pt.x && g_row-width >= g_i_start_pt.y)
487 {
488 direction = ShowDotDirection::LOWER_RIGHT;
489 start_x = g_col-width;
490 stop_x = g_col;
491 start_y = g_row-width;
492 stop_y = g_row-1;
493 }
494 else if (g_col+width <= g_i_stop_pt.x && g_row-width >= g_i_start_pt.y)
495 {
496 direction = ShowDotDirection::LOWER_LEFT;
497 start_x = g_col;
498 stop_x = g_col+width;
499 start_y = g_row-width;
500 stop_y = g_row-1;
501 }
502 }
503 show_dot_save_restore(start_x, stop_x, start_y, stop_y, direction, ShowDotAction::SAVE);
504 if (g_orbit_delay > 0)
505 {
506 sleep_ms(g_orbit_delay);
507 }
508 const int out = s_calc_type_tmp();
509 show_dot_save_restore(start_x, stop_x, start_y, stop_y, direction, ShowDotAction::RESTORE);
510 return out;
511}

519{

522 g_basin = 0;
523 g_put_color = put_color_a;

532 std::string light_path{get_save_path(WriteFile::IMAGE, g_light_name).string()};
533 check_write_file(light_path, ".tga");
534 if (!start_targa(light_path))
535 {

537 g_std_calc_mode = CalcMode::ONE_PASS;
538 g_user.std_calc_mode = g_std_calc_mode;
539 g_put_color = put_true_color_disk;
540 }
541 else

543 g_true_color = false;
544 }

547 init_misc(); // set up some variables in parser.c
548
549 // following delta values useful only for types with rotation disabled
550 // currently used only by bifurcation
551 g_param_z1.x = g_params[0];
552 g_param_z1.y = g_params[1];
553 g_param_z2.x = g_params[2];
554 g_param_z2.y = g_params[3];
555
556 if (g_log_map_flag && g_colors < 16)
557 {
558 stop_msg("Need at least 16 colors to use logmap");
559 g_log_map_flag = 0;
560 }
561
562 if (g_use_old_periodicity)
563 {
564 g_periodicity_next_saved_incr = 1;
565 g_first_saved_and = 1;
566 }
567 else
568 {
569 // works better than log()
570 g_periodicity_next_saved_incr = static_cast<int>(std::log10(static_cast<double>(g_max_iterations)));
571 // maintains image with low iterations
572 g_periodicity_next_saved_incr = std::max(g_periodicity_next_saved_incr, 4);
573 g_first_saved_and = static_cast<long>(g_periodicity_next_saved_incr * 2 + 1);
574 }
575
576 g_log_map_table.clear();
577 g_log_map_table_max_size = g_max_iterations;
578 g_log_map_calculate = false;
579 // below, 32767 is used as the allowed value for maximum iteration count for
580 // historical reasons. TODO: increase this limit
581 if (g_log_map_flag && (g_max_iterations > 32767 || g_log_map_fly_calculate == LogMapCalculate::ON_THE_FLY))
582 {
583 g_log_map_calculate = true; // calculate on the fly
584 setup_log_table();
585 }
586 else if (g_log_map_flag && g_log_map_fly_calculate == LogMapCalculate::USE_LOG_TABLE)
587 {
588 g_log_map_table_max_size = 32767;
589 g_log_map_calculate = false; // use logtable
590 }
591 else if (!g_iteration_ranges.empty() && g_max_iterations >= 32767)
592 {
593 g_log_map_table_max_size = 32766;
594 }
595
596 if ((g_log_map_flag || !g_iteration_ranges.empty()) && !g_log_map_calculate)
597 {
598 bool resized = false;
599 try
600 {
601 g_log_map_table.resize(g_log_map_table_max_size + 1);
602 resized = true;
603 }
604 catch (const std::bad_alloc &)
605 {
606 }
607
608 if (!resized)
609 {
610 if (!g_iteration_ranges.empty() || g_log_map_fly_calculate == LogMapCalculate::USE_LOG_TABLE)
611 {
612 stop_msg("Insufficient memory for logmap/ranges with this maxiter");
613 }
614 else
615 {
616 stop_msg("Insufficient memory for logTable, using on-the-fly routine");
617 g_log_map_fly_calculate = LogMapCalculate::ON_THE_FLY;
618 g_log_map_calculate = true; // calculate on the fly
619 setup_log_table();
620 }
621 }
622 else if (!g_iteration_ranges.empty())
623 {
624 // Can't do ranges if MaxLTSize > 32767
625 int l = 0;
626 int k = 0;
627 int i = 0;
628 g_log_map_flag = 0; // ranges overrides logmap
629 while (i < static_cast<int>(g_iteration_ranges.size()))
630 {
631 int flip = 0;
632 int m = 0;
633 int altern = 32767;
634 int num_val = g_iteration_ranges[i++];
635 if (num_val < 0)
636 {
637 altern = g_iteration_ranges[i++]; // sub-range iterations
638 num_val = g_iteration_ranges[i++];
639 }
640 if (num_val > static_cast<int>(g_log_map_table_max_size) || i >= static_cast<int>(g_iteration_ranges.size()))
641 {
642 num_val = static_cast<int>(g_log_map_table_max_size);
643 }
644 while (l <= num_val)
645 {
646 g_log_map_table[l++] = static_cast<Byte>(k + flip);
647 if (++m >= altern)
648 {
649 flip ^= 1; // Alternate colors
650 m = 0;
651 }

2120// Continuous potential calculation for Mandelbrot and Julia
2121// Reference: Science of Fractal Images p. 190.
2122// Special thanks to Mark Peterson for his "MtMand" program that
2123// beautifully approximates plate 25 (same reference) and spurred
2124// on the inclusion of similar capabilities.
2125//
2126// The purpose of this function is to calculate a color value
2127// for a fractal that varies continuously with the screen pixels
2128// locations for better rendering in 3D.
2129//
2130// Here "magnitude" is the modulus of the orbit value at
2131// "iterations". The potparms[] are user-entered paramters
2132// controlling the level and slope of the continuous potential
2133// surface. Returns color.
2134//
2135int potential(const double mag, const long iterations)
2136{
2137 float pot;
2138 long l_pot;
2139
2140 if (iterations < g_max_iterations)
2141 {
2142 l_pot = iterations+2;
2143 pot = static_cast<float>(l_pot);
2144 if (l_pot <= 0 || mag <= 1.0)
2145 {
2146 pot = 0.0F;
2147 }
2148 else // pot = log(mag) / pow(2.0, (double)pot);
2149 {
2150 if (const double d_tmp = std::log(mag) / std::pow(2.0, static_cast<double>(pot));
2151 d_tmp > FLT_MIN) // prevent float type underflow
2152 {
2153 pot = static_cast<float>(d_tmp);
2154 }
2155 else
2156 {
2157 pot = 0.0F;
2158 }
2159 }
2160
2161 // following transformation strictly for aesthetic reasons
2162 // meaning of parameters:
2163 // potparam[0] -- zero potential level - highest color -
2164 // potparam[1] -- slope multiplier -- higher is steeper
2165 // potparam[2] -- rqlim value if changeable (bailout for modulus)
2166 if (pot > 0.0)
2167 {
2168 pot = std::sqrt(pot);
2169 pot = static_cast<float>(g_potential.params[0] - pot * g_potential.params[1] - 1.0);
2170 }
2171 else
2172 {
2173 pot = static_cast<float>(g_potential.params[0] - 1.0);
2174 }
2175 pot = std::max(pot, 1.0f); // avoid color 0
2176 }
2177 else if (g_inside_method == ColorMethod::COLOR)
2178 {
2179 pot = static_cast<float>(g_inside_color);
2180 }
2181 else // inside < 0 implies inside = maxit, so use 1st pot param instead
2182 {
2183 pot = static_cast<float>(g_potential.params[0]);
2184 }
2185
2186 l_pot = static_cast<long>(pot * 256);
2187 int i_pot = static_cast<int>(l_pot >> 8);
2188 if (i_pot >= g_colors)
2189 {
2190 i_pot = g_colors - 1;
2191 l_pot = 255;
2192 }
2193
2194 if (g_potential.store_16bit)
2195 {
2196 if (!driver_is_disk()) // if putcolor won't be doing it for us
2197 {
2198 disk_write_pixel(g_col+g_logical_screen.x_offset, g_row+g_logical_screen.y_offset, i_pot);
2199 }
2200 disk_write_pixel(g_col+g_logical_screen.x_offset, g_row+g_screen_y_dots+g_logical_screen.y_offset, static_cast<int>(l_pot));
2201 }
2202
2203 return i_pot;
2204}
