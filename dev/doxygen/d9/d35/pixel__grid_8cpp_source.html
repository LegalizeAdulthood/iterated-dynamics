22/*
23 * The following functions calculate the real and imaginary complex
24 * coordinates of the point in the complex plane corresponding to
25 * the screen coordinates (col,row) at the current zoom corners
26 * settings. The functions come in two flavors. One looks up the pixel
27 * values using the precalculated grid arrays dx0, dx1, dy0, and dy1,
28 * which has a speed advantage but is limited to MAX_PIXELS image
29 * dimensions. The other calculates the complex coordinates at a
30 * cost of two additions and two multiplications for each component,
31 * but works at any resolution.
32 *
33 * The function call overhead
34 * appears to be negligible. It also appears that the speed advantage
35 * of the lookup vs the calculation is negligible on machines with
36 * coprocessors. Bert Tyler's original implementation was designed for
37 * machines with no coprocessor; on those machines the saving was
38 * significant. For the time being, the table lookup capability will
39 * be maintained.
40 */
41
42// Real component, grid lookup version - requires dx0/dx1 arrays
43double dx_pixel()
44{
45 return s_grid_x0[g_col]+s_grid_x1[g_row];
46}
47
48// Imaginary component, grid lookup version - requires dy0/dy1 arrays
49double dy_pixel()
50{
51 return s_grid_y0[g_row]+s_grid_y1[g_col];
52}

71void fill_pixel_grid()
72{
73 s_grid_x0[0] = g_image_region.m_min.x; // fill up the x, y grids
74 s_grid_y0[0] = g_image_region.m_max.y;
75 s_grid_y1[0] = 0;
76 s_grid_x1[0] = 0;
77 for (int i = 1; i < g_logical_screen.x_dots; i++)
78 {
79 s_grid_x0[i] = s_grid_x0[0] + i * static_cast<double>(g_delta_x);
80 s_grid_y1[i] = s_grid_y1[0] - i * static_cast<double>(g_delta_y2);
81 }
82 for (int i = 1; i < g_logical_screen.y_dots; i++)
83 {
84 s_grid_y0[i] = s_grid_y0[0] - i * static_cast<double>(g_delta_y);
85 s_grid_x1[i] = s_grid_x1[0] + i * static_cast<double>(g_delta_x2);
86 }
87}
