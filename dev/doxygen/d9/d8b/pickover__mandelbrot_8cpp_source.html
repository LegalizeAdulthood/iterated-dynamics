31int mandel_trig_plus_exponent_orbit()
32{
33 // another Scientific American biomorph type
34 // z(n+1) = e**z(n) + trig(z(n)) + C
35
36 if (std::abs(g_old_z.x) >= 6.4e2)
37 {
38 return 1; // DOMAIN errors
39 }
40 const double tmp_exp = std::exp(g_old_z.x);
41 double sin_y;
42 double cos_y;
43 sin_cos(g_old_z.y, &sin_y, &cos_y);
44 cmplx_trig0(g_old_z, g_new_z);
45
46 //new = trig(old) + e**old + C
47 g_new_z.x += tmp_exp*cos_y + g_float_param->x;
48 g_new_z.y += tmp_exp*sin_y + g_float_param->y;
49 return g_bailout_float();

60// Generate Pascal's Triangle coefficients
61void pascal_triangle()
62{
63 long c = 1L;
64
65 for (long j = 0; j <= g_c_exponent; j++)
66 {
67 if (j == 0)
68 {
69 c = 1;
70 }
71 else
72 {
73 c = c * (g_c_exponent - j + 1) / j;
74 }
75 s_pascal_triangle[j] = c;
76 }
77}

121void mandel_z_power_perturb(
122 const std::complex<double> &ref, std::complex<double> &delta_n, const std::complex<double> &delta0)
123{
124 if (g_c_exponent == 3)
125 {
126 const double r{ref.real()};
127 const double i{ref.imag()};
128 const double a{delta_n.real()};
129 const double b{delta_n.imag()};
130 const double a0{delta0.real()};
131 const double b0{delta0.imag()};
132 const double dnr{ //
133 3 * r * r * a //
134 - 6 * r * i * b //
135 - 3 * i * i * a //
136 + 3 * r * a * a //
137 - 3 * r * b * b //
138 - 3 * i * 2 * a * b //
139 + a * a * a //
140 - 3 * a * b * b //
141 + a0};
142 const double dni{ //
143 3 * r * r * b //
144 + 6 * r * i * a //
145 - 3 * i * i * b //
146 + 3 * r * 2 * a * b //
147 + 3 * i * a * a //
148 - 3 * i * b * b //
149 + 3 * a * a * b //
150 - b * b * b //
151 + b0};
152 delta_n.imag(dni);
153 delta_n.real(dnr);
154 }
155 else
156 {
157 std::complex<double> zp(1.0, 0.0);
158 std::complex<double> sum(0.0, 0.0);
159 for (int j = 0; j < g_c_exponent; j++)
160 {
161 sum += zp * static_cast<double>(s_pascal_triangle[j]);
162 sum *= delta_n;
163 zp *= ref;
164 }
165 delta_n = sum;
166 delta_n += delta0;
167 }
168}
