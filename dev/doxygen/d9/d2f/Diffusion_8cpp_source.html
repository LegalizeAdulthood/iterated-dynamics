36Diffusion::Diffusion() :
37 m_border(std::max(static_cast<int>(g_params[0]), 10)),
38 m_mode(std::clamp(static_cast<DiffusionMode>(g_params[1]), DiffusionMode::CENTRAL, DiffusionMode::SQUARE_CAVITY)),
39 m_color_shift(static_cast<int>(g_params[2])),
40 m_color_count(static_cast<int>(g_params[2]))
41{
42 set_random_seed();
43
44 switch (m_mode)
45 {
46 case DiffusionMode::CENTRAL:
47 m_x_max = g_logical_screen.x_dots / 2 + m_border; // Initial box
48 m_x_min = g_logical_screen.x_dots / 2 - m_border;
49 m_y_max = g_logical_screen.y_dots / 2 + m_border;
50 m_y_min = g_logical_screen.y_dots / 2 - m_border;
51 break;
52
53 case DiffusionMode::FALLING_PARTICLES:
54 m_x_max = g_logical_screen.x_dots / 2 + m_border; // Initial box
55 m_x_min = g_logical_screen.x_dots / 2 - m_border;
56 m_y_min = g_logical_screen.y_dots - m_border;
57 break;
58
59 case DiffusionMode::SQUARE_CAVITY:
60 if (g_logical_screen.x_dots > g_logical_screen.y_dots)
61 {
62 m_radius = static_cast<float>(g_logical_screen.y_dots - m_border);
63 }
64 else
65 {
66 m_radius = static_cast<float>(g_logical_screen.x_dots - m_border);
67 }
68 break;
69 }
70
71 if (g_resuming) // restore work list, if we can't the above will stay in place
72 {
73 start_resume();
74 if (m_mode != DiffusionMode::SQUARE_CAVITY)
75 {
76 get_resume(m_x_max, m_x_min, m_y_max, m_y_min);
77 }
78 else
79 {
80 get_resume(m_x_max, m_x_min, m_y_max, m_radius);
81 }
82 end_resume();
83 }
84
85 switch (m_mode)
86 {
87 case DiffusionMode::CENTRAL:
88 g_put_color(g_logical_screen.x_dots / 2, g_logical_screen.y_dots / 2, m_current_color);
89 break;
90
91 case DiffusionMode::FALLING_PARTICLES:
92 for (int i = 0; i <= g_logical_screen.x_dots; i++)
93 {
94 g_put_color(i, g_logical_screen.y_dots - 1, m_current_color);
95 }
96 break;
97
98 case DiffusionMode::SQUARE_CAVITY:
99 if (g_logical_screen.x_dots > g_logical_screen.y_dots)
100 {
101 for (int i = 0; i < g_logical_screen.y_dots; i++)
102 {
103 g_put_color(g_logical_screen.x_dots / 2 - g_logical_screen.y_dots / 2, i, m_current_color);
104 g_put_color(g_logical_screen.x_dots / 2 + g_logical_screen.y_dots / 2, i, m_current_color);
105 g_put_color(g_logical_screen.x_dots / 2 - g_logical_screen.y_dots / 2 + i, 0, m_current_color);
106 g_put_color(g_logical_screen.x_dots / 2 - g_logical_screen.y_dots / 2 + i,
107 g_logical_screen.y_dots - 1, m_current_color);
108 }
109 }
110 else
111 {
112 for (int i = 0; i < g_logical_screen.x_dots; i++)
113 {
114 g_put_color(0, g_logical_screen.y_dots / 2 - g_logical_screen.x_dots / 2 + i, m_current_color);
115 g_put_color(g_logical_screen.x_dots - 1,
116 g_logical_screen.y_dots / 2 - g_logical_screen.x_dots / 2 + i, m_current_color);
117 g_put_color(i, g_logical_screen.y_dots / 2 - g_logical_screen.x_dots / 2, m_current_color);
118 g_put_color(i, g_logical_screen.y_dots / 2 + g_logical_screen.x_dots / 2, m_current_color);
119 }
120 }
121 break;
122 }

125void Diffusion::release_new_particle()
126{
127 if (!m_particle_needed)
128 {
129 return;
130 }
131
132 switch (m_mode)
133 {
134 case DiffusionMode::CENTRAL:
135 {
136 // Release new point on a circle inside the box
137 const double angle = 2 * static_cast<double>(std::rand()) / (RAND_MAX / PI);
138 double cosine;
139 double sine;
140 sin_cos(angle, sine, cosine);
141 m_x = static_cast<int>(cosine * (m_x_max - m_x_min) + g_logical_screen.x_dots);
142 m_y = static_cast<int>(sine * (m_y_max - m_y_min) + g_logical_screen.y_dots);
143 m_x /= 2;
144 m_y /= 2;
145 break;
146 }
147
148 case DiffusionMode::FALLING_PARTICLES:
149 // Release new point on the line ymin somewhere between xmin and xmax
150 m_y = m_y_min;
151 m_x = random(m_x_max - m_x_min) + (g_logical_screen.x_dots - m_x_max + m_x_min) / 2;
152 break;
153
154 case DiffusionMode::SQUARE_CAVITY:
155 {
156 // Release new point on a circle inside the box with radius given by the radius variable
157 const double angle = 2 * static_cast<double>(std::rand()) / (RAND_MAX / PI);
158 double cosine;
159 double sine;
160 sin_cos(angle, sine, cosine);
161 m_x = static_cast<int>(cosine * m_radius + g_logical_screen.x_dots);
162 m_y = static_cast<int>(sine * m_radius + g_logical_screen.y_dots);
163 m_x /= 2;
164 m_y /= 2;
165 break;
166 }
167 }
168
169 m_particle_needed = false;
170}

196bool Diffusion::move_particle()
197{
198 // Loop as long as the point (x,y) is surrounded by color 0 on all eight sides
199 while (get_color(m_x + 1, m_y + 1) == 0 && get_color(m_x + 1, m_y) == 0 && get_color(m_x + 1, m_y - 1) == 0 //
200 && get_color(m_x, m_y + 1) == 0 && get_color(m_x, m_y - 1) == 0 && get_color(m_x - 1, m_y + 1) == 0 //
201 && get_color(m_x - 1, m_y) == 0 && get_color(m_x - 1, m_y - 1) == 0)
202 {
203 // Erase moving point
204 if (g_show_orbit)
205 {
206 g_put_color(m_x, m_y, 0);
207 }
208
209 if (m_mode == DiffusionMode::CENTRAL)
210 {
211 // Make sure point is inside the box
212 if (m_x == m_x_max)
213 {
214 m_x--;
215 }
216 else if (m_x == m_x_min)
217 {
218 m_x++;
219 }
220 if (m_y == m_y_max)
221 {
222 m_y--;
223 }
224 else if (m_y == m_y_min)
225 {
226 m_y++;
227 }
228 }
229
230 // Make sure point is on the screen below ymin, but
231 // we need a 1 pixel margin because of the next random step.
232 if (m_mode == DiffusionMode::FALLING_PARTICLES)
233 {
234 if (m_x >= g_logical_screen.x_dots - 1)
235 {
236 m_x--;
237 }
238 else if (m_x <= 1)
239 {
240 m_x++;
241 }
242 if (m_y < m_y_min)
243 {
244 m_y++;
245 }
246 }
247
248 // Take one random step
249 m_x += random(3) - 1;
250 m_y += random(3) - 1;
251
252 // Show the moving point
253 if (g_show_orbit)
254 {
255 g_put_color(m_x, m_y, random(g_colors - 1) + 1);
256 }
257
258 if (keyboard_check_needed())
259 {
260 return true;
261 }
262 }
263
264 m_particle_needed = true;
265
266 return false;
267}

269void Diffusion::color_particle()
270{
271 // If we're doing color shifting then check to see if we need to shift
272 if (m_color_shift)
273 {
274 // If we're doing color shifting then use current color
275 g_put_color(m_x, m_y, m_current_color);
276 if (!--m_color_count)
277 {
278 // If the counter reaches zero then shift
279 m_current_color++; // Increase the current color and wrap
280 m_current_color %= g_colors; // around skipping zero
281 if (!m_current_color)
282 {
283 m_current_color++;
284 }
285 m_color_count = m_color_shift; // and reset the counter
286 }
287 }
288 else
289 {
290 // pick a color at random
291 g_put_color(m_x, m_y, random(g_colors - 1) + 1);
292 }
293}

295// If the new point is close to an edge, we may need to increase
296// some limits so that the limits expand to match the growing
297// fractal.
298bool Diffusion::adjust_limits()
299{
300 switch (m_mode)
301 {
302 case DiffusionMode::CENTRAL:
303 if (m_x + m_border > m_x_max ||
304 m_x - m_border < m_x_min ||
305 m_y - m_border < m_y_min ||
306 m_y + m_border > m_y_max)
307 {
308 // Increase box size, but not past the edge of the screen
309 m_y_min--;
310 m_y_max++;
311 m_x_min--;
312 m_x_max++;
313 if (m_y_min == 0 || m_x_min == 0)
314 {
315 return true;
316 }
317 }
318 break;
319
320 case DiffusionMode::FALLING_PARTICLES:
321 // Decrease ymin, but not past top of screen
322 if (m_y - m_border < m_y_min)
323 {
324 m_y_min--;
325 }
326 if (m_y_min == 0)
327 {
328 return true;
329 }
330 break;
331
332 case DiffusionMode::SQUARE_CAVITY:
333 {
334 // Decrease the radius where points are released to stay away
335 // from the fractal. It might be decreased by 1 or 2
336 const double r = sqr(static_cast<float>(m_x) - g_logical_screen.x_dots / 2) +
337 sqr(static_cast<float>(m_y) - g_logical_screen.y_dots / 2);
338 if (r <= m_border * m_border)
339 {
340 return true;
341 }
342 while ((m_radius - m_border) * (m_radius - m_border) > r)
343 {
344 m_radius--;
345 }
346 break;
347 }
348 }
349 return false;
350}
