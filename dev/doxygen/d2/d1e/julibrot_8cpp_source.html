106bool julibrot_per_image()
107{
108 const char *map_name;
109
110 if (g_colors < 255)
111 {
112 stop_msg("Sorry, but Julibrots require a 256-color video mode");
113 return false;
114 }
115
116 s_jb.x_offset = g_image_region.center_x(); // Calculate average
117 s_jb.y_offset = g_image_region.center_y(); // Calculate average
118 s_jb.delta_mx = (g_julibrot_x_max - g_julibrot_x_min) / g_julibrot_z_dots;
119 s_jb.delta_my = (g_julibrot_y_max - g_julibrot_y_min) / g_julibrot_z_dots;
120 g_float_param = &s_jb.jb_c;
121 s_jb.x_per_inch = (g_image_region.m_min.x - g_image_region.m_max.x) / g_julibrot_width;
122 s_jb.y_per_inch = g_image_region.height() / g_julibrot_height;
123 s_jb.inch_per_x_dot = g_julibrot_width / g_logical_screen.x_dots;
124 s_jb.inch_per_y_dot = g_julibrot_height / g_logical_screen.y_dots;
125 s_jb.init_z = g_julibrot_origin - g_julibrot_depth / 2;
126 if (g_julibrot_3d_mode == Julibrot3DMode::MONOCULAR)
127 {
128 s_jb.right_eye.x = 0.0;
129 }
130 else
131 {
132 s_jb.right_eye.x = g_eyes / 2;
133 }
134 s_jb.left_eye.x = -s_jb.right_eye.x;
135 s_jb.right_eye.y = 0;
136 s_jb.left_eye.y = 0;
137 s_jb.right_eye.zx = g_julibrot_dist;
138 s_jb.left_eye.zx = s_jb.right_eye.zx;
139 s_jb.right_eye.zy = g_julibrot_dist;
140 s_jb.left_eye.zy = s_jb.right_eye.zy;
141 s_b_base = 128;
142
143 if (g_julibrot_3d_mode == Julibrot3DMode::RED_BLUE)
144 {
145 g_save_dac = SaveDAC::NO;
146 map_name = GLASSES1_MAP_NAME.c_str();
147 }
148 else
149 {
150 map_name = ALTERN_MAP_NAME.c_str();
151 }
152 if (g_save_dac != SaveDAC::YES)
153 {
154 if (validate_luts(map_name))
155 {
156 return false;
157 }
158 refresh_dac();
159 if (g_save_dac == SaveDAC::NEXT_TIME)
160 {
161 g_save_dac = SaveDAC::YES;
162 }
163 }
164 return true;
165}

167int julibrot_per_pixel()
168{
169 s_jb.jx = ((s_jb.per->x - s_jb.x_pixel) * s_jb.init_z / g_julibrot_dist - s_jb.x_pixel) * s_jb.x_per_inch;
170 s_jb.jx += s_jb.x_offset;
171 s_jb.delta_jx = g_julibrot_depth / g_julibrot_dist * (s_jb.per->x - s_jb.x_pixel) * s_jb.x_per_inch / g_julibrot_z_dots;
172
173 s_jb.jy = ((s_jb.per->y - s_jb.y_pixel) * s_jb.init_z / g_julibrot_dist - s_jb.y_pixel) * s_jb.y_per_inch;
174 s_jb.jy += s_jb.y_offset;
175 s_jb.delta_jy = g_julibrot_depth / g_julibrot_dist * (s_jb.per->y - s_jb.y_pixel) * s_jb.y_per_inch / g_julibrot_z_dots;
176
177 return 1;
178}

180static void z_line(const double x, const double y)
181{
182 s_jb.x_pixel = x;
183 s_jb.y_pixel = y;
184 s_jb.mx = g_julibrot_x_min;
185 s_jb.my = g_julibrot_y_min;
186 switch (g_julibrot_3d_mode)
187 {
188 case Julibrot3DMode::MONOCULAR:
189 case Julibrot3DMode::LEFT_EYE:
190 s_jb.per = &s_jb.left_eye;
191 break;
192 case Julibrot3DMode::RIGHT_EYE:
193 s_jb.per = &s_jb.right_eye;
194 break;
195 case Julibrot3DMode::RED_BLUE:
196 if (g_row + g_col & 1)
197 {
198 s_jb.per = &s_jb.left_eye;
199 }
200 else
201 {
202 s_jb.per = &s_jb.right_eye;
203 }
204 break;
205 }
206 julibrot_per_pixel();
207 for (s_z_pixel = 0; s_z_pixel < g_julibrot_z_dots; s_z_pixel++)
208 {
209 // Special initialization for Mandelbrot types
210 if (g_new_orbit_type == FractalType::QUAT || g_new_orbit_type == FractalType::HYPER_CMPLX)
211 {
212 g_old_z.x = 0.0;
213 g_old_z.y = 0.0;
214 s_jb.jb_c.x = 0.0;
215 s_jb.jb_c.y = 0.0;
216 g_quaternion_c = s_jb.jx;
217 g_quaternion_ci = s_jb.jy;
218 g_quaternion_cj = s_jb.mx;
219 g_quaternion_ck = s_jb.my;
220 }
221 else
222 {
223 g_old_z.x = s_jb.jx;
224 g_old_z.y = s_jb.jy;
225 s_jb.jb_c.x = s_jb.mx;
226 s_jb.jb_c.y = s_jb.my;
227 g_quaternion_c = g_params[0];
228
229 g_quaternion_ci = g_params[1];
230 g_quaternion_cj = g_params[2];
231 g_quaternion_ck = g_params[3];
232 }
233 g_temp_sqr_x = sqr(g_old_z.x);
234 g_temp_sqr_y = sqr(g_old_z.y);
235
236 for (s_n = 0; s_n < g_max_iterations; s_n++)
237 {
238 if (get_fractal_specific(g_new_orbit_type)->orbit_calc())
239 {
240 break;
241 }
242 }
243 if (s_n == g_max_iterations)
244 {
245 if (g_julibrot_3d_mode == Julibrot3DMode::RED_BLUE)
246 {
247 g_color = static_cast<int>(128L * s_z_pixel / g_julibrot_z_dots);
248 if (g_row + g_col & 1)
249 {
250 g_plot(g_col, g_row, 127 - g_color);
251 }
252 else
253 {
254 g_color = static_cast<int>(g_color * s_br_ratio);
255 g_color = std::max(g_color, 1);
256 g_color = std::min(g_color, 127);
257 g_plot(g_col, g_row, 127 + s_b_base - g_color);
258 }
259 }
260 else
261 {
262 g_color = static_cast<int>(254L * s_z_pixel / g_julibrot_z_dots);
263 g_plot(g_col, g_row, g_color + 1);
264 }
265 s_plotted = 1;
266 break;
267 }
268 s_jb.mx += s_jb.delta_mx;
269 s_jb.my += s_jb.delta_my;
270 s_jb.jx += s_jb.delta_jx;
271 s_jb.jy += s_jb.delta_jy;
272 }
273}

275Standard4D::Standard4D()
276{
277 g_c_exponent = static_cast<int>(g_params[2]);
278
279 if (g_new_orbit_type == FractalType::JULIA_Z_POWER)
280 {
281 if (g_params[3] == 0.0 && g_debug_flag != DebugFlags::FORCE_COMPLEX_POWER &&
282 static_cast<double>(g_c_exponent) == g_params[2])
283 {

285 }

289 }
290 get_julia_attractor(g_params[0], g_params[1]); // another attractor?
291 }
292 m_y_dot = (g_logical_screen.y_dots >> 1) - 1;
293 m_x_dot = 0;
294 s_plotted = false;
295 m_x = -g_julibrot_width / 2.0;
296}

298bool Standard4D::iterate()
299{
300 if (m_y_dot < 0)
301 {
302 return false;
303 }
304
305 g_col = m_x_dot;
306 g_row = m_y_dot;
307 z_line(m_x, m_y);
308 g_col = g_logical_screen.x_dots - g_col - 1;
309 g_row = g_logical_screen.y_dots - g_row - 1;
310 z_line(-m_x, -m_y);
311 ++m_x_dot;
312 m_x += s_jb.inch_per_x_dot;
313 if (m_x_dot == g_logical_screen.x_dots)
314 {
315 // next scanline
316 if (!s_plotted && m_y != 0.0)
317 {
318 return false;
319 }
320 --m_y_dot;
321 m_y -= s_jb.inch_per_y_dot;
322 s_plotted = false;
323 m_x_dot = 0;
324 m_x = -g_julibrot_width / 2.0;
325 }
326
327 return true;
328}
