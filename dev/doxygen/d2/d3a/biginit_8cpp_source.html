122/*********************************************************************/
123// given g_bn_length, calc_lengths will calculate all the other lengths
124void calc_lengths()
125{
126 g_bn_step = 4; // use 4 in all cases
127
128 if (g_bn_length % g_bn_step != 0)
129 {
130 g_bn_length = (g_bn_length / g_bn_step + 1) * g_bn_step;
131 }
132 if (g_bn_length == g_bn_step)
133 {
134 g_padding = g_bn_length;
135 }
136 else
137 {
138 g_padding = 2*g_bn_step;
139 }
140 g_r_length = g_bn_length + g_padding;
141
142 // This shift factor assumes non-full multiplications will be performed.
143 // Change to g_bn_length-g_int_length for full multiplications.
144 g_shift_factor = g_padding - g_int_length;
145
146 g_bf_length = g_bn_length+g_bn_step; // one extra step for added precision
147 g_r_bf_length = g_bf_length + g_padding;
148 g_bf_decimals = static_cast<int>((g_bf_length - 2) * LOG10_256);
149}

161{
162 long ptr;
163 save_bf_vars(); // copy corners values for conversion
164
165 calc_lengths();
166
167 s_bn_root = reinterpret_cast<BigFloat>(&s_storage[0]);
168
169 /* at present time one call would suffice, but this logic allows

171 if (int i = find_alternate_math(g_fractal_type, BFMathType::BIG_NUM); i > -1)
172 {
173 g_bf_math = g_alternate_math[i].math;
174 }
175 else
176 {
177 i = find_alternate_math(g_fractal_type, BFMathType::BIG_FLT);
178 if (i > -1)
179 {
180 g_bf_math = g_alternate_math[i].math;
181 }

184 g_bf_math = BFMathType::BIG_NUM; // maybe called from cmdfiles.c and g_fractal_type not set
185 }
186 }
187
188 // Now split up the memory among the pointers
189 const auto alloc_size = [&ptr](const int size)
190 {
191 BigNum result = s_bn_root + ptr;
192 ptr += size;
193 return result;
194 };
195 ptr = 0;
196 g_bn_tmp1 = alloc_size(g_r_length);
197 g_bn_tmp2 = alloc_size(g_r_length);
198 g_bn_tmp3 = alloc_size(g_r_length);
199 g_bn_tmp4 = alloc_size(g_r_length);
200 g_bn_tmp5 = alloc_size(g_r_length);
201 g_bn_tmp6 = alloc_size(g_r_length);
202
203 g_bf_tmp1 = alloc_size(g_r_bf_length+2);
204 g_bf_tmp2 = alloc_size(g_r_bf_length+2);
205 g_bf_tmp3 = alloc_size(g_r_bf_length+2);
206 g_bf_tmp4 = alloc_size(g_r_bf_length+2);

213 g_bn_tmp_copy1 = alloc_size(g_r_length * 2);
214 g_bn_tmp_copy2 = alloc_size(g_r_length * 2);
215
216 if (g_bf_math == BFMathType::BIG_NUM)
217 {
218 g_x_min_bn = alloc_size(g_bn_length);
219 g_x_max_bn = alloc_size(g_bn_length);
220 g_y_min_bn = alloc_size(g_bn_length);
221 g_y_max_bn = alloc_size(g_bn_length);
222 g_x_3rd_bn = alloc_size(g_bn_length);

248 g_old_z_bf.y = alloc_size(g_r_bf_length+2);

260 }
261 g_bf10_tmp = alloc_size(g_bf_decimals+4);
262
263 // ptr needs to be 16-bit aligned on some systems
264 ptr = ptr + 1 &~1;
265
266 s_stack_ptr = s_bn_root + ptr;
267 g_start_stack = ptr;
268
269 // max stack offset from s_bn_root
270 g_max_stack = 0x10000L-(g_bf_length+2)*22;
271
272 // sanity check
273 // leave room for NUMVARS variables allocated from stack
274 // also leave room for the safe area at top of segment
275 if (ptr + NUM_VARS*(g_bf_length+2) > g_max_stack)
276 {
277 stop_msg("Requested precision of " + std::to_string(g_decimals) + " too high, aborting");
278 goodbye();
279 }
280
281 // room for 6 corners + 6 save corners + 10 params at top of extraseg
282 // this area is safe - use for variables that are used outside fractal
283 // generation - e.g. zoom box variables
284 ptr = g_max_stack;
285 g_bf_x_min = alloc_size(g_bf_length+2);
286 g_bf_x_max = alloc_size(g_bf_length+2);
287 g_bf_y_min = alloc_size(g_bf_length+2);
288 g_bf_y_max = alloc_size(g_bf_length+2);
289 g_bf_x_3rd = alloc_size(g_bf_length+2);
290 g_bf_y_3rd = alloc_size(g_bf_length+2);
291 for (BigFloat &param : g_bf_params)
292 {
293 param = s_bn_root + ptr;
294 ptr += g_bf_length + 2;
295 }
296 g_bf_save_x_min = alloc_size(g_bf_length+2);
297 g_bf_save_x_max = alloc_size(g_bf_length+2);
298 g_bf_save_y_min = alloc_size(g_bf_length+2);
299 g_bf_save_y_max = alloc_size(g_bf_length+2);
300 g_bf_save_x_3rd = alloc_size(g_bf_length+2);
301 g_bf_save_y_3rd = s_bn_root+ptr;
302 // end safe vars
303
304 // good citizens initialize variables
305 if (g_bf_save_len > 0) // leave save area
306 {
307 std::memset(s_bn_root+(g_bf_save_len+2)*22, 0, static_cast<unsigned>(g_start_stack - (g_bf_save_len + 2) * 22));
308 }
309 else // first time through - nothing saved
310 {
311 // high variables
312 std::memset(s_bn_root+g_max_stack, 0, (g_bf_length+2)*22);
313 // low variables
314 std::memset(s_bn_root, 0, static_cast<unsigned>(g_start_stack));
315 }
316
317 restore_bf_vars();
318}

320/**********************************************************/
321// save current corners and parameters to start of s_bn_root
322// to preserve values across calls to init_bf()
323static int save_bf_vars()
324{
325 int ret;
326 if (s_bn_root != nullptr)
327 {
328 const unsigned int mem = (g_bf_length+2)*22; // 6 corners + 6 save corners + 10 params
329 g_bf_save_len = g_bf_length;
330 std::memcpy(s_bn_root, g_bf_x_min, mem);
331 // scrub old high area
332 std::memset(g_bf_x_min, 0, mem);
333 ret = 0;
334 }
335 else
336 {
337 g_bf_save_len = 0;
338 ret = -1;
339 }
340 return ret;
341}
342
343/************************************************************************/
344// copy current corners and parameters from save location
345static int restore_bf_vars()
346{
347 if (g_bf_save_len == 0)
348 {
349 return -1;
350 }
351 BigFloat ptr = s_bn_root;
352 convert_bf(g_bf_x_min, ptr, g_bf_length, g_bf_save_len);
353 ptr += g_bf_save_len+2;
354 convert_bf(g_bf_x_max, ptr, g_bf_length, g_bf_save_len);
355 ptr += g_bf_save_len+2;
356 convert_bf(g_bf_y_min, ptr, g_bf_length, g_bf_save_len);
357 ptr += g_bf_save_len+2;
358 convert_bf(g_bf_y_max, ptr, g_bf_length, g_bf_save_len);
359 ptr += g_bf_save_len+2;
360 convert_bf(g_bf_x_3rd, ptr, g_bf_length, g_bf_save_len);
361 ptr += g_bf_save_len+2;
362 convert_bf(g_bf_y_3rd, ptr, g_bf_length, g_bf_save_len);
363 ptr += g_bf_save_len+2;
364 for (BigFloat &param : g_bf_params)
365 {
366 convert_bf(param, ptr, g_bf_length, g_bf_save_len);
367 ptr += g_bf_save_len + 2;
368 }
369 convert_bf(g_bf_save_x_min, ptr, g_bf_length, g_bf_save_len);
370 ptr += g_bf_save_len+2;
371 convert_bf(g_bf_save_x_max, ptr, g_bf_length, g_bf_save_len);
372 ptr += g_bf_save_len+2;
373 convert_bf(g_bf_save_y_min, ptr, g_bf_length, g_bf_save_len);
374 ptr += g_bf_save_len+2;
375 convert_bf(g_bf_save_y_max, ptr, g_bf_length, g_bf_save_len);
376 ptr += g_bf_save_len+2;
377 convert_bf(g_bf_save_x_3rd, ptr, g_bf_length, g_bf_save_len);
378 ptr += g_bf_save_len+2;
379 convert_bf(g_bf_save_y_3rd, ptr, g_bf_length, g_bf_save_len);
380
381 // scrub save area
382 std::memset(s_bn_root, 0, (g_bf_save_len+2)*22);
383 return 0;
384}

404/************************************************************************/
405// Memory allocator routines start here.
406/************************************************************************/
407// Allocates a BigNum variable on stack
408BigNum alloc_stack(const size_t size)
409{
410 if (g_bf_math == BFMathType::NONE)
411 {
412 stop_msg("alloc_stack called with g_bf_math==0");
413 return nullptr;
414 }
415 const long stack_addr = static_cast<long>(s_stack_ptr - s_bn_root + size); // part of s_bn_root
416
417 if (stack_addr > g_max_stack)
418 {
419 stop_msg("Aborting, Out of Bignum Stack Space");
420 goodbye();
421 }
422 // keep track of max ptr
423 g_bignum_max_stack_addr = std::max(stack_addr, g_bignum_max_stack_addr);
424 s_stack_ptr += size; // increment stack pointer
425 return s_stack_ptr - size;
426}
427
428/************************************************************************/
429// Returns stack pointer offset so it can be saved.
430int save_stack()
431{
432 return static_cast<int>(s_stack_ptr - s_bn_root);
433}
434
435/************************************************************************/
436// Restores stack pointer, effectively freeing local variables
437// allocated since save_stack()
438void restore_stack(const int old_offset)
439{
440 s_stack_ptr = s_bn_root+old_offset;
441}

448// initialize bignumber global variables
449// dec = decimal places after decimal point
450// intl = bytes for integer part (1, 2, or 4)
451
452void init_bf_dec(const int dec)
453{
454 if (g_bf_digits > 0)
455 {
456 g_decimals = g_bf_digits; // blindly force
457 }
458 else
459 {
460 g_decimals = dec;
461 }
462 if (g_user.bailout_value > 10) // arbitrary value
463 {
464 // using 2 doesn't gain much and requires another test
465 g_int_length = 4;
466 }
467 else if (g_fractal_type == FractalType::MANDEL_Z_POWER //
468 || g_fractal_type == FractalType::JULIA_Z_POWER //
469 || g_fractal_type == FractalType::DIVIDE_BROT5)
470 {
471 g_int_length = 4; // 2 leaves artifacts in the center of the lakes
472 }
473 // the bailout tests need greater dynamic range
474 else if (g_bailout_test == Bailout::REAL //
475 || g_bailout_test == Bailout::IMAG //
476 || g_bailout_test == Bailout::AND //
477 || g_bailout_test == Bailout::MANR)
478 {
479 g_int_length = 2;
480 }
481 else
482 {
483 g_int_length = 1;
484 }
485 // conservative estimate
486 g_bn_length = g_int_length + static_cast<int>(g_decimals / LOG10_256) + 1; // round up
487 init_bf2();
488}
