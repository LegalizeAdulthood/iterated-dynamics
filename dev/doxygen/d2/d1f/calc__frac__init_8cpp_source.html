63/* This function calculates the precision needed to distinguish adjacent
64 pixels at the maximum resolution of MAX_PIXELS by MAX_PIXELS
65 (if res==Resolution::MAX) or at current resolution (if res==Resolution::CURRENT) */
66static int get_prec_dbl(const ResolutionFlag flag)
67{
68 LDouble res;
69 if (flag == ResolutionFlag::MAX)
70 {
71 res = OLD_MAX_PIXELS - 1;
72 }
73 else
74 {
75 res = g_logical_screen.x_dots - 1;
76 }
77
78 const LDouble x_delta = (static_cast<LDouble>(g_image_region.m_max.x) - static_cast<LDouble>(g_image_region.m_3rd.x)) / res;
79 const LDouble y_delta2 = static_cast<LDouble>(g_image_region.height3()) / res;
80
81 if (flag == ResolutionFlag::CURRENT)
82 {
83 res = g_logical_screen.y_dots - 1;
84 }
85
86 const LDouble y_delta = (static_cast<LDouble>(g_image_region.m_max.y) - static_cast<LDouble>(g_image_region.m_3rd.y)) / res;
87 const LDouble x_delta2 = static_cast<LDouble>(g_image_region.width3()) / res;
88
89 LDouble del1 = std::abs(x_delta) + std::abs(x_delta2);
90 const LDouble del2 = std::abs(y_delta) + std::abs(y_delta2);
91 del1 = std::min(del2, del1);
92 if (del1 == 0)
93 {
94#ifndef NDEBUG
95 show_corners_dbl("getprecdbl");
96#endif
97 return -1;
98 }
99 int digits = 1;
100 while (del1 < 1.0)
101 {
102 digits++;
103 del1 *= 10;
104 }
105 digits = std::max(digits, 3);
106 return digits;
107}

185 fractal_float_to_bf();
186 }
187 else
188 {
189 free_bf_vars();
190 }

192 {
193 g_user.std_calc_mode = CalcMode::ONE_PASS;

197 int tries = 0;

199 driver_check_memory();

201 /* the following variables may be forced to a different setting due to

205 g_periodicity_check = g_user.periodicity_value;
206 g_distance_estimator = g_user.distance_estimator_value;

215 g_potential.flag = true;
216 g_user.distance_estimator_value = 0;
217 g_distance_estimator = 0; // can't do distest too
218 }
219
220 assert(g_cur_fractal_specific == get_fractal_specific(g_fractal_type));
221 g_cur_fractal_specific = get_fractal_specific(g_fractal_type);
222
223 if (g_potential.flag && g_potential.params[2] != 0.0)
224 {
225 g_magnitude_limit = g_potential.params[2];
226 }
227 else if (g_user.bailout_value) // user input bailout
228 {
229 g_magnitude_limit = g_user.bailout_value;
230 }
231 else if (g_biomorph != -1) // biomorph benefits from larger bailout
232 {
233 g_magnitude_limit = 100;
234 }
235 else
236 {
237 g_magnitude_limit = g_cur_fractal_specific->orbit_bailout;
238 }
239
240 if (bit_set(g_cur_fractal_specific->flags, FractalFlags::NO_ROTATE))
241 {
242 // ensure min<max and unrotated rectangle
243 if (g_image_region.m_min.x > g_image_region.m_max.x)
244 {
245 std::swap(g_image_region.m_min.x, g_image_region.m_max.x);
246 }
247 if (g_image_region.m_min.y > g_image_region.m_max.y)
248 {
249 std::swap(g_image_region.m_min.y, g_image_region.m_max.y);
250 }
251 g_image_region.m_3rd = g_image_region.m_min;
252 }
253
254 g_attractor.radius = 1.0/32768L;
255
256 // now setup arrays of real coordinates corresponding to each pixel
257 if (g_bf_math != BFMathType::NONE)
258 {
259 adjust_to_limits_bf(1.0); // make sure all corners in valid range
260 }
261 else
262 {
263 adjust_to_limits(1.0); // make sure all corners in valid range
264 g_delta_x = static_cast<LDouble>(g_image_region.m_max.x - g_image_region.m_3rd.x) / static_cast<LDouble>(g_logical_screen.x_size_dots); // calculate stepsizes
265 g_delta_y = static_cast<LDouble>(g_image_region.m_max.y - g_image_region.m_3rd.y) / static_cast<LDouble>(g_logical_screen.y_size_dots);
266 const DComplex size3{g_image_region.size3()};
267 g_delta_x2 = static_cast<LDouble>(size3.x) / static_cast<LDouble>(g_logical_screen.y_size_dots);
268 g_delta_y2 = static_cast<LDouble>(size3.y) / static_cast<LDouble>(g_logical_screen.x_size_dots);
269 fill_pixel_grid();
270 }
271
272 // skip this if plasma to avoid 3d problems
273 // skip if g_bf_math to avoid extraseg conflict with dx0 arrays
274 // skip if ifs, ifs3d, or lsystem to avoid crash when mathtolerance
275 // is set. These types don't auto switch between float and integer math
276 if (g_fractal_type != FractalType::PLASMA
277 && g_bf_math == BFMathType::NONE
278 && g_fractal_type != FractalType::IFS
279 && g_fractal_type != FractalType::IFS_3D
280 && g_fractal_type != FractalType::L_SYSTEM)
281 {
282expand_retry:
283 // set up dx0 and dy0 analogs of lx0 and ly0
284 // put fractal parameters in doubles
285 double dx0 = g_image_region.m_min.x; // fill up the x, y grids
286 double dy0 = g_image_region.m_max.y;
287 double dy1 = 0;
288 double dx1 = 0;
289 /* this way of defining the dx and dy arrays is not the most
290 accurate, but it is kept because it is used to determine
291 the limit of resolution */
292 for (int i = 1; i < g_logical_screen.x_dots; i++)
293 {
294 dx0 = dx0 + static_cast<double>(g_delta_x);
295 dy1 = dy1 - static_cast<double>(g_delta_y2);
296 }
297 for (int i = 1; i < g_logical_screen.y_dots; i++)
298 {
299 dy0 = dy0 - static_cast<double>(g_delta_y);
300 dx1 = dx1 + static_cast<double>(g_delta_x2);
301 }
302 if (g_bf_math == BFMathType::NONE) // redundant test, leave for now
303 {
304 /* Following is the old logic for detecting failure of double
305 precision. It has two advantages: it is independent of the
306 representation of numbers, and it is sensitive to resolution
307 (allows deeper zooms at lower resolution). However, it fails
308 for rotations of exactly 90 degrees, so we added a safety net
309 by using the magnification. */
310 if (++tries < 2) // for safety
311 {
312 if (tries > 1)
313 {
314 stop_msg("precision-detection error");
315 }
316 /* Previously there were four tests of distortions in the
317 zoom box used to detect precision limitations. In some
318 cases of rotated/skewed zoom boxes, this causes the algorithm
319 to bail out to arbitrary precision too soon. The logic
320 now only tests the larger of the two deltas in an attempt
321 to repair this bug. This should never make the transition
322 to arbitrary precision sooner, but always later.*/
323 double test_x_try;
324 double test_x_exact;
325 if (std::abs(g_image_region.m_max.x - g_image_region.m_3rd.x) > std::abs(g_image_region.width3()))
326 {
327 test_x_exact = g_image_region.m_max.x - g_image_region.m_3rd.x;
328 test_x_try = dx0 - g_image_region.m_min.x;
329 }
330 else
331 {
332 test_x_exact = g_image_region.width3();
333 test_x_try = dx1;
334 }
335 double testy_try;
336 double testy_exact;
337 if (std::abs(g_image_region.m_3rd.y - g_image_region.m_max.y) > std::abs(g_image_region.m_min.y - g_image_region.m_3rd.y))
338 {
339 testy_exact = g_image_region.m_3rd.y - g_image_region.m_max.y;
340 testy_try = dy0 - g_image_region.m_max.y;
341 }
342 else
343 {
344 testy_exact = g_image_region.m_min.y - g_image_region.m_3rd.y;
345 testy_try = dy1;
346 }
347 if (ratio_bad(test_x_try, test_x_exact) || ratio_bad(testy_try, testy_exact))
348 {
349 if (bit_set(g_cur_fractal_specific->flags, FractalFlags::BF_MATH))
350 {
351 fractal_float_to_bf();
352 goto init_restart;
353 }
354 goto expand_retry;
355 } // end if ratio_bad etc.
356 } // end if tries < 2
357 } // end if bf_math == 0
358
359 // if long double available, this is more accurate
360 fill_pixel_grid(); // fill up the x, y grids
361
362 // re-set corners to match reality
363 g_image_region.m_max.x = static_cast<double>(
364 g_image_region.m_min.x + (g_logical_screen.x_dots - 1) * g_delta_x + (g_logical_screen.y_dots - 1) * g_delta_x2);
365 g_image_region.m_min.y = static_cast<double>(
366 g_image_region.m_max.y - (g_logical_screen.y_dots - 1) * g_delta_y - (g_logical_screen.x_dots - 1) * g_delta_y2);
367 g_image_region.m_3rd.x = static_cast<double>(g_image_region.m_min.x + (g_logical_screen.y_dots - 1) * g_delta_x2);
368 g_image_region.m_3rd.y = static_cast<double>(g_image_region.m_max.y - (g_logical_screen.y_dots - 1) * g_delta_y);
369 // end else
370 } // end if not plasma
371
372 // for periodicity close-enough, and for unity:
373 // min(max(delx,delx2),max(dely,dely2))
374 g_delta_min = std::abs(g_delta_x);
375 g_delta_min = std::max(std::abs(static_cast<double>(g_delta_x2)), g_delta_min);
376 if (std::abs(g_delta_y) > std::abs(g_delta_y2))
377 {
378 g_delta_min = std::min(std::abs(static_cast<double>(g_delta_y)), g_delta_min);
379 }
380 else if (std::abs(g_delta_y2) < g_delta_min)
381 {
382 g_delta_min = std::abs(g_delta_y2);
383 }
384
385 // calculate factors which plot real values to screen co-ords
386 // calcfrac.c plot_orbit routines have comments about this
387 const double tmp = static_cast<double>(
388 (0.0 - g_delta_y2) * g_delta_x2 * g_logical_screen.x_size_dots * g_logical_screen.y_size_dots -
389 (g_image_region.m_max.x - g_image_region.m_3rd.x) * (g_image_region.m_3rd.y - g_image_region.m_max.y));
390 if (tmp != 0)
391 {
392 g_plot_mx1 = static_cast<double>(
393 g_delta_x2 * g_logical_screen.x_size_dots * g_logical_screen.y_size_dots / tmp);
394 g_plot_mx2 = (g_image_region.m_3rd.y-g_image_region.m_max.y) * g_logical_screen.x_size_dots / tmp;
395 g_plot_my1 = static_cast<double>(
396 (0.0 - g_delta_y2) * g_logical_screen.x_size_dots * g_logical_screen.y_size_dots / tmp);
397 g_plot_my2 = (g_image_region.m_max.x-g_image_region.m_3rd.x) * g_logical_screen.y_size_dots / tmp;
398 }
399 if (g_bf_math == BFMathType::NONE)
400 {
401 free_bf_vars();
402 }

483{

533static void adjust_to_limits_bf(const double expand)
534{
535 BigStackSaver saved;
536 BigFloat b_corner_x[4];
537 BigFloat b_corner_y[4];
538 b_corner_x[0] = alloc_stack(g_r_bf_length+2);
539 b_corner_x[1] = alloc_stack(g_r_bf_length+2);
540 b_corner_x[2] = alloc_stack(g_r_bf_length+2);

780static void adjust_to_limits(const double expand)
781{
782 double corner_x[4];
783 double corner_y[4];
784 constexpr double limit = 32767.99;
785 const DComplex center{g_image_region.center()};
786
787 if (g_image_region.m_min.x == center.x)
788 {
789 // ohoh, infinitely thin, fix it
790 smallest_add(&g_image_region.m_max.x);
791 g_image_region.m_min.x -= g_image_region.m_max.x- center.x;
792 }
793
794 if (g_image_region.m_min.y == center.y)
795 {
796 smallest_add(&g_image_region.m_max.y);
797 g_image_region.m_min.y -= g_image_region.m_max.y- center.y;
798 }
799
800 if (g_image_region.m_3rd.x == center.x)
801 {
802 smallest_add(&g_image_region.m_3rd.x);
803 }
804
805 if (g_image_region.m_3rd.y == center.y)
806 {
807 smallest_add(&g_image_region.m_3rd.y);
808 }
809
810 // setup array for easier manipulation
811 corner_x[0] = g_image_region.m_min.x;
812 corner_x[1] = g_image_region.m_max.x;
813 corner_x[2] = g_image_region.m_3rd.x;
814 corner_x[3] = g_image_region.m_min.x+(g_image_region.m_max.x-g_image_region.m_3rd.x);
815
816 corner_y[0] = g_image_region.m_max.y;
817 corner_y[1] = g_image_region.m_min.y;
818 corner_y[2] = g_image_region.m_3rd.y;
819 corner_y[3] = g_image_region.m_min.y+(g_image_region.m_max.y-g_image_region.m_3rd.y);
820
821 // if caller wants image size adjusted, do that first
822 if (expand != 1.0)
823 {
824 for (int i = 0; i < 4; ++i)
825 {
826 corner_x[i] = center.x + (corner_x[i]- center.x)*expand;
827 corner_y[i] = center.y + (corner_y[i]- center.y)*expand;
828 }
829 }
830 // get min/max x/y values
831 double high_x = corner_x[0];
832 double low_x = corner_x[0];
833 double high_y = corner_y[0];
834 double low_y = corner_y[0];
835
836 for (int i = 1; i < 4; ++i)
837 {
838 low_x = std::min(corner_x[i], low_x);
839 high_x = std::max(corner_x[i], high_x);
840 low_y = std::min(corner_y[i], low_y);
841 high_y = std::max(corner_y[i], high_y);
842 }
843
844 // if image is too large, downsize it maintaining center
845 double f_temp = high_x - low_x;
846 f_temp = std::max(high_y - low_y, f_temp);
847
848 // if image is too large, downsize it maintaining center
849 f_temp = limit*2/f_temp;
850 if (f_temp < 1.0)
851 {
852 for (int i = 0; i < 4; ++i)
853 {
854 corner_x[i] = center.x + (corner_x[i]- center.x)*f_temp;
855 corner_y[i] = center.y + (corner_y[i]- center.y)*f_temp;
856 }
857 }
858
859 // if any corner has x or y past limit, move the image
860 double adj_y = 0;
861 double adj_x = 0;
862
863 for (int i = 0; i < 4; ++i)
864 {
865 if (corner_x[i] > limit && (f_temp = corner_x[i] - limit) > adj_x)
866 {
867 adj_x = f_temp;
868 }
869 if (corner_x[i] < 0.0-limit && (f_temp = corner_x[i] + limit) < adj_x)
870 {
871 adj_x = f_temp;
872 }
873 if (corner_y[i] > limit && (f_temp = corner_y[i] - limit) > adj_y)
874 {
875 adj_y = f_temp;
876 }
877 if (corner_y[i] < 0.0-limit && (f_temp = corner_y[i] + limit) < adj_y)
878 {
879 adj_y = f_temp;
880 }
881 }
882 if (g_calc_status == CalcStatus::RESUMABLE && (adj_x != 0 || adj_y != 0) && g_zoom_box_width == 1.0)
883 {
884 g_calc_status = CalcStatus::PARAMS_CHANGED;
885 }
886 g_image_region.m_min.x = corner_x[0] - adj_x;
887 g_image_region.m_max.x = corner_x[1] - adj_x;
888 g_image_region.m_3rd.x = corner_x[2] - adj_x;
889 g_image_region.m_max.y = corner_y[0] - adj_y;
890 g_image_region.m_min.y = corner_y[1] - adj_y;
891 g_image_region.m_3rd.y = corner_y[2] - adj_y;
892
893 adjust_corner(); // make 3rd corner exact if very near other co-ords
894}
