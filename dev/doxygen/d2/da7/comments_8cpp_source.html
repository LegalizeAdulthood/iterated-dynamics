34static std::string get_cpu_id()
35{
36 static std::string cpu_id;
37 if (cpu_id.empty())
38 {
39 if (!cpuid_present())
40 {
41 cpu_id = "(Unknown CPU)";
42 }
43 const auto check = [](const int result, const char *label)
44 {
45 if (result < 0)
46 {
47 throw std::runtime_error(std::string{label} + ": " + cpuid_error());
48 }
49 };
50 cpu_id_t data;
51 check(cpu_identify(nullptr, &data), "cpu_identify");
52 std::string brand{data.brand_str};
53 const auto drop = [&](const std::string_view text)
54 {
55 for (std::string::size_type pos = brand.find(text); pos != std::string::npos;
56 pos = brand.find(text))
57 {
58 brand.erase(pos, text.length());
59 }
60 };
61 drop("(R)");
62 drop("(TM)");
63 cpu_id = brand;
64 }
65 return cpu_id;
66}

150// extract comments from the comments= command
151void parse_comments(char *value)
152{
153 for (char *elem : g_par_comment)
154 {
155 char save = '\0';
156 if (*value == 0)
157 {
158 break;
159 }
160 char *next = std::strchr(value, '/');
161 if (*value != '/')
162 {
163 if (next != nullptr)
164 {
165 save = *next;
166 *next = '\0';
167 }
168 std::strncpy(elem, value, MAX_COMMENT_LEN);
169 }
170 if (next == nullptr)
171 {
172 break;
173 }
174 if (save != '\0')
175 {
176 *next = save;
177 }
178 value = next+1;
179 }
180}

182// expands comments from the comments= command
183static std::string expand_comments(const std::string_view source, const std::time_t local_time)
184{
185 constexpr char QUOTE{'\\'}; // used to quote the next character (_, \ or $) from special interpretation
186 constexpr char DELIMITER{'$'}; // delimits variable names
187 bool in_variable{};
188 std::string var_name;
189 char last_c{};
190 std::string target;
191 for (char c : source)
192 {
193 if (c == QUOTE && last_c != QUOTE)
194 {
195 last_c = c;
196 continue;
197 }
198 // expand underscores to blanks
199 if (c == '_' && last_c != QUOTE)
200 {
201 c = ' ';
202 }
203 // DELIMITER marks start and end of variable names
204 if (c == DELIMITER && last_c != QUOTE)
205 {
206 in_variable = !in_variable;
207 }
208 if (c != DELIMITER && in_variable) // if true, building variable name
209 {
210 var_name += c;
211 }
212 // got variable name
213 else if (c == DELIMITER && !in_variable && last_c != QUOTE)
214 {
215 target += expand_var(var_name, local_time);
216 }
217 else if ((c != DELIMITER || last_c == QUOTE) && !in_variable)
218 {
219 target += c;
220 }
221 last_c = c == QUOTE && last_c == QUOTE ? '\0' : c;
222 }
223 return target;
224}
