33static std::string get_cpu_id()
34{
35 static std::string cpu_id;
36 if (cpu_id.empty())
37 {
38 if (!cpuid_present())
39 {
40 cpu_id = "(Unknown CPU)";
41 }
42 const auto check = [](const int result, const char *label)
43 {
44 if (result < 0)
45 {
46 throw std::runtime_error(std::string{label} + ": " + cpuid_error());
47 }
48 };
49 cpu_id_t data;
50 check(cpu_identify(nullptr, &data), "cpu_identify");
51 std::string brand{data.brand_str};
52 const auto drop = [&](const std::string_view text)
53 {
54 for (std::string::size_type pos = brand.find(text); pos != std::string::npos;
55 pos = brand.find(text))
56 {
57 brand.erase(pos, text.length());
58 }
59 };
60 drop("(R)");
61 drop("(TM)");
62 cpu_id = brand;
63 }
64 return cpu_id;
65}

149// extract comments from the comments= command
150void parse_comments(std::string_view value)
151{
152 for (std::string &elem : g_par_comment)
153 {
154 if (value.empty())
155 {
156 break;
157 }
158 const std::size_t next{value.find('/')};
159 if (value.front() != '/')
160 {
161 const std::size_t len{std::min(next, static_cast<std::size_t>(MAX_COMMENT_LEN - 1))};
162 elem = value.substr(0, len);
163 }
164 if (next == std::string_view::npos)
165 {
166 break;
167 }
168 value.remove_prefix(next + 1);
169 }
170}

172// expands comments from the comments= command
173static std::string expand_comments(const std::string_view source, const std::time_t local_time)
174{
175 constexpr char QUOTE{'\\'}; // used to quote the next character (_, \ or $) from special interpretation
176 constexpr char DELIMITER{'$'}; // delimits variable names
177 bool in_variable{};
178 std::string var_name;
179 char last_c{};
180 std::string target;
181 for (char c : source)
182 {
183 if (c == QUOTE && last_c != QUOTE)
184 {
185 last_c = c;
186 continue;
187 }
188 // expand underscores to blanks
189 if (c == '_' && last_c != QUOTE)
190 {
191 c = ' ';
192 }
193 // DELIMITER marks start and end of variable names
194 if (c == DELIMITER && last_c != QUOTE)
195 {
196 in_variable = !in_variable;
197 }
198 if (c != DELIMITER && in_variable) // if true, building variable name
199 {
200 var_name += c;
201 }
202 // got variable name
203 else if (c == DELIMITER && !in_variable && last_c != QUOTE)
204 {
205 target += expand_var(var_name, local_time);
206 }
207 else if ((c != DELIMITER || last_c == QUOTE) && !in_variable)
208 {
209 target += c;
210 }
211 last_c = c == QUOTE && last_c == QUOTE ? '\0' : c;
212 }
213 return target;
214}
