21int burning_ship_orbit()
22{
23 DComplex q;
24 const int degree = static_cast<int>(g_params[2]);
25 q.x = g_init.x;
26 q.y = g_init.y;
27 if (degree == 2)
28 {
29 // (|x| + i|y|)^2 = (|x| + i|y|)*(|x| + i|y|)
30 // = (|x|*|x| - |y|*|y|) + i(|x|*|y| + |x|*|y|)
31 // = (|x|^2 - |y|^2) + i(2*|x|*|y|)
32 g_temp_sqr_x = sqr(g_old_z.x); // real(z)^2 = |real(z)|^2
33 g_temp_sqr_y = sqr(g_old_z.y); // imag(z)^2 = |imag(z)|^2
34 const double real_imag = std::abs(g_old_z.x * g_old_z.y); // |real(z)*imag(z)| = |real(z)|*|imag(z)|
35 g_new_z.x = g_temp_sqr_x - g_temp_sqr_y + q.x; // real(z)^2 - imag(z)^2 + real(c)
36 g_new_z.y = real_imag + real_imag - q.y; // |real(z)*imag(z)|*2 - imag(c)
37 }
38 else if (degree > 2)
39 {
40 DComplex z;
41 z.x = std::abs(g_old_z.x);
42 z.y = -std::abs(g_old_z.y);
43 pow(&z, degree, &z);
44 g_new_z.x = z.x + q.x;
45 g_new_z.y = z.y + q.y;
46 }
47 return g_bailout_float();
48}

50static double diff_abs(const double c, const double d)
51{
52 const double cd = c + d;
53
54 if (c >= 0.0)
55 {
56 return cd >= 0.0 ? d : -d - 2.0 * c;
57 }
58 return cd > 0.0 ? d + 2.0 * c : -d;
59}

61void burning_ship_perturb(
62 const std::complex<double> &ref, std::complex<double> &delta_n, const std::complex<double> &delta0)
63{
64 const int degree = static_cast<int>(g_params[2]);
65 const double r{ref.real()};
66 const double i{ref.imag()};
67 const double r2 = r * r;
68 const double i2 = i * i;
69 const double a{delta_n.real()};
70 const double b{delta_n.imag()};
71 const double a2 = a * a;
72 const double b2 = b * b;
73 const double a0{delta0.real()};
74 const double b0{delta0.imag()};
75
76 switch (degree)
77 {
78 case 2:
79 delta_n.real(2.0 * a * r + a2 - 2.0 * b * i - b2);
80 delta_n.imag(diff_abs(r * i, r * b + i * a + a * b) * 2);
81 delta_n += delta0;
82 break;
83
84 case 3:
85 {
86 double dnr = diff_abs(r, a);
87 double ab = r + a;
88 dnr = (r * r - 3 * i * i) * dnr //
89 + (2 * a * r + a2 - 6 * i * b - 3 * b2) * std::abs(ab) //
90 + a0;
91 double dni = diff_abs(i, b);
92 ab = i + b;
93 dni = (3 * r * r - i * i) * dni //
94 + (6 * r * a + 3 * a2 - 2 * i * b - b2) * std::abs(ab) //
95 + b0;
96 delta_n.imag(dni);
97 delta_n.real(dnr);
98 break;
99 }
100
101 case 4:
102 {
103 const double dnr = 4 * r2 * r * a //
104 + 6 * r2 * a2 //
105 + 4 * r * a2 * a //
106 + a2 * a2 //
107 + 4 * i2 * i * b //
108 + 6 * i2 * b2 //
109 + 4 * i * b2 * b //
110 + b2 * b2 //
111 - 12 * r2 * i * b //
112 - 6 * r2 * b2 //
113 - 12 * r * a * i2 //
114 - 24 * r * a * i * b //
115 - 12 * r * a * b2 //
116 - 6 * a2 * i2 //
117 - 12 * a2 * i * b //
118 - 6 * a2 * b2 //
119 + a0;
120 double dni = diff_abs(r * i, r * b + a * i + a * b);
121 dni = 4 * (r2 - i2) * dni //
122 + 4 * std::abs(r * i + r * b + a * i + a * b) * (2 * a * r + a2 - 2 * b * i - b2) //
123 + b0;
124 delta_n.imag(dni);
125 delta_n.real(dnr);
126 break;
127 }
128
129 case 5:
130 {
131 double dnr = diff_abs(r, a);
132 dnr = dnr * (r * r * r * r - 10 * r * r * i * i + 5 * i * i * i * i) //
133 + std::abs(r + a) *
134 (4 * r * r * r * a //
135 + 6 * r * r * a2 //
136 + 4 * r * a2 * a //
137 + a2 * a2 //
138 - 20 * r2 * i * b //
139 - 10 * r2 * b2 //
140 - 20 * r * a * i2 //
141 - 40 * r * a * i * b //
142 - 20 * r * a * b2 //
143 - 10 * a2 * i2 //
144 - 20 * a2 * i * b //
145 - 10 * a2 * b2 //
146 + 20 * i2 * i * b //
147 + 30 * i2 * b2 //
148 + 20 * i * b2 * b //
149 + 5 * b2 * b2) //
150 + a0;
151 const double dni = diff_abs(i, b) * (5 * r2 * r2 - 10 * r2 * i2 + i2 * i2) + //
152 std::abs(i + b) *
153 (20 * r2 * r * a //
154 + 30 * r2 * a2 //
155 + 20 * r * a2 * a //
156 + 5 * a2 * a2 //
157 - 20 * r2 * i * b //
158 - 10 * r2 * b2 //
159 - 20 * r * a * i2 //
160 - 40 * r * a * i * b //
161 - 20 * r * a * b2 //
162 - 10 * a2 * i2 //
163 - 20 * a2 * i * b //
164 - 10 * a2 * b2 //
165 + 4 * i2 * i * b //
166 + 6 * i2 * b2 //
167 + 4 * i * b2 * b //
168 + b2 * b2) //
169 + b0;
170 delta_n.imag(dni);
171 delta_n.real(dnr);
172 break;
173 }
174
175 default:
176 throw std::runtime_error("Unexpected degree " + std::to_string(degree));
177 }
178}

180void burning_ship_ref_pt(const std::complex<double> &center, std::complex<double> &z)
181{
182 if (const int degree = static_cast<int>(g_params[2]); degree == 2)
183 {
184 const double real_sqr = sqr(z.real());
185 const double imag_sqr = sqr(z.imag());
186 const double real = real_sqr - imag_sqr + center.real();
187 const double temp = std::abs(2.0 * z.real() * z.imag());
188 const double imag = temp + center.imag();
189 z.real(real);
190 z.imag(imag);
191 }
192 else if (degree > 2)
193 {
194 DComplex temp;
195 DComplex temp_z;
196 temp.x = std::abs(z.real());
197 temp.y = std::abs(z.imag());
198 pow(&temp, degree, &temp_z);
199 z.real(temp_z.x);
200 z.imag(temp_z.y);
201 z += center;
202 }
203}

205void burning_ship_ref_pt_bf(const BFComplex &center, BFComplex &z)
206{
207 const int degree = static_cast<int>(g_params[2]);
208 BigStackSaver saved;
209 BigFloat temp_real = alloc_stack(g_r_bf_length + 2);
210 BigFloat real_sqr = alloc_stack(g_r_bf_length + 2);
211 BigFloat imag_sqr = alloc_stack(g_r_bf_length + 2);
212 BigFloat real_imag = alloc_stack(g_r_bf_length + 2);
213 BFComplex z2;
214 z2.x = alloc_stack(g_r_bf_length + 2);
215 z2.y = alloc_stack(g_r_bf_length + 2);
216 double_bf(z2.x, z.x);
217 double_bf(z2.y, z.y);
218 if (degree == 2)
219 {
220 square_bf(real_sqr, z.x);
221 square_bf(imag_sqr, z.y);
222 sub_bf(temp_real, real_sqr, imag_sqr);
223 add_bf(z.x, temp_real, center.x);
224 mult_bf(real_imag, z2.x, z.y);
225 abs_bf(real_imag, real_imag);
226 add_bf(z.y, real_imag, center.y);
227 }
228 else if (degree > 2)
229 {
230 abs_bf(z2.x, z.x);
231 abs_bf(z2.y, z.y);
232 power(g_new_z_bf, z2, degree);
233 add_bf(z.x, g_new_z_bf.x, center.x);
234 add_bf(z.y, g_new_z_bf.y, center.y);
235 }
236}

238int burning_ship_bf_fractal()
239{
240 const int degree = static_cast<int>(g_params[2]);
241 BigStackSaver saved;
242
243 if (degree == 2) // Burning Ship
244 {
245 square_bf(g_tmp_sqr_x_bf, g_old_z_bf.x);
246 square_bf(g_tmp_sqr_y_bf, g_old_z_bf.y);

248 add_bf(g_new_z_bf.x, g_bf_tmp, g_param_z_bf.x);
249 mult_bf(g_bf_tmp, g_old_z_bf.x, g_old_z_bf.y);
250 double_a_bf(g_bf_tmp);
251 abs_a_bf(g_bf_tmp);
252 sub_bf(g_new_z_bf.y, g_bf_tmp, g_param_z_bf.y);
253 }
254 else if (degree > 2) // Burning Ship to higher power
255 {
256 BFComplex big_z;
257 big_z.x = alloc_stack(g_bf_length + 2);
258 big_z.y = alloc_stack(g_bf_length + 2);
259 abs_bf(big_z.x, g_old_z_bf.x);
260 abs_bf(big_z.y, g_old_z_bf.y);
261 neg_a_bf(big_z.y);
262 power(g_new_z_bf, big_z, degree);
263 add_a_bf(g_new_z_bf.x, g_param_z_bf.x);
264 add_a_bf(g_new_z_bf.y, g_param_z_bf.y);
265 }
266 return g_bailout_bigfloat();
267}
