97void trig_details(char *buf)
98{
99 int num_fn;
100 if (g_fractal_type == FractalType::JULIBROT)
101 {
102 num_fn = +get_fractal_specific(g_new_orbit_type)->flags >> 6 & 7;
103 }
104 else
105 {
106 num_fn = +g_cur_fractal_specific->flags >> 6 & 7;
107 }
108 if (g_fractal_type == FractalType::FORMULA)
109 {
110 num_fn = g_max_function;
111 }
112 *buf = 0; // null string if none
113 if (num_fn > 0)
114 {
115 std::strcpy(buf, g_trig_fn[+g_trig_index[0]].name);
116 int i = 0;
117 while (++i < num_fn)
118 {
119 std::strcat(buf, "/");
120 std::strcat(buf, g_trig_fn[+g_trig_index[i]].name);
121 }
122 }
123}

125// set array of trig function indices according to "function=" command
126int set_trig_array(const int k, const char *name)
127{
128 char trig_name[10];
129 std::strncpy(trig_name, name, 6);
130 trig_name[6] = 0; // safety first
131
132 if (char *slash = std::strchr(trig_name, '/'); slash != nullptr)
133 {
134 *slash = 0;
135 }
136
137 string_lower(trig_name);
138
139 for (int i = 0; i < g_num_trig_functions; i++)
140 {
141 if (std::strcmp(trig_name, g_trig_fn[i].name) == 0)
142 {
143 g_trig_index[k] = static_cast<TrigFn>(i);
144 set_trig_pointers(k);
145 break;
146 }
147 }
148 return 0;
149}

151void set_trig_pointers(const int which)
152{
153 // set trig variable functions to avoid array lookup time
154 switch (which)
155 {
156 case 0:
157 g_d_trig0 = g_trig_fn[+g_trig_index[0]].d_fn;
158 break;
159 case 1:
160 g_d_trig1 = g_trig_fn[+g_trig_index[1]].d_fn;
161 break;
162 case 2:
163 g_d_trig2 = g_trig_fn[+g_trig_index[2]].d_fn;
164 break;
165 case 3:
166 g_d_trig3 = g_trig_fn[+g_trig_index[3]].d_fn;
167 break;
168 default: // do 'em all
169 for (int i = 0; i < 4; i++)
170 {
171 set_trig_pointers(i);
172 }
173 break;
174 }
175}
