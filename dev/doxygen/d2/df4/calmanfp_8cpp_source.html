34long mandelbrot_orbit()
35{
36 double x;
37 double y;
38 double x2;
39 double y2;
40 double xy;
41 double c_x;
42 double c_y;
43
44 if (g_periodicity_check == 0)
45 {
46 g_old_color_iter = 0; // don't check periodicity
47 }
48 else if (g_reset_periodicity)
49 {
50 g_old_color_iter = g_max_iterations - 255;
51 }
52
53 const long tmp_fsd = g_max_iterations - g_first_saved_and;
54 // this defeats checking periodicity immediately
55 // but matches the code in standard_fractal()
56 g_old_color_iter = std::min(g_old_color_iter, tmp_fsd);
57
58 // initparms
59 double saved_x = 0;
60 double saved_y = 0;
61 g_orbit_save_index = 0;
62 long saved_and = g_first_saved_and;
63 int saved_incr = 1; // start checking the very first time
64
65 long cx = g_max_iterations;
66 if (g_fractal_type != FractalType::JULIA)
67 {
68 // Mandelbrot_87
69 c_x = g_init.x;
70 c_y = g_init.y;
71 x = g_param_z1.x+c_x;
72 y = g_param_z1.y+c_y;
73 }
74 else
75 {
76 // dojulia_87
77 c_x = g_param_z1.x;
78 c_y = g_param_z1.y;
79 x = g_init.x;
80 y = g_init.y;
81 x2 = x*x;
82 y2 = y*y;
83 xy = x*y;
84 x = x2-y2+c_x;
85 y = 2*xy+c_y;
86 }
87 x2 = x*x;
88 y2 = y*y;
89 xy = x*y;
90
91 // top_of_cs_loop_87
92 while (--cx > 0)
93 {
94 x = x2-y2+c_x;
95 y = 2*xy+c_y;
96 x2 = x*x;
97 y2 = y*y;
98 xy = x*y;
99 g_magnitude = x2+y2;
100
101 if (g_magnitude >= g_magnitude_limit)
102 {
103 goto over_bailout_87;
104 }
105
106 // no_save_new_xy_87
107 if (cx < g_old_color_iter) // check periodicity
108 {
109 if ((g_max_iterations - cx & saved_and) == 0)
110 {
111 saved_x = x;
112 saved_y = y;
113 saved_incr--;
114 if (saved_incr == 0)
115 {
116 saved_and = (saved_and << 1) + 1;
117 saved_incr = g_periodicity_next_saved_incr;
118 }
119 }
120 else
121 {
122 if (std::abs(saved_x-x) < g_close_enough && std::abs(saved_y-y) < g_close_enough)
123 {
124 // oldcoloriter = 65535;
125 g_old_color_iter = g_max_iterations;
126 g_real_color_iter = g_max_iterations;
127 g_keyboard_check_interval = g_keyboard_check_interval -(g_max_iterations-cx);
128 g_color_iter = s_periodicity_color;
129 goto pop_stack;
130 }
131 }
132 }
133 // no_periodicity_check_87
134 if (g_show_orbit)
135 {
136 plot_orbit(x, y, -1);
137 }
138 // no_show_orbit_87
139 } // while (--cx > 0)
140
141 // reached maxit
142 // check periodicity immediately next time, remember we count down from maxit
143 g_old_color_iter = g_max_iterations;
144 g_keyboard_check_interval -= g_max_iterations;
145 g_real_color_iter = g_max_iterations;
146 g_color_iter = s_inside_color;
147
148pop_stack:
149 if (g_orbit_save_index)
150 {
151 scrub_orbit();
152 }
153 return g_color_iter;
154
155over_bailout_87:
156 if (g_outside_method <= ColorMethod::REAL)
157 {
158 g_new_z.x = x;
159 g_new_z.y = y;
160 }
161 if (cx-10 > 0)
162 {
163 g_old_color_iter = cx-10;
164 }
165 else
166 {
167 g_old_color_iter = 0;
168 }
169 g_real_color_iter = g_max_iterations-cx;
170 g_color_iter = g_real_color_iter;
171 if (g_color_iter == 0)
172 {
173 g_color_iter = 1;
174 }
175 g_keyboard_check_interval -= g_real_color_iter;
176 if (g_outside_method == ColorMethod::ITER)
177 {
178 }
179 else if (g_outside_method > ColorMethod::REAL)
180 {
181 g_color_iter = g_outside_color;
182 }
183 else
184 {
185 // special_outside
186 if (g_outside_method == ColorMethod::REAL)
187 {
188 g_color_iter += static_cast<long>(g_new_z.x) + 7;
189 }
190 else if (g_outside_method == ColorMethod::IMAG)
191 {
192 g_color_iter += static_cast<long>(g_new_z.y) + 7;
193 }
194 else if (g_outside_method == ColorMethod::MULT && g_new_z.y != 0.0)
195 {
196 g_color_iter = static_cast<long>(static_cast<double>(g_color_iter) * (g_new_z.x / g_new_z.y));
197 }
198 else if (g_outside_method == ColorMethod::SUM)
199 {
200 g_color_iter += static_cast<long>(g_new_z.x + g_new_z.y);
201 }
202 else if (g_outside_method == ColorMethod::ATAN)
203 {
204 g_color_iter = static_cast<long>(std::abs(std::atan2(g_new_z.y, g_new_z.x) * g_atan_colors / PI));
205 }
206 // check_color
207 if ((g_color_iter <= 0 || g_color_iter > g_max_iterations) && g_outside_method != ColorMethod::FMOD)
208 {
209 g_color_iter = 1;
210 }
211 }
212
213 goto pop_stack;
214}
