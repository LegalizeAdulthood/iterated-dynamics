40// extern short out_line(pixels, linelen)
41// UBYTE pixels[];
42// short linelen;
43//
44// - This function takes a full line of pixels (one byte per pixel) and
45// displays them (or does whatever your program wants with them...). It
46// should return zero, or negative if an error or some other event occurs
47// which would require aborting the decode process... Note that the length
48// passed will almost always be equal to the line length passed to the
49// decoder function, with the sole exception occurring when an ending code
50// occurs in an odd place in the GIF file... In any case, linelen will be
51// equal to the number of pixels passed...

102// The reason we have these separated like this instead of using
103// a structure like the original Wilhite code did, is because this
104// stuff generally produces significantly faster code when compiled...
105// This code is full of similar speedups... (For a good book on writing
106// C for speed or for space optimization, see Efficient C by Tom Plum,
107// published by Plum-Hall Associates...)

109// short decoder(linewidth)
110// short linewidth; * Pixels per line of image *
111//
112// - This function decodes an LZW image, according to the method used
113// in the GIF spec. Every *linewidth* "characters" (i.e. pixels) decoded
114// will generate a call to out_line(), which is a user specific function
115// to display a line of pixels. The function gets its codes from
116// get_next_code() which is responsible for reading blocks of data and
117// separating them into the proper size codes. Finally, get_byte() is
118// the global routine to read the next byte from the GIF file.
119//
120// It is generally a good idea to have linewidth correspond to the actual
121// width of a line (as specified in the Image header) to make your own
122// code a bit simpler, but it isn't absolutely necessary.
123//
124// Returns: 0 if successful, else negative. (See ERRS.H)
125//
126//
127
128// moved sizeofstring here for possible re-use elsewhere
129static short s_sizeof_string[MAX_CODES + 1]{}; // size of string list

131short decoder(const short line_width)
132{
133 U16 prefix[MAX_CODES+1]{}; // Prefix linked list
134 short ret;
135 short c;
136
137 // Initialize for decoding a new image...
138
139 const short size = static_cast<short>(get_byte());
140 if (size < 0)
141 {
142 return size;
143 }
144 if (size < 2 || 9 < size)
145 {
146 return BAD_CODE_SIZE;
147 }
148
149 s_curr_size = static_cast<short>(size + 1);
150 short top_slot = static_cast<short>(1 << s_curr_size); // Highest code for current size
151 const short clear = static_cast<short>(1 << size); // Value for a clear code
152 const short ending = static_cast<short>(clear + 1); // Value for an ending code
153 const short new_codes = static_cast<short>(ending + 1); // First available code
154 short slot = new_codes; // Last read code
155 short old_code = 0;
156 short y_skip = 0;
157 short x_skip = 0;
158 s_sizeof_string[slot] = 0;
159 s_num_bits_left = 0;
160 s_num_avail_bytes = 0;
161 Byte out_value = 0;
162 for (short i = 0; i < slot; i++)
163 {
164 s_sizeof_string[i] = 0;
165 }
166
167 // Initialize in case they forgot to put in a clear code. (This shouldn't
168 // happen, but we'll try and decode it anyway...)
169
170 // Set up the stack pointer and decode buffer pointer
171 Byte decode_stack[4096]{};
172 Byte *sp = decode_stack;
173 Byte *buf_ptr = s_decoder_line;
174 short buf_cnt = line_width; // how many empty spaces left in buffer
175
176 // This is the main loop. For each code we get we pass through the linked
177 // list of prefix codes, pushing the corresponding "character" for each
178 // code onto the stack. When the list reaches a single "character" we
179 // push that on the stack too, and then start unstacking each character
180 // for output in the correct order. Special handling is included for the
181 // clear code, and the whole thing ends when we get an ending code.
182 while ((c = get_next_code()) != ending)
183 {
184
185 // If we had a file error, return without completing the decode
186 if (c < 0)
187 {
188 return 0;
189 }
190
191 // If the code is a clear code, reinitialize all necessary items.
192 if (c == clear)
193 {
194 s_curr_size = static_cast<short>(size + 1);
195 slot = new_codes;
196 s_sizeof_string[slot] = 0;
197 top_slot = static_cast<short>(1 << s_curr_size);
198
199 // Continue reading codes until we get a non-clear code (Another
200 // unlikely, but possible case...)
201 do
202 {
203 c = get_next_code();
204 }
205 while (c == clear);
206
207 // If we get an ending code immediately after a clear code (Yet
208 // another unlikely case), then break out of the loop.
209 if (c == ending)
210 {
211 break;
212 }
213
214 // Finally, if the code is beyond the range of already set codes,
215 // (This one had better NOT happen... I have no idea what will
216 // result from this, but I doubt it will look good...) then set it
217 // to color zero.
218 if (c >= slot)
219 {
220 c = 0;
221 }
222
223 old_code = c;
224 out_value = static_cast<Byte>(old_code);
225
226 // And let us not forget to put the char into the buffer...
227 *sp++ = static_cast<Byte>(c);
228 }
229 else
230 {
231 // In this case, it's not a clear code or an ending code, so it must
232 // be a code code... So we can now decode the code into a stack of
233 // character codes. (Clear as mud, right?)
234 short code = c;
235
236 // Here we go again with one of those off chances... If, on the off
237 // chance, the code we got is beyond the range of those already set
238 // up (Another thing which had better NOT happen...) we trick the
239 // decoder into thinking it actually got the next slot avail.
240
241 if (code >= slot)
242 {
243 if (code > slot)
244 {
245 ++s_bad_code_count;
246 c = slot;
247 }
248 code = old_code;
249 *sp++ = out_value;
250 }
251
252 // Here we scan back along the linked list of prefixes. If they can
253 // fit into the output buffer than transfer them direct. ELSE push
254 // them into the stack until we are down to enough characters that
255 // they do fit. Output the line then fall through to unstack the
256 // ones that would not fit.
257 short fast_loop = NOPE;
258 while (code >= new_codes)
259 {
260 int i = s_sizeof_string[code];
261 short j = i;
262 if (i > 0 && buf_cnt - i > 0 && g_skip_x_dots == 0)
263 {
264 fast_loop = YUP;
265
266 do
267 {
268 *(buf_ptr + j) = s_suffix[code];
269 code = prefix[code];
270 }
271 while (--j > 0);
272 *buf_ptr = static_cast<Byte>(code);
273 buf_ptr += ++i;
274 buf_cnt -= i;
275 if (buf_cnt == 0) // finished an input row?
276 {
277 if (--y_skip < 0)
278 {
279 ret = static_cast<short>(
280 g_out_line(s_decoder_line, static_cast<int>(buf_ptr - s_decoder_line)));
281 if (ret < 0)
282 {
283 return ret;
284 }
285 y_skip = g_skip_y_dots;
286 }
287 buf_ptr = s_decoder_line;
288 buf_cnt = line_width;
289 x_skip = 0;
290 }
291 }
292 else
293 {
294 *sp++ = s_suffix[code];
295 code = prefix[code];
296 }
297 }
298
299 // Push the last character on the stack, and set up the new prefix
300 // and suffix, and if the required slot number is greater than that
301 // allowed by the current bit size, increase the bit size. (NOTE -
302 // If we are all full, we *don't* save the new suffix and prefix...
303 // I'm not certain if this is correct... it might be more proper to
304 // overwrite the last code...)
305 if (fast_loop == NOPE)
306 {
307 *sp++ = static_cast<Byte>(code);
308 }
309
310 if (slot < top_slot)
311 {
312 s_sizeof_string[slot] = static_cast<short>(s_sizeof_string[old_code] + 1);
313 out_value = static_cast<Byte>(code);
314 s_suffix[slot] = out_value;
315 prefix[slot++] = old_code;
316 old_code = c;
317 }
318 if (slot >= top_slot)
319 {
320 if (s_curr_size < 12)
321 {
322 top_slot <<= 1;
323 ++s_curr_size;
324 }
325 }
326 }
327 while (sp > decode_stack)
328 {
329 --sp;
330 if (--x_skip < 0)
331 {
332 x_skip = g_skip_x_dots;
333 *buf_ptr++ = *sp;
334 }
335 if (--buf_cnt == 0) // finished an input row?
336 {
337 if (--y_skip < 0)
338 {
339 ret = static_cast<short>(g_out_line(s_decoder_line, static_cast<int>(buf_ptr - s_decoder_line)));
340 if (ret < 0)
341 {
342 return ret;
343 }
344 y_skip = g_skip_y_dots;
345 }
346 buf_ptr = s_decoder_line;
347 buf_cnt = line_width;
348 x_skip = 0;
349 }
350 }
351 }
352 return 0;
353}

355// get_next_code()
356// - gets the next code from the GIF file. Returns the code, or else
357// a negative number in case of file errors...
358//
359static short get_next_code()
360{
361 static Byte b1; // Current byte
362 static unsigned short ret_code;
363
364 if (s_num_bits_left == 0)
365 {
366 if (s_num_avail_bytes <= 0)
367 {
368
369 // Out of bytes in current block, so read next block
370 s_ptr_bytes = s_byte_buff;
371 s_num_avail_bytes = static_cast<short>(get_byte());
372 if (s_num_avail_bytes < 0)
373 {
374 return s_num_avail_bytes;
375 }
376 if (s_num_avail_bytes)
377 {
378 get_bytes(s_byte_buff, s_num_avail_bytes);
379 }
380 }
381 b1 = *s_ptr_bytes++;
382 s_num_bits_left = 8;
383 --s_num_avail_bytes;
384 }
385
386 ret_code = static_cast<short>(b1 >> (8 - s_num_bits_left));
387 while (s_curr_size > s_num_bits_left)
388 {
389 if (s_num_avail_bytes <= 0)
390 {
391
392 // Out of bytes in current block, so read next block
393 s_ptr_bytes = s_byte_buff;
394 s_num_avail_bytes = static_cast<short>(get_byte());
395 if (s_num_avail_bytes < 0)
396 {
397 return s_num_avail_bytes;
398 }
399 if (s_num_avail_bytes)
400 {
401 get_bytes(s_byte_buff, s_num_avail_bytes);
402 }
403 }
404 b1 = *s_ptr_bytes++;
405 ret_code |= b1 << s_num_bits_left;
406 s_num_bits_left += 8;
407 --s_num_avail_bytes;
408 }
409 s_num_bits_left -= s_curr_size;
410 return static_cast<short>(ret_code & s_code_mask[s_curr_size]);
411}
