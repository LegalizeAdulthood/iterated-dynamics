307template <typename T, size_t N>
308struct JsonArray
309{
310 explicit JsonArray(const T (&value)[N])
311 {
312 for (size_t i = 0; i < N; ++i)
313 {
314 array[i] = value[i];
315 }
316 }
317
318 explicit JsonArray(const std::array<T, N> &value) :
319 array(value)
320 {
321 }
322
323 std::array<T, N> array{};
324};
325
326template <typename T, size_t N>
327std::ostream &operator<<(std::ostream &str, const JsonArray<T, N> &value)
328{
329 str << '[';
330 bool first{true};
331 for (T elem : value.array)
332 {
333 if (!first)
334 {
335 str << ',';
336 }
337 str << elem;
338 first = false;
339 }
340 str << ']';
341 return str;
342}

361struct DacBox
362{
363 explicit DacBox(const unsigned char dac[256][3])
364 {
365 for (int i = 0; i < 256; ++i)
366 {
367 value[i][0] = dac[i][0];
368 value[i][1] = dac[i][1];
369 value[i][2] = dac[i][2];
370 }
371 }
372
373 unsigned char value[256][3];
374};

376std::ostream &operator<<(std::ostream &str, const DacBox &value)
377{
378 str << '[';
379 for (int i = 0; i < 256; ++i)
380 {
381 if (i > 0)
382 {
383 str << ',';
384 }
385 str << '[' << static_cast<int>(value.value[i][0]) << ',' << static_cast<int>(value.value[i][1]) << ','
386 << static_cast<int>(value.value[i][2]) << ']';
387 }
388 str << ']';
389 return str;
390}
