33/* fill_dta
34 *
35 * Use data in s_find_data to fill in DTA.filename, DTA.attribute and DTA.path
36 */
37static void fill_dta()
38{
39 g_dta.path = s_search.it->path().string();
40 g_dta.filename = s_search.it->path().filename().string();
41 g_dta.attribute = is_directory(*s_search.it) ? SUB_DIR : 0;
42}

44static bool next_match()
45{
46 if (s_search.pos == DirPos::NONE)
47 {
48 g_dta.path = (s_search.it->path() / ".").string();
49 g_dta.filename = ".";
50 g_dta.attribute = SUB_DIR;
51 s_search.pos = DirPos::DOT;
52 return true;
53 }
54 if (s_search.pos == DirPos::DOT)
55 {
56 g_dta.path = (s_search.it->path() / "..").string();
57 g_dta.filename = "..";
58 g_dta.attribute = SUB_DIR;
59 s_search.pos = DirPos::DOT_DOT;
60 return true;
61 }
62 s_search.pos = DirPos::FILES;
63
64 while (s_search.it != fs::directory_iterator() && !s_search.path_matcher(*s_search.it))
65 {
66 ++s_search.it;
67 }
68 if (s_search.it == fs::directory_iterator())
69 {
70 s_search.pos = DirPos::NONE;
71 return false;
72 }
73
74 fill_dta();
75 return true;
76}

78bool fr_find_first(const char *path) // Find 1st file (or subdir) meeting path/filespec
79{
80 const fs::path search{path};
81 const fs::path search_dir{is_directory(search) ? search : search.has_parent_path() ? search.parent_path()
82 : "."};
83 std::error_code err;
84 s_search.it = fs::directory_iterator(search_dir, err);
85 if (err)
86 {
87 return false;
88 }
89
90 s_search.path_matcher = match_fn(path);
91 if (is_directory(search) || search.filename() == "*" || search.filename() == "*.*")
92 {
93 s_search.pos = DirPos::NONE;
94 }
95 else
96 {
97 s_search.pos = DirPos::FILES;
98 }
99 return next_match();
100}

102bool fr_find_next()
103{
104 if (s_search.pos == DirPos::FILES)
105 {
106 ++s_search.it;
107 }
108 return next_match();
