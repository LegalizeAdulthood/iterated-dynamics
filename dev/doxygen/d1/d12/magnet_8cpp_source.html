45// Z = ((Z**2 + C - 1)/(2Z + C - 2))**2
46int magnet1_orbit()
47{
48 // In "Beauty of Fractals", code by Kev Allen.
49 DComplex top;
50 DComplex bot;
51 DComplex tmp;
52
53 top.x = g_temp_sqr_x - g_temp_sqr_y + g_float_param->x - 1; // top = Z**2+C-1
54 top.y = g_old_z.x * g_old_z.y;
55 top.y = top.y + top.y + g_float_param->y;
56
57 bot.x = g_old_z.x + g_old_z.x + g_float_param->x - 2; // bot = 2*Z+C-2
58 bot.y = g_old_z.y + g_old_z.y + g_float_param->y;
59
60 const double div = bot.x * bot.x + bot.y * bot.y; // tmp = top/bot
61 if (div < std::numeric_limits<float>::min())
62 {
63 return 1;
64 }
65 tmp.x = (top.x*bot.x + top.y*bot.y)/div;
66 tmp.y = (top.y*bot.x - top.x*bot.y)/div;
67
68 g_new_z.x = (tmp.x + tmp.y) * (tmp.x - tmp.y); // Z = tmp**2
69 g_new_z.y = tmp.x * tmp.y;
70 g_new_z.y += g_new_z.y;
71
72 return g_bailout_float();
73}

75// Z = ((Z**3 + 3(C-1)Z + (C-1)(C-2) ) /
76// (3Z**2 + 3(C-2)Z + (C-1)(C-2)+1) )**2
77int magnet2_orbit()
78{
79 // In "Beauty of Fractals", code by Kev Allen.
80 DComplex top;
81 DComplex bot;
82 DComplex tmp;
83
84 top.x = g_old_z.x * (g_temp_sqr_x-g_temp_sqr_y-g_temp_sqr_y-g_temp_sqr_y + s_t_cm1.x)
85 - g_old_z.y * s_t_cm1.y + s_t_cm1_cm2.x;
86 top.y = g_old_z.y * (g_temp_sqr_x+g_temp_sqr_x+g_temp_sqr_x-g_temp_sqr_y + s_t_cm1.x)
87 + g_old_z.x * s_t_cm1.y + s_t_cm1_cm2.y;
88
89 bot.x = g_temp_sqr_x - g_temp_sqr_y;
90 bot.x = bot.x + bot.x + bot.x //
91 + g_old_z.x * s_t_cm2.x - g_old_z.y * s_t_cm2.y //
92 + s_t_cm1_cm2.x + 1.0;
93 bot.y = g_old_z.x * g_old_z.y;
94 bot.y += bot.y;
95 bot.y = bot.y + bot.y + bot.y //
96 + g_old_z.x * s_t_cm2.y + g_old_z.y * s_t_cm2.x //
97 + s_t_cm1_cm2.y;
98
99 const double div = bot.x * bot.x + bot.y * bot.y; // tmp = top/bot
100 if (div < std::numeric_limits<float>::min())
101 {
102 return 1;
103 }

105 tmp.y = (top.y*bot.x - top.x*bot.y)/div;
