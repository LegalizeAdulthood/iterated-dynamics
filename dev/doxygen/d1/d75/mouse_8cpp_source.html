36static int s_subscriber_id{};
37static std::map<int, MouseNotificationPtr> s_subscribers;
38static bool s_inside_notification{};
39static std::vector<int> s_pending_unsubscribe;
40
41int mouse_subscribe(MouseNotificationPtr subscriber)
42{
43 assert(!s_inside_notification);
44 const int result{s_subscriber_id++};
45 s_subscribers[result] = std::move(subscriber);
46 return result;
47}

49void mouse_unsubscribe(const int id)
50{
51 if (s_inside_notification)
52 {
53 s_pending_unsubscribe.push_back(id);
54 return;
55 }
56 const auto it = s_subscribers.find(id);
57 assert(it != s_subscribers.end());
58 s_subscribers.erase(id);
59}

61static void notify(const std::function<void(const MouseNotificationPtr &)> &handler)
62{
63 {
64 ValueSaver saved_inside_notification(s_inside_notification, true);
65 for (const auto &entry : s_subscribers)
66 {
67 handler(entry.second);
68 }
69 }
70 if (!s_inside_notification)
71 {
72 for (const int id : s_pending_unsubscribe)
73 {
74 mouse_unsubscribe(id);
75 }
76 }
77}
