28/* make_mig() takes a collection of individual GIF images (all
29 presumably the same resolution and all presumably generated
30 by this program and its "divide and conquer" algorithm) and builds
31 a single multiple-image GIF out of them. This routine is
32 invoked by the "batch=stitchmode/x/y" option, and is called
33 with the 'x' and 'y' parameters
34*/
35
36void make_mig(unsigned int x_mult, unsigned int y_mult)
37{
38 unsigned int x_res;
39 unsigned int y_res;
40 unsigned int x_tot;
41 unsigned int y_tot;
42 unsigned int x_loc;
43 unsigned int y_loc;
44 unsigned int i;
45 std::string gif_in;
46 unsigned char *temp;
47
48 int error_flag = 0; // no errors so
49 int input_error_flag = 0;
50 unsigned int all_i_tbl = 0;
51 unsigned int all_y_res = all_i_tbl;
52 unsigned int all_x_res = all_y_res;
53 std::FILE *in = nullptr;
54 std::FILE *out = in;
55
56 std::string gif_out{"fractmig.gif"};
57
58 temp = &g_old_dac_box[0][0]; // a safe place for our temp data
59
60 // process each input image, one at a time
61 for (unsigned y_step = 0U; y_step < y_mult; y_step++)
62 {
63 for (unsigned x_step = 0U; x_step < x_mult; x_step++)
64 {
65 if (x_step == 0 && y_step == 0) // first time through?
66 {
67 std::printf("\n"
68 " Generating multi-image GIF file %s using %u X and %u Y components\n"
69 "\n",
70 gif_out.c_str(), x_mult, y_mult);
71 // attempt to create the output file
72 const std::filesystem::path path{get_save_path(WriteFile::IMAGE, gif_out)};
73 assert(!path.empty());
74 out = std::fopen(path.string().c_str(), "wb");
75 if (out == nullptr)
76 {
77 std::printf("Cannot create output file %s!\n", gif_out.c_str());
78 std::exit(1);
79 }
80 }
81
82 gif_in = fmt::format("frmig_{:c}{:c}.gif", par_key(x_step), par_key(y_step));
83 const std::filesystem::path path{find_file(ReadFile::IMAGE, gif_in)};
84 if (path.empty())
85 {
86 fmt::print("Can't locate file {:s}!\n", gif_in);
87 std::exit(1);
88 }
89 in = std::fopen(path.string().c_str(), "rb");
90 if (in == nullptr)
91 {
92 std::printf("Can't open file %s!\n", gif_in.c_str());
93 std::exit(1);
94 }
95
96 // (read, but only copy this if it's the first time through)
97 if (std::fread(temp, 13, 1, in) != 1) // read the header and LDS
98 {
99 input_error_flag = 1;
100 }
101 std::memcpy(&x_res, &temp[6], 2); // X-resolution
102 std::memcpy(&y_res, &temp[8], 2); // Y-resolution
103
104 if (x_step == 0 && y_step == 0) // first time through?
105 {
106 all_x_res = x_res; // save the "master" resolution
107 all_y_res = y_res;
108 x_tot = x_res * x_mult; // adjust the image size
109 y_tot = y_res * y_mult;
110 std::memcpy(&temp[6], &x_tot, 2);
111 std::memcpy(&temp[8], &y_tot, 2);
112 temp[12] = 0; // reserved
113 if (std::fwrite(temp, 13, 1, out) != 1) // write out the header
114 {
115 error_flag = 1;
116 }
117 } // end of first-time-through
118
119 unsigned char i_char = static_cast<char>(temp[10] & 0x07); // find the color table size
120 unsigned int i_tbl = 1 << ++i_char;
121 i_char = static_cast<char>(temp[10] & 0x80); // is there a global color table?
122 if (x_step == 0 && y_step == 0) // first time through?
123 {
124 all_i_tbl = i_tbl; // save the color table size
125 }
126 if (i_char != 0) // yup
127 {
128 // (read, but only copy this if it's the first time through)
129 if (std::fread(temp, 3*i_tbl, 1, in) != 1) // read the global color table
130 {
131 input_error_flag = 2;
132 }
133 if (x_step == 0 && y_step == 0) // first time through?
134 {
135 if (std::fwrite(temp, 3*i_tbl, 1, out) != 1) // write out the GCT
136 {
137 error_flag = 2;
138 }
139 }
140 }
141
142 if (x_res != all_x_res || y_res != all_y_res || i_tbl != all_i_tbl)
143 {
144 // Oops - our pieces don't match
145 std::printf("File %s doesn't have the same resolution as its predecessors!\n", gif_in.c_str());
146 std::exit(1);
147 }
148
149 while (true) // process each information block
150 {
151 std::memset(temp, 0, 10);
152 if (std::fread(temp, 1, 1, in) != 1) // read the block identifier
153 {
154 input_error_flag = 3;
155 }
156
157 if (temp[0] == 0x2c) // image descriptor block
158 {
159 if (std::fread(&temp[1], 9, 1, in) != 1) // read the Image Descriptor
160 {
161 input_error_flag = 4;
162 }
163 std::memcpy(&x_loc, &temp[1], 2); // X-location
164 std::memcpy(&y_loc, &temp[3], 2); // Y-location
165 x_loc += x_step * x_res; // adjust the locations
166 y_loc += y_step * y_res;
167 std::memcpy(&temp[1], &x_loc, 2);
168 std::memcpy(&temp[3], &y_loc, 2);
169 if (std::fwrite(temp, 10, 1, out) != 1) // write out the Image Descriptor
170 {
171 error_flag = 4;
172 }
173
174 i_char = static_cast<char>(temp[9] & 0x80); // is there a local color table?
175 if (i_char != 0) // yup
176 {
177 if (std::fread(temp, 3*i_tbl, 1, in) != 1) // read the local color table
178 {
179 input_error_flag = 5;
180 }
181 if (std::fwrite(temp, 3*i_tbl, 1, out) != 1) // write out the LCT
182 {
183 error_flag = 5;
184 }
185 }
186
187 if (std::fread(temp, 1, 1, in) != 1) // LZH table size
188 {
189 input_error_flag = 6;
190 }
191 if (std::fwrite(temp, 1, 1, out) != 1)
192 {
193 error_flag = 6;
194 }
195 while (true)
196 {
197 if (error_flag != 0 || input_error_flag != 0) // oops - did something go wrong?
198 {
199 break;
200 }
201 if (std::fread(temp, 1, 1, in) != 1) // block size
202 {
203 input_error_flag = 7;
204 }
205 if (std::fwrite(temp, 1, 1, out) != 1)
206 {
207 error_flag = 7;
208 }
209 i = temp[0];
210 if (i == 0)
211 {
212 break;
213 }
214 if (std::fread(temp, i, 1, in) != 1) // LZH data block
215 {
216 input_error_flag = 8;
217 }
218 if (std::fwrite(temp, i, 1, out) != 1)
219 {
220 error_flag = 8;
221 }
222 }
223 }
224
225 if (temp[0] == 0x21) // extension block
226 {
227 // (read, but only copy this if it's the last time through)
228 if (std::fread(&temp[2], 1, 1, in) != 1) // read the block type
229 {
230 input_error_flag = 9;
231 }
232 if (x_step == x_mult-1 && y_step == y_mult-1)
233 {
234 if (std::fwrite(temp, 2, 1, out) != 1)
235 {
236 error_flag = 9;
237 }
238 }
239 while (true)
240 {
241 if (error_flag != 0 || input_error_flag != 0) // oops - did something go wrong?
242 {
243 break;
244 }
245 if (std::fread(temp, 1, 1, in) != 1) // block size
246 {
247 input_error_flag = 10;
248 }
249 if (x_step == x_mult-1 && y_step == y_mult-1)
250 {
251 if (std::fwrite(temp, 1, 1, out) != 1)
252 {
253 error_flag = 10;
254 }
255 }
256 i = temp[0];
257 if (i == 0)
258 {
259 break;
260 }
261 if (std::fread(temp, i, 1, in) != 1) // data block
262 {
263 input_error_flag = 11;
264 }
265 if (x_step == x_mult-1 && y_step == y_mult-1)
266 {
267 if (std::fwrite(temp, i, 1, out) != 1)
268 {
269 error_flag = 11;
270 }
271 }
272 }
273 }
274
275 if (temp[0] == 0x3b) // end-of-stream indicator
276 {
277 break; // done with this file
278 }
279
280 if (error_flag != 0 || input_error_flag != 0) // oops - did something go wrong?
281 {
282 break;
283 }
284 }
285 std::fclose(in); // done with an input GIF
286
287 if (error_flag != 0 || input_error_flag != 0) // oops - did something go wrong?
288 {
289 break;
290 }
291 }
292
293 if (error_flag != 0 || input_error_flag != 0) // oops - did something go wrong?
294 {
295 break;
296 }
297 }
298
299 temp[0] = 0x3b; // end-of-stream indicator
300 if (std::fwrite(temp, 1, 1, out) != 1)
301 {
302 error_flag = 12;
303 }
304 std::fclose(out); // done with the output GIF
305
306 if (input_error_flag != 0) // uh-oh - something failed
307 {
308 std::printf("\007 Process failed = early EOF on input file %s\n", gif_in.c_str());
309 }
310
311 if (error_flag != 0) // uh-oh - something failed
312 {
313 std::printf("\007 Process failed = out of disk space?\n");
314 }
315
316 // now delete each input image, one at a time
317 if (error_flag == 0 && input_error_flag == 0)
318 {
319 for (unsigned y_step = 0U; y_step < y_mult; y_step++)
320 {
321 for (unsigned x_step = 0U; x_step < x_mult; x_step++)
322 {
323 gif_in = fmt::format("frmig_{:c}{:c}.gif", par_key(x_step), par_key(y_step));
324 std::filesystem::remove(gif_in);
325 }
326 }
327 }
328
329 // tell the world we're done
330 if (error_flag == 0 && input_error_flag == 0)
331 {
332 std::printf("File %s has been created (and its component files deleted)\n", gif_out.c_str());
333 }
334}
