183static void validate_canvas_size(const MigMetadata &metadata, const unsigned int x_mult, const unsigned int y_mult)
184{
185 constexpr std::uint64_t MAX_GIF_DIMENSION{std::numeric_limits<std::uint16_t>::max()};
186 const std::uint64_t width = static_cast<std::uint64_t>(metadata.width) * x_mult;
187 const std::uint64_t height = static_cast<std::uint64_t>(metadata.height) * y_mult;
188 if (width > MAX_GIF_DIMENSION || height > MAX_GIF_DIMENSION)
189 {
190 fmt::print("MIG output dimensions {:d} x {:d} exceed GIF limits!\n", width, height);
191 std::exit(1);
192 }

238static bool same_color_map(const ColorMapObject *lhs, const ColorMapObject *rhs)
239{
240 if (lhs == nullptr || rhs == nullptr)
241 {
242 return lhs == rhs;
243 }
244 return lhs->ColorCount == rhs->ColorCount &&
245 std::memcmp(lhs->Colors, rhs->Colors, lhs->ColorCount * sizeof(GifColorType)) == 0;
246}

248static bool same_extension_blocks(
249 const int lhs_count,
250 const ExtensionBlock *lhs_blocks,
251 const int rhs_count,
252 const ExtensionBlock *rhs_blocks)
253{
254 if (lhs_count != rhs_count)
255 {
256 return false;
257 }
258 for (int i = 0; i < lhs_count; i++)
259 {
260 if (lhs_blocks[i].Function != rhs_blocks[i].Function ||
261 lhs_blocks[i].ByteCount != rhs_blocks[i].ByteCount ||
262 std::memcmp(lhs_blocks[i].Bytes, rhs_blocks[i].Bytes, lhs_blocks[i].ByteCount) != 0)
263 {
264 return false;
265 }
266 }
267 return true;
268}

270static bool same_image(const SavedImage &lhs, const SavedImage &rhs)
271{
272 return lhs.ImageDesc.Left == rhs.ImageDesc.Left && lhs.ImageDesc.Top == rhs.ImageDesc.Top &&
273 lhs.ImageDesc.Width == rhs.ImageDesc.Width && lhs.ImageDesc.Height == rhs.ImageDesc.Height &&
274 lhs.ImageDesc.Interlace == rhs.ImageDesc.Interlace &&
275 same_color_map(lhs.ImageDesc.ColorMap, rhs.ImageDesc.ColorMap) &&
276 same_extension_blocks(
277 lhs.ExtensionBlockCount, lhs.ExtensionBlocks, rhs.ExtensionBlockCount, rhs.ExtensionBlocks) &&
278 std::memcmp(lhs.RasterBits, rhs.RasterBits, raster_size(lhs)) == 0;
279}

281static bool same_image_at_offset(const SavedImage &lhs, const SavedImage &rhs, const int left, const int top)
282{
283 return lhs.ImageDesc.Left == rhs.ImageDesc.Left + left && lhs.ImageDesc.Top == rhs.ImageDesc.Top + top &&
284 lhs.ImageDesc.Width == rhs.ImageDesc.Width && lhs.ImageDesc.Height == rhs.ImageDesc.Height &&
285 lhs.ImageDesc.Interlace == rhs.ImageDesc.Interlace &&
286 same_color_map(lhs.ImageDesc.ColorMap, rhs.ImageDesc.ColorMap) &&
287 same_extension_blocks(
288 lhs.ExtensionBlockCount, lhs.ExtensionBlocks, rhs.ExtensionBlockCount, rhs.ExtensionBlocks) &&
289 std::memcmp(lhs.RasterBits, rhs.RasterBits, raster_size(lhs)) == 0;
290}

345static void verify_single_image_output(
346 const std::filesystem::path &path,
347 const GifFileType *first_input,
348 const unsigned int x_mult,
349 const unsigned int y_mult)
350{
351 InputGif output{read_input_gif(path)};
352 if (output.get() == nullptr || output.slurp_result() != GIF_OK)
353 {
354 std::printf("Cannot read output file %s!\n", path.filename().string().c_str());
355 std::exit(1);
356 }
357 const MigMetadata input_metadata{get_metadata(first_input)};
358 const MigMetadata output_metadata{get_metadata(output.get())};
359 if (output_metadata.width != input_metadata.width * static_cast<int>(x_mult) ||
360 output_metadata.height != input_metadata.height * static_cast<int>(y_mult) ||
361 output_metadata.color_resolution != input_metadata.color_resolution ||
362 output_metadata.background_color != input_metadata.background_color ||
363 output_metadata.aspect_byte != input_metadata.aspect_byte ||
364 !same_color_map(output.get()->SColorMap, first_input->SColorMap) || output.get()->ImageCount != 1 ||
365 !same_image(output.get()->SavedImages[0], first_input->SavedImages[0]))
366 {
367 std::printf("Output file %s failed GIF verification!\n", path.filename().string().c_str());
368 std::exit(1);
369 }
370}

411static int total_tile_image_count(const unsigned int x_mult, const unsigned int y_mult)
412{
413 int image_count{};
414 for (unsigned y_step = 0U; y_step < y_mult; y_step++)

425static void verify_tiled_images_output(
426 const std::filesystem::path &path,
427 const GifFileType *first_input,
428 const unsigned int x_mult,
429 const unsigned int y_mult)
430{
431 InputGif output{read_input_gif(path)};
432 if (output.get() == nullptr || output.slurp_result() != GIF_OK)
433 {
434 std::printf("Cannot read output file %s!\n", path.filename().string().c_str());
435 std::exit(1);
436 }
437 const MigMetadata input_metadata{get_metadata(first_input)};
438 const MigMetadata output_metadata{get_metadata(output.get())};
439 if (output_metadata.width != input_metadata.width * static_cast<int>(x_mult) ||
440 output_metadata.height != input_metadata.height * static_cast<int>(y_mult) ||
441 !same_color_map(output.get()->SColorMap, first_input->SColorMap) ||
442 output.get()->ImageCount != total_tile_image_count(x_mult, y_mult))
443 {
444 std::printf("Output file %s failed GIF layout verification!\n", path.filename().string().c_str());
445 std::exit(1);
446 }
447 int image_number{};
448 for (unsigned y_step = 0U; y_step < y_mult; y_step++)
449 {
450 for (unsigned x_step = 0U; x_step < x_mult; x_step++)
451 {
452 InputGif input{read_tile_gif(x_step, y_step)};
453 const bool final_tile{x_step == x_mult - 1 && y_step == y_mult - 1};
454 const int left{static_cast<int>(x_step) * input_metadata.width};
455 const int top{static_cast<int>(y_step) * input_metadata.height};
456 for (int i = 0; i < input.get()->ImageCount; i++)
457 {
458 SavedImage expected{input.get()->SavedImages[i]};
459 if (!final_tile)
460 {
461 expected.ExtensionBlockCount = 0;
462 expected.ExtensionBlocks = nullptr;
463 }
464 if (!same_image_at_offset(output.get()->SavedImages[image_number++], expected, left, top))
465 {
466 std::printf("Output file %s failed GIF offset verification!\n", path.filename().string().c_str());
467 std::exit(1);
468 }
469 }
470 }
471 }
472 InputGif final_input{read_tile_gif(x_mult - 1, y_mult - 1)};
473 if (!same_extension_blocks(
474 output.get()->ExtensionBlockCount,
475 output.get()->ExtensionBlocks,
476 final_input.get()->ExtensionBlockCount,
477 final_input.get()->ExtensionBlocks))
478 {
479 std::printf("Output file %s failed GIF extension verification!\n", path.filename().string().c_str());
480 std::exit(1);
481 }
482}

484static void write_tiled_images_gif(
485 const std::filesystem::path &path, const unsigned int x_mult, const unsigned int y_mult)
486{
487 InputGif first_input{read_tile_gif(0, 0)};
488 const MigMetadata input_metadata{get_metadata(first_input.get())};
489 OutputGif output{create_output_gif(path, first_input.get(), x_mult, y_mult)};
490 for (unsigned y_step = 0U; y_step < y_mult; y_step++)
491 {
492 for (unsigned x_step = 0U; x_step < x_mult; x_step++)
493 {
494 InputGif input{read_tile_gif(x_step, y_step)};
495 const bool final_tile{x_step == x_mult - 1 && y_step == y_mult - 1};
496 const int left{static_cast<int>(x_step) * input_metadata.width};
497 const int top{static_cast<int>(y_step) * input_metadata.height};
498 for (int i = 0; i < input.get()->ImageCount; i++)
499 {
500 copy_saved_image_at_offset(output.get(), input.get()->SavedImages[i], left, top, final_tile);
501 }
502 if (final_tile)
503 {
504 copy_extension_blocks(output.get(), input.get());
505 }
506 }
507 }
508 if (output.spew() != GIF_OK)
509 {
510 std::printf("Cannot write output file %s!\n", path.filename().string().c_str());
511 std::exit(1);
512 }
513 verify_tiled_images_output(path, first_input.get(), x_mult, y_mult);
514}

516static MigMetadata validate_input_gifs(const unsigned int x_mult, const unsigned int y_mult)
517{
518 MigMetadata first{};
519 bool have_first{false};
520 for (unsigned y_step = 0U; y_step < y_mult; y_step++)
521 {
522 for (unsigned x_step = 0U; x_step < x_mult; x_step++)
523 {
524 const std::string gif_in{input_gif_name(x_step, y_step)};
525 const std::filesystem::path path{find_file(ReadFile::IMAGE, gif_in)};
526 if (path.empty())
527 {
528 fmt::print("Can't locate file {:s}!\n", gif_in);
529 std::exit(1);
530 }
531 InputGif gif{read_input_gif(path)};
532 if (gif.get() == nullptr)
533 {
534 std::printf("Can't open file %s!\n", gif_in.c_str());
535 std::exit(1);
536 }
537 if (gif.slurp_result() != GIF_OK)
538 {
539 std::printf("\007 Process failed = early EOF on input file %s\n", gif_in.c_str());
540 std::exit(1);
541 }
542 const MigMetadata metadata{get_metadata(gif.get())};
543 if (!have_first)
544 {
545 first = metadata;
546 validate_canvas_size(first, x_mult, y_mult);
547 have_first = true;
548 }
549 if (metadata.width != first.width || metadata.height != first.height ||
550 metadata.color_count != first.color_count)
551 {
552 std::printf("File %s doesn't have the same resolution as its predecessors!\n", gif_in.c_str());
553 std::exit(1);
554 }
555 }
556 }
557 return first;
558}

571/* make_mig() takes a collection of individual GIF images (all
572 presumably the same resolution and all presumably generated
573 by this program and its "divide and conquer" algorithm) and builds
574 a single multiple-image GIF out of them. This routine is
575 invoked by the "batch=stitchmode/x/y" option, and is called
576 with the 'x' and 'y' parameters
577*/
578
579void make_mig(unsigned int x_mult, unsigned int y_mult)
580{
581 validate_input_gifs(x_mult, y_mult);
582
583 std::string gif_out{"fractmig.gif"};
584 std::printf("\n"
585 " Generating multi-image GIF file %s using %u X and %u Y components\n"
586 "\n",
587 gif_out.c_str(), x_mult, y_mult);
588 const std::filesystem::path path{get_checked_save_path(WriteFile::IMAGE, gif_out)};
589 assert(!path.empty());
590
591 write_tiled_images_gif(path, x_mult, y_mult);
592 delete_tile_gifs(x_mult, y_mult);
593 std::printf("File %s has been created (and its component files deleted)\n", gif_out.c_str());
594}
