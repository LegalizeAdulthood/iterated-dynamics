23int popcorn_type()

28 start_resume();

33 g_plot = no_plot;
34 g_temp_sqr_x = 0;
35 for (g_row = start_row; g_row <= g_i_stop_pt.y; g_row++)

45 }
46 g_reset_periodicity = false;
47 }
48 }
49 g_calc_status = CalcStatus::COMPLETED;
50 return 0;
51}

53int popcorn_fractal()
54{
55 g_tmp_z = g_old_z;
56 g_tmp_z.x *= 3.0;
57 g_tmp_z.y *= 3.0;
58 sin_cos(g_tmp_z.x, g_sin_x, g_cos_x);
59 double sin_y;
60 double cos_y;
61 sin_cos(g_tmp_z.y, sin_y, cos_y);
62 g_tmp_z.x = g_sin_x/g_cos_x + g_old_z.x;
63 g_tmp_z.y = sin_y/cos_y + g_old_z.y;
64 sin_cos(g_tmp_z.x, g_sin_x, g_cos_x);
65 sin_cos(g_tmp_z.y, sin_y, cos_y);
66 g_new_z.x = g_old_z.x - g_param_z1.x*sin_y;
67 g_new_z.y = g_old_z.y - g_param_z1.x*g_sin_x;
68 if (g_plot == no_plot)
69 {
70 plot_orbit(g_new_z.x, g_new_z.y, 1+g_row%g_colors);
71 g_old_z = g_new_z;
72 }
73 g_temp_sqr_x = sqr(g_new_z.x);
74 g_temp_sqr_y = sqr(g_new_z.y);
75 g_magnitude = g_temp_sqr_x + g_temp_sqr_y;
76 if (g_magnitude >= g_magnitude_limit
77 || std::abs(g_new_z.x) > g_magnitude_limit2
78 || std::abs(g_new_z.y) > g_magnitude_limit2)
79 {
80 return 1;
81 }
82 g_old_z = g_new_z;
83 return 0;
84}

88int popcorn_orbit()
89{
90 DComplex tmp_x;
91 DComplex tmp_y;
92
93 // tmpx contains the generalized value of the old real "x" equation
94 g_tmp_z = g_param_z2 * g_old_z.y; // tmp = (C * old.y)
95 cmplx_trig1(g_tmp_z, tmp_x); // tmpx = trig1(tmp)
96 tmp_x.x += g_old_z.y; // tmpx = old.y + trig1(tmp)
97 cmplx_trig0(tmp_x, g_tmp_z); // tmp = trig0(tmpx)
98 cmplx_mult(g_tmp_z, g_param_z1, tmp_x); // tmpx = tmp * h
99
100 // tmpy contains the generalized value of the old real "y" equation
101 g_tmp_z = g_param_z2 * g_old_z.x; // tmp = (C * old.x)
102 cmplx_trig3(g_tmp_z, tmp_y); // tmpy = trig3(tmp)
103 tmp_y.x += g_old_z.x; // tmpy = old.x + trig1(tmp)
104 cmplx_trig2(tmp_y, g_tmp_z); // tmp = trig2(tmpy)
105 cmplx_mult(g_tmp_z, g_param_z1, tmp_y); // tmpy = tmp * h
106
107 g_new_z.x = g_old_z.x - tmp_x.x - tmp_y.y;
108 g_new_z.y = g_old_z.y - tmp_y.x - tmp_x.y;
109
110 if (g_plot == no_plot)
111 {
112 plot_orbit(g_new_z.x, g_new_z.y, 1+g_row%g_colors);
113 g_old_z = g_new_z;
114 }
115
116 g_temp_sqr_x = sqr(g_new_z.x);
117 g_temp_sqr_y = sqr(g_new_z.y);
118 g_magnitude = g_temp_sqr_x + g_temp_sqr_y;
119 if (g_magnitude >= g_magnitude_limit
120 || std::abs(g_new_z.x) > g_magnitude_limit2
121 || std::abs(g_new_z.y) > g_magnitude_limit2)
122 {
123 return 1;
124 }
125 g_old_z = g_new_z;
126 return 0;
127}
