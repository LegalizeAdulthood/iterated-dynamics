28/* Showing orbit requires converting real co-ords to screen co-ords.
29 Define:
30 Xs == xxmax-xx3rd Ys == yy3rd-yymax
31 W == xdots-1 D == ydots-1
32 We know that:
33 realx == lx0[col] + lx1[row]
34 realy == ly0[row] + ly1[col]
35 lx0[col] == (col/width) * Xs + xxmin
36 lx1[row] == row * delxx
37 ly0[row] == (row/D) * Ys + yymax
38 ly1[col] == col * (0-delyy)
39 so:
40 realx == (col/W) * Xs + xxmin + row * delxx
41 realy == (row/D) * Ys + yymax + col * (0-delyy)
42 and therefore:
43 row == (realx-xxmin - (col/W)*Xs) / Xv (1)
44 col == (realy-yymax - (row/D)*Ys) / Yv (2)
45 substitute (2) into (1) and solve for row:
46 row == ((realx-xxmin)*(0-delyy2)*W*D - (realy-yymax)*Xs*D)
47 / ((0-delyy2)*W*delxx2*D-Ys*Xs)
48 */

116void OrbitPlot::plot_d_orbit(const double dx, const double dy)
117{
118 int i = static_cast<int>(dy * g_plot_mx1 - dx * g_plot_mx2);
119 i += g_logical_screen.x_offset;
120 if (i < 0 || i >= g_screen_x_dots)
121 {
122 return;
123 }
124 int j = static_cast<int>(dx * g_plot_my1 - dy * g_plot_my2);
125 j += g_logical_screen.y_offset;
126 if (j < 0 || j >= g_screen_y_dots)
127 {
128 return;
129 }
130 {
131 ValueSaver save_screen_x_offset{g_logical_screen.x_offset, 0};
132 ValueSaver save_screen_y_offset{g_logical_screen.y_offset, 0};
133 switch (m_mode)
134 {
135 case PlotMode::IMAGE:
136 plot_image_point(i, j);
137 break;
138
139 case PlotMode::OVERLAY:
140 plot_overlay_point(i, j);
141 break;
142 }
143 }
144 update_orbit_sound(i, j);
145}

159void OrbitPlot::update_orbit_sound(const int x, const int y)
160{
161 if (g_debug_flag == DebugFlags::FORCE_SCALED_SOUND_FORMULA)
162 {
163 if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_X) // sound = x
164 {
165 write_sound(x * 1000 / g_logical_screen.x_dots + g_base_hertz);
166 }
167 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) > SOUNDFLAG_X) // sound = y or z
168 {
169 write_sound(y * 1000 / g_logical_screen.y_dots + g_base_hertz);
170 }
171 else if (g_orbit_delay > 0)
172 {
173 m_delay_pending = true;
174 }
175 }
176 else
177 {
178 if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_X) // sound = x
179 {
180 write_sound(x + g_base_hertz);
181 }
182 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_Y) // sound = y
183 {
184 write_sound(y + g_base_hertz);
185 }
186 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_Z) // sound = z
187 {
188 write_sound(x + y + g_base_hertz);
189 }
190 else if (g_orbit_delay > 0)
191 {
192 m_delay_pending = true;
193 }
194 }
195
196 // placing sleepms here delays each dot
197}

245void OrbitPlot::scrub()
246{
247 driver_mute();
248 if (!g_orbit_save_flag || m_saved_overlay.empty())
249 {
250 m_saved_overlay.clear();
251 g_orbit_save_flag = false;
252 return;
253 }
254 ValueSaver save_screen_x_offset{g_logical_screen.x_offset, 0};
255 ValueSaver save_screen_y_offset{g_logical_screen.y_offset, 0};
256 while (!m_saved_overlay.empty())
257 {
258 const SavedOverlayPoint point{m_saved_overlay.back()};
259 m_saved_overlay.pop_back();
260 g_put_color(point.x, point.y, point.color);
261 update_saved_overlay_count();
262 }
263}
