34/* Showing orbit requires converting real co-ords to screen co-ords.
35 Define:
36 Xs == xxmax-xx3rd Ys == yy3rd-yymax
37 W == xdots-1 D == ydots-1
38 We know that:
39 realx == lx0[col] + lx1[row]
40 realy == ly0[row] + ly1[col]
41 lx0[col] == (col/width) * Xs + xxmin
42 lx1[row] == row * delxx
43 ly0[row] == (row/D) * Ys + yymax
44 ly1[col] == col * (0-delyy)
45 so:
46 realx == (col/W) * Xs + xxmin + row * delxx
47 realy == (row/D) * Ys + yymax + col * (0-delyy)
48 and therefore:
49 row == (realx-xxmin - (col/W)*Xs) / Xv (1)
50 col == (realy-yymax - (row/D)*Ys) / Yv (2)
51 substitute (2) into (1) and solve for row:
52 row == ((realx-xxmin)*(0-delyy2)*W*D - (realy-yymax)*Xs*D)
53 / ((0-delyy2)*W*delxx2*D-Ys*Xs)
54 */
55
56static void plot_d_orbit(const double dx, const double dy, const int color)
57{
58 if (g_orbit_save_index >= NUM_SAVE_ORBIT-3)
59 {
60 return;
61 }
62 int i = static_cast<int>(dy * g_plot_mx1 - dx * g_plot_mx2);
63 i += g_logical_screen.x_offset;
64 if (i < 0 || i >= g_screen_x_dots)
65 {
66 return;
67 }
68 int j = static_cast<int>(dx * g_plot_my1 - dy * g_plot_my2);
69 j += g_logical_screen.y_offset;
70 if (j < 0 || j >= g_screen_y_dots)
71 {
72 return;
73 }
74 int save_screen_x_offset = g_logical_screen.x_offset;
75 int save_screen_y_offset = g_logical_screen.y_offset;
76 g_logical_screen.y_offset = 0;
77 g_logical_screen.x_offset = 0;
78 // save orbit value
79 if (color == -1)
80 {
81 s_save_orbit[g_orbit_save_index++] = i;
82 s_save_orbit[g_orbit_save_index++] = j;
83 const int c = get_color(i, j);
84 s_save_orbit[g_orbit_save_index++] = c;
85 g_put_color(i, j, c^g_orbit_color);
86 }
87 else
88 {
89 g_put_color(i, j, color);
90 }
91 g_logical_screen.x_offset = save_screen_x_offset;
92 g_logical_screen.y_offset = save_screen_y_offset;
93 if (g_debug_flag == DebugFlags::FORCE_SCALED_SOUND_FORMULA)
94 {
95 if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_X) // sound = x
96 {
97 write_sound(i * 1000 / g_logical_screen.x_dots + g_base_hertz);
98 }
99 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) > SOUNDFLAG_X) // sound = y or z
100 {
101 write_sound(j * 1000 / g_logical_screen.y_dots + g_base_hertz);
102 }
103 else if (g_orbit_delay > 0)
104 {
105 wait_until(g_orbit_delay);
106 }
107 }
108 else
109 {
110 if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_X) // sound = x
111 {
112 write_sound(i + g_base_hertz);
113 }
114 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_Y) // sound = y
115 {
116 write_sound(j + g_base_hertz);
117 }
118 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) == SOUNDFLAG_Z) // sound = z
119 {
120 write_sound(i + j + g_base_hertz);
121 }
122 else if (g_orbit_delay > 0)
123 {
124 wait_until(g_orbit_delay);
125 }
126 }
127
128 // placing sleepms here delays each dot
129}
