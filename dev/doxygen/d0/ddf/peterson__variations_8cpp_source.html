25bool marks_julia_per_image()
26{
27 g_params[2] = std::max(g_params[2], 1.0);
28 g_c_exponent = static_cast<int>(g_params[2]);
29 g_float_param = &g_param_z1;
30 g_old_z = g_param_z1;
31 if (g_c_exponent > 3)
32 {
33 pow(&g_old_z, g_c_exponent-1, &g_marks_coefficient);
34 }
35 else if (g_c_exponent == 3)
36 {
37 g_marks_coefficient.x = sqr(g_old_z.x) - sqr(g_old_z.y);
38 g_marks_coefficient.y = g_old_z.x * g_old_z.y * 2;
39 }
40 else if (g_c_exponent == 2)
41 {
42 g_marks_coefficient = g_old_z;
43 }
44 else if (g_c_exponent < 2)
45 {
46 g_marks_coefficient.x = 1.0;
47 g_marks_coefficient.y = 0.0;
48 }
49 get_julia_attractor(0.0, 0.0); // an attractor?
50 return true;
51}

53int marks_lambda_orbit()
54{
55 // Mark Peterson's variation of "lambda" function
56
57 // Z1 = (C^(exp-1) * Z**2) + C
58 g_tmp_z.x = g_temp_sqr_x - g_temp_sqr_y;
59 g_tmp_z.y = g_old_z.x * g_old_z.y *2;
60
61 g_new_z.x = g_marks_coefficient.x * g_tmp_z.x - g_marks_coefficient.y * g_tmp_z.y + g_float_param->x;
62 g_new_z.y = g_marks_coefficient.x * g_tmp_z.y + g_marks_coefficient.y * g_tmp_z.x + g_float_param->y;
63
64 return g_bailout_float();
65}
