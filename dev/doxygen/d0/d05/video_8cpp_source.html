33// read_span(int row, int startcol, int stopcol, Byte *pixels)
34//
35// This routine is a 'span' analog of 'getcolor()', and gets a horizontal
36// span of pixels from the screen and stores it in pixels[] at one byte per
37// pixel
38//
39void read_span(const int row, const int start_col, const int stop_col, Byte *pixels)
40{
41 if (start_col + g_logical_screen.x_offset >= g_screen_x_dots || row + g_logical_screen.y_offset >= g_screen_y_dots)
42 {
43 return;
44 }
45 assert(s_read_span);
46 s_read_span(row + g_logical_screen.y_offset, start_col + g_logical_screen.x_offset, stop_col + g_logical_screen.x_offset, pixels);

118// Return the color on the screen at the (xdot, ydot) point
119//
120int get_color(const int x, const int y)
121{
122 const int x1 = x + g_logical_screen.x_offset;
123 const int y1 = y + g_logical_screen.y_offset;
124 if (x1 < 0 || y1 < 0 || x1 >= g_screen_x_dots || y1 >= g_screen_y_dots)
125 {
126 // this can happen in boundary trace
127 return 0;
128 }
129 assert(s_read_pixel);
130 return s_read_pixel(x1, y1);
131}

133// write the color on the screen at the (xdot, ydot) point
134//
135void put_color_a(const int x, const int y, const int color)

139 assert(x1 >= 0 && x1 <= g_screen_x_dots);
140 assert(y1 >= 0 && y1 <= g_screen_y_dots);
141 assert(s_write_pixel);

145// This routine is a 'line' analog of 'putcolor()', and sends an
146// entire line of pixels to the screen (0 <= xdot < xdots) at a clip
147// Called by the GIF decoder
148//
149int out_line(Byte *pixels, const int line_len)
150{
151 driver_check_memory();
152 if (g_row_count + g_logical_screen.y_offset >= g_screen_y_dots)
153 {
154 return 0;
155 }
156 assert(s_write_span);
157 s_write_span(g_row_count + g_logical_screen.y_offset, g_logical_screen.x_offset, line_len + g_logical_screen.x_offset - 1, pixels);
158 g_row_count++;
159 return 0;
160}
