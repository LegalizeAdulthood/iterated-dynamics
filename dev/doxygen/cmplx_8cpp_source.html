12DComplex complex_sqrt_float(const double x, const double y)
13{
14 DComplex result;
15
16 if (x == 0.0 && y == 0.0)
17 {
18 result.x = 0.0;
19 result.y = 0.0;
20 }
21 else
22 {
23 const double mag = std::sqrt(std::sqrt(x*x + y*y));
24 const double theta = std::atan2(y, x) / 2;
25 sin_cos(theta, &result.y, &result.x);
26 result.x *= mag;
27 result.y *= mag;
28 }
29 return result;
30}

32DComplex complex_power(const DComplex xx, const DComplex yy)
33{
34 DComplex z;
35 DComplex c_log;
36 DComplex t;
37
38 if (xx.x == 0 && xx.y == 0)
39 {
40 z.y = 0.0;
41 z.x = z.y;
42 return z;
43 }
44
45 fpu_cmplx_log(&xx, &c_log);
46 fpu_cmplx_mul(&c_log, &yy, &t);
47 fpu_cmplx_exp(&t, &z);
48 return z;
49}

57// rz=Arcsin(z)=-i*Log{i*z+sqrt(1-z*z)}
58void asin_z(const DComplex z, DComplex *rz)
59{
60 DComplex temp_z1;
61 DComplex temp_z2;
62
63 fpu_cmplx_mul(&z, &z, &temp_z1);
64 temp_z1.x = 1 - temp_z1.x;
65 temp_z1.y = -temp_z1.y; // tempz1 = 1 - tempz1
66 temp_z1 = complex_sqrt_float(temp_z1);
67
68 temp_z2.x = -z.y;
69 temp_z2.y = z.x; // tempz2 = i*z
70 temp_z1.x += temp_z2.x;
71 temp_z1.y += temp_z2.y; // tempz1 += tempz2
72 fpu_cmplx_log(&temp_z1, &temp_z1);
73 rz->x = temp_z1.y;
74 rz->y = -temp_z1.x; // rz = (-i)*tempz1
75} // end. Arcsinz

78void acos_z(const DComplex z, DComplex *rz)
79{
80 DComplex temp;
81
82 fpu_cmplx_mul(&z, &z, &temp);
83 temp.x -= 1; // temp = temp - 1
84 temp = complex_sqrt_float(temp);

106// rz=Arccosh(z)=Log(z+sqrt(z*z-1)}
107void acosh_z(const DComplex z, DComplex *rz)
108{
109 DComplex temp_z;
110 fpu_cmplx_mul(&z, &z, &temp_z);
111 temp_z.x -= 1; // tempz = tempz - 1
112 temp_z = complex_sqrt_float(temp_z);
113 temp_z.x = z.x + temp_z.x;
114 temp_z.y = z.y + temp_z.y; // tempz = z + tempz

118// rz=Arctanh(z)=1/2*Log{(1+z)/(1-z)}
119void atanh_z(const DComplex z, DComplex *rz)
120{
121 DComplex temp0;
122 DComplex temp1;
123 DComplex temp2;
124
125 if (z.x == 0.0)
126 {
127 rz->x = 0;
128 rz->y = std::atan(z.y);
129 }
130 else
131 {
132 if (std::abs(z.x) == 1.0 && z.y == 0.0)
133 {
134 }
135 else if (std::abs(z.x) < 1.0 && z.y == 0.0)
136 {
137 rz->x = std::log((1+z.x)/(1-z.x))/2;
138 rz->y = 0;
139 }
140 else
141 {
142 temp0.x = 1 + z.x;
143 temp0.y = z.y; // temp0 = 1 + z
144 temp1.x = 1 - z.x;
145 temp1.y = -z.y; // temp1 = 1 - z
146 fpu_cmplx_div(&temp0, &temp1, &temp2);
147 fpu_cmplx_log(&temp2, &temp2);
148 rz->x = .5*temp2.x;
149 rz->y = .5*temp2.y; // rz = .5*temp2
150 }
151 }
152} // end. Arctanhz

154// rz=Arctan(z)=i/2*Log{(1-i*z)/(1+i*z)}
155void atan_z(const DComplex z, DComplex *rz)
156{
157 DComplex temp0;
158 DComplex temp1;
159 DComplex temp2;
160 DComplex temp3;
161 if (z.x == 0.0 && z.y == 0.0)
162 {
163 rz->y = 0;
164 rz->x = 0;
165 }
166 else if (z.x != 0.0 && z.y == 0.0)
167 {
168 rz->x = std::atan(z.x);
169 rz->y = 0;
170 }
171 else if (z.x == 0.0 && z.y != 0.0)
172 {
173 temp0.x = z.y;
174 temp0.y = 0.0;
175 atanh_z(temp0, &temp0);
176 rz->x = -temp0.y;
177 rz->y = temp0.x; // i*temp0
178 }
179 else if (z.x != 0.0 && z.y != 0.0)
180 {
181
182 temp0.x = -z.y;
183 temp0.y = z.x; // i*z
184 temp1.x = 1 - temp0.x;
185 temp1.y = -temp0.y; // temp1 = 1 - temp0
186 temp2.x = 1 + temp0.x;
187 temp2.y = temp0.y; // temp2 = 1 + temp0
188
189 fpu_cmplx_div(&temp1, &temp2, &temp3);
190 fpu_cmplx_log(&temp3, &temp3);
191 rz->x = -temp3.y*.5;
192 rz->y = .5*temp3.x; // .5*i*temp0
193 }
194} // end. Arctanz
