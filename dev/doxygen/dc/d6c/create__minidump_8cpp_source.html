38void create_minidump(EXCEPTION_POINTERS *ep)
39{
40 HMODULE debug_hlp = LoadLibraryA("dbghelp.dll");
41 if (debug_hlp == nullptr)
42 {
43 MessageBoxA(nullptr,
44 "An unexpected error occurred while loading dbghelp.dll.\n" ID_PROGRAM_NAME " will now exit.",
45 ID_PROGRAM_NAME ": Unexpected Error", MB_OK);
46 return;
47 }
48 MiniDumpWriteDumpProc *dumper{(MiniDumpWriteDumpProc *) GetProcAddress(debug_hlp, "MiniDumpWriteDump")};
49 if (dumper == nullptr)
50 {
51 MessageBoxA(
52 nullptr, "Could not locate MiniDumpWriteDump", ID_PROGRAM_NAME ": Unexpected Error", MB_OK);
53 ::FreeLibrary(debug_hlp);
54 return;
55 }
56
57 char minidump[MAX_PATH]{"id-" ID_GIT_HASH ".dmp"};
58 int i{1};
59 fs::path path{g_save_dir};
60 while (fs::exists(path / minidump))
61 {
62 std::sprintf(minidump, "id-" ID_GIT_HASH "-%d.dmp", i++);
63 }
64 path /= minidump;
65 HANDLE dump_file{CreateFileA(path.string().c_str(), GENERIC_READ | GENERIC_WRITE, 0, nullptr, CREATE_NEW,
66 FILE_ATTRIBUTE_NORMAL, nullptr)};
67 _ASSERTE(dump_file != INVALID_HANDLE_VALUE);
68 if (dump_file == INVALID_HANDLE_VALUE)
69 {
70 MessageBoxA(nullptr, ("Could not open dump file " + path.string() + " for writing.").c_str(),
71 ID_PROGRAM_NAME ": Unexpected Error", MB_OK);
72 ::FreeLibrary(debug_hlp);
73 return;
74 }
75
76 MINIDUMP_EXCEPTION_INFORMATION dump_info{GetCurrentThreadId(), ep, FALSE};
77 BOOL status{(*dumper)(
78 GetCurrentProcess(), GetCurrentProcessId(), dump_file, MiniDumpNormal, &dump_info, nullptr, nullptr)};
79 _ASSERTE(status);
80 if (!status)
81 {
82 char msg[100];
83 std::sprintf(msg, "MiniDumpWriteDump failed with %08lx", GetLastError());
84 MessageBoxA(nullptr, msg, ID_PROGRAM_NAME ": Unexpected Error", MB_OK);
85 }
86 else
87 {
88 status = CloseHandle(dump_file);
89 _ASSERTE(status);
90 }
91 dumper = nullptr;
92 status = FreeLibrary(debug_hlp);
93 _ASSERTE(status);
94
95 if (g_init_batch != BatchMode::NORMAL)
96 {
97 char msg[MAX_PATH * 2];
98 std::sprintf(msg,
99 "Unexpected error, crash dump saved to %s.\n"
100 "Please include this file with your bug report.",
101 minidump);
102 MessageBoxA(nullptr, msg, ID_PROGRAM_NAME ": Unexpected Error", MB_OK);
103 }
104}
