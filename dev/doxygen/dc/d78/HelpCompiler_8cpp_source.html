187// calculate topic_num/topic_off for each link.
188void HelpCompiler::make_hot_links()
189{
190 MSG_MSG("Making hot-links.");
191
192 // Calculate topic_num for all entries in DocContents. Also set
193 // "TF_IN_DOC" flag for all topics included in the document.
194 for (Content &c : g_src.contents)
195 {
196 for (int ctr = 0; ctr < c.num_topic; ctr++)
197 {
198 if (c.is_label[ctr])
199 {
200 c.label_topic(ctr);
201 }
202 else
203 {
204 c.content_topic(ctr);
205 }
206 }
207 }
208
209 // Find topic_num and topic_off for all hot-links. Also flag all hot-
210 // links which will (probably) appear in the document.
211 for (Link &l : g_src.all_links)
212 {
213 // name is the title of the topic
214 if (l.type == LinkTypes::LT_TOPIC)
215 {
216 l.link_topic();
217 }
218 // name is the name of a label
219 else if (l.type == LinkTypes::LT_LABEL)
220 {
221 l.link_label();
222 }
223 // it's a "special" link; topic_off already has the value
224 else if (l.type == LinkTypes::LT_SPECIAL)
225 {
226 }
227 }
228}

230// online help pagination stuff
231void HelpCompiler::paginate_online() // paginate the text for on-line help
232{
233 int size;
234 int width;
235
236 MSG_MSG("Paginating online help.");
237
238 for (Topic &t : g_src.topics)
239 {
240 if (bit_set(t.flags, TopicFlags::DATA))
241 {
242 continue; // don't paginate data topics
243 }
244
245 const char *text = t.get_topic_text();
246 const char *curr = text;
247 unsigned int len = t.text_len;
248
249 const char *start = curr;
250 bool skip_blanks = false;
251 int line_num = 0;
252 int num_links = 0;
253 int col = 0;
254 int start_margin = -1;
255
256 while (len > 0)
257 {
258 switch (TokenType tok = find_token_length(TokenMode::ONLINE, curr, len, &size, &width); tok)
259 {
260 case TokenType::TOK_PARA:
261 {
262 ++curr;
263 const int indent = *curr++;
264 const int margin = *curr++;
265 len -= 3;
266 col = indent;
267 while (true)
268 {
269 tok = find_token_length(TokenMode::ONLINE, curr, len, &size, &width);
270
271 if (tok == TokenType::TOK_DONE || tok == TokenType::TOK_NL || tok == TokenType::TOK_FF)
272 {
273 break;
274 }
275
276 if (tok == TokenType::TOK_PARA)
277 {
278 col = 0; // fake a nl
279 ++line_num;
280 break;
281 }
282
283 if (tok == TokenType::TOK_XONLINE || tok == TokenType::TOK_XDOC)
284 {
285 curr += size;
286 len -= size;
287 continue;
288 }
289
290 // now tok is SPACE or LINK or WORD
291 if (col+width > SCREEN_WIDTH)
292 {
293 // go to next line...
294 if (++line_num >= SCREEN_DEPTH)
295 {
296 // go to next page...
297 t.add_page_break(start_margin, text, start, curr, num_links);
298 start = curr + (tok == TokenType::TOK_SPACE ? size : 0);
299 start_margin = margin;
300 line_num = 0;
301 num_links = 0;
302 }
303 if (tok == TokenType::TOK_SPACE)
304 {
305 width = 0; // skip spaces at start of a line
306 }
307
308 col = margin;
309 }
310
311 col += width;
312 curr += size;
313 len -= size;
314 }
315
316 skip_blanks = false;
317 size = 0;
318 width = 0;
319 break;
320 }
321
322 case TokenType::TOK_NL:
323 if (skip_blanks && col == 0)
324 {
325 start += size;
326 break;
327 }
328 ++line_num;
329 if (line_num >= SCREEN_DEPTH || (col == 0 && line_num == SCREEN_DEPTH-1))
330 {
331 t.add_page_break(start_margin, text, start, curr, num_links);
332 start = curr + size;
333 start_margin = -1;
334 line_num = 0;
335 num_links = 0;
336 skip_blanks = true;
337 }
338 col = 0;
339 break;
340
341 case TokenType::TOK_FF:
342 col = 0;
343 if (skip_blanks)
344 {
345 start += size;
346 break;
347 }
348 t.add_page_break(start_margin, text, start, curr, num_links);
349 start_margin = -1;
350 start = curr + size;
351 line_num = 0;
352 num_links = 0;
353 break;
354
355 case TokenType::TOK_DONE:
356 case TokenType::TOK_XONLINE: // skip
357 case TokenType::TOK_XDOC: // ignore
358 case TokenType::TOK_CENTER: // ignore
359 break;
360
361 case TokenType::TOK_LINK:
362 ++num_links;
363 // fall-through
364
365 default: // SPACE, LINK, WORD
366 skip_blanks = false;
367 break;
368 } // switch
369
370 curr += size;
371 len -= size;
372 col += width;
373 } // while
374
375 if (!skip_blanks)
376 {
377 t.add_page_break(start_margin, text, start, curr, num_links);
378 }
379
380 m_max_pages = std::max(m_max_pages, t.num_page);
381
382 t.release_topic_text(false);
383 } // for
384}

386static Label *find_next_label_by_topic(const int t)
387{
388 Label *g = nullptr;
389 for (Label &l : g_src.labels)
390 {
391 if (l.topic_num == t && l.doc_page == -1)
392 {
393 g = &l;
394 break;
395 }
396 if (l.topic_num > t)
397 {
398 break;
399 }
400 }
401
402 Label *p = nullptr;
403 for (Label &pl : g_src.private_labels)
404 {
405 if (pl.topic_num == t && pl.doc_page == -1)
406 {
407 p = &pl;
408 break;
409 }
410 if (pl.topic_num > t)
411 {
412 break;
413 }
414 }
415
416 if (p == nullptr)
417 {
418 return g;
419 }
420
421 if (g == nullptr)
422 {
423 return p;
424 }
425 return g->topic_off < p->topic_off ? g : p;
426}

428// Find doc_page for all hot-links.
429void HelpCompiler::set_hot_link_doc_page()
430{
431 for (Link &l : g_src.all_links)
432 {
433 switch (l.type)
434 {
435 case LinkTypes::LT_TOPIC:
436 if (const int t = find_topic_title(l.name.c_str()); t == -1)
437 {
438 g_current_src_filename = l.src_file;
439 g_src_line = l.src_line; // pretend we are still in the source...
440 MSG_ERROR(0, "Cannot find implicit hot-link \"%s\".", l.name.c_str());
441 g_src_line = -1; // back to reality
442 }
443 else
444 {
445 l.doc_page = g_src.topics[t].doc_page;
446 }
447 break;
448
449 case LinkTypes::LT_LABEL:
450 if (const Label *const lbl = g_src.find_label(l.name.c_str()); lbl == nullptr)
451 {
452 g_current_src_filename = l.src_file;
453 g_src_line = l.src_line; // pretend again
454 MSG_ERROR(0, "Cannot find explicit hot-link \"%s\".", l.name.c_str());
455 g_src_line = -1;
456 }
457 else
458 {
459 l.doc_page = lbl->doc_page;
460 }
461 break;
462
463 case LinkTypes::LT_SPECIAL:
464 // special topics don't appear in the document
465 break;
466 }
467 }

470// insert page #'s in the DocContents
471void HelpCompiler::set_content_doc_page()
472{
473 const int topic_num = find_topic_title(DOC_CONTENTS_TITLE);
474 assert(topic_num >= 0);
475 Topic &t = g_src.topics[topic_num];
476 char *base = t.get_topic_text();
477 for (const Content &c : g_src.contents)
478 {
479 assert(c.doc_page >= 1);
480 std::string doc_page{std::to_string(c.doc_page)};
481 const int len = static_cast<int>(doc_page.size());
482 assert(len <= 3);
483 std::memcpy(base + c.page_num_pos + (3 - len), doc_page.c_str(), len);
484 }
485
486 t.release_topic_text(true);
487}

489// this function also used by print_document()
490static bool pd_get_info(const PrintDocCommand cmd, ProcessDocumentInfo *pd, void *context)
491{
492 auto &[content_num, topic_num, link_dest_warn] = *static_cast<DocInfo *>(context);
493 const Content *c;
494
495 switch (cmd)
496 {
497 case PrintDocCommand::PD_GET_CONTENT:
498 if (++content_num >= static_cast<int>(g_src.contents.size()))
499 {
500 return false;
501 }
502 c = &g_src.contents[content_num];
503 topic_num = -1;
504 pd->id = c->id.c_str();
505 pd->title = c->name.c_str();
506 pd->new_page = (c->flags & CF_NEW_PAGE) != 0;
507 return true;
508
509 case PrintDocCommand::PD_GET_TOPIC:
510 c = &g_src.contents[content_num];
511 if (++topic_num >= c->num_topic)
512 {
513 return false;
514 }
515 pd->curr = g_src.topics[c->topic_num[topic_num]].get_topic_text();
516 pd->len = g_src.topics[c->topic_num[topic_num]].text_len;
517 return true;
518
519 case PrintDocCommand::PD_GET_LINK_PAGE:
520 {
521 const Link &link = g_src.all_links[get_int(pd->s)];
522 if (link.doc_page == -1)
523 {
524 if (link_dest_warn)
525 {
526 g_current_src_filename = link.src_file;
527 g_src_line = link.src_line;
528 MSG_WARN(0, "Hot-link destination is not in the document.");
529 g_src_line = -1;
530 }
531 return false;
532 }
533 pd->i = link.doc_page;
534 pd->link_page = fmt::format("(p. {:d})", pd->i);
535 return true;
536 }
537
538 case PrintDocCommand::PD_RELEASE_TOPIC:
539 c = &g_src.contents[content_num];
540 g_src.topics[c->topic_num[topic_num]].release_topic_text(false);
541 return true;
542
543 default:
544 return false;
545 }
546}

548static bool paginate_doc_output(const PrintDocCommand cmd, ProcessDocumentInfo *pd, void *context)
549{
550 PaginateDocIno *info = static_cast<PaginateDocIno *>(context);
551 switch (cmd)
552 {
553 case PrintDocCommand::PD_FOOTING:
554 case PrintDocCommand::PD_PRINT:
555 case PrintDocCommand::PD_PRINT_N:
556 case PrintDocCommand::PD_PRINT_SEC:
557 return true;
558
559 case PrintDocCommand::PD_HEADING:
560 ++s_num_doc_pages;
561 return true;
562
563 case PrintDocCommand::PD_START_SECTION:
564 info->c = &g_src.contents[info->content_num];
565 return true;
566
567 case PrintDocCommand::PD_START_TOPIC:
568 info->start = pd->curr;
569 info->lbl = find_next_label_by_topic(info->c->topic_num[info->topic_num]);
570 return true;
571
572 case PrintDocCommand::PD_SET_SECTION_PAGE:
573 info->c->doc_page = pd->page_num;
574 return true;
575
576 case PrintDocCommand::PD_SET_TOPIC_PAGE:
577 g_src.topics[info->c->topic_num[info->topic_num]].doc_page = pd->page_num;
578 return true;
579
580 case PrintDocCommand::PD_PERIODIC:
581 while (info->lbl != nullptr && static_cast<unsigned>(pd->curr - info->start) >= info->lbl->topic_off)
582 {
583 info->lbl->doc_page = pd->page_num;
584 info->lbl = find_next_label_by_topic(info->c->topic_num[info->topic_num]);
585 }
586 return true;
587
588 default:
589 return false;
590 }
591}

593void HelpCompiler::paginate_document()
594{
595 if (g_src.contents.empty())
596 {
597 return;
598 }
599
600 MSG_MSG("Paginating document.");
601
602 PaginateDocIno info;
603 info.topic_num = -1;
604 info.content_num = info.topic_num;
605 info.link_dest_warn = true;
606
607 process_document(TokenMode::DOC, true, pd_get_info, paginate_doc_output, &info);
608
609 set_hot_link_doc_page();
610 set_content_doc_page();
611}

615// returns true if different
616static bool compare_files(std::FILE *f1, std::FILE *f2)
617{
618 if (filelength(fileno(f1)) != filelength(fileno(f2)))
619 {
620 return true; // different if sizes are not the same
621 }
622
623 while (!std::feof(f1) && !std::feof(f2))
624 {
625 if (std::getc(f1) != std::getc(f2))
626 {
627 return true;
628 }
629 }
630
631 return !(std::feof(f1) && std::feof(f2));
632}

684void HelpCompiler::write_header()
685{
686 const std::string &fname{g_src.hdr_filename};
687
688 std::FILE *hdr = std::fopen(fname.c_str(), "rt");
689
690 if (hdr == nullptr)
691 {
692 // if no prev. hdr file generate a new one
693 hdr = std::fopen(fname.c_str(), "wt");
694 if (hdr == nullptr)
695 {
696 throw std::runtime_error(R"msg(Cannot create ")msg" + fname + R"msg(".)msg");
697 }
698 MSG_MSG("Writing: %s", fname.c_str());
699 write_header_file(fname.c_str(), hdr);
700 std::fclose(hdr);
701 MSG_NOTICE("Id must be re-compiled.");
702 return;
703 }
704
705 MSG_MSG("Comparing: %s", fname.c_str());
706
707 std::FILE *temp = std::fopen(TEMP_FNAME, "wt");
708
709 if (temp == nullptr)
710 {
711 throw std::runtime_error(R"msg(Cannot create temporary file: ")msg" + std::string{TEMP_FNAME} + R"msg(".)msg");
712 }
713
714 write_header_file(fname.c_str(), temp);
715
716 std::fclose(temp);
717 temp = std::fopen(TEMP_FNAME, "rt");
718
719 if (temp == nullptr)
720 {
721 throw std::runtime_error(R"msg(Cannot open temporary file: ")msg" + std::string{TEMP_FNAME} + R"msg(".)msg");
722 }
723
724 if (compare_files(temp, hdr)) // if they are different...
725 {
726 MSG_MSG("Updating: %s", fname.c_str());
727 std::fclose(temp);
728 std::fclose(hdr);
729 std::remove(fname.c_str()); // delete the old hdr file
730 std::rename(TEMP_FNAME, fname.c_str()); // rename the temp to the hdr file
731 MSG_NOTICE("Id must be re-compiled.");
732 }
733 else
734 {
735 // if they are the same leave the original alone.
736 std::fclose(temp);
737 std::fclose(hdr);
738 std::remove(TEMP_FNAME); // delete the temp
739 }
740}

742void HelpCompiler::write_link_source()
743{
744 std::vector<std::pair<int, Label>> topics;
745 for (const Label &label : g_src.labels)
746 {
747 if (label.topic_off == 0)
748 {
749 const Topic &topic{g_src.topics[label.topic_num]};
750 if (bit_set(topic.flags, TopicFlags::IN_DOC))
751 {
752 topics.emplace_back(label.topic_num, label);
753 }
754 }
755 }
756
757 {
758 MSG_MSG("Writing: help_links.h");
759 std::ofstream hdr{"help_links.h"};
760 hdr << //
761 "// SPDX-License-Identifier: GPL-3.0-only\n"
762 "//\n"
763 "#pragma once\n"
764 "\n"
765 R"(#include "helpdefs.h")"
766 "\n"
767 "#include <array>\n"
768 "#include <string_view>\n"
769 "\n"
770 "namespace hc\n"
771 "{\n"
772 "\n"
773 "struct HelpLink\n"
774 "{\n"
775 " id::help::HelpLabels label;\n"
776 " std::string_view link;\n"
777 "};\n"
778 "\n"
779 "extern std::array<HelpLink, "
780 << topics.size()
781 << "> g_help_links;\n"
782 "\n"
783 "} // namespace hc\n";
784 }
785
786 {
787 MSG_MSG("Writing: help_links.cpp");
788 std::ofstream src{"help_links.cpp"};
789 src << //
790 "// SPDX-License-Identifier: GPL-3.0-only\n"
791 "//\n"
792 "#include \"helpdefs.h\"\n"
793 "#include \"help_links.h\"\n"
794 "\n"
795 "namespace hc\n"
796 "{\n"
797 "\n"
798 "std::array<HelpLink, "
799 << topics.size() << "> g_help_links = {\n";
800 const auto link_for_title = [](std::string text)
801 {
802 for (auto pos = text.find_first_of("/\""); pos != std::string::npos;
803 pos = text.find_first_of("/\""))
804 {
805 text.erase(pos, 1);
806 }
807 std::transform(text.begin(), text.end(), text.begin(),
808 [](const char c)
809 {
810 const unsigned char test = static_cast<unsigned char>(c);
811 return std::isalnum(test) ? static_cast<char>(std::tolower(test)) : '_';
812 });
813 for (auto pos = text.find("__"); pos != std::string::npos; pos = text.find("__"))
814 {
815 text.erase(pos, 1);
816 }
817 if (const auto pos = text.find_last_not_of('_') + 1; pos != text.length())
818 {
819 text.erase(pos);
820 }
821 return "id.html#_" + text;
822 };
823 for (const auto &[topic_num, label] : topics)
824 {
825 src << " HelpLink{ id::help::HelpLabels::" << label.name << ", \""
826 << link_for_title(g_src.topics[topic_num].title) << "\" },\n";
827 }
828 src << "};\n"
829 "\n"
830 "} // namespace hc\n";
831 }
832}

834void HelpCompiler::calc_offsets() // calc file offset to each topic
835{
836 // NOTE: offsets do NOT include 6 bytes for signature & version!
837 long offset = static_cast<long>(sizeof(int) + // max_pages
838 sizeof(int) + // max_links
839 sizeof(int) + // num_topic
840 sizeof(int) + // num_label
841 sizeof(int) + // num_contents
842 sizeof(int) + // num_doc_pages
843 g_src.topics.size() * sizeof(long) + // offsets to each topic
844 g_src.labels.size() * 2 * sizeof(int)); // topic_num/topic_off for all public labels
845
846 offset = std::accumulate(g_src.contents.begin(), g_src.contents.end(), offset,
847 [](long value, const Content &cp)
848 {
849 return value += sizeof(int) + // flags
850 1 + // id length
851 static_cast<int>(cp.id.length()) + // id text
852 1 + // name length
853 static_cast<int>(cp.name.length()) + // name text
854 1 + // number of topics
855 cp.num_topic * sizeof(int); // topic numbers
856 });
857
858 for (Topic &tp : g_src.topics)
859 {
860 tp.offset = offset;
861 offset += static_cast<long>(sizeof(int) + // topic flags
862 sizeof(int) + // number of pages
863 tp.num_page * 3 * sizeof(int) + // page offset, length & starting margin
864 1 + // length of title
865 tp.title_len + // title
866 sizeof(int) + // length of text
867 tp.text_len); // text
868 }

871// Replaces link indexes in the help text with topic_num, topic_off and
872// doc_page info.
873static void insert_real_link_info(char *curr, unsigned int len)
874{
875 while (len > 0)
876 {
877 int size = 0;
878
879 if (TokenType tok = find_token_length(TokenMode::NONE, curr, len, &size, nullptr);
880 tok == TokenType::TOK_LINK)
881 {
882 const Link &l = g_src.all_links[ get_int(curr+1) ];
883 set_int(curr+1, l.topic_num);
884 set_int(curr+1+sizeof(int), l.topic_off);
885 set_int(curr+1+2*sizeof(int), l.doc_page);
886 }
887
888 len -= size;
889 curr += size;
890 }

893void HelpCompiler::write_help(std::FILE *file)
894{
895 // write the signature and version
896 HelpSignature hs{};
897 hs.sig = HELP_SIG;
898 hs.version = g_src.version;
899
900 std::fwrite(&hs, sizeof(HelpSignature), 1, file);
901
902 // write max_pages & max_links
903
904 putw(m_max_pages, file);
905 putw(g_src.max_links, file);
906
907 // write num_topic, num_label and num_contents
908
909 putw(static_cast<int>(g_src.topics.size()), file);
910 putw(static_cast<int>(g_src.labels.size()), file);
911 putw(static_cast<int>(g_src.contents.size()), file);
912
913 // write num_doc_page
914
915 putw(s_num_doc_pages, file);
916
917 // write the offsets to each topic
918 for (const Topic &t : g_src.topics)
919 {
920 std::fwrite(&t.offset, sizeof(long), 1, file);
921 }
922
923 // write all public labels
924 for (const Label &l : g_src.labels)
925 {
926 putw(l.topic_num, file);
927 putw(l.topic_off, file);
928 }
929
930 // write contents
931 for (const Content &cp : g_src.contents)
932 {
933 putw(cp.flags, file);
934
935 int t = static_cast<int>(cp.id.length());
936 putc(static_cast<Byte>(t), file);
937 std::fwrite(cp.id.c_str(), 1, t, file);
938
939 t = static_cast<int>(cp.name.length());
940 putc(static_cast<Byte>(t), file);
941 std::fwrite(cp.name.c_str(), 1, t, file);
942
943 putc(static_cast<Byte>(cp.num_topic), file);
944 std::fwrite(cp.topic_num, sizeof(int), cp.num_topic, file);
945 }
946
947 // write topics
948 for (Topic &tp : g_src.topics)
949 {
950 // write the topics flags
951 putw(+tp.flags, file);
952
953 // write offset, length and starting margin for each page
954
955 putw(tp.num_page, file);
956 for (const auto &[offset, length, margin] : tp.page)
957 {
958 putw(offset, file);
959 putw(length, file);
960 putw(margin, file);
961 }
962
963 // write the help title
964
965 putc(static_cast<Byte>(tp.title_len), file);
966 std::fwrite(tp.title.c_str(), 1, tp.title_len, file);
967
968 // insert hot-link info & write the help text
969
970 char *text = tp.get_topic_text();
971
972 if (!bit_set(tp.flags, TopicFlags::DATA)) // don't process data topics...
973 {
974 insert_real_link_info(text, tp.text_len);
975 }
976
977 putw(tp.text_len, file);
978 std::fwrite(text, 1, tp.text_len, file);
979
980 tp.release_topic_text(false); // don't save the text even though
981 // insert_real_link_info() modified it
982 // because we don't access the info after
983 // this.
984 }

1003static void printer_ch(PrintDocInfo *info, const int c, int n)
1004{
1005 while (n-- > 0)
1006 {
1007 if (c == ' ')
1008 {
1009 ++info->spaces;
1010 }
1011 else if (c == '\n' || c == '\f')
1012 {
1013 info->start_of_line = true;
1014 info->spaces = 0; // strip spaces before a new-line
1015 putc(c, info->file);
1016 }
1017 else
1018 {
1019 if (info->start_of_line)
1020 {
1021 info->spaces += info->margin;
1022 info->start_of_line = false;
1023 }
1024
1025 while (info->spaces > 0)
1026 {
1027 std::fputc(' ', info->file);
1028 --info->spaces;
1029 }
1030
1031 std::fputc(c, info->file);
1032 }
1033 }
1034}

1247// add/delete help from .EXE functions.
1248void HelpCompiler::add_hlp_to_exe()
1249{
1250 const char *hlp_fname{m_options.fname1.empty() ? DEFAULT_HLP_FNAME : m_options.fname1.c_str()};
1251 const char *exe_fname{m_options.fname2.empty() ? DEFAULT_EXE_FNAME : m_options.fname2.c_str()};
1252
1253 HelpSignature hs{};
1254
1255 const int exe = open(exe_fname, O_RDWR | O_BINARY);
1256 if (exe == -1)
1257 {
1258 throw std::runtime_error(R"msg(Unable to open ")msg" + std::string{exe_fname} + '"');
1259 }
1260
1261 const int hlp = open(hlp_fname, O_RDONLY | O_BINARY);
1262 if (hlp == -1)
1263 {
1264 throw std::runtime_error("Unable to open \"" + std::string{hlp_fname} + '"');
1265 }
1266
1267 MSG_MSG("Appending %s to %s", hlp_fname, exe_fname);
1268
1269 // first, check and see if any help is currently installed
1270
1271 lseek(exe, filelength(exe) - sizeof(HelpSignature), SEEK_SET);
1272
1273 if (read(exe, &hs, sizeof(HelpSignature)) != sizeof(HelpSignature))
1274 {
1275 close(hlp);
1276 close(exe);
1277 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed read");
1278 }
1279
1280 if (hs.sig == HELP_SIG)
1281 {
1282 MSG_WARN(0, "Overwriting previous help. (Version=%d)", static_cast<int>(hs.version));
1283 }
1284 else
1285 {
1286 hs.base = static_cast<std::uint32_t>(filelength(exe));
1287 }
1288
1289 // now, let's see if their help file is for real (and get the version)
1290
1291 if (read(hlp, &hs, sizeof(HelpSignature)) != sizeof(HelpSignature))
1292 {
1293 close(hlp);
1294 close(exe);
1295 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed read2");
1296 }
1297
1298 if (hs.sig != HELP_SIG)
1299 {
1300 throw std::runtime_error("Help signature not found in " + std::string{hlp_fname});
1301 }
1302
1303 MSG_MSG("Help file %s Version=%d", hlp_fname, hs.version);
1304
1305 // append the help stuff, overwriting old help (if any)
1306
1307 lseek(exe, hs.base, SEEK_SET);
1308
1309 const long len = filelength(hlp) - sizeof(HelpSignature); // adjust for the file signature & version
1310
1311 for (int count = 0; count < len;)
1312 {
1313 const int size = static_cast<int>(std::min(static_cast<long>(BUFFER_SIZE), len - count));
1314 if (read(hlp, g_src.buffer.data(), size) != size)
1315 {
1316 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed read");
1317 }
1318 if (write(exe, g_src.buffer.data(), size) != size)
1319 {
1320 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed write");
1321 }
1322 count += size;
1323 }
1324
1325 // add on the signature, version and offset
1326
1327 if (write(exe, &hs, sizeof(HelpSignature)) != sizeof(HelpSignature))
1328 {
1329 close(hlp);
1330 close(exe);
1331 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed std::fwrite");
1332 }
1333
1334 if (const off_t offset = lseek(exe, 0L, SEEK_CUR);
1335 chsize(exe, offset) != offset) // truncate if old help was longer
1336 {
1337 close(hlp);
1338 close(exe);
1339 throw std::system_error(errno, std::system_category(), "add_hlp_to_exe failed chsize");
1340 }
1341
1342 close(exe);
1343 close(hlp);
1344}
