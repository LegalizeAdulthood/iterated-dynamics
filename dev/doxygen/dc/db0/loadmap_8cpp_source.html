92bool validate_luts(const std::string &map_name)
93{
94 std::filesystem::path map_path{map_name};
95 if (!map_path.has_extension())
96 {
97 map_path.replace_extension(".map");
98 }
99 map_path = find_file(ReadFile::MAP, map_path);
100 std::FILE *f = std::fopen(map_path.string().c_str(), "r"); // search the dos path
101 if (f == nullptr)
102 {
103 stop_msg(fmt::format("Could not load color map {:s}", map_name));
104 return true;
105 }
106 int index;
107 for (index = 0; index < 256; index++)
108 {
109 char line[160];
110 if (std::fgets(line, std::size(line), f) == nullptr)
111 {
112 break;
113 }
114 unsigned int r{};
115 unsigned int g{};
116 unsigned int b{};
117 if (int count = std::sscanf(line, "%u %u %u", &r, &g, &b); count != 3)
118 {
119 stop_msg(fmt::format("Malformed map entry on line {:d}, only {:d} channels but expected 3.\n"
120 "Skipping remaining lines.",
121 index, count));
122 break;
123 }
124 //* load global dac values *
125 DAC[index].red = static_cast<Byte>(r % 256);
126 DAC[index].green = static_cast<Byte>(g % 256);
127 DAC[index].blue = static_cast<Byte>(b % 256);
128 }
129 std::fclose(f);
130 while (index < 256)
131 {
132 // zap unset entries
133 DAC[index].green = 40;
134 DAC[index].blue = 40;
135 DAC[index].red = 40;
136 ++index;
137 }
138 backwards_id1_3_map();
139 g_color_state = ColorState::MAP_FILE;
140 g_last_map_name = map_name;
141 return false;
142}
