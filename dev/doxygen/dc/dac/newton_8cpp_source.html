43// transform points with reciprocal function
44void invertz2(DComplex *z)
45{
46 z->x = dx_pixel();
47 z->y = dy_pixel();
48 *z -= g_inversion.center; // Normalize values to center of circle
49
50 g_temp_sqr_x = sqr(z->x) + sqr(z->y); // Get old radius
51 if (std::abs(g_temp_sqr_x) > std::numeric_limits<float>::min())
52 {
53 g_temp_sqr_x = g_inversion.radius / g_temp_sqr_x;
54 }
55 else
56 {
57 g_temp_sqr_x = std::numeric_limits<float>::max(); // a big number, but not TOO big
58 }
59 z->x *= g_temp_sqr_x;
60 z->y *= g_temp_sqr_x; // Perform inversion
61 *z += g_inversion.center; // Renormalize
62}

70static int complex_mult(const DComplex arg1, const DComplex arg2, DComplex *pz)
71{
72 pz->x = arg1.x * arg2.x - arg1.y * arg2.y;
73 pz->y = arg1.x * arg2.y + arg1.y * arg2.x;
74 return 0;

77int newton_orbit()
78{
79 static char start = 1;
80 if (start)
81 {
82 start = 0;
83 }
84 pow(&g_old_z, g_degree-1, &g_tmp_z);
85 complex_mult(g_tmp_z, g_old_z, &g_new_z);
86
87 if (distance1(g_new_z) < s_threshold)
88 {
89 if (g_fractal_type == FractalType::NEWT_BASIN)
90 {
91 long tmp_color = -1;
92 /* this code determines which degree-th root of root the
93 Newton formula converges to. The roots of a 1 are
94 distributed on a circle of radius 1 about the origin. */
95 for (int i = 0; i < g_degree; i++)
96 {
97 /* color in alternating shades with iteration according to
98 which root of 1 it converged to */
99 if (distance(s_roots[i], g_old_z) < s_threshold)
100 {
101 if (g_basin == 2)
102 {
103 tmp_color = 1+(i&7)+((g_color_iter&1) << 3);
104 }
105 else
106 {
107 tmp_color = 1+i;
108 }
109 break;
110 }
111 }
112 if (tmp_color == -1)
113 {
114 g_color_iter = g_max_color;
115 }
116 else
117 {
118 g_color_iter = tmp_color;
119 }
120 }
121 return 1;
122 }
123 g_new_z.x = s_degree_minus_1_over_degree * g_new_z.x + s_newton_r_over_d;
124 g_new_z.y *= s_degree_minus_1_over_degree;
125
126 // Watch for divide underflow
127 double t2 = g_tmp_z.x * g_tmp_z.x + g_tmp_z.y * g_tmp_z.y;
128 if (t2 < std::numeric_limits<float>::min())
129 {
130 return 1;
131 }
132 t2 = 1.0 / t2;
133 g_old_z.x = t2 * (g_new_z.x * g_tmp_z.x + g_new_z.y * g_tmp_z.y);
134 g_old_z.y = t2 * (g_new_z.y * g_tmp_z.x - g_new_z.x * g_tmp_z.y);
135 return 0;
136}

157int complex_newton_orbit()
158{
159 DComplex cd1;
160
161 /* new = ((cdegree-1) * old**cdegree) + croot
162 ----------------------------------
163 cdegree * old**(cdegree-1) */
164
165 cd1.x = s_c_degree.x - 1.0;
166 cd1.y = s_c_degree.y;
167
168 s_temp = pow(g_old_z, cd1);
169 fpu_cmplx_mul(s_temp, g_old_z, g_new_z);
170
171 g_tmp_z.x = g_new_z.x - s_c_root.x;
172 g_tmp_z.y = g_new_z.y - s_c_root.y;
173 if (sqr(g_tmp_z.x) + sqr(g_tmp_z.y) < s_threshold)
174 {
175 return 1;
176 }
177
178 fpu_cmplx_mul(g_new_z, cd1, g_tmp_z);
179 g_tmp_z.x += s_c_root.x;
180 g_tmp_z.y += s_c_root.y;
181
182 fpu_cmplx_mul(s_temp, s_c_degree, cd1);
183 fpu_cmplx_div(g_tmp_z, cd1, g_old_z);
184 if (g_overflow)
185 {
186 return 1;
187 }
188 g_new_z = g_old_z;
189 return 0;
190}

192int complex_basin_orbit()
193{
194 DComplex cd1;
195
196 /* new = ((cdegree-1) * old**cdegree) + croot
197 ----------------------------------
198 cdegree * old**(cdegree-1) */
199
200 cd1.x = s_c_degree.x - 1.0;
201 cd1.y = s_c_degree.y;
202
203 s_temp = pow(g_old_z, cd1);
204 fpu_cmplx_mul(s_temp, g_old_z, g_new_z);
205
206 g_tmp_z.x = g_new_z.x - s_c_root.x;
207 g_tmp_z.y = g_new_z.y - s_c_root.y;
208 if (sqr(g_tmp_z.x) + sqr(g_tmp_z.y) < s_threshold)
209 {
210 if (std::abs(g_old_z.y) < .01)
211 {
212 g_old_z.y = 0.0;
213 }
214 cmplx_log(g_old_z, s_temp);
215 fpu_cmplx_mul(s_temp, s_c_degree, g_tmp_z);
216 const double mod = g_tmp_z.y/s_two_pi;
217 g_color_iter = static_cast<long>(mod);
218 if (std::abs(mod - g_color_iter) > 0.5)
219 {
220 if (mod < 0.0)
221 {
222 g_color_iter--;
223 }
224 else
225 {
226 g_color_iter++;
227 }
228 }
229 g_color_iter += 2;
230 if (g_color_iter < 0)
231 {
232 g_color_iter += 128;
233 }
234 return 1;
235 }
236
237 fpu_cmplx_mul(g_new_z, cd1, g_tmp_z);
238 g_tmp_z.x += s_c_root.x;
239 g_tmp_z.y += s_c_root.y;
240
241 fpu_cmplx_mul(s_temp, s_c_degree, cd1);
242 fpu_cmplx_div(g_tmp_z, cd1, g_old_z);
243 if (g_overflow)
244 {
245 return 1;
246 }
247 g_new_z = g_old_z;
248 return 0;
249}

251// Newton/NewtBasin Routines
252bool newton_per_image()
253{
254 // TODO: is it necessary to update g_cur_fractal_specific?
255 assert(g_cur_fractal_specific == get_fractal_specific(g_fractal_type));
256 g_cur_fractal_specific = get_fractal_specific(g_fractal_type);
257 // set up table of roots of 1 along unit circle
258 g_degree = static_cast<int>(g_param_z1.x);
259 if (g_degree < 2)
260 {
261 g_degree = 3; // defaults to 3, but 2 is possible
262 }
263
264 // precalculated values
265 s_newton_r_over_d = 1.0 / static_cast<double>(g_degree);
266 s_degree_minus_1_over_degree = static_cast<double>(g_degree - 1) / static_cast<double>(g_degree);
267 g_max_color = 0;
268 s_threshold = .3*PI/g_degree; // less than half distance between roots
269
270 g_basin = 0;
271 s_roots.resize(16);
272 if (g_fractal_type == FractalType::NEWT_BASIN)
273 {
274 if (g_param_z1.y)
275 {
276 g_basin = 2; //stripes
277 }
278 else
279 {
280 g_basin = 1;
281 }
282 s_roots.resize(g_degree);
283
284 // list of roots to discover where we converged for newtbasin
285 for (int i = 0; i < g_degree; i++)
286 {
287 s_roots[i].x = std::cos(i*PI*2.0/ static_cast<double>(g_degree));
288 s_roots[i].y = std::sin(i*PI*2.0/ static_cast<double>(g_degree));
289 }
290 }
291
292 g_params[0] = static_cast<double>(g_degree);
293 if (g_degree%4 == 0)
294 {
295 g_symmetry = SymmetryType::XY_AXIS;
296 }
297 else
298 {
299 g_symmetry = SymmetryType::X_AXIS;
300 }
301
302 g_calc_type = standard_fractal_type;
303 return true;
304}
