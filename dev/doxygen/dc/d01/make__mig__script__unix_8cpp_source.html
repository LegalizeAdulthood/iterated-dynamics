21std::string shell_quote(const std::string_view text)
22{
23 std::string result{"'"};
24 for (const char ch : text)
25 {
26 if (ch == '\'')
27 {
28 result += "'\\''";
29 }
30 else
31 {
32 result += ch;
33 }
34 }
35 result += "'";
36 return result;
37}

50void set_make_mig_script_executable(const char *path)
51{
52 std::error_code err;
53 const fs::path self{fs::read_symlink("/proc/self/exe", err)};
54 if (!err && !self.empty())
55 {
56 s_executable = self;
57 }
58 else if (path != nullptr && *path != 0)
59 {
60 s_executable = path;
61 }
62}
