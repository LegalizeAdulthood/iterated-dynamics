34fs::path current_executable_path()
35{
36#if defined(__APPLE__)
37 std::uint32_t size{1024};
38 std::vector<char> path(size);
39 if (_NSGetExecutablePath(path.data(), &size) != 0)
40 {
41 if (size == 0)
42 {
43 return {};
44 }
45 path.resize(size);
46 if (_NSGetExecutablePath(path.data(), &size) != 0)
47 {
48 return {};
49 }
50 }
51 return canonical_path(path.data());
52#elif defined(__linux__)
53 std::error_code err;
54 const fs::path self{fs::read_symlink("/proc/self/exe", err)};
55 return err ? fs::path{} : self;
56#else
57 return {};
58#endif
59}

61std::string shell_quote(const std::string_view text)
62{
63 std::string result{"'"};
64 for (const char ch : text)
65 {
66 if (ch == '\'')
67 {
68 result += "'\\''";
69 }
70 else
71 {
72 result += ch;
73 }
74 }
75 result += "'";
76 return result;
77}
