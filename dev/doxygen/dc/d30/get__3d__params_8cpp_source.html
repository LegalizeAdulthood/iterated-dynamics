361static bool get_light_params()
362{
363 ChoiceBuilder<13> builder;
364
365 // defaults go here
366 if (illumine() || g_raytrace_format != RayTraceFormat::NONE)
367 {
368 builder.int_number("X value light vector", g_light_x)
369 .int_number("Y value light vector", g_light_y)
370 .int_number("Z value light vector", g_light_z);
371
372 if (g_raytrace_format == RayTraceFormat::NONE)
373 {
374 builder.int_number("Light Source Smoothing Factor", g_light_avg);
375 builder.int_number("Ambient", g_ambient);
376 }
377 }
378
379 if (g_targa_out && g_raytrace_format == RayTraceFormat::NONE)
380 {
381 builder.int_number("Haze Factor (0 - 100, '0' disables)", g_haze);
382
383 if (!g_targa_overlay)
384 {
385 check_write_file(g_light_name, ".tga");
386 }
387 builder.string("Targa File Name (Assume .tga)", g_light_name.c_str())
388 .comment("Background Color (0 - 255)")
389 .int_number(" Red", g_background_color[0])
390 .int_number(" Green", g_background_color[1])
391 .int_number(" Blue", g_background_color[2])
392 .yes_no("Overlay Targa File? (Y/N)", g_targa_overlay);
393 }
394 builder.comment("");
395
396 {
397 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_3D_LIGHT};
398 if (builder.prompt("Light Source Parameters") < 0)
399 {
400 return true;
401 }
402 }
403
404 if (illumine())
405 {
406 g_light_x = builder.read_int_number();
407 g_light_y = builder.read_int_number();
408 g_light_z = builder.read_int_number();
409 if (g_raytrace_format == RayTraceFormat::NONE)
410 {
411 g_light_avg = builder.read_int_number();
412 g_ambient = builder.read_int_number();
413 g_ambient = std::min(g_ambient, 100);
414 g_ambient = std::max(g_ambient, 0);
415 }
416 }
417
418 if (g_targa_out && g_raytrace_format == RayTraceFormat::NONE)
419 {
420 g_haze = builder.read_int_number();
421 g_haze = std::min(g_haze, 100);
422 g_haze = std::max(g_haze, 0);
423 g_light_name = builder.read_string();
424 /* In case light_name conflicts with an existing name it is checked again in line3d */
425 g_background_color[0] = static_cast<char>(builder.read_int_number() % 255);
426 g_background_color[1] = static_cast<char>(builder.read_int_number() % 255);
427 g_background_color[2] = static_cast<char>(builder.read_int_number() % 255);
428 g_targa_overlay = builder.read_yes_no();
429 }
430 return false;
431}

435static bool check_map_file()
436{
437 bool ask_flag = false;
438 if (!g_read_color)
439 {
440 return false;
441 }
442 std::filesystem::path map_name{"*"};
443 if (g_map_set)
444 {
445 map_name = g_map_name;
446 }
447 if (!glasses_alternating_or_superimpose())
448 {
449 ask_flag = true;
450 }
451 else
452 {
453 map_name = s_funny_glasses_map_name;
454 }
455
456 while (true)
457 {
458 if (ask_flag)
459 {
460 char buff[80]{};
461 std::strcpy(buff, map_name.string().c_str());
462 ValueSaver saved_help_mode{g_help_mode, HelpLabels::NONE};
463 const int i = field_prompt("Enter name of .map file to use,\n"
464 "or '*' to use palette from the image to be loaded.",
465 nullptr, buff, 60, nullptr);
466 if (i < 0)
467 {
468 return true;
469 }
470 map_name = buff;
471 if (buff[0] == '*')
472 {
473 g_map_set = false;
474 break;
475 }
476 }
477 std::memcpy(g_old_dac_box, g_dac_box, 256*3); // save the DAC
478 const bool valid = validate_luts(map_name.string());
479 std::memcpy(g_dac_box, g_old_dac_box, 256*3); // restore the DAC
480 if (valid) // Oops, something's wrong
481 {
482 ask_flag = true;
483 continue;
484 }
485 g_map_set = true;
486 g_map_name = map_name.filename().string();
487 break;
488 }
489 return false;
490}

492static bool get_funny_glasses_params()
493{
494 // defaults
495 if (g_viewer_z == 0)
496 {
497 g_viewer_z = 150;
498 }
499 if (g_eye_separation == 0)
500 {
501 if (g_fractal_type == FractalType::IFS_3D || g_fractal_type == FractalType::LORENZ_3D)
502 {
503 g_eye_separation = 2;
504 g_converge_x_adjust = -2;
505 }
506 else
507 {
508 g_eye_separation = 3;
509 g_converge_x_adjust = 0;
510 }
511 }
512
513 if (g_glasses_type == GlassesType::ALTERNATING)
514 {
515 s_funny_glasses_map_name = GLASSES1_MAP_NAME;
516 }
517 else if (g_glasses_type == GlassesType::SUPERIMPOSE)
518 {
519 if (g_fill_type == FillType::SURFACE_GRID)
520 {
521 s_funny_glasses_map_name = "grid.map";
522 }
523 else
524 {
525 std::string glasses2_map{GLASSES1_MAP_NAME};
526 glasses2_map.replace(glasses2_map.find('1'), 1, "2");
527 s_funny_glasses_map_name = glasses2_map;
528 }
529 }
530
531 ChoiceBuilder<10> builder;
532 builder.int_number("Interocular distance (as % of screen)", g_eye_separation)
533 .int_number("Convergence adjust (positive = spread greater)", g_converge_x_adjust)
534 .int_number("Left red image crop (% of screen)", g_red_crop_left)
535 .int_number("Right red image crop (% of screen)", g_red_crop_right)
536 .int_number("Left blue image crop (% of screen)", g_blue_crop_left)
537 .int_number("Right blue image crop (% of screen)", g_blue_crop_right)
538 .int_number("Red brightness factor (%)", g_red_bright)
539 .int_number("Blue brightness factor (%)", g_blue_bright);
540 if (glasses_alternating_or_superimpose())
541 {
542 builder.string("Map file name", s_funny_glasses_map_name.c_str());
543 }
544
545 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_3D_GLASSES};
546 int k = builder.prompt("Funny Glasses Parameters");
547 if (k < 0)
548 {
549 return true;
550 }
551
552 g_eye_separation = builder.read_int_number();
553 g_converge_x_adjust = builder.read_int_number();
554 g_red_crop_left = builder.read_int_number();
555 g_red_crop_right = builder.read_int_number();
556 g_blue_crop_left = builder.read_int_number();
557 g_blue_crop_right = builder.read_int_number();
558 g_red_bright = builder.read_int_number();
559 g_blue_bright = builder.read_int_number();
560
561 if (glasses_alternating_or_superimpose())
562 {
563 s_funny_glasses_map_name = builder.read_string();
564 }
565 return false;
566}

568int get_fract3d_params() // prompt for 3D fractal parameters
569{
570 driver_stack_screen();
571 ChoiceBuilder<7> builder;
572 builder.int_number("X-axis rotation in degrees", g_x_rot)
573 .int_number("Y-axis rotation in degrees", g_y_rot)
574 .int_number("Z-axis rotation in degrees", g_z_rot)
575 .int_number("Perspective distance [1 - 999, 0 for no persp]", g_viewer_z)
576 .int_number("X shift with perspective (positive = right)", g_shift_x)
577 .int_number("Y shift with perspective (positive = up )", g_shift_y)
578 .int_number("Stereo (R/B 3D)? (0=no,1=alternate,2=superimpose,3=photo,4=stereo pair)",
579 static_cast<int>(g_glasses_type));
580
581 int i;
582 {
583 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_3D_FRACT};
584 i = builder.prompt("3D Parameters");
585 }
586
587 int ret{};
588 if (i < 0)
589 {
590 ret = -1;
591 goto get_f3d_exit;
592 }
593
594 g_x_rot = builder.read_int_number();
595 g_y_rot = builder.read_int_number();
596 g_z_rot = builder.read_int_number();
597 g_viewer_z = builder.read_int_number();
598 g_shift_x = builder.read_int_number();
599 g_shift_y = builder.read_int_number();
600 g_glasses_type = static_cast<GlassesType>(builder.read_int_number());
601 if (g_glasses_type < GlassesType::NONE || g_glasses_type > GlassesType::STEREO_PAIR)
602 {
603 g_glasses_type = GlassesType::NONE;
604 }
605 if (g_glasses_type != GlassesType::NONE)
606 {
607 if (get_funny_glasses_params() || check_map_file())
608 {
609 ret = -1;
610 }
611 }
612
613get_f3d_exit:
614 driver_unstack_screen();
615 return ret;
616}
