44/* This function calculates the precision needed to distinguish adjacent
45 pixels at maximum resolution of MAX_PIXELS by MAX_PIXELS
46 (if res==Resolution::MAX) or at current resolution (if res==Resolution::CURRENT) */
47int get_prec_bf(const ResolutionFlag flag)
48{
49 BigStackSaver saved;
50 int res;
51 BigFloat del1 = alloc_stack(g_bf_length + 2);
52 BigFloat del2 = alloc_stack(g_bf_length + 2);
53 BigFloat one = alloc_stack(g_bf_length + 2);
54 BigFloat bf_x_delta = alloc_stack(g_bf_length + 2);
55 BigFloat bf_x_delta2 = alloc_stack(g_bf_length + 2);
56 BigFloat bf_y_delta = alloc_stack(g_bf_length + 2);
57 BigFloat bf_y_delta2 = alloc_stack(g_bf_length + 2);
58 float_to_bf(one, 1.0);
59 if (flag == ResolutionFlag::MAX)
60 {
61 res = OLD_MAX_PIXELS -1;
62 }
63 else
64 {
65 res = g_logical_screen.x_dots-1;
66 }
67
68 // bfxxdel = (bfxmax - bfx3rd)/(xdots-1)
69 sub_bf(bf_x_delta, g_bf_x_max, g_bf_x_3rd);
70 div_a_bf_int(bf_x_delta, static_cast<U16>(res));
71
72 // bfyydel2 = (bfy3rd - bfymin)/(xdots-1)
73 sub_bf(bf_y_delta2, g_bf_y_3rd, g_bf_y_min);
74 div_a_bf_int(bf_y_delta2, static_cast<U16>(res));
75
76 if (flag == ResolutionFlag::CURRENT)
77 {
78 res = g_logical_screen.y_dots-1;
79 }
80
81 // bfyydel = (bfymax - bfy3rd)/(ydots-1)
82 sub_bf(bf_y_delta, g_bf_y_max, g_bf_y_3rd);
83 div_a_bf_int(bf_y_delta, static_cast<U16>(res));
84
85 // bfxxdel2 = (bfx3rd - bfxmin)/(ydots-1)
86 sub_bf(bf_x_delta2, g_bf_x_3rd, g_bf_x_min);
87 div_a_bf_int(bf_x_delta2, static_cast<U16>(res));
88
89 abs_a_bf(add_bf(del1, bf_x_delta, bf_x_delta2));
90 abs_a_bf(add_bf(del2, bf_y_delta, bf_y_delta2));
91 if (cmp_bf(del2, del1) < 0)
92 {
93 copy_bf(del1, del2);
94 }
95 if (cmp_bf(del1, clear_bf(del2)) == 0)
96 {
97 return -1;
98 }
99 int digits = 1;
100 while (cmp_bf(del1, one) < 0)
101 {
102 digits++;
103 mult_a_bf_int(del1, 10);
104 }
105 digits = std::max(digits, 3);
106 return std::max(digits, get_prec_bf_mag());
107}

109// get_power10(x) returns the magnitude of x.
110static int get_power10(const LDouble x)
111{
112 // Special case for zero
113 if (x == 0.0)
114 {
115 return 0;
116 }
117 return static_cast<int>(std::floor(std::log10(std::abs(x))));
118}
