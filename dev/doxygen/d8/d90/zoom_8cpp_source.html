142 int saved = 0;
143 BigFloat bf_f_x_width;
144 BigFloat bf_f_x_skew;
145 BigFloat bf_f_y_height;
146 BigFloat bf_f_y_skew;
147 BigFloat bf_f_x_adj;
148 if (g_bf_math != BFMathType::NONE)
149 {
150 saved = save_stack();
151 bf_f_x_width = alloc_stack(g_r_bf_length+2);
152 bf_f_x_skew = alloc_stack(g_r_bf_length+2);
153 bf_f_y_height = alloc_stack(g_r_bf_length+2);
154 bf_f_y_skew = alloc_stack(g_r_bf_length+2);
155 bf_f_x_adj = alloc_stack(g_r_bf_length+2);
156 }
157 double f_temp1 = PI * g_zoom_box_rotation / 72; // convert to radians
158 const double rot_cos = std::cos(f_temp1); // sin & cos of rotation
159 const double rot_sin = std::sin(f_temp1);
160
161 // do some calcs just once here to reduce fp work a bit
162 const double f_x_width = g_save_image_region.m_max.x - g_save_image_region.m_3rd.x;
163 const double f_x_skew = g_save_image_region.width3();
164 const double f_y_height = g_save_image_region.m_3rd.y - g_save_image_region.m_max.y;
165 const double f_y_skew = g_save_image_region.m_min.y - g_save_image_region.m_3rd.y;
166 const double f_x_adj = g_zoom_box_width * g_zoom_box_skew;
167
168 if (g_bf_math != BFMathType::NONE)
169 {
170 // do some calcs just once here to reduce fp work a bit
171 sub_bf(bf_f_x_width, g_bf_save_x_max, g_bf_save_x_3rd);
172 sub_bf(bf_f_x_skew, g_bf_save_x_3rd, g_bf_save_x_min);
173 sub_bf(bf_f_y_height, g_bf_save_y_3rd, g_bf_save_y_max);
174 sub_bf(bf_f_y_skew, g_bf_save_y_min, g_bf_save_y_3rd);
175 float_to_bf(bf_f_x_adj, f_x_adj);
176 }
177
178 // calc co-ords of topleft & botright corners of box
179 double tmp_x = g_zoom_box_width / -2 + f_x_adj; // from zoombox center as origin, on xdots scale
180 double tmp_y = g_zoom_box_height * g_viewport.final_aspect_ratio / 2;
181 double dx = rot_cos * tmp_x - rot_sin * tmp_y - tmp_x; // delta x to rotate topleft
182 double dy = tmp_y - (rot_sin * tmp_x + rot_cos * tmp_y); // delta y to rotate topleft
183
184 // calc co-ords of topleft
185 f_temp1 = g_zoom_box_x + dx + f_x_adj;
186 double f_temp2 = g_zoom_box_y + dy / g_viewport.final_aspect_ratio;
187
188 Coord top_left;
189 top_left.x = static_cast<int>(f_temp1 * (g_logical_screen.x_size_dots + PIXEL_ROUND)); // screen co-ords
190 top_left.y = static_cast<int>(f_temp2 * (g_logical_screen.y_size_dots + PIXEL_ROUND));
191 g_image_region.m_min.x = g_save_image_region.m_min.x + f_temp1*f_x_width + f_temp2*f_x_skew; // real co-ords
192 g_image_region.m_max.y = g_save_image_region.m_max.y + f_temp2*f_y_height + f_temp1*f_y_skew;
193 if (g_bf_math != BFMathType::NONE)
194 {
195 calc_corner(g_bf_x_min, g_bf_save_x_min, f_temp1, bf_f_x_width, f_temp2, bf_f_x_skew);
196 calc_corner(g_bf_y_max, g_bf_save_y_max, f_temp2, bf_f_y_height, f_temp1, bf_f_y_skew);
197 }
198
199 // calc co-ords of bottom right
200 f_temp1 = g_zoom_box_x + g_zoom_box_width - dx - f_x_adj;
201 f_temp2 = g_zoom_box_y - dy/g_viewport.final_aspect_ratio + g_zoom_box_height;
202 Coord bot_right;
203 bot_right.x = static_cast<int>(f_temp1 * (g_logical_screen.x_size_dots + PIXEL_ROUND));
204 bot_right.y = static_cast<int>(f_temp2 * (g_logical_screen.y_size_dots + PIXEL_ROUND));
205 g_image_region.m_max.x = g_save_image_region.m_min.x + f_temp1*f_x_width + f_temp2*f_x_skew;
206 g_image_region.m_min.y = g_save_image_region.m_max.y + f_temp2*f_y_height + f_temp1*f_y_skew;
207 if (g_bf_math != BFMathType::NONE)
208 {
209 calc_corner(g_bf_x_max, g_bf_save_x_min, f_temp1, bf_f_x_width, f_temp2, bf_f_x_skew);
210 calc_corner(g_bf_y_min, g_bf_save_y_max, f_temp2, bf_f_y_height, f_temp1, bf_f_y_skew);
211 }
212
213 // do the same for botleft & topright
214 tmp_x = g_zoom_box_width/-2 - f_x_adj;
215 tmp_y = 0.0-tmp_y;
216 dx = rot_cos * tmp_x - rot_sin * tmp_y - tmp_x;
217 dy = tmp_y - (rot_sin*tmp_x + rot_cos*tmp_y);
218 f_temp1 = g_zoom_box_x + dx - f_x_adj;
219 f_temp2 = g_zoom_box_y + dy/g_viewport.final_aspect_ratio + g_zoom_box_height;
220 Coord bot_left;
221 bot_left.x = static_cast<int>(f_temp1 * (g_logical_screen.x_size_dots + PIXEL_ROUND));
222 bot_left.y = static_cast<int>(f_temp2 * (g_logical_screen.y_size_dots + PIXEL_ROUND));
223 g_image_region.m_3rd.x = g_save_image_region.m_min.x + f_temp1*f_x_width + f_temp2*f_x_skew;
224 g_image_region.m_3rd.y = g_save_image_region.m_max.y + f_temp2*f_y_height + f_temp1*f_y_skew;
225 if (g_bf_math != BFMathType::NONE)
226 {
227 calc_corner(g_bf_x_3rd, g_bf_save_x_min, f_temp1, bf_f_x_width, f_temp2, bf_f_x_skew);
228 calc_corner(g_bf_y_3rd, g_bf_save_y_max, f_temp2, bf_f_y_height, f_temp1, bf_f_y_skew);
229 restore_stack(saved);
230 }
231 f_temp1 = g_zoom_box_x + g_zoom_box_width - dx + f_x_adj;
232 f_temp2 = g_zoom_box_y - dy/g_viewport.final_aspect_ratio;
233 Coord top_right;
234 top_right.x = static_cast<int>(f_temp1 * (g_logical_screen.x_size_dots + PIXEL_ROUND));
235 top_right.y = static_cast<int>(f_temp2 * (g_logical_screen.y_size_dots + PIXEL_ROUND));
236
237 if (g_box_count != 0)
238 {
239 // remove the old box from display
240 clear_box();
241 g_box_count = 0;
242 }
243
244 if (draw_it)
245 {
246 // caller wants box drawn as well as co-ords calc'd
247 // build the list of zoom box pixels
248 add_box(top_left);
249 add_box(top_right); // corner pixels
250 add_box(bot_left);
251 add_box(bot_right);
252 draw_lines(top_left, top_right, bot_left.x-top_left.x, bot_left.y-top_left.y); // top & bottom lines
253 draw_lines(top_left, bot_left, top_right.x-top_left.x, top_right.y-top_left.y); // left & right lines
254 display_box();
255 }

258void draw_lines(Coord fr, Coord to, const int dx, const int dy)
259{
260 if (std::abs(to.x-fr.x) > std::abs(to.y-fr.y))
261 {
262 // delta.x > delta.y
263 if (fr.x > to.x)
264 {
265 // swap so from.x is < to.x
266 const Coord tmp_p = fr;
267 fr = to;
268 to = tmp_p;
269 }
270 const int x_incr = (to.x-fr.x)*4/g_screen_x_dots+1; // do every 1st, 2nd, 3rd, or 4th dot
271 int ctr = (to.x - fr.x - 1) / x_incr;
272 const int alt_dec = std::abs(to.y - fr.y) * x_incr;
273 const int alt_inc = to.x-fr.x;
274 int alt_ctr = alt_inc / 2;
275 const int y_incr = to.y > fr.y ?1:-1;
276 Coord line1;
277 line1.y = fr.y;
278 line1.x = fr.x;
279 Coord line2;
280 line2.x = line1.x + dx;
281 line2.y = line1.y + dy;
282 while (--ctr >= 0)
283 {
284 line1.x += x_incr;
285 line2.x += x_incr;
286 alt_ctr -= alt_dec;
287 while (alt_ctr < 0)
288 {
289 alt_ctr += alt_inc;
290 line1.y += y_incr;
291 line2.y += y_incr;
292 }
293 add_box(line1);
294 add_box(line2);
295 }
296 }
297 else
298 {
299 // delta.y > delta.x
300 if (fr.y > to.y)
301 {
302 // swap so from.y is < to.y
303 const Coord tmp_p = fr;
304 fr = to;
305 to = tmp_p;
306 }
307 const int y_incr = (to.y-fr.y)*4/g_screen_y_dots+1; // do every 1st, 2nd, 3rd, or 4th dot
308 int ctr = (to.y - fr.y - 1) / y_incr;
309 const int alt_dec = std::abs(to.x - fr.x) * y_incr;
310 const int alt_inc = to.y-fr.y;
311 int alt_ctr = alt_inc / 2;
312 const int x_incr = to.x > fr.x ? 1 : -1;
313 Coord line1;
314 line1.x = fr.x;
315 line1.y = fr.y;
316 Coord line2;
317 line2.x = line1.x + dx;
318 line2.y = line1.y + dy;
319 while (--ctr >= 0)
320 {
321 line1.y += y_incr;
322 line2.y += y_incr;
323 alt_ctr -= alt_dec;
324 while (alt_ctr < 0)
325 {
326 alt_ctr += alt_inc;
327 line1.x += x_incr;
328 line2.x += x_incr;
329 }
330 add_box(line1);
331 add_box(line2);
332 }
333 }

350void move_box(const double dx, const double dy)
351{
352 const int align = check_pan();
353 if (dx != 0.0)
354 {
355 if ((g_zoom_box_x += dx) + g_zoom_box_width/2 < 0) // center must stay onscreen
356 {
357 g_zoom_box_x = g_zoom_box_width/-2;
358 }
359 if (g_zoom_box_x + g_zoom_box_width/2 > 1)
360 {
361 g_zoom_box_x = 1.0 - g_zoom_box_width/2;
362 }
363 if (align != 0)
364 {
365 if (int col = static_cast<int>(g_zoom_box_x * (g_logical_screen.x_size_dots + PIXEL_ROUND));
366 (col & align - 1) != 0)
367 {
368 if (dx > 0)
369 {
370 col += align;
371 }
372 col -= col & align - 1; // adjust col to pass alignment
373 g_zoom_box_x = static_cast<double>(col) / g_logical_screen.x_size_dots;
374 }
375 }
376 }
377 if (dy != 0.0)
378 {
379 if ((g_zoom_box_y += dy) + g_zoom_box_height/2 < 0)
380 {
381 g_zoom_box_y = g_zoom_box_height/-2;
382 }
383 if (g_zoom_box_y + g_zoom_box_height/2 > 1)
384 {
385 g_zoom_box_y = 1.0 - g_zoom_box_height/2;
386 }
387 if (align != 0)
388 {
389 if (int row = static_cast<int>(g_zoom_box_y * (g_logical_screen.y_size_dots + PIXEL_ROUND));
390 (row & align - 1) != 0)
391 {
392 if (dy > 0)
393 {
394 row += align;
395 }
396 row -= row & align - 1;
397 g_zoom_box_y = static_cast<double>(row) / g_logical_screen.y_size_dots;
398 }
399 }
400 }

404{
405 if (g_zoom_box_width + delta_width > 1.0)
406 {
407 delta_width = 1.0 - g_zoom_box_width;

415 {
416 delta_height = 1.0 - g_zoom_box_height;
417 }
418 if (g_zoom_box_height + delta_height < 0.05)
419 {
420 delta_height = 0.05 - g_zoom_box_height;
421 }
422 g_zoom_box_height += delta_height;
423 move_box(delta_width / -2.0, delta_height / -2.0); // keep it centered & check limits

427{
428 double delta_x;
429 double delta_y;

431 {
432 // box larger on y axis

434 delta_x = g_zoom_box_width * delta_y / g_zoom_box_height;
435 }
436 else
437 {
438 // box larger on x axis
439 delta_x = steps * 0.036;
440 delta_y = g_zoom_box_height * delta_x / g_zoom_box_width;
441 }
442 change_box(delta_x, delta_y);
443}

451static void zoom_out_calc(BigFloat bf_dx, BigFloat bf_dy, //
452 BigFloat bf_new_x, BigFloat bf_new_y, //
453 BigFloat bf_plot_mx1, BigFloat bf_plot_mx2, //
454 BigFloat bf_plot_my1, BigFloat bf_plot_my2, //
455 BigFloat bf_f_temp)
456{
457 BigStackSaver saved;
458 BigFloat b_tmp1 = alloc_stack(g_r_bf_length + 2);
459 BigFloat b_tmp2 = alloc_stack(g_r_bf_length + 2);
460 BigFloat b_tmp3 = alloc_stack(g_r_bf_length + 2);
461 BigFloat b_tmp4 = alloc_stack(g_r_bf_length + 2);
462 BigFloat b_tmp2a = alloc_stack(g_r_bf_length + 2);
463 BigFloat b_tmp4a = alloc_stack(g_r_bf_length + 2);
464 BigFloat b_temp_x = alloc_stack(g_r_bf_length + 2);
465 BigFloat b_temp_y = alloc_stack(g_r_bf_length + 2);
466
467 /* calc cur screen corner relative to zoombox, when zoombox co-ords
468 are taken as (0,0) topleft through (1,1) bottom right */
469
470 // tempx = dy * plotmx1 - dx * plotmx2;
471 mult_bf(b_tmp1, bf_dy, bf_plot_mx1);
472 mult_bf(b_tmp2, bf_dx, bf_plot_mx2);
473 sub_bf(b_temp_x, b_tmp1, b_tmp2);
474
475 // tempy = dx * plotmy1 - dy * plotmy2;
476 mult_bf(b_tmp1, bf_dx, bf_plot_my1);
477 mult_bf(b_tmp2, bf_dy, bf_plot_my2);
478 sub_bf(b_temp_y, b_tmp1, b_tmp2);
479
480 // calc new corner by extending from current screen corners
481 // *newx = sxmin + tempx*(sxmax-sx3rd)/ftemp + tempy*(sx3rd-sxmin)/ftemp;
482 sub_bf(b_tmp1, g_bf_save_x_max, g_bf_save_x_3rd);
483 mult_bf(b_tmp2, b_temp_x, b_tmp1);
484 div_bf(b_tmp2a, b_tmp2, bf_f_temp);
485 sub_bf(b_tmp3, g_bf_save_x_3rd, g_bf_save_x_min);
486 mult_bf(b_tmp4, b_temp_y, b_tmp3);
487 div_bf(b_tmp4a, b_tmp4, bf_f_temp);
488 add_bf(bf_new_x, g_bf_save_x_min, b_tmp2a);
489 add_a_bf(bf_new_x, b_tmp4a);
490
491 // *newy = symax + tempy*(sy3rd-symax)/ftemp + tempx*(symin-sy3rd)/ftemp;
492 sub_bf(b_tmp1, g_bf_save_y_3rd, g_bf_save_y_max);
493 mult_bf(b_tmp2, b_temp_y, b_tmp1);
494 div_bf(b_tmp2a, b_tmp2, bf_f_temp);
495 sub_bf(b_tmp3, g_bf_save_y_min, g_bf_save_y_3rd);
496 mult_bf(b_tmp4, b_temp_x, b_tmp3);
497 div_bf(b_tmp4a, b_tmp4, bf_f_temp);
498 add_bf(bf_new_y, g_bf_save_y_max, b_tmp2a);
499 add_a_bf(bf_new_y, b_tmp4a);
500}

514static void zoom_out_bf() // for ctl-enter, calc corners for zooming out
515{
516 /* (xxmin,yymax), etc., are already set to zoombox corners;
517 (sxmin,symax), etc., are still the screen's corners;
518 use the same logic as plot_orbit stuff to first calculate current screen
519 corners relative to the zoombox, as if the zoombox were a square with

606void aspect_ratio_crop(const float old_aspect, const float new_aspect)
607{
608 double f_temp;
609 double x_margin;
610 double y_margin;
611 if (new_aspect > old_aspect)
612 {
613 // new ratio is taller, crop x
614 f_temp = (1.0 - old_aspect / new_aspect) / 2;
615 x_margin = (g_image_region.m_max.x - g_image_region.m_3rd.x) * f_temp;
616 y_margin = (g_image_region.m_min.y - g_image_region.m_3rd.y) * f_temp;
617 g_image_region.m_3rd.x += x_margin;
618 g_image_region.m_3rd.y += y_margin;
619 }
620 else
621 {
622 // new ratio is wider, crop y
623 f_temp = (1.0 - new_aspect / old_aspect) / 2;
624 const DComplex size3{g_image_region.size3()};
625 x_margin = size3.x * f_temp;
626 y_margin = size3.y * f_temp;
627 g_image_region.m_3rd.x -= x_margin;
628 g_image_region.m_3rd.y -= y_margin;
629 }
630 g_image_region.m_min.x += x_margin;
631 g_image_region.m_max.y += y_margin;
632 g_image_region.m_max.x -= x_margin;
633 g_image_region.m_min.y -= y_margin;
634}

636static int check_pan() // return 0 if can't, alignment requirement if can
637{
638 if ((g_calc_status != CalcStatus::RESUMABLE && g_calc_status != CalcStatus::COMPLETED) ||
639 g_evolving != EvolutionModeFlags::NONE)
640 {
641 return 0; // not resumable, not complete
642 }
643 if (g_cur_fractal_specific->calc_type != standard_fractal_type //
644 && g_cur_fractal_specific->calc_type != calc_mandelbrot_type //
645 && g_cur_fractal_specific->calc_type != lyapunov_type //
646 && g_cur_fractal_specific->calc_type != froth_type)
647 {
648 return 0; // not a worklist-driven type
649 }
650 if (g_zoom_box_width != 1.0 || g_zoom_box_height != 1.0
651 || g_zoom_box_skew != 0.0 || g_zoom_box_rotation != 0.0)
652 {
653 return 0; // not a full size unrotated unskewed zoom box
654 }
655 if (g_std_calc_mode == CalcMode::TESSERAL)
656 {
657 return 0; // tesseral, can't do it
658 }
659 if (g_std_calc_mode == CalcMode::DIFFUSION)
660 {
661 return 0; // diffusion scan: can't do it either
662 }
663 if (g_std_calc_mode == CalcMode::ORBIT)
664 {
665 return 0; // orbits, can't do it
666 }
667
668 // can pan if we get this far
669
670 if (g_calc_status == CalcStatus::COMPLETED)
671 {
672 return 1; // image completed, align on any pixel
673 }
674 if (g_potential.flag && g_potential.store_16bit)
675 {
676 return 1; // 1 pass forced so align on any pixel
677 }
678 if (g_std_calc_mode == CalcMode::BOUNDARY_TRACE)
679 {
680 return 1; // boundary trace method, align on any pixel
681 }
682 if (g_std_calc_mode != CalcMode::SOLID_GUESS ||
683 bit_set(g_cur_fractal_specific->flags, FractalFlags::NO_GUESS))
684 {
685 // align on even pixel for 2pass
686 if (g_std_calc_mode == CalcMode::TWO_PASS || g_std_calc_mode == CalcMode::THREE_PASS)
687 {
688 return 2;
689 }
690 return 1; // assume 1pass
691 }
692 // solid guessing
693 start_resume();
694 get_resume_len(sizeof(g_num_work_list), &g_num_work_list, sizeof(g_work_list), g_work_list, 0);
695 // don't do end_resume! we're just looking
696 int i = 9;
697 for (int j = 0; j < g_num_work_list; ++j) // find the lowest pass in any pending window
698 {
699 i = std::min(g_work_list[j].pass, i);
700 }
701 int j = ssg_block_size(); // worst-case alignment requirement
702 while (--i >= 0)
703 {
704 j = j >> 1; // reduce requirement
705 }
706 return j;
707}

732// decide to recalc, or to chg worklist & pan
733void init_pan_or_recalc(const bool do_zoom_out)
734{
735 // no zoombox, leave g_calc_status as is
736 if (g_zoom_box_width == 0.0)
737 {
738 return;
739 }
740 // got a zoombox
741 const int align_mask = check_pan() - 1;
742
743 // can't pan, trigger recalc
744 if (align_mask < 0 || g_evolving != EvolutionModeFlags::NONE)
745 {
746 g_calc_status = CalcStatus::PARAMS_CHANGED;
747 return;
748 }
749
750 // box is full screen, leave g_calc_status as is
751 if (g_zoom_box_x == 0.0 && g_zoom_box_y == 0.0)
752 {
753 clear_box();
754 return;
755 }
756
757 int col = static_cast<int>(
758 g_zoom_box_x * (g_logical_screen.x_size_dots + PIXEL_ROUND)); // calc dest col,row of topleft pixel
759 int row = static_cast<int>(g_zoom_box_y * (g_logical_screen.y_size_dots + PIXEL_ROUND));
760 if (do_zoom_out)
761 {
762 // invert row and col
763 row = -row;
764 col = -col;
765 }
766 if ((row&align_mask) != 0 || (col&align_mask) != 0)
767 {
768 g_calc_status = CalcStatus::PARAMS_CHANGED; // not on useable pixel alignment, trigger recalc
769 return;
770 }
771
772 // pan
773 g_num_work_list = 0;
774 if (g_calc_status == CalcStatus::RESUMABLE)
775 {
776 start_resume();
777 get_resume_len(sizeof(g_num_work_list), &g_num_work_list, sizeof(g_work_list), g_work_list, 0);
778 // don't do end_resume! we might still change our mind
779 }
780 // adjust existing worklist entries
781 for (int i = 0; i < g_num_work_list; ++i)
782 {
783 g_work_list[i].start.y -= row;
784 g_work_list[i].stop.y -= row;
785 g_work_list[i].begin.y -= row;
786 g_work_list[i].start.x -= col;
787 g_work_list[i].stop.x -= col;
788 g_work_list[i].begin.x -= col;
789 }
790 // add worklist entries for the new edges
791 bool list_full{};
792 int i = 0;
793 int j = g_logical_screen.y_dots-1;
794 if (row < 0)
795 {
796 list_full = add_work_list(0, 0, g_logical_screen.x_dots - 1, 0 - row - 1, 0, 0, 0, 0) || list_full;
797 i = -row;
798 }
799 if (row > 0)
800 {
801 list_full = add_work_list(0, g_logical_screen.y_dots - row, g_logical_screen.x_dots - 1,
802 g_logical_screen.y_dots - 1, 0, g_logical_screen.y_dots - row, 0, 0) || list_full;
803 j = g_logical_screen.y_dots - row - 1;
804 }
805 if (col < 0)
806 {
807 list_full = add_work_list(0, i, 0 - col - 1, j, 0, i, 0, 0) || list_full;
808 }
809 if (col > 0)
810 {
811 list_full = add_work_list(g_logical_screen.x_dots - col, i, g_logical_screen.x_dots - 1, j,
812 g_logical_screen.x_dots - col, i, 0, 0) || list_full;
813 }
814 if (list_full)
815 {
816 if (stop_msg(StopMsgFlags::CANCEL,
817 "Tables full, can't pan current image.\n"
818 "Cancel resumes old image, continue pans and calculates a new one."))
819 {
820 g_zoom_box_width = 0; // cancel the zoombox
821 draw_box(true);
822 }
823 else
824 {
825 g_calc_status = CalcStatus::PARAMS_CHANGED; // trigger recalc
826 }
827 return;
828 }
829 // now we're committed
830 g_calc_status = CalcStatus::RESUMABLE;
831 clear_box();
832 if (row > 0) // move image up
833 {
834 for (int y = 0; y < g_logical_screen.y_dots; ++y)
835 {
836 move_row(y+row, y, col);
837 }
838 }
839 else // move image down
840 {
841 for (int y = g_logical_screen.y_dots; --y >=0;)
842 {
843 move_row(y+row, y, col);
844 }
845 }
846 fix_work_list(); // fixup any out of bounds worklist entries
847 alloc_resume(sizeof(g_work_list)+20, 2); // post the new worklist
848 put_resume_len(sizeof(g_num_work_list), &g_num_work_list, sizeof(g_work_list), g_work_list, 0);
849}

853{
854 const int y_start = std::max(0, g_work_list[index].start.y);
855 const int x_start = std::max(0, g_work_list[index].start.x);

869// fix out of bounds and symmetry related stuff
870static void fix_work_list()
871{
872 for (int i = 0; i < g_num_work_list; ++i)
873 {
874 WorkList *wk = &g_work_list[i];
875 if (wk->start.y >= g_logical_screen.y_dots || wk->start.y < 0
876 || wk->start.x >= g_logical_screen.x_dots || wk->stop.x < 0)
877 {
878 // offscreen, delete
879 for (int j = i+1; j < g_num_work_list; ++j)
880 {
881 g_work_list[j-1] = g_work_list[j];
882 }
883 --g_num_work_list;
884 --i;
885 continue;
886 }
887 if (wk->start.y < 0)
888 {
889 // partly off top edge
890 if ((wk->symmetry&1) == 0)
891 {
892 // no sym, easy
893 wk->start.y = 0;
894 wk->begin.x = 0;
895 }
896 else
897 {
898 // xaxis symmetry
899 int j = wk->start.y + wk->start.y;
900 if (j > 0 && g_num_work_list < MAX_CALC_WORK)
901 {
902 // split the sym part
903 g_work_list[g_num_work_list] = g_work_list[i];
904 g_work_list[g_num_work_list].start.y = 0;
905 g_work_list[g_num_work_list++].stop.y = j;
906 wk->start.y = j+1;
907 }
908 else
909 {
910 wk->start.y = 0;
911 }
912 restart_window(i); // restart the no-longer sym part
913 }
914 }
915 if (wk->start.y >= g_logical_screen.y_dots)
916 {
917 // partly off bottom edge
918 int j = g_logical_screen.y_dots-1;
919 if ((wk->symmetry&1) != 0)
920 {
921 // uses xaxis symmetry
922 int k = wk->start.y + (wk->start.y - j);
923 if (k < j)
924 {
925 if (g_num_work_list >= MAX_CALC_WORK) // no room to split
926 {
927 restart_window(i);
928 }
929 else
930 {
931 // split it
932 g_work_list[g_num_work_list] = g_work_list[i];
933 g_work_list[g_num_work_list].start.y = k;
934 g_work_list[g_num_work_list++].stop.y = j;
935 j = k-1;
936 }
937 }
938 wk->symmetry &= -1 - 1;
939 }
940 wk->start.y = j;
941 }
942 if (wk->start.x < 0)
943 {
944 // partly off left edge
945 if ((wk->symmetry&2) == 0) // no sym, easy
946 {
947 wk->start.x = 0;
948 }
949 else
950 {
951 // yaxis symmetry
952 int j = wk->stop.x + wk->start.x;
953 if (j > 0 && g_num_work_list < MAX_CALC_WORK)
954 {
955 // split the sym part
956 g_work_list[g_num_work_list] = g_work_list[i];
957 g_work_list[g_num_work_list].start.x = 0;
958 g_work_list[g_num_work_list++].stop.x = j;
959 wk->start.x = j+1;
960 }
961 else
962 {
963 wk->start.x = 0;
964 }
965 restart_window(i); // restart the no-longer sym part
966 }
967 }
968 if (wk->stop.x >= g_logical_screen.x_dots)
969 {
970 // partly off right edge
971 int j = g_logical_screen.x_dots-1;
972 if ((wk->symmetry&2) != 0)
973 {
974 // uses xaxis symmetry
975 int k = wk->start.x + (wk->stop.x - j);
976 if (k < j)
977 {
978 if (g_num_work_list >= MAX_CALC_WORK) // no room to split
979 {
980 restart_window(i);
981 }
982 else
983 {
984 // split it
985 g_work_list[g_num_work_list] = g_work_list[i];
986 g_work_list[g_num_work_list].start.x = k;
987 g_work_list[g_num_work_list++].stop.x = j;
988 j = k-1;
989 }
990 }
991 wk->symmetry &= -1 - 2;
992 }
993 wk->stop.x = j;
994 }
995 wk->begin.y = std::max(wk->begin.y, wk->start.y);
996 wk->begin.y = std::min(wk->begin.y, wk->start.y);
997 wk->begin.x = std::max(wk->begin.x, wk->start.x);
998 wk->begin.x = std::min(wk->begin.x, wk->stop.x);
999 }
1000 tidy_work_list(); // combine where possible, re-sort
1001}
