170void StandardFractal::suspend()
171{
172 m_standard_pass.suspend();
173 s_standard_pass_status = standard_pass_status();
174 if (m_perturbation_active)
175 {
176 m_pert_engine.finish();
177 if (m_perturbation_retry_active || !m_perturbation_retry_points.empty())
178 {
179 add_work_list(g_start_pt, g_stop_pt, g_begin_pt, g_work_pass, g_work_symmetry);
180 }
181 m_perturbation_active = false;
182 m_perturbation_retry_active = false;
183 m_perturbation_retry_index = 0;
184 m_perturbation_retry_points.clear();
185 }
186 if (g_num_work_list > 0)
187 {
188 alloc_resume(sizeof(g_work_list) + 20, 2);
189 put_resume_len(sizeof(g_num_work_list), &g_num_work_list, sizeof(g_work_list), g_work_list, 0);
190 }
191 else
192 {
193 g_calc_status = CalcStatus::COMPLETED;
194 }
195 if (m_work_item_active)
196 {
197 cleanup_standard_fractal_show_dot();
198 m_work_item_active = false;
199 }
200 m_standard_pass.reset();
201 clear_standard_pixel();
202 m_perturbation_active = false;
203 complete();
204}

335void StandardFractal::run_current_work_item()
336{
337 if (!m_work_list_started)
338 {
339 start_work_list();
340 }
341 if (!m_work_item_active && g_num_work_list <= 0)
342 {
343 finish_work_list();
344 return;
345 }
346
347 if (!m_work_item_active)
348 {
349 fractals::g_dispatch.init_calc_type(*fractals::g_cur_fractal_specific); // per_image can override
350 g_symmetry = fractals::g_cur_fractal_specific->symmetry; // table symmetry
351 g_plot = g_put_color; // defaults when set symmetry not called or does nothing
352
353 g_start_pt = g_work_list[0].start;
354 g_i_start_pt = g_work_list[0].start;
355 g_stop_pt = g_work_list[0].stop;
356 g_i_stop_pt = g_work_list[0].stop;
357 g_begin_pt = g_work_list[0].begin;
358 g_work_pass = g_work_list[0].pass;
359 g_work_symmetry = g_work_list[0].symmetry;
360 pop_work_list_front();
361
362 g_calc_status = CalcStatus::IN_PROGRESS;
363
364 fractals::per_image();
365 m_perturbation_requested = perturbation_requested(m_requested_calc_mode);
366 setup_standard_fractal_show_dot();
367
368 g_close_enough = g_delta_min * std::pow(2.0, -static_cast<double>(std::abs(g_periodicity_check)));
369 g_keyboard_check_interval = g_max_keyboard_check_interval;
370
371 set_symmetry(g_symmetry, true);
372
373 if (!g_resuming && (std::abs(g_log_map_flag) == 2 || (g_log_map_flag && g_log_map_auto_calculate)))
374 {
375 g_log_map_flag = auto_log_map() * (g_log_map_flag / std::abs(g_log_map_flag));
376 setup_log_table();
377 }
378 m_work_item_active = true;
379 }
380
381 run_current_work_item_mode();
382
383 if (done())
384 {
385 return;
386 }
387 if (m_work_item_yielded)
388 {
389 return;
390 }
391 cleanup_standard_fractal_show_dot();
392 m_standard_pass.reset();
393 m_work_item_active = false;
394 if (g_num_work_list == 0)
395 {
396 finish_work_list();
397 }
398}

400void StandardFractal::run_current_work_item_mode()
401{
402 if (m_perturbation_retry_active || !m_perturbation_retry_points.empty())
403 {
404 run_perturbation_glitch_retries();
405 return;
406 }
407 if (m_perturbation_requested && perturbation_supported() && !m_perturbation_active)
408 {
409 start_perturbation_frame();
410 fractals::g_dispatch.set_calc_type(standard_fractal_type);
411 }
412
413 m_standard_pass.select(g_std_calc_mode);
414 const bool pass_completed{m_standard_pass.iterate()};
415 s_standard_pass_status = standard_pass_status();
416 if (!pass_completed)
417 {
418 m_work_item_yielded = true;
419 }
420 else
421 {
422 m_standard_pass.reset();
423 if (m_perturbation_active)
424 {
425 m_perturbation_retry_points = m_pert_engine.take_glitch_points();
426 m_perturbation_retry_index = 0;
427 run_perturbation_glitch_retries();
428 }
429 }

432void StandardFractal::run_perturbation_glitch_retries()
433{
434 if (!m_perturbation_retry_active)
435 {
436 if (!m_pert_engine.select_retry_reference(m_perturbation_retry_points))
437 {
438 m_pert_engine.finish();
439 m_perturbation_active = false;
440 m_perturbation_retry_index = 0;
441 m_perturbation_retry_points.clear();
442 return;
443 }
444 m_perturbation_retry_active = true;
445 }
446
447 const std::size_t stop_index{
448 std::min(m_perturbation_retry_index + PERTURBATION_RETRY_POINTS_PER_CHUNK, m_perturbation_retry_points.size())};
449 for (; m_perturbation_retry_index < stop_index; ++m_perturbation_retry_index)
450 {
451 m_pert_engine.calculate_pixel(m_perturbation_retry_points[m_perturbation_retry_index]);
452 }
453
454 if (m_perturbation_retry_index < m_perturbation_retry_points.size())
455 {
456 m_work_item_yielded = true;
457 return;
458 }
459
460 m_perturbation_retry_points = m_pert_engine.take_glitch_points();
461 m_perturbation_retry_index = 0;
462 m_perturbation_retry_active = false;
463 if (m_pert_engine.retry_needed(m_perturbation_retry_points))
464 {
465 m_work_item_yielded = true;
466 return;
467 }
468
469 m_pert_engine.finish();
470 m_perturbation_active = false;
471 m_perturbation_retry_points.clear();
472}

474void StandardFractal::start_next_pass()
475{
476 if (m_requested_calc_mode == CalcMode::THREE_PASS)
477 {

486 g_std_calc_mode = followup_calc_mode();

492 g_three_pass = false;

494 m_after_work_list = AfterWorkList::COMPLETE;
495 }
496
497 m_work_list_started = false;
498 m_phase = Phase::WORK_LIST;

510void StandardFractal::start_work_list()
511{
512 if (!m_dispatch_saved)
513 {
514 m_saved_dispatch = fractals::g_dispatch;
515 m_dispatch_saved = true;
516 }
517 select_alternate_math_dispatch();
518
519 if (g_potential.flag && g_potential.store_16bit)
520 {
521 const CalcMode tmp_calc_mode = g_std_calc_mode;
522
523 g_std_calc_mode = CalcMode::ONE_PASS;
524 if (!g_resuming)
525 {
526 if (ui::pot_start_disk() < 0)
527 {
528 g_potential.store_16bit = false;
529 g_std_calc_mode = tmp_calc_mode;
530 }
531 }
532 }
533 if (g_std_calc_mode == CalcMode::BOUNDARY_TRACE &&
534 bit_set(fractals::g_cur_fractal_specific->flags, fractals::FractalFlags::NO_TRACE))
535 {
536 g_std_calc_mode = CalcMode::ONE_PASS;
537 }
538 if (g_std_calc_mode == CalcMode::SOLID_GUESS &&
539 bit_set(fractals::g_cur_fractal_specific->flags, fractals::FractalFlags::NO_GUESS))
540 {
541 g_std_calc_mode = CalcMode::ONE_PASS;
542 }
543 if (g_std_calc_mode == CalcMode::ORBIT && !calc_type_supports_orbit_mode())
544 {
545 g_std_calc_mode = CalcMode::ONE_PASS;
546 }
547
548 g_num_work_list = 0;
549 add_work_list({0, 0}, {g_logical_screen.x_dots - 1, g_logical_screen.y_dots - 1}, {0, 0}, 0, 0);
550 if (g_resuming)
551 {
552 const int version = start_resume();
553 get_resume_len(sizeof(g_num_work_list), &g_num_work_list, sizeof(g_work_list), g_work_list, 0);
554 end_resume();
555 if (version < 2)
556 {
557 g_begin_pt.x = 0;
558 }
559 }
560
561 if (g_distance_estimator)
562 {
563 setup_standard_fractal_distance_estimator();
564 }
565 m_work_list_started = true;
566}
