43void set_default_params()
44{
45 g_image_region.m_min.x = g_cur_fractal_specific->x_min;
46 g_image_region.m_max.x = g_cur_fractal_specific->x_max;
47 g_image_region.m_min.y = g_cur_fractal_specific->y_min;
48 g_image_region.m_max.y = g_cur_fractal_specific->y_max;
49 g_image_region.m_3rd.x = g_image_region.m_min.x;
50 g_image_region.m_3rd.y = g_image_region.m_min.y;
51
52 if (g_viewport.crop && g_viewport.final_aspect_ratio != g_screen_aspect)
53 {
54 aspect_ratio_crop(g_screen_aspect, g_viewport.final_aspect_ratio);
55 }
56 for (int i = 0; i < 4; i++)
57 {
58 g_params[i] = g_cur_fractal_specific->params[i];
59 if (g_fractal_type != FractalType::CELLULAR //
60 && g_fractal_type != FractalType::FROTHY_BASIN //
61 && g_fractal_type != FractalType::ANT)
62 {
63 round_float_double(&g_params[i]); // don't round cellular, frothybasin or ant
64 }
65 }
66 int extra = find_extra_param(g_fractal_type);
67 if (extra > -1)
68 {
69 for (int i = 0; i < MAX_PARAMS - 4; i++)
70 {
71 g_params[i + 4] = g_more_fractal_params[extra].params[i];
72 }
73 }
74 if (g_debug_flag != DebugFlags::FORCE_ARBITRARY_PRECISION_MATH)
75 {
76 g_bf_math = BFMathType::NONE;
77 }
78 else if (g_bf_math != BFMathType::NONE)
79 {
80 fractal_float_to_bf();
81 }
82}
