93static void which_disk_error(const int status)
94{
95 // Set status == 1 after a file create, status == 2 after a file set value
96 // Set status == 3 after a file write, status == 4 after a file read
97 if (g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
98 {
99 const char *names[4]{"Create", "Set", "Write", "Read"};
100 if (stop_msg(StopMsgFlags::CANCEL | StopMsgFlags::NO_BUZZER,
101 fmt::format("{:s} file error {:d}: {:s}",
102 names[status >= 1 && status <= 4 ? status - 1 : 0], errno, std::strerror(errno))))
103 {
104 goodbye(); // bailout if ESC
105 }
106 }
107}

109// buffer is a pointer to local memory
110// Always start moving from the beginning of buffer
111// offset is the number of units from the start of the allocated "Memory"
112// to start moving the contents of buffer to
113// size is the size of the unit, count is the number of units to move
114// Returns true if successful, false if failure
115bool MemoryHandle::from_memory(const Byte *buffer, const U16 size, const long count, const long offset)
116{
117 Byte disk_buff[DISK_WRITE_LEN];
118 U16 num_written;
119 const long start = offset * size; // offset to first location to move to
120 long to_move = count * size; // number of bytes to move
121 if (g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
122 {
123 if (check_bounds(start, to_move, index))
124 {
125 return false; // out of bounds, don't do it
126 }
127 }
128 bool success = false;
129 switch (s_handles[index].stored_at)
130 {
131 case MemoryLocation::NOWHERE: // MoveToMemory
132 display_handle(index);
133 break;
134
135 case MemoryLocation::MEMORY: // MoveToMemory
136#if defined(_WIN32)
137 _ASSERTE(s_handles[index].size >= size * count + start);
138#endif
139 std::memcpy(s_handles[index].linear.memory + start, buffer, size * count);
140 success = true; // No way to gauge success or failure
141 break;
142
143 case MemoryLocation::DISK: // MoveToMemory
144 std::fseek(s_handles[index].disk.file, start, SEEK_SET);
145 while (to_move > DISK_WRITE_LEN)
146 {
147 std::memcpy(disk_buff, buffer, DISK_WRITE_LEN);
148 num_written = static_cast<U16>(std::fwrite(disk_buff, DISK_WRITE_LEN, 1, s_handles[index].disk.file));
149 if (num_written != 1)
150 {
151 which_disk_error(3);
152 goto disk_error;
153 }
154 to_move -= DISK_WRITE_LEN;
155 buffer += DISK_WRITE_LEN;
156 }
157 std::memcpy(disk_buff, buffer, static_cast<U16>(to_move));
158 num_written = static_cast<U16>(std::fwrite(disk_buff, static_cast<U16>(to_move), 1, s_handles[index].disk.file));
159 if (num_written != 1)
160 {
161 which_disk_error(3);
162 break;
163 }
164 success = true;
165disk_error:
166 break;
167 } // end of switch
168 if (!success && g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
169 {
170 display_handle(index);
171 }
172 return success;
173}

175// buffer points is the location to move the data to
176// offset is the number of units from the beginning of buffer to start moving
177// size is the size of the unit, count is the number of units to move
178// Returns true if successful, false if failure
179bool MemoryHandle::to_memory(Byte *buffer, const U16 size, const long count, const long offset)
180{
181 Byte disk_buff[DISK_WRITE_LEN];
182 U16 num_read;
183 long start = offset * size;// first location to move
184 long to_move = count * size; // number of bytes to move
185 if (g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
186 {
187 if (check_bounds(start, to_move, index))
188 {
189 return false; // out of bounds, don't do it
190 }
191 }
192 bool success = false;
193 switch (s_handles[index].stored_at)
194 {
195 case MemoryLocation::NOWHERE: // MoveFromMemory
196 display_handle(index);
197 break;
198
199 case MemoryLocation::MEMORY: // MoveFromMemory
200 for (int i = 0; i < size; i++)
201 {
202 std::memcpy(buffer, s_handles[index].linear.memory + start, static_cast<U16>(count));
203 start += count;
204 buffer += count;
205 }
206 success = true; // No way to gauge success or failure
207 break;
208
209 case MemoryLocation::DISK: // MoveFromMemory
210 std::fseek(s_handles[index].disk.file, start, SEEK_SET);
211 while (to_move > DISK_WRITE_LEN)
212 {
213 num_read = static_cast<U16>(std::fread(disk_buff, DISK_WRITE_LEN, 1, s_handles[index].disk.file));
214 if (num_read != 1 && !std::feof(s_handles[index].disk.file))
215 {
216 which_disk_error(4);
217 goto disk_error;
218 }
219 std::memcpy(buffer, disk_buff, DISK_WRITE_LEN);
220 to_move -= DISK_WRITE_LEN;
221 buffer += DISK_WRITE_LEN;
222 }
223 num_read = static_cast<U16>(std::fread(disk_buff, static_cast<U16>(to_move), 1, s_handles[index].disk.file));
224 if (num_read != 1 && !std::feof(s_handles[index].disk.file))
225 {
226 which_disk_error(4);
227 break;
228 }
229 std::memcpy(buffer, disk_buff, static_cast<U16>(to_move));
230 success = true;
231disk_error:
232 break;
233 } // end of switch
234 if (!success && g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
235 {
236 display_handle(index);
237 }
238 return success;
239}

320// This function returns an adjusted stored_at value.
321// This is where the memory requested can be allocated.
322static MemoryLocation check_for_mem(MemoryLocation where, const std::uint64_t size)
323{
324 if (g_debug_flag == DebugFlags::FORCE_MEMORY_FROM_DISK)
325 {
326 where = MemoryLocation::DISK;
327 }
328 if (g_debug_flag == DebugFlags::FORCE_MEMORY_FROM_MEMORY)
329 {
330 where = MemoryLocation::MEMORY;
331 }
332
333 MemoryLocation result{MemoryLocation::NOWHERE};
334 switch (where)
335 {
336 case MemoryLocation::MEMORY:
337 if (void *temp = std::malloc(size); temp != nullptr)
338 {
339 std::free(temp);
340 result = MemoryLocation::MEMORY;
341 break;
342 }
343 // failed, fall through, not enough heap available
344
345 case MemoryLocation::DISK: // NOLINT(clang-diagnostic-implicit-fallthrough)
346 if (check_disk_space(size))
347 {
348 result = MemoryLocation::DISK;
349 break;
350 }
351 // failed, fall through, no memory available
352
353 case MemoryLocation::NOWHERE: // NOLINT(clang-diagnostic-implicit-fallthrough)
354 result = MemoryLocation::NOWHERE;
355 break;
356 }
357
358 return result;
359}

372static int check_bounds(const long start, const long length, const U16 handle)
373{
374 if (s_handles[handle].size < static_cast<std::uint64_t>(start - length))
375 {
376 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER, "Memory reference out of bounds.");
377 display_handle(handle);
378 return 1;
379 }
380 if (length > static_cast<long>(std::numeric_limits<unsigned short>::max()))
381 {
382 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER, "Tried to move > 65,535 bytes.");
383 display_handle(handle);
384 return 1;
385 }
386 if (s_handles[handle].stored_at == MemoryLocation::DISK && stack_avail() <= DISK_WRITE_LEN)
387 {
388 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER, "Stack space insufficient for disk memory.");
389 display_handle(handle);
390 return 1;
391 }
392 if (length <= 0)
393 {
394 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER, "Zero or negative length.");
395 display_handle(handle);
396 return 1;
397 }
398 if (start < 0)
399 {
400 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER, "Negative offset.");
401 display_handle(handle);
402 return 1;
403 }
404 return 0;
405}

461// Returns handle number if successful, 0 or nullptr if failure
462MemoryHandle memory_alloc(const U16 size, const long count, const MemoryLocation stored_at)
463{
464 std::uint64_t to_allocate = count * size;
465
466 /* check structure for requested memory type (add em up) to see if
467 sufficient amount is available to grant request */
468
469 MemoryLocation use_this_type = check_for_mem(stored_at, to_allocate);
470 if (use_this_type == MemoryLocation::NOWHERE)
471 {
472 display_error(stored_at, to_allocate);
473 goodbye();
474 }
475
476 // get next available handle
477
478 U16 handle = next_handle();
479
480 if (handle >= MAX_HANDLES || handle == 0)
481 {
482 display_handle(handle);
483 return {};
484 // Oops, do something about this! ?????
485 }
486
487 // attempt to allocate requested memory type
488 bool success = false;
489 switch (use_this_type)
490 {
491 default:
492 case MemoryLocation::NOWHERE: // MemoryAlloc
493 use_this_type = MemoryLocation::NOWHERE; // in case nonsense value is passed
494 break;
495
496 case MemoryLocation::MEMORY: // MemoryAlloc
497 // Availability of memory checked in check_for_mem()
498 s_handles[handle].linear.memory = static_cast<Byte *>(malloc(to_allocate));
499 s_handles[handle].size = to_allocate;
500 s_handles[handle].stored_at = MemoryLocation::MEMORY;
501 s_num_total_handles++;
502 success = true;
503 break;
504
505 case MemoryLocation::DISK: // MemoryAlloc
506 if (g_disk_targa)
507 {
508 s_handles[handle].disk.file = dir_fopen(g_working_dir, g_light_name, "a+b");
509 }
510 else
511 {
512 s_handles[handle].disk.file = dir_fopen(g_temp_dir, mem_filename(handle), "w+b");
513 }
514 std::fseek(s_handles[handle].disk.file, 0, SEEK_SET);
515 if (std::fseek(s_handles[handle].disk.file, to_allocate, SEEK_SET) != 0)
516 {
517 s_handles[handle].disk.file = nullptr;
518 }
519 if (s_handles[handle].disk.file == nullptr)
520 {
521 s_handles[handle].stored_at = MemoryLocation::NOWHERE;
522 use_this_type = MemoryLocation::NOWHERE;
523 which_disk_error(1);
524 display_memory();
525 driver_buzzer(Buzzer::PROBLEM);
526 break;
527 }
528 s_num_total_handles++;
529 success = true;
530 std::fclose(s_handles[handle].disk.file); // so clusters aren't lost if we crash while running
531 s_handles[handle].disk.file = g_disk_targa ?
532 dir_fopen(g_working_dir, g_light_name, "r+b") :
533 dir_fopen(g_temp_dir, mem_filename(handle), "r+b");
534 // cppcheck-suppress useClosedFile
535 std::fseek(s_handles[handle].disk.file, 0, SEEK_SET);
536 s_handles[handle].size = to_allocate;
537 s_handles[handle].stored_at = MemoryLocation::DISK;
538 use_this_type = MemoryLocation::DISK;
539 break;
540 } // end of switch
541
542 if (stored_at != use_this_type && g_debug_flag == DebugFlags::DISPLAY_MEMORY_STATISTICS)
543 {
544 stop_msg(StopMsgFlags::INFO_ONLY | StopMsgFlags::NO_BUZZER,
545 fmt::format("Asked for {:s}, allocated {:d} bytes of {:s}, handle = {:d}.", //
546 memory_type(stored_at), to_allocate, memory_type(use_this_type), handle));
547 display_memory();
548 }
549
550 if (success)
551 {
552 return {handle};
553 }
554 return {};
555}

557void memory_release(const MemoryHandle handle)
558{
559 const U16 index{handle.index};
560 switch (s_handles[index].stored_at)
561 {
562 case MemoryLocation::NOWHERE: // MemoryRelease
563 break;
564
565 case MemoryLocation::MEMORY: // MemoryRelease
566 free(s_handles[index].linear.memory);
567 s_handles[index].linear.memory = nullptr;
568 s_handles[index].size = 0;
569 s_handles[index].stored_at = MemoryLocation::NOWHERE;
570 s_num_total_handles--;
571 break;
572
573 case MemoryLocation::DISK: // MemoryRelease
574 std::fclose(s_handles[index].disk.file);
575 dir_remove(g_temp_dir, mem_filename(index));
576 s_handles[index].disk.file = nullptr;
577 s_handles[index].size = 0;
578 s_handles[index].stored_at = MemoryLocation::NOWHERE;
579 s_num_total_handles--;
580 break;
581 } // end of switch
