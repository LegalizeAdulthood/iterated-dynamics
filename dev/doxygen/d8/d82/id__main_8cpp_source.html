164static bool main_restore_start(MainContext &context)
165{
166 driver_check_memory();
167 if (g_colors_preloaded)
168 {
169 std::memcpy(g_dac_box, g_old_dac_box, 256 * 3); // restore in case colors= present
170 }
171 g_look_at_mouse = MouseLook::IGNORE_MOUSE;
172 while (g_show_file <= ShowFile::LOAD_IMAGE) // image is to be loaded
173 {
174 g_tab_enabled = false;
175 if (!g_browse.browsing) /*RB*/
176 {
177 const char *hdg;
178 if (g_overlay_3d)
179 {
180 hdg = "Select File for 3D Overlay";
181 g_help_mode = HelpLabels::HELP_3D_OVERLAY;
182 }
183 else if (g_display_3d != Display3DMode::NONE)
184 {
185 hdg = "Select File for 3D Transform";
186 g_help_mode = HelpLabels::HELP_3D;
187 }
188 else
189 {
190 hdg = "Select File to Restore";
191 g_help_mode = HelpLabels::HELP_SAVE_RESTORE;
192 }
193 if (g_show_file == ShowFile::REQUEST_IMAGE &&
194 driver_get_filename(hdg, "Image File", g_image_filename_mask.c_str(), g_read_filename))
195 {
196 g_show_file = ShowFile::IMAGE_LOADED; // cancelled
197 g_init_mode = -1;
198 break;
199 }
200
201 g_browse.stack.clear(); // 'r' reads first filename for browsing
202 g_browse.stack.push_back(g_browse.name);
203 }
204
205 g_evolving = EvolutionModeFlags::NONE;
206 g_viewport.enabled = false;
207 g_show_file = ShowFile::LOAD_IMAGE;
208 g_help_mode = HelpLabels::NONE;
209 g_tab_enabled = true;
210 if (context.stacked)
211 {
212 driver_discard_screen();
213 driver_set_for_text();
214 context.stacked = false;
215 }
216 if (read_overlay() == 0) // read hdr, get video mode
217 {
218 break; // got it, exit
219 }
220 if (g_browse.browsing) // break out of infinite loop, but lose your mind
221 {
222 g_show_file = ShowFile::IMAGE_LOADED;
223 }
224 else
225 {
226 g_show_file = ShowFile::REQUEST_IMAGE; // retry
227 }
228 }
229 g_help_mode = HelpLabels::HELP_MENU; // now use this help mode
230 g_tab_enabled = true;
231 g_look_at_mouse = MouseLook::IGNORE_MOUSE;
232 if (((g_overlay_3d && g_init_batch == BatchMode::NONE) || context.stacked) &&
233 g_init_mode < 0) // overlay command failed
234 {
235 driver_unstack_screen(); // restore the graphics screen
236 context.stacked = false;
237 g_overlay_3d = false; // forget overlays
238 g_display_3d = Display3DMode::NONE;
239 if (g_calc_status == CalcStatus::NON_RESUMABLE)
240 {
241 g_calc_status = CalcStatus::PARAMS_CHANGED;
242 }
243 context.resume = true;
244 return true;
245 }
246 g_save_dac = SaveDAC::NO;
247 return false;
248}

250static MainState main_image_start(MainContext &context)
251{
252 driver_check_memory();
253 if (context.stacked)
254 {
255 driver_discard_screen();
256 context.stacked = false;
257 }
258 g_passes = Passes::NONE; // for tab_display
259
260 if (g_show_file != ShowFile::LOAD_IMAGE)
261 {
262 // goto image_start implies re-calc
263 g_calc_status = std::min(g_calc_status, CalcStatus::PARAMS_CHANGED);
264 }
265
266 g_cycle_limit = g_init_cycle_limit; // default cycle limit
267 g_adapter = g_init_mode; // set the video adapter up
268 g_init_mode = -1; // (once)
269
270 while (g_adapter < 0) // cycle through instructions
271 {
272 if (g_init_batch != BatchMode::NONE) // batch, nothing to do
273 {
274 g_init_batch = BatchMode::BAILOUT_INTERRUPTED_TRY_SAVE; // exit with error condition set
275 goodbye();
276 }
277 int key = main_menu(false);
278 if (key == ID_KEY_INSERT)
279 {
280 return MainState::RESTART; // restart pgm on Insert Key
281 }
282 if (key == ID_KEY_DELETE) // select video mode list
283 {
284 key = select_video_mode(-1);
285 }
286 g_adapter = check_vid_mode_key(key);
287 if (g_adapter >= 0)
288 {
289 break; // got a video mode now
290 }
291 key = std::tolower(key);
292 if (key == 'd')
293 {
294 // shell to DOS
295 driver_set_clear();
296 driver_shell();
297 return MainState::IMAGE_START;
298 }
299
300 if (key == '@' || key == '2')
301 {
302 // execute commands
303 if (!bit_set(get_commands(), CmdArgFlags::YES_3D))
304 {
305 return MainState::IMAGE_START;
306 }
307 key = '3'; // 3d=y so fall thru '3' code
308 }
309 if (key == 'r' || key == '3' || key == '#')
310 {
311 g_display_3d = Display3DMode::NONE;
312 if (key == '3' || key == '#' || key == ID_KEY_F3)
313 {
314 g_display_3d = Display3DMode::YES;
315 }
316 if (g_colors_preloaded)
317 {
318 std::memcpy(g_old_dac_box, g_dac_box, 256*3); // save in case colors= present
319 }
320 driver_set_for_text(); // switch to text mode
321 g_show_file = ShowFile::REQUEST_IMAGE;
322 return MainState::RESTORE_START;
323 }
324 if (key == 't')
325 {
326 // set fractal type
327 g_julibrot = false;
328 get_fract_type();
329 return MainState::IMAGE_START;
330 }
331 if (key == 'x')
332 {
333 // generic toggle switch
334 get_toggles();
335 return MainState::IMAGE_START;
336 }
337 if (key == 'y')
338 {
339 // generic toggle switch
340 get_toggles2();
341 return MainState::IMAGE_START;
342 }
343 if (key == 'z')
344 {
345 // type specific params
346 get_fract_params(true);
347 return MainState::IMAGE_START;
348 }
349 if (key == 'v')
350 {
351 // view parameters
352 get_view_params();
353 return MainState::IMAGE_START;
354 }
355 if (key == ID_KEY_CTL_B)
356 {
357 /* ctrl B = browse params*/
358 get_browse_params();
359 return MainState::IMAGE_START;
360 }
361 if (key == ID_KEY_CTL_F)
362 {
363 /* ctrl f = sound params*/
364 get_sound_params();
365 return MainState::IMAGE_START;
366 }
367 if (key == 'i')
368 {
369 // set 3d fractal params
370 get_fract3d_params(); // get the parameters
371 return MainState::IMAGE_START;
372 }
373 if (key == 'g')
374 {
375 get_cmd_string(); // get command string
376 return MainState::IMAGE_START;
377 }
378 }
379
380 g_zoom_enabled = true;
381 g_help_mode = HelpLabels::HELP_MAIN; // now use this help mode
382 context.resume = false; // allows taking goto inside big_while_loop()
383
384 return MainState::CONTINUE;
385}

405int id_main(int argc, char *argv[])
406{
407 set_search_dirs();
408
409 // this traps non-math library floating point errors
410 std::signal(SIGFPE, my_floating_point_err);
411
412 init_asm_vars(); // initialize ASM stuff
413 init_memory();
414
415 // let drivers add their video modes
416 if (! init_drivers(&argc, argv))
417 {
418 init_failure("Sorry, I couldn't find any working video drivers for your system\n");
419 std::exit(-1);
420 }
421 // load id.cfg, match against driver supplied modes
422 load_config();
423 init_help();
424
425 MainState state{MainState::RESTART};
426 MainContext context;
427 bool done{};
428 while (!done)
429 {
430 driver_check_memory();
431 switch (state)
432 {
433 case MainState::RESTART:
434 // insert key re-starts here
435 main_restart(argc, argv, context);
436 state = MainState::RESTORE_START;
437 break;
438
439 case MainState::RESTORE_START:
440 state = main_restore_start(context) ? MainState::RESUME_LOOP : MainState::IMAGE_START;
441 break;
442
443 case MainState::IMAGE_START:
444 state = main_image_start(context);
445 if (state != MainState::RESTORE_START && state != MainState::IMAGE_START &&
446 state != MainState::RESTART)
447 {
448 state = MainState::RESUME_LOOP;
449 }
450 break;
451
452 case MainState::RESUME_LOOP:
453 save_param_history();
454 state = big_while_loop(context);
455 break;
456
457 case MainState::CONTINUE:
458 case MainState::NOTHING:
459 done = true;
460 break;
461 }
462 }
463
464 return 0;
465}
