43void load_config(const std::string &cfg_path)
44{
45 std::FILE *cfg_file;
46 VideoInfo video_entry;
47 char *fields[5]{};
48
49 if (cfg_path.empty() // can't find the file
50 || (cfg_file = std::fopen(cfg_path.c_str(), "r")) == nullptr) // can't open it
51 {
52 g_bad_config = ConfigStatus::BAD_NO_MESSAGE;
53 return;
54 }
55
56 int line_num = 0;
57 char temp_string[150];
58 std::fill_n(std::begin(g_cfg_line_nums), g_video_table_len, -1);
59 while (g_video_table_len < MAX_VIDEO_MODES
60 && std::fgets(temp_string, std::size(temp_string), cfg_file))
61 {
62 if (std::strchr(temp_string, '\n') == nullptr)
63 {
64 // finish reading the line
65 while (std::fgetc(cfg_file) != '\n' && !std::feof(cfg_file))
66 {
67 }
68 }
69 ++line_num;
70 if (temp_string[0] == ';')
71 {
72 continue; // comment line
73 }
74 temp_string[120] = 0;
75 temp_string[std::strlen(temp_string) -1] = 0; // zap trailing \n
76 int j = -1;
77 int i = j;
78 // key, 0: mode name, 1: x, 2: y, 3: colors, 4: driver, 5: comments
79 while (true)
80 {
81 if (temp_string[++i] < ' ')
82 {
83 if (temp_string[i] == 0)
84 {
85 break;
86 }
87 temp_string[i] = ' '; // convert tab (or whatever) to blank
88 }
89 else if (temp_string[i] == ',' && ++j < 6)
90 {
91 assert(j >= 0 && j < 11);
92 fields[j] = &temp_string[i + 1]; // remember start of next field
93 temp_string[i] = 0; // make field a separate string
94 }
95 }
96 const int key = check_vid_mode_key_name(temp_string);
97 assert(fields[0]);
98 const long x_dots = std::atol(fields[0]);
99 assert(fields[1]);
100 const long y_dots = std::atol(fields[1]);
101 assert(fields[2]);
102 const int colors = std::atoi(fields[2]);
103
104 if (j < 4 ||
105 key < 0 ||
106 x_dots < MIN_PIXELS || x_dots > MAX_PIXELS ||
107 y_dots < MIN_PIXELS || y_dots > MAX_PIXELS ||
108 (colors != 0 && colors != 2 && colors != 4 && colors != 16 &&
109 colors != 256)
110 )
111 {
112 g_bad_config = ConfigStatus::BAD_NO_MESSAGE;
113 return;
114 }
115 g_cfg_line_nums[g_video_table_len] = line_num; // for update_id_cfg
116
117 std::memset(&video_entry, 0, sizeof(video_entry));
118 std::strncpy(&video_entry.comment[0], fields[4], std::size(video_entry.comment));
119 video_entry.comment[25] = 0;
120 video_entry.key = key;
121 video_entry.x_dots = static_cast<short>(x_dots);
122 video_entry.y_dots = static_cast<short>(y_dots);
123 video_entry.colors = colors;
124
125 // if valid, add to supported modes
126 video_entry.driver = driver_find_by_name(fields[3]);
127 if (video_entry.driver != nullptr && video_entry.driver->validate_mode(video_entry))
128 {
129 // look for a synonym mode and if found, overwrite its key
130 VideoInfo *begin{&g_video_table[0]};
131 VideoInfo *end{&g_video_table[g_video_table_len]};
132 const auto it = std::find_if(begin, end,
133 [&](const VideoInfo &mode)
134 {
135 return mode.driver == video_entry.driver //
136 && mode.colors == video_entry.colors //
137 && mode.x_dots == video_entry.x_dots //
138 && mode.y_dots == video_entry.y_dots;
139 });
140 if (it != end && it->key == 0)
141 {
142 it->key = video_entry.key;
143 }
144 else
145 {
146 add_video_mode(video_entry.driver, &video_entry);
147 }
148 }
149 }
150 std::fclose(cfg_file);
151}
