42{
43 if (g_std_calc_mode == CalcMode::PERTURBATION &&
44 bit_set(g_cur_fractal_specific->flags, FractalFlags::PERTURB))

71static bool use_calc_mandelbrot()
72{
73 return g_debug_flag != DebugFlags::FORCE_STANDARD_FRACTAL //
74 && !g_distance_estimator //
75 && g_decomp[0] == 0 //
76 && g_biomorph == -1 //
77 && g_inside_method >= ColorMethod::ITER //
78 && g_outside_method >= ColorMethod::ATAN //
79 && g_use_init_orbit != InitOrbitMode::VALUE //
80 && (g_sound_flag & SOUNDFLAG_ORBIT_MASK) < SOUNDFLAG_X //
81 && !g_attractor.enabled //
82 && !g_using_jiim //
83 && g_bailout_test == Bailout::MOD //
84 && (g_orbit_save_flags & OSF_MIDI) == 0;
85}

87bool mandel_per_image()
88{
89 g_bf_math = BFMathType::NONE;
90 g_c_exponent = static_cast<int>(g_params[2]);
91 g_power_z.x = g_params[2] - 1.0;
92 g_power_z.y = g_params[3];
93 g_float_param = &g_init;
94 switch (g_fractal_type)
95 {
96 case FractalType::MARKS_MANDEL:
97 if (g_c_exponent < 1)
98 {
99 g_c_exponent = 1;
100 g_params[2] = 1;
101 }
102 // odd/even exponents
103 g_symmetry = g_c_exponent & 1 ? SymmetryType::X_AXIS_NO_PARAM : SymmetryType::XY_AXIS_NO_PARAM;
104 break;
105
106 case FractalType::MANDEL:
107 /*
108 floating point code could probably be altered to handle many of
109 the situations that otherwise are using standard_fractal().
110 calcmandfp() can currently handle invert, any rqlim, potflag
111 zmag, epsilon cross, and all the current outside options
112 */
113 if (g_std_calc_mode == CalcMode::PERTURBATION &&
114 bit_set(g_cur_fractal_specific->flags, FractalFlags::PERTURB))
115 {
116 return mandel_perturbation_per_image();
117 }
118 if (use_calc_mandelbrot())
119 {
120 g_calc_type = calc_mandelbrot_type; // the normal case
121 calc_mandelbrot_init();
122 }
123 else
124 {
125 // special case: use the main processing loop
126 g_calc_type = standard_fractal_type;
127 }
128 break;
129
130 case FractalType::MANDEL_Z_POWER:
131 if (g_std_calc_mode == CalcMode::PERTURBATION &&
132 bit_set(g_cur_fractal_specific->flags, FractalFlags::PERTURB))
133 {
134 if (g_c_exponent == 2)
135 {
136 return mandel_perturbation_per_image();
137 }
138 if (g_c_exponent > 2)
139 {
140 return mandel_z_power_perturbation_per_image();
141 }
142 }
143 if (static_cast<double>(g_c_exponent) == g_params[2] && g_c_exponent & 1) // odd exponents
144 {
145 g_symmetry = SymmetryType::XY_AXIS_NO_PARAM;
146 }
147 if (g_params[3] != 0)
148 {
149 g_symmetry = SymmetryType::NONE;
150 }
151 if (g_params[3] == 0.0 && g_debug_flag != DebugFlags::FORCE_COMPLEX_POWER && static_cast<double>(g_c_exponent) == g_params[2])
152 {
153 get_fractal_specific(g_fractal_type)->orbit_calc = mandel_z_power_orbit;
154 }
155 else
156 {
157 get_fractal_specific(g_fractal_type)->orbit_calc = mandel_z_power_cmplx_orbit;
158 }
159 break;
160 case FractalType::MAGNET_1M:
161 case FractalType::MAGNET_2M:
162 g_attractor.z[0].x = 1.0; // 1.0 + 0.0i always attracts
163 g_attractor.z[0].y = 0.0; // - both MAGNET1 and MAGNET2
164 g_attractor.period[0] = 1;
165 g_attractor.count = 1;
166 break;
167 case FractalType::SPIDER:
168 if (g_periodicity_check == 1) // if not user set
169 {
170 g_periodicity_check = 4;
171 }
172 break;
173 case FractalType::FN_PLUS_FN:
174 g_symmetry = SymmetryType::X_AXIS_NO_PARAM;
175 break;
176 case FractalType::MANDEL_FN_PLUS_EXP:
177 case FractalType::MANDEL_FN_PLUS_Z_SQRD:
178 if (g_param_z1.y == 0.0)
179 {
180 g_symmetry = SymmetryType::X_AXIS;
181 }
182 else
183 {
184 g_symmetry = SymmetryType::NONE;
185 }
186 if (g_trig_index[0] == TrigFn::LOG || g_trig_index[0] == TrigFn::FLIP)
187 {
188 g_symmetry = SymmetryType::NONE;
189 }
190 break;
191 case FractalType::QUAT:
192 g_float_param = &g_tmp_z;
193 g_attractor.count = 0;
194 g_periodicity_check = 0;
195 break;
196 case FractalType::HYPER_CMPLX:
197 g_float_param = &g_tmp_z;
198 g_attractor.count = 0;
199 g_periodicity_check = 0;
200 if (g_params[2] != 0)
201 {
202 g_symmetry = SymmetryType::NONE;
203 }
204 if (g_trig_index[0] == TrigFn::FLIP)
205 {
206 g_symmetry = SymmetryType::NONE;
207 }
208 break;
209 case FractalType::TIMS_ERROR:
210 if (g_trig_index[0] == TrigFn::FLIP)
211 {
212 g_symmetry = SymmetryType::NONE;
213 }
214 break;
215 case FractalType::MARKS_MANDEL_PWR:
216 if (g_trig_index[0] == TrigFn::FLIP)
217 {
218 g_symmetry = SymmetryType::NONE;
219 }
220 break;
221 default:
222 break;
223 }
224 return true;
225}

227bool julia_per_image()
228{
229 g_c_exponent = static_cast<int>(g_params[2]);
230 g_float_param = &g_param_z1;
231 if (g_fractal_type == FractalType::COMPLEX_MARKS_JUL)
232 {
233 g_power_z.x = g_params[2] - 1.0;
234 g_power_z.y = g_params[3];
235 g_marks_coefficient = pow(*g_float_param, g_power_z);
236 }
237 switch (g_fractal_type)
238 {
239 case FractalType::JULIA:
240 /*
241 floating point code could probably be altered to handle many of
242 the situations that otherwise are using standard_fractal().
243 calcmandfp() can currently handle invert, any rqlim, potflag
244 zmag, epsilon cross, and all the current outside options
245 */
246 if (use_calc_mandelbrot())
247 {
248 g_calc_type = calc_mandelbrot_type; // the normal case
249 calc_mandelbrot_init();
250 }
251 else
252 {
253 // special case: use the main processing loop
254 g_calc_type = standard_fractal_type;
255 get_julia_attractor(0.0, 0.0); // another attractor?
256 }
257 break;
258 case FractalType::JULIA_Z_POWER:
259 if (g_c_exponent & 1 || g_params[3] != 0.0 || static_cast<double>(g_c_exponent) != g_params[2])
260 {
261 g_symmetry = SymmetryType::NONE;
262 }
263 if (g_params[3] == 0.0 && g_debug_flag != DebugFlags::FORCE_COMPLEX_POWER && static_cast<double>(g_c_exponent) == g_params[2])
264 {
265 get_fractal_specific(g_fractal_type)->orbit_calc = mandel_z_power_orbit;
266 }
267 else
268 {
269 get_fractal_specific(g_fractal_type)->orbit_calc = mandel_z_power_cmplx_orbit;
270 }
271 get_julia_attractor(g_params[0], g_params[1]); // another attractor?
272 break;
273 case FractalType::MAGNET_2J:
274 float_pre_calc_magnet2();
275 case FractalType::MAGNET_1J:
276 g_attractor.z[0].x = 1.0; // 1.0 + 0.0i always attracts
277 g_attractor.z[0].y = 0.0; // - both MAGNET1 and MAGNET2
278 g_attractor.period[0] = 1;
279 g_attractor.count = 1;
280 get_julia_attractor(0.0, 0.0); // another attractor?
281 break;
282 case FractalType::LAMBDA:
283 get_julia_attractor(0.0, 0.0); // another attractor?
284 get_julia_attractor(0.5, 0.0); // another attractor?
285 break;
286 case FractalType::MAN_O_WAR:
287 if (g_param_z1.y == 0.0)
288 {
289 g_symmetry = SymmetryType::X_AXIS;
290 }
291 get_julia_attractor(0.0, 0.0); // another attractor?
292 break;
293 case FractalType::JULIA_FN_PLUS_EXP:
294 case FractalType::JULIA_FN_PLUS_Z_SQRD:
295 if (g_param_z1.y == 0.0)
296 {
297 g_symmetry = SymmetryType::X_AXIS;
298 }
299 else
300 {
301 g_symmetry = SymmetryType::NONE;
302 }
303 if (g_trig_index[0] == TrigFn::LOG || g_trig_index[0] == TrigFn::FLIP)
304 {
305 g_symmetry = SymmetryType::NONE;
306 }
307 get_julia_attractor(0.0, 0.0); // another attractor?
308 break;
309 case FractalType::HYPER_CMPLX_J:
310 if (g_params[2] != 0)
311 {
312 g_symmetry = SymmetryType::NONE;
313 }
314 if (g_trig_index[0] != TrigFn::SQR)
315 {
316 g_symmetry = SymmetryType::NONE;
317 }
318 case FractalType::QUAT_JUL:
319 g_attractor.count = 0; // attractors broken since code checks r,i not j,k
320 g_periodicity_check = 0;
321 if (g_params[4] != 0.0 || g_params[5] != 0)
322 {
323 g_symmetry = SymmetryType::NONE;
324 }
325 break;
326 case FractalType::POPCORN:
327 case FractalType::POPCORN_JUL:
328 {
329 bool default_functions = false;
330 if (g_trig_index[0] == TrigFn::SIN
331 && g_trig_index[1] == TrigFn::TAN
332 && g_trig_index[2] == TrigFn::SIN
333 && g_trig_index[3] == TrigFn::TAN
334 && std::abs(g_param_z2.x - 3.0) < .0001
335 && g_param_z2.y == 0
336 && g_param_z1.y == 0)
337 {
338 default_functions = true;
339 if (g_fractal_type == FractalType::POPCORN_JUL)
340 {
341 g_symmetry = SymmetryType::ORIGIN;
342 }
343 }
344 if (default_functions && g_debug_flag == DebugFlags::FORCE_REAL_POPCORN)
345 {
346 g_cur_fractal_specific->orbit_calc = popcorn_fractal;
347 }
348 else
349 {
350 g_cur_fractal_specific->orbit_calc = popcorn_orbit;
351 }
352 get_julia_attractor(0.0, 0.0); // another attractor?
353 break;
354 }
355
356 case FractalType::CIRCLE:
357 if (g_inside_method == ColorMethod::STAR_TRAIL) // CIRCLE locks up when used with STAR_TRAIL
358 {
359 g_inside_method = ColorMethod::COLOR;
360 g_inside_color = 0; // arbitrarily set inside = 0
361 }
362 get_julia_attractor(0.0, 0.0); // another attractor?
363 break;
364 default:
365 get_julia_attractor(0.0, 0.0); // another attractor?
366 break;
367 }
368 return true;
369}
