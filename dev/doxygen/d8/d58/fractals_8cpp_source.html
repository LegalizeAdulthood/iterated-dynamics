3/*
4Actually calculate the fractal images (well, SOMEBODY had to do it!).
5
6The modules are set up so that all logic that is independent of any
7fractal-specific code is in calcfrac, the code that IS fractal-specific
8is in fractals, and the structure that ties (we hope!) everything together
9is in fractalp.
10
11Original author is Tim Wegner, but just about ALL the authors have
12contributed SOME code to this routine at one time or another, or
13contributed to one of the many massive restructurings.
14
15The Fractal-specific routines are divided into three categories:
16
171. Routines that are called once-per-orbit to calculate the orbit
18 value. These have names like "XxxxFractal", and their function
19 pointers are stored in fractalspecific[fractype].orbitcalc. EVERY
20 new fractal type needs one of these. Return 0 to continue iterations,
21 1 if we're done. Results for integer fractals are left in 'lnew.x' and
22 'lnew.y', for floating point fractals in 'new.x' and 'new.y'.
23
242. Routines that are called once per pixel to set various variables
25 prior to the orbit calculation. These have names like xxx_per_pixel
26 and are fairly generic - chances are one is right for your new type.
27 They are stored in fractalspecific[fractype].per_pixel.
28
293. Routines that are called once per screen to set various variables.
30 These have names like XxxxSetup, and are stored in
31 fractalspecific[fractype].per_image.
32
334. The main fractal routine. Usually this will be standard_fractal(),
34 but if you have written a stand-alone fractal routine independent
35 of the standard_fractal mechanisms, your routine name goes here,
36 stored in fractalspecific[fractype].calctype.per_image.
37
38Adding a new fractal type should be simply a matter of adding an item
39to the 'fractalspecific' structure, writing (or re-using one of the existing)
40an appropriate setup, per_image, per_pixel, and orbit routines.
41
42-------------------------------------------------------------------- */

86void pow(DComplex *base, int exp, DComplex *result)
87{
88 if (exp < 0)
89 {
90 pow(base, -exp, result);
91 cmplx_recip(*result, *result);
92 return;
93 }
94
95 double xt = base->x;
96 double yt = base->y;
97
98 if (exp & 1)
99 {
100 result->x = xt;
101 result->y = yt;
102 }
103 else
104 {
105 result->x = 1.0;
106 result->y = 0.0;
107 }
108
109 exp >>= 1;
110 while (exp)
111 {
112 double t2 = xt * xt - yt * yt;
113 yt = 2 * xt * yt;
114 xt = t2;
115
116 if (exp & 1)
117 {
118 t2 = xt * result->x - yt * result->y;
119 result->y = result->y * xt + yt * result->x;
120 result->x = t2;
121 }
122 exp >>= 1;
123 }
124}

145int mandel_per_pixel()
146{
147 // floating point mandelbrot
148 // mandelfp
149 // burning ship
150
151 if (g_inversion.invert != 0)
152 {
153 invertz2(&g_init);
154 }
155 else
156 {
157 g_init.x = dx_pixel();
158 g_init.y = dy_pixel();
159 }
160 switch (g_fractal_type)
161 {
162 case FractalType::BURNING_SHIP:
163 g_old_z.x = 0.0;
164 g_old_z.y = 0.0;
165 break;
166 case FractalType::MAGNET_2M:
167 float_pre_calc_magnet2();
168 case FractalType::MAGNET_1M: // NOLINT(clang-diagnostic-implicit-fallthrough)
169 g_old_z.y = 0.0; // Critical Val Zero both, but neither
170 g_old_z.x = g_old_z.y; // is of the form f(Z,C) = Z*g(Z)+C
171 break;
172 case FractalType::MANDEL_LAMBDA: // Critical Value 0.5 + 0.0i
173 g_old_z.x = 0.5;
174 g_old_z.y = 0.0;
175 break;
176 default:
177 g_old_z = g_init;
178 break;
179 }
180
181 // alter init value
182 if (g_use_init_orbit == InitOrbitMode::VALUE)
183 {
184 g_old_z = g_init_orbit;
185 }
186 else if (g_use_init_orbit == InitOrbitMode::PIXEL)
187 {
188 g_old_z = g_init;
189 }
190
191 if ((g_inside_method == ColorMethod::BOF60 || g_inside_method == ColorMethod::BOF61) && g_bof_match_book_images)
192 {
193 /* kludge to match "Beauty of Fractals" picture since we start
194 Mandelbrot iteration with init rather than 0 */
195 g_old_z.x = g_param_z1.x; // initial pertubation of parameters set
196 g_old_z.y = g_param_z1.y;
197 g_color_iter = -1;
198 }
199 else
200 {
201 g_old_z.x += g_param_z1.x;
202 g_old_z.y += g_param_z1.y;
203 }
204 g_tmp_z = g_init; // for spider
205 g_temp_sqr_x = sqr(g_old_z.x); // precalculated value for regular Mandelbrot
206 g_temp_sqr_y = sqr(g_old_z.y);
207 return 1; // 1st iteration has been done
208}

210void mandel_ref_pt(const std::complex<double> &center, std::complex<double> &z)
211{
212 const double real_sqr = sqr(z.real());
213 const double imag_sqr = sqr(z.imag());
214 const double real = real_sqr - imag_sqr + center.real();
215 const double imag = 2.0 * z.real() * z.imag() + center.imag();
216 z.real(real);
217 z.imag(imag);
218}

220void mandel_ref_pt_bf(const BFComplex &center, BFComplex &z)
221{
222 BigStackSaver saved;
223 BigFloat temp_real = alloc_stack(g_r_bf_length + 2);
224 BigFloat real_sqr = alloc_stack(g_r_bf_length + 2);
225 BigFloat imag_sqr = alloc_stack(g_r_bf_length + 2);
226 BigFloat real_imag = alloc_stack(g_r_bf_length + 2);
227 BFComplex z2;
228 z2.x = alloc_stack(g_r_bf_length + 2);
229 z2.y = alloc_stack(g_r_bf_length + 2);
230 double_bf(z2.x, z.x);
231 double_bf(z2.y, z.y);
232
233 square_bf(real_sqr, z.x);
234 square_bf(imag_sqr, z.y);
235 sub_bf(temp_real, real_sqr, imag_sqr);
236 add_bf(z.x, temp_real, center.x);
237 mult_bf(real_imag, z2.x, z.y);
238 add_bf(z.y, real_imag, center.y);
239}

241void mandel_perturb(
242 const std::complex<double> &ref, std::complex<double> &delta_n, const std::complex<double> &delta0)
243{
244 const double r{ref.real()};
245 const double i{ref.imag()};
246 const double a{delta_n.real()};
247 const double b{delta_n.imag()};
248 const double a0{delta0.real()};
249 const double b0{delta0.imag()};
250
251 const double dnr{(2 * r + a) * a - (2 * i + b) * b + a0};
252 const double dni{2 * ((r + a) * b + i * a) + b0};
253 delta_n.imag(dni);
254 delta_n.real(dnr);
255}
