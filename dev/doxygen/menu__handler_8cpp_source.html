44MainState request_fractal_type(MainContext &context)
45{

59 g_evolve_discrete_y_parameter_offset = 0;

61 g_evolve_max_random_mutation = 1; // reset param evolution stuff
62 g_set_orbit_corners = false;
63 save_param_history();
64 if (type == 0)
65 {
66 g_init_mode = g_adapter;
67 context.from_mandel = false;
68 }
69 else if (g_init_mode < 0) // it is supposed to be...
70 {
71 driver_set_for_text(); // reset to text mode
72 }
73 return MainState::IMAGE_START;
74 }
75 driver_unstack_screen();
76 return MainState::NOTHING;
77}

79MainState get_history(const int kbd_char)
80{
81 if (g_max_image_history <= 0 || g_bf_math != BFMathType::NONE)
82 {
83 return MainState::NOTHING;
84 }
85
86 if (kbd_char == '\\' || kbd_char == 'h')
87 {
88 if (--g_history_ptr < 0)
89 {
90 g_history_ptr = g_max_image_history - 1;
91 }
92 }
93 if (kbd_char == ID_KEY_CTL_BACKSLASH || kbd_char == ID_KEY_BACKSPACE)
94 {
95 if (++g_history_ptr >= g_max_image_history)
96 {
97 g_history_ptr = 0;
98 }
99 }
100 restore_history_info(g_history_ptr);
101 g_zoom_enabled = true;
102 g_init_mode = g_adapter;
103 g_history_flag = true; // avoid re-store parms due to rounding errs
104 return MainState::IMAGE_START;
105}

107MainState color_cycle(MainContext &context)
108{
109 clear_zoom_box();
110 std::memcpy(g_old_dac_box, g_dac_box, 256 * 3);
111 if (context.key == 'c')
112 {
113 rotate(SpinDirection::NONE);
114 }
115 else
116 {
117 rotate(context.key == '+' ? SpinDirection::FORWARD : SpinDirection::BACKWARD);
118 }
119 if (std::memcmp(g_old_dac_box, g_dac_box, 256 * 3) != 0)
120 {
121 g_color_state = ColorState::UNKNOWN_MAP;
122 save_history_info();
123 }
124 return MainState::CONTINUE;
125}

157MainState restore_from_image(MainContext &context)
158{
159 g_compare_gif = false;
160 context.from_mandel = false;
161 g_browse.browsing = false;
162 if (context.key == 'r')
163 {
164 if (g_debug_flag == DebugFlags::FORCE_DISK_RESTORE_NOT_SAVE)
165 {
166 g_compare_gif = true;
167 g_overlay_3d = true;
168 if (g_init_batch == BatchMode::SAVE)
169 {
170 driver_stack_screen(); // save graphics image
171 g_read_filename = g_save_filename;
172 g_show_file = ShowFile::LOAD_IMAGE;
173 return MainState::RESTORE_START;
174 }
175 }
176 else
177 {
178 g_compare_gif = false;
179 g_overlay_3d = false;
180 }
181 g_display_3d = Display3DMode::NONE;
182 }
183 driver_stack_screen(); // save graphics image
184 context.stacked = !g_overlay_3d;
185 if (g_resave_flag != TimedSave::NONE)
186 {
187 update_save_name(g_save_filename); // do the pending increment
188 g_resave_flag = TimedSave::NONE;
189 g_started_resaves = false;
190 }
191 g_show_file = ShowFile::REQUEST_IMAGE;
192 return MainState::RESTORE_START;
193}
