49// next has a skip bit for each maxblock unit;
50// 1st pass sets bit [1]... off only if block's contents guessed;
51// at end of 1st pass [0]... bits are set if any surrounding block not guessed;
52// bits are numbered [..][y/16+1][x+1]&(1<<(y&15))
53// size of next puts a limit of MAX_PIXELS pixels across on solid guessing logic

63int diffusion_scan()
64{
65
66 g_passes = Passes::DIFFUSION;
67
68 // note: the max size of 2048x2048 gives us a 22 bit counter that will
69 // fit any 32 bit architecture, the maximum limit for this case would
70 // be 65536x65536
71
72 g_diffusion_bits =
73 static_cast<unsigned>(std::min( //
74 std::log(static_cast<double>(g_i_stop_pt.y - g_i_start_pt.y + 1)),
75 std::log(static_cast<double>(g_i_stop_pt.x - g_i_start_pt.x + 1))) /
76 LOG2);
77 g_diffusion_bits <<= 1; // double for two axes
78 g_diffusion_limit = 1UL << g_diffusion_bits;
79
80 if (diffusion_engine() == -1)
81 {
82 add_work_list(g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, g_start_pt.x,
83 static_cast<int>(g_diffusion_counter >> 16), // high,
84 static_cast<int>(g_diffusion_counter & 0xffff), // low order words
85 g_work_symmetry);
86 return -1;
87 }
88
89 return 0;
90}

92// little function that plots a filled square of color c, size s with
93// top left cornet at (x,y) with optimization from sym_fill_line
94static void plot_block(const int x, const int y, const int s, const int c)
95{
96 std::memset(s_stack, c, s);
97 for (int ty = y; ty < y + s; ty++)
98 {
99 sym_fill_line(ty, x, x + s - 1, s_stack);
100 }
101}

103// function that does the same as above, but checks the limits in x and y
104static void plot_block_lim(const int x, const int y, const int s, const int c)
105{
106 std::memset(s_stack, c, s);
107 for (int ty = y; ty < std::min(y + s, g_i_stop_pt.y + 1); ty++)
108 {
109 sym_fill_line(ty, x, std::min(x + s - 1, g_i_stop_pt.x), s_stack);
110 }
111}

113// count_to_int(dif_counter, colo, rowo, dif_offset)
114static void count_to_int(const unsigned long c, int &x, int &y, const int dif_offset)
115{
116 long unsigned t_c = c;
117 x = DIF_LA[t_c & 0xFF];
118 y = DIF_LB[t_c & 0xFF];
119 t_c >>= 8;
120 x <<= 4;
121 x += DIF_LA[t_c & 0xFF];
122 y <<= 4;
123 y += DIF_LB[t_c & 0xFF];
124 t_c >>= 8;
125 x <<= 4;
126 x += DIF_LA[t_c & 0xFF];
127 y <<= 4;
128 y += DIF_LB[t_c & 0xFF];
129 x >>= dif_offset;
130 y >>= dif_offset;
131}

143 int i;
144 int j;
145 int orig_col;
146 int orig_row;
147 const int s = 1 << (g_diffusion_bits/2); // size of the square
148 // number of tiles to build in x and y dirs
149 const int nx = static_cast<int>(floor(static_cast<double>((g_i_stop_pt.x - g_i_start_pt.x + 1) / s)));
150 const int ny = static_cast<int>(floor(static_cast<double>((g_i_stop_pt.y - g_i_start_pt.y + 1) / s)));
151 // made this to complete the area that is not
152 // a square with sides like 2 ** n
153 // what is left on the last tile to draw
154 const int rem_x = g_i_stop_pt.x - g_i_start_pt.x + 1 - nx * s;
155 const int rem_y = g_i_stop_pt.y - g_i_start_pt.y + 1 - ny * s;
156
157 if (g_begin_pt.y == g_i_start_pt.y && g_work_pass == 0)
158 {
159 // if restarting on pan:
160 g_diffusion_counter = 0L;
161 }
162 else
163 {
164 // yybegin and passes contain data for resuming the type:
165 g_diffusion_counter =
166 static_cast<long>(static_cast<unsigned>(g_begin_pt.y)) << 16 | static_cast<unsigned>(g_work_pass);
167 }
168
169 const int dif_offset = 12 - g_diffusion_bits / 2; // offset to adjust coordinates
170 // (*) for 4 bytes use 16 for 3 use 12 etc.
171
172 // only the points (dithering only) :
173 if (g_fill_color == 0)
174 {
175 while (g_diffusion_counter < g_diffusion_limit >> 1)

261 j = 0;
262 do

264 g_row = g_i_start_pt.y + orig_row + j * s; // get the right tiles
265
266 CALCULATE;
