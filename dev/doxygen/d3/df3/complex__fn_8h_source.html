21// Cube: z^3 = (a + jb) * (a + jb) * (a + jb)
22inline std::complex<double> cube(const std::complex<double> &z)
23{
24 const double r2{z.real() * z.real()};
25 const double i2{z.imag() * z.imag()};
26 return {z.real() * (r2 - (i2 + i2 + i2)), z.imag() * (r2 + r2 + r2 - i2)};
27}

51// base^exp
52inline std::complex<double> power(const std::complex<double> &base, int exp)
53{
54 double xt = base.real();
55 double yt = base.imag();
56
57 std::complex<double> result;
58 exp = std::max(exp, 0);
59 if (exp & 1)
60 {
61 result.real(xt);
62 result.imag(yt);
63 }
64 else
65 {
66 result.real(1.0);
67 result.imag(0.0);
68 }
69
70 exp >>= 1;
71 while (exp)
72 {
73 double t2 = (xt + yt) * (xt - yt);
74 yt = xt * yt;
75 yt = yt + yt;
76 xt = t2;
77
78 if (exp & 1)
79 {
80 t2 = xt * result.real() - yt * result.imag();
81 result.imag(result.imag() * xt + yt * result.real());
82 result.real(t2);
83 }
84 exp >>= 1;
85 }
86 return result;
87}

97
98 degree = std::max(degree, 0);

101 copy_bf(t2, z.y); // BigTemp2 = yt
102
103 if (degree & 1)
104 {
105 copy_bf(result.x, t1); // new.x = result real
106 copy_bf(result.y, t2); // new.y = result imag
107 }
108 else
109 {
110 int_to_bf(result.x, 1);
111 int_to_bf(result.y, 0);
112 }
113
114 degree >>= 1;
115 while (degree)
116 {
117 sub_bf(t, t1, t2); // (xt - yt)
118 add_bf(t3, t1, t2); // (xt + yt)
119 mult_bf(t4, t, t3); // t2 = (xt + yt) * (xt - yt)
120 copy_bf(t, t2);
121 mult_bf(t3, t, t1); // yt = xt * yt
122 add_bf(t2, t3, t3); // yt = yt + yt
123 copy_bf(t1, t4);
124
125 if (degree & 1)
126 {
127 mult_bf(t, t1, result.x); // xt * result->x
128 mult_bf(t3, t2, result.y); // yt * result->y
129 sub_bf(t4, t, t3); // t2 = xt * result->x - yt * result->y
130 mult_bf(t, t1, result.y); // xt * result->y
131 mult_bf(t3, t2, result.x); // yt * result->x
132 add_bf(result.y, t, t3); // result->y = result->y * xt + yt * result->x
133 copy_bf(result.x, t4); // result->x = t2
134 }
135 degree >>= 1;
136 }
137}
