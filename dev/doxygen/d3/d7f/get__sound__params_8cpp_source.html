31int get_sound_params()
32{
33 ChoiceBuilder<15> builder;
34 /* routine to get sound settings */
35 const char *sound_modes[]{"off", "beep", "x", "y", "z"};
36 int i;
37
38 int old_sound_flag = g_sound_flag;
39 int old_orbit_delay = g_orbit_delay;
40 bool old_start_show_orbit = g_start_show_orbit;
41
42 /* g_sound_flag bits 0..7 used as thus:
43 bit 0,1,2 controls sound beep/off and x,y,z
44 (0 == off 1 == beep, 2 == x, 3 == y, 4 == z)
45 bit 3 controls PC speaker
46 bit 4 controls sound card OPL3 FM sound
47 bit 5 controls midi output
48 bit 6 controls pitch quantise
49 bit 7 free! */
50get_sound_restart:
51 s_menu2 = 0;
52 builder.reset()
53 .list("Sound (off, beep, x, y, z)", 5, 4, sound_modes, g_sound_flag & SOUNDFLAG_ORBIT_MASK)
54 .yes_no("Use PC internal speaker?", (g_sound_flag & SOUNDFLAG_SPEAKER) != 0)
55 .yes_no("Use sound card output?", (g_sound_flag & SOUNDFLAG_OPL3_FM) != 0)
56 .yes_no("Midi...not implemented yet", (g_sound_flag & SOUNDFLAG_MIDI) != 0)
57 .yes_no("Quantize note pitch ?", (g_sound_flag & SOUNDFLAG_QUANTIZED) != 0)
58 .int_number("Orbit delay in ms (0 = none)", g_orbit_delay)
59 .int_number("Base Hz Value", g_base_hertz)
60 .yes_no("Show orbits?", g_start_show_orbit)
61 .comment("")
62 .comment("Press F6 for FM synth parameters, F7 for scale mappings")
63 .comment("Press F4 to reset to default values");
64
65 {
66 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_SOUND};
67 i = builder.prompt("Sound Control Screen", 255);
68 }
69 if (i < 0)
70 {
71 g_sound_flag = old_sound_flag;
72 g_orbit_delay = old_orbit_delay;
73 g_start_show_orbit = old_start_show_orbit;
74 return -1; /*escaped */
75 }
76
77 g_sound_flag = builder.read_list();
78 g_sound_flag |= builder.read_yes_no() ? SOUNDFLAG_SPEAKER : 0;
79 g_sound_flag |= builder.read_yes_no() ? SOUNDFLAG_OPL3_FM : 0;
80 g_sound_flag |= builder.read_yes_no() ? SOUNDFLAG_MIDI : 0;
81 g_sound_flag |= builder.read_yes_no() ? SOUNDFLAG_QUANTIZED : 0;
82
83 g_orbit_delay = builder.read_int_number();
84 g_base_hertz = builder.read_int_number();
85 g_start_show_orbit = builder.read_yes_no();
86
87 /* now do any initialization needed and check for soundcard */
88 if (g_sound_flag & SOUNDFLAG_OPL3_FM && !(old_sound_flag & SOUNDFLAG_OPL3_FM))
89 {
90 driver_init_fm();
91 }
92
93 if (i == ID_KEY_F6)
94 {
95 get_music_params(); /* see below, for controlling fmsynth */
96 goto get_sound_restart;
97 }
98
99 if (i == ID_KEY_F7)
100 {
101 get_scale_map(); /* see below, for setting scale mapping */
102 goto get_sound_restart;
103 }
104
105 if (i == ID_KEY_F4)
106 {
107 g_sound_flag = SOUNDFLAG_SPEAKER | SOUNDFLAG_BEEP; /* reset to default */
108 g_orbit_delay = 0;
109 g_base_hertz = 440;
110 g_start_show_orbit = false;
111 goto get_sound_restart;
112 }
113
114 return g_sound_flag != old_sound_flag &&
115 ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) > 1 || (old_sound_flag & SOUNDFLAG_ORBIT_MASK) > 1)
116 ? 1
117 : 0;
118}

120static int get_scale_map()
121{
122 ChoiceBuilder<15> builder;
123 int i;
124
125 ++s_menu2;
126get_map_restart:
127 builder.reset()
128 .int_number("Scale map C (1)", s_scale_map[0])
129 .int_number("Scale map C#(2)", s_scale_map[1])
130 .int_number("Scale map D (3)", s_scale_map[2])
131 .int_number("Scale map D#(4)", s_scale_map[3])
132 .int_number("Scale map E (5)", s_scale_map[4])
133 .int_number("Scale map F (6)", s_scale_map[5])
134 .int_number("Scale map F#(7)", s_scale_map[6])
135 .int_number("Scale map G (8)", s_scale_map[7])
136 .int_number("Scale map G#(9)", s_scale_map[8])
137 .int_number("Scale map A (10)", s_scale_map[9])
138 .int_number("Scale map A#(11)", s_scale_map[10])
139 .int_number("Scale map B (12)", s_scale_map[11])
140 .comment("")
141 .comment("Press F6 for FM synth parameters")
142 .comment("Press F4 to reset to default values");
143
144 {
145 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_MUSIC};
146 i = builder.prompt("Scale Mapping Screen", 255);
147 }
148 if (i < 0)
149 {
150 return -1;
151 }
152
153 for (int j = 0; j <= 11; j++)
154 {
155 s_scale_map[j] = std::min(12, std::abs(builder.read_int_number()));
156 }
157
158 if (i == ID_KEY_F6 && s_menu2 == 1)
159 {
160 get_music_params(); /* see below, for controlling fmsynth */
161 goto get_map_restart;
162 }
163 if (i == ID_KEY_F6 && s_menu2 == 2)
164 {
165 s_menu2--;
166 }
167
168 if (i == ID_KEY_F4)
169 {
170 for (int j = 0; j <= 11; j++)
171 {
172 s_scale_map[j] = j + 1;
173 }
174 goto get_map_restart;
175 }
176
177 return 0;
178}

180static int get_music_params()
181{
182 ChoiceBuilder<11> builder;
183 const char *attenuation_modes[] = {"none", "low", "mid", "high"};
184
185 s_menu2++;
186get_music_restart:
187 builder.reset()
188 .int_number("polyphony 1..9", g_polyphony + 1)
189 .int_number("Wave type 0..7", g_fm_wave_type)
190 .int_number("Note attack time 0..15", g_fm_attack)
191 .int_number("Note decay time 0..15", g_fm_decay)
192 .int_number("Note sustain level 0..15", g_fm_sustain)
193 .int_number("Note release time 0..15", g_fm_release)
194 .int_number("Soundcard volume? 0..63", g_fm_volume)
195 .list("Hi pitch attenuation", 4, 4, attenuation_modes, g_hi_attenuation)
196 .comment("")
197 .comment("Press F7 for scale mappings")
198 .comment("Press F4 to reset to default values");
199
200 HelpLabels old_help_mode = g_help_mode; /* this prevents HELP from activating */
201 g_help_mode = HelpLabels::HELP_MUSIC;
202 const int i = builder.prompt("FM Synth Card Control Screen", 255);
203 g_help_mode = old_help_mode; /* re-enable HELP */
204 if (i < 0)
205 {
206 return -1;
207 }
208
209 g_polyphony = std::min(8, std::abs(builder.read_int_number() - 1));
210 g_fm_wave_type = builder.read_int_number() & 0x07;
211 g_fm_attack = builder.read_int_number() & 0x0F;
212 g_fm_decay = builder.read_int_number() & 0x0F;
213 g_fm_sustain = builder.read_int_number() & 0x0F;
214 g_fm_release = builder.read_int_number() & 0x0F;
215 g_fm_volume = builder.read_int_number() & 0x3F;
216 g_hi_attenuation = builder.read_list();
217 if (g_sound_flag & SOUNDFLAG_OPL3_FM)
218 {
219 driver_init_fm();
220 }
221
222 if (i == ID_KEY_F7 && s_menu2 == 1)
223 {
224 get_scale_map(); /* see above, for setting scale mapping */
225 goto get_music_restart;
226 }
227 if (i == ID_KEY_F7 && s_menu2 == 2)
228 {
229 s_menu2--;
230 }
231
232 if (i == ID_KEY_F4)
233 {
234 g_polyphony = 0;
235 g_fm_wave_type = 0;
236 g_fm_attack = 5;
237 g_fm_decay = 10;
238 g_fm_sustain = 13;
239 g_fm_release = 5;
240 g_fm_volume = 63;
241 g_hi_attenuation = 0;
242 if (g_sound_flag & SOUNDFLAG_OPL3_FM)
243 {
244 driver_init_fm();
245 }
246 goto get_music_restart;
247 }
248
249 return 0;
250}
