64// Bresenham's algorithm for drawing line
65void draw_line(const int x1, const int y1, const int x2, const int y2, const int color)
66{
67 // uses Bresenham algorithm to draw a line
68 // vector components
69 int row;
70 int col;
71 int final; // final row or column number
72 int g; // used to test for new row or column
73 int inc1; // G increment when row or column doesn't change
74 int inc2; // G increment when row or column changes
75
76 const int dx = x2 - x1; // find vector components
77 const int dy = y2 - y1;
78 bool pos_slope = dx > 0; // is slope positive?
79 if (dy < 0)
80 {
81 pos_slope = !pos_slope;
82 }
83 if (std::abs(dx) > std::abs(dy)) // shallow line case
84 {
85 if (dx > 0) // determine start point and last column
86 {
87 col = x1;
88 row = y1;
89 final = x2;
90 }
91 else
92 {
93 col = x2;
94 row = y2;
95 final = x1;
96 }
97 inc1 = 2 * std::abs(dy); // determine increments and initial G
98 g = inc1 - std::abs(dx);
99 inc2 = 2 * (std::abs(dy) - std::abs(dx));
100 if (pos_slope)
101 {
102 while (col <= final) // step through columns checking for new row
103 {
104 g_plot(col, row, color);
105 col++;
106 if (g >= 0) // it's time to change rows
107 {
108 row++; // positive slope so increment through the rows
109 g += inc2;
110 }
111 else // stay at the same row
112 {
113 g += inc1;
114 }
115 }
116 }
117 else
118 {
119 while (col <= final) // step through columns checking for new row
120 {
121 g_plot(col, row, color);
122 col++;
123 if (g > 0) // it's time to change rows
124 {
125 row--; // negative slope so decrement through the rows
126 g += inc2;
127 }
128 else // stay at the same row
129 {
130 g += inc1;
131 }
132 }
133 }
134 } // if |dX| > |dY|
135 else // steep line case
136 {
137 if (dy > 0) // determine start point and last row
138 {
139 col = x1;
140 row = y1;
141 final = y2;
142 }
143 else
144 {
145 col = x2;
146 row = y2;
147 final = y1;
148 }
149 inc1 = 2 * std::abs(dx); // determine increments and initial G
150 g = inc1 - std::abs(dy);
151 inc2 = 2 * (std::abs(dx) - std::abs(dy));
152 if (pos_slope)
153 {
154 while (row <= final) // step through rows checking for new column
155 {
156 g_plot(col, row, color);
157 row++;
158 if (g >= 0) // it's time to change columns
159 {
160 col++; // positive slope so increment through the columns
161 g += inc2;
162 }
163 else // stay at the same column
164 {
165 g += inc1;
166 }
167 }
168 }
169 else
170 {
171 while (row <= final) // step through rows checking for new column
172 {
173 g_plot(col, row, color);
174 row++;
175 if (g > 0) // it's time to change columns
176 {
177 col--; // negative slope so decrement through the columns
178 g += inc2;
179 }
180 else // stay at the same column
181 {
182 g += inc1;
183 }
184 }
185 }
186 }
187} // draw_line

227static void plot3d_superimpose256(const int x, const int y, int color)
228{
229 int tmp;
230
231 const Byte t_c = static_cast<Byte>(255 - color);
232
233 if (color != 0) // Keeps index 0 still 0
234 {
235 color = g_colors - color; // Reverses color order
236 color = 1 + color / 18; // Maps colors 1-255 to 15 even ranges
237 }
238
239 tmp = get_color(x, y);
240 // map to 16 colors
241 if (g_which_image == StereoImage::RED) // RED
242 {
243 if (s_red_local_left < x && x < s_red_local_right)
244 {
245 // Overwrite prev Red don't mess w/blue
246 g_put_color(x, y, color| tmp & 240);
247 if (g_targa_out)
248 {
249 if (!illumine())

251 targa_color(x, y, color| tmp & 240);

253 else
254 {
255 targa_write_disk(x+g_logical_screen.x_offset, y+g_logical_screen.y_offset, t_c, 0, 0);

261 {
262 if (s_blue_local_left < x && x < s_blue_local_right)
263 {
264 // Overwrite previous blue, don't mess with existing red

427void plot_setup()
428{
429 double d_red_bright = 0;
430 double d_blue_bright = 0;
431
432 // set funny glasses plot function
433 switch (g_glasses_type)
434 {
435 case GlassesType::ALTERNATING:
436 g_standard_plot = plot3d_alternate;
437 break;
438
439 case GlassesType::SUPERIMPOSE:
440 if (g_colors == 256)
441 {
442 if (g_fractal_type != FractalType::IFS_3D)
443 {
444 g_standard_plot = plot3d_superimpose256;
445 }
446 else
447 {
448 g_standard_plot = plot_ifs3d_superimpose256;
449 }
450 }
451 else
452 {
453 g_standard_plot = plot3d_superimpose16;
454 }
455 break;
456
457 case GlassesType::STEREO_PAIR: // crosseyed mode
458 if (g_screen_x_dots < 2*g_logical_screen.x_dots)
459 {
460 if (g_x_rot == 0 && g_y_rot == 0)
461 {
462 g_standard_plot = plot3d_cross_eyed_a; // use hidden surface kludge
463 }
464 else
465 {
466 g_standard_plot = plot3d_cross_eyed_b;
467 }
468 }
469 else if (g_x_rot == 0 && g_y_rot == 0)
470 {
471 g_standard_plot = plot3d_cross_eyed_c; // use hidden surface kludge
472 }
473 else
474 {
475 g_standard_plot = g_put_color;
476 }
477 break;
478
479 default:
480 g_standard_plot = g_put_color;
481 break;
482 }
483
484 g_x_shift = static_cast<int>(g_shift_x * static_cast<double>(g_logical_screen.x_dots) / 100);
485 g_x_shift1 = g_x_shift;
486 g_y_shift = static_cast<int>(g_shift_y * static_cast<double>(g_logical_screen.y_dots) / 100);
487 g_y_shift1 = g_y_shift;
488
489 if (g_glasses_type != GlassesType::NONE)
490 {
491 s_red_local_left = static_cast<int>(g_red_crop_left * static_cast<double>(g_logical_screen.x_dots) / 100.0);
492 s_red_local_right = static_cast<int>((100 - g_red_crop_right) * static_cast<double>(g_logical_screen.x_dots) / 100.0);
493 s_blue_local_left = static_cast<int>(g_blue_crop_left * static_cast<double>(g_logical_screen.x_dots) / 100.0);
494 s_blue_local_right = static_cast<int>((100 - g_blue_crop_right) * static_cast<double>(g_logical_screen.x_dots) / 100.0);
495 d_red_bright = static_cast<double>(g_red_bright) /100.0;
496 d_blue_bright = static_cast<double>(g_blue_bright) /100.0;
497
498 switch (g_which_image)
499 {
500 case StereoImage::RED:
501 g_x_shift += static_cast<int>(g_eye_separation * static_cast<double>(g_logical_screen.x_dots) / 200);
502 g_xx_adjust = static_cast<int>(
503 (g_adjust_3d.x + g_converge_x_adjust) * static_cast<double>(g_logical_screen.x_dots) / 100);
504 g_x_shift1 -= static_cast<int>(g_eye_separation * static_cast<double>(g_logical_screen.x_dots) / 200);
505 g_xx_adjust1 = static_cast<int>(
506 (g_adjust_3d.x - g_converge_x_adjust) * static_cast<double>(g_logical_screen.x_dots) / 100);
507 if (g_glasses_type == GlassesType::STEREO_PAIR && g_screen_x_dots >= 2*g_logical_screen.x_dots)
508 {
509 g_logical_screen.x_offset = g_screen_x_dots / 2 - g_logical_screen.x_dots;
510 }
511 break;
512
513 case StereoImage::BLUE:
514 g_x_shift -= static_cast<int>(g_eye_separation * static_cast<double>(g_logical_screen.x_dots) / 200);
515 g_xx_adjust = static_cast<int>(
516 (g_adjust_3d.x - g_converge_x_adjust) * static_cast<double>(g_logical_screen.x_dots) / 100);
517 if (g_glasses_type == GlassesType::STEREO_PAIR && g_screen_x_dots >= 2*g_logical_screen.x_dots)
518 {
519 g_logical_screen.x_offset = g_screen_x_dots / 2;
520 }
521 break;
522
523 default:
524 break;
525 }
526 }
527 else
528 {
529 g_xx_adjust = static_cast<int>(g_adjust_3d.x * static_cast<double>(g_logical_screen.x_dots) / 100);
530 }
531 g_yy_adjust = static_cast<int>(-(g_adjust_3d.y * static_cast<double>(g_logical_screen.y_dots)) / 100);
532
533 if (g_map_set)
534 {
535 validate_luts(g_map_name); // read the palette file
536 if (glasses_alternating_or_superimpose())
537 {
538 if (g_glasses_type == GlassesType::SUPERIMPOSE && g_colors < 256)
539 {
540 g_dac_box[PAL_RED ][0] = 255;
541 g_dac_box[PAL_RED ][1] = 0;
542 g_dac_box[PAL_RED ][2] = 0;
543
544 g_dac_box[PAL_BLUE ][0] = 0;
545 g_dac_box[PAL_BLUE ][1] = 0;
546 g_dac_box[PAL_BLUE ][2] = 255;
547
548 g_dac_box[PAL_MAGENTA][0] = 255;
549 g_dac_box[PAL_MAGENTA][1] = 0;
550 g_dac_box[PAL_MAGENTA][2] = 255;
551 }
552 for (auto &elem : g_dac_box)
553 {
554 elem[0] = static_cast<Byte>(elem[0] * d_red_bright);
555 elem[2] = static_cast<Byte>(elem[2] * d_blue_bright);
556 }
557 }
558 refresh_dac();
559 }
