67Cellular::Cellular()
68{
69 set_cellular_palette();
70
71 rand_param = static_cast<S32>(g_params[0]);
72 kr = static_cast<U16>(g_params[2]);
73 line_num = static_cast<U32>(g_params[3]);
74
75 switch (kr)
76 {
77 case 21:
78 case 22:
79 case 23:
80 case 24:
81 case 25:
82 case 26:
83 case 27:
84 case 31:
85 case 32:
86 case 33:
87 case 41:
88 case 42:
89 case 51:
90 case 61:
91 break;
92 default:
93 throw CellularError(*this, TYPE_KR);
94 }
95
96 m_s_r = static_cast<S16>(kr % 10); // Number of nearest neighbors to sum
97 k = static_cast<U16>(kr / 10); // Number of different states, k=3 has states 0,1,2
98 m_k_1 = static_cast<S16>(k - 1); // Highest state value, k=3 has highest state value of 2
99 m_rule_digits = static_cast<S16>((m_s_r * 2 + 1) * m_k_1 + 1); // Number of digits in the rule
100
101 if (!g_random_seed_flag && rand_param == -1)
102 {
103 --g_random_seed;
104 }
105
106 if (rand_param != 0 && rand_param != -1)
107 {
108 double n = g_params[0];
109 const std::string buf{fmt::format("{:.16g}", n)}; // # of digits in initial string
110 const S16 t = static_cast<S16>(buf.length());
111 if (t>16 || t <= 0)
112 {
113 throw CellularError(*this, STRING1);
114 }
115 std::fill(std::begin(init_string), std::end(init_string), 0);
116 const S16 t2 = static_cast<S16>((16 - t) / 2);
117 for (int i = 0; i < t; i++)
118 {
119 // center initial string in array
120 init_string[i+t2] = from_digit(buf[i]);
121 if (init_string[i+t2]> static_cast<U16>(m_k_1))
122 {
123 throw CellularError(*this, STRING2);
124 }
125 }
126 }
127
128 set_random_seed();
129
130 // generate rule table from parameter 1
131 double n = g_params[1];
132 if (n == 0.0)
133 {
134 // calculate a random rule
135 n = std::rand() % static_cast<int>(k);
136 for (int i = 1; i < m_rule_digits; i++)
137 {
138 n *= 10;
139 n += std::rand() % static_cast<int>(k);
140 }
141 g_params[1] = n;
142 }
143 const std::string buf{fmt::format("{:.{}g}", n, m_rule_digits)};
144 const S16 t = static_cast<S16>(buf.length());
145 if (m_rule_digits < t || t < 0)
146 {
147 // leading 0s could make t smaller
148 throw CellularError(*this, RULE_LENGTH);
149 }
150 for (int i = 0; i < m_rule_digits; i++) // zero the table
151 {
152 cell_table[i] = 0;
153 }
154 for (int i = 0; i < t; i++)
155 {
156 // reverse order
157 cell_table[i] = from_digit(buf[t - i - 1]);
158 if (cell_table[i]> static_cast<U16>(m_k_1))
159 {
160 throw CellularError(*this, TABLE_K);
161 }
162 }
163
164 m_cell_array[0].resize(g_i_stop_pt.x+1);
165 m_cell_array[1].resize(g_i_stop_pt.x+1);
166
167 // g_cellular_next_screen toggled by space bar, true for continuous
168 // false to stop on next screen
169
170 if (g_resuming && !g_cellular_next_screen && !m_last_screen_flag)
171 {
172 start_resume();
173 get_resume(start_row);
174 end_resume();
175 read_span(start_row, 0, g_i_stop_pt.x, m_cell_array[filled].data());
176 }
177 else if (g_cellular_next_screen && !m_last_screen_flag)
178 {
179 start_resume();
180 end_resume();
181 read_span(g_i_stop_pt.y, 0, g_i_stop_pt.x, m_cell_array[filled].data());
182 g_params[3] += g_i_stop_pt.y + 1;
183 start_row = -1; // after 1st iteration its = 0
184 }
185 else
186 {
187 if (g_random_seed_flag || rand_param == 0 || rand_param == -1)
188 {
189 for (g_col = 0; g_col <= g_i_stop_pt.x; g_col++)
190 {
191 m_cell_array[filled][g_col] = static_cast<Byte>(std::rand() % static_cast<int>(k));
192 }
193 } // end of if random
194 else
195 {
196 for (g_col = 0; g_col <= g_i_stop_pt.x; g_col++)
197 {
198 // Clear from end to end
199 m_cell_array[filled][g_col] = 0;
200 }
201 int i = 0;
202 for (g_col = (g_i_stop_pt.x-16)/2; g_col < (g_i_stop_pt.x+16)/2; g_col++)
203 {
204 // insert initial
205 m_cell_array[filled][g_col] = static_cast<Byte>(init_string[i++]); // string
206 }
207 } // end of if not random
208 m_last_screen_flag = line_num != 0;
209 write_span(start_row, 0, g_i_stop_pt.x, m_cell_array[filled].data());
210 }
211 start_row++;
212
213 g_row = start_row;
214}

216// This section does all the work
217bool Cellular::iterate()
218{
219 // This section calculates the starting line when it is not zero
220 // This section can't be resumed since no screen output is generated
221 // calculates the (line_num - 1) generation
222 if (m_last_screen_flag) // line number != 0 & not resuming & not continuing
223 {
224 if (g_row < static_cast<int>(line_num))
225 {
226 thinking("Cellular thinking (higher start row takes longer)");
227 if (g_random_seed_flag || rand_param == 0 || rand_param == -1)
228 {
229 // Use a random border
230 for (int i = 0; i <= m_s_r; i++)
231 {
232 m_cell_array[not_filled][i] = static_cast<Byte>(std::rand() % static_cast<int>(k));
233 m_cell_array[not_filled][g_i_stop_pt.x-i] = static_cast<Byte>(std::rand() % static_cast<int>(k));
234 }
235 }
236 else
237 {
238 // Use a zero border
239 for (int i = 0; i <= m_s_r; i++)
240 {
241 m_cell_array[not_filled][i] = 0;
242 m_cell_array[not_filled][g_i_stop_pt.x-i] = 0;
243 }
244 }
245
246 S16 t = 0; // do first cell
247 const U16 two_r = static_cast<U16>(m_s_r + m_s_r);
248 for (int i = 0; i <= two_r; i++)
249 {
250 t = static_cast<S16>(t + static_cast<S16>(m_cell_array[filled][i]));
251 }
252 if (t > m_rule_digits || t < 0)
253 {
254 thinking_end();
255 throw CellularError(*this, BAD_T, t);
256 }
257 m_cell_array[not_filled][m_s_r] = static_cast<Byte>(cell_table[t]);
258
259 // use a rolling sum in t
260 for (g_col = m_s_r+1; g_col < g_i_stop_pt.x-m_s_r; g_col++)
261 {
262 // now do the rest
263 t = static_cast<S16>(
264 t + m_cell_array[filled][g_col + m_s_r] - m_cell_array[filled][g_col - m_s_r - 1]);
265 if (t > m_rule_digits || t < 0)
266 {
267 thinking_end();
268 throw CellularError(*this, BAD_T, t);
269 }
270 m_cell_array[not_filled][g_col] = static_cast<Byte>(cell_table[t]);
271 }
272
273 filled = not_filled;
274 not_filled = static_cast<S16>(1 - filled);
275 ++g_row;
276 return true;
277 }
278 start_row = 0;
279 thinking_end();
280 m_last_screen_flag = false;
281 }
282
283 if (g_row <= g_i_stop_pt.y)
284 {
285 if (g_random_seed_flag || rand_param == 0 || rand_param == -1)
286 {
287 // Use a random border
288 for (int i = 0; i <= m_s_r; i++)
289 {
290 m_cell_array[not_filled][i] = static_cast<Byte>(std::rand() % static_cast<int>(k));
291 m_cell_array[not_filled][g_i_stop_pt.x-i] = static_cast<Byte>(std::rand() % static_cast<int>(k));
292 }
293 }
294 else
295 {
296 // Use a zero border
297 for (int i = 0; i <= m_s_r; i++)
298 {
299 m_cell_array[not_filled][i] = 0;
300 m_cell_array[not_filled][g_i_stop_pt.x-i] = 0;
301 }
302 }
303
304 S16 t = 0; // do first cell
305 const U16 two_r = static_cast<U16>(m_s_r + m_s_r);
306 for (int i = 0; i <= two_r; i++)
307 {
308 t = static_cast<S16>(t + static_cast<S16>(m_cell_array[filled][i]));
309 }
310 if (t > m_rule_digits || t < 0)
311 {
312 thinking_end();
313 throw CellularError(*this, BAD_T, t);
314 }
315 m_cell_array[not_filled][m_s_r] = static_cast<Byte>(cell_table[t]);
316
317 // use a rolling sum in t
318 for (g_col = m_s_r+1; g_col < g_i_stop_pt.x-m_s_r; g_col++)
319 {
320 // now do the rest
321 t = static_cast<S16>(
322 t + m_cell_array[filled][g_col + m_s_r] - m_cell_array[filled][g_col - m_s_r - 1]);
323 if (t > m_rule_digits || t < 0)
324 {
325 thinking_end();
326 throw CellularError(*this, BAD_T, t);
327 }
328 m_cell_array[not_filled][g_col] = static_cast<Byte>(cell_table[t]);
329 }
330
331 filled = not_filled;
332 not_filled = static_cast<S16>(1 - filled);
333 write_span(g_row, 0, g_i_stop_pt.x, m_cell_array[filled].data());
334 ++g_row;
335 return true;
336 }
337
338 if (g_cellular_next_screen)
339 {
340 g_params[3] += g_i_stop_pt.y + 1;
341 start_row = 0;
342 return true;
343 }
344
345 return false;
346}

359std::string Cellular::error(const int err, int t) const
360{
361 switch (err)
362 {
363 case BAD_T:
364 return fmt::format("Bad t={:d}, aborting\n", t);
365
366 case BAD_MEM:
367 return "Insufficient free memory for calculation";
368
369 case STRING1:
370 return "String can be a maximum of 16 digits";
371
372 case STRING2:
373 {
374 static char msg[] = {"Make string of 0's through 's"};
375 msg[27] = to_digit(m_k_1);
376 return msg;
377 }
378
379 case TABLE_K:
380 {
381 static char msg[] = {"Make Rule with 0's through 's"};
382 msg[27] = to_digit(m_k_1);
383 return msg;
384 }
385
386 case TYPE_KR:
387 return "Type must be 21, 31, 41, 51, 61, 22, 32, 42, 23, 33, 24, 25, 26, 27";
388
389 case RULE_LENGTH:
390 {
391 static char msg[] = {"Rule must be digits long"};
392 if (int i = m_rule_digits / 10; i == 0)
393 {
394 msg[14] = to_digit(m_rule_digits);
395 }
396 else
397 {
398 msg[13] = to_digit(i);
399 msg[14] = to_digit(m_rule_digits % 10);
400 }
401 return msg;
402 }
403
404 case INTERRUPT:
405 return "Interrupted, can't resume";
406 }
407
408 return "Unknown error";
409}
