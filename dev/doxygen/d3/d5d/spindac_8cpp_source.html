30// Rotate the colormap in the (plus or minus) "direction"
31// in "rstep" increments - or, if "direction" is 0, just replace it.
32void spin_dac(const SpinDirection dir, const int inc)
33{
34 if (g_colors < 16)
35 {
36 return;
37 }
38 if (g_is_true_color && g_true_mode != TrueColorMode::DEFAULT_COLOR)
39 {
40 return;
41 }
42 if (dir != SpinDirection::NONE && //
43 g_color_cycle_range_lo < g_colors && //
44 g_color_cycle_range_lo < g_color_cycle_range_hi)
45 {
46 const int top = g_color_cycle_range_hi > g_colors ? g_colors - 1 : g_color_cycle_range_hi;
47 unsigned char *dac_bot = reinterpret_cast<unsigned char *>(g_dac_box) + 3 * g_color_cycle_range_lo;
48 const int len = (top - g_color_cycle_range_lo)*3*sizeof(unsigned char);
49 if (dir > SpinDirection::NONE)
50 {
51 for (int i = 0; i < inc; i++)
52 {
53 unsigned char tmp[3];
54 std::memcpy(tmp, dac_bot, 3*sizeof(unsigned char));
55 std::memcpy(dac_bot, dac_bot + 3*sizeof(unsigned char), len);
56 std::memcpy(dac_bot + len, tmp, 3*sizeof(unsigned char));
57 }
58 }
59 else
60 {
61 for (int i = 0; i < inc; i++)
62 {
63 unsigned char tmp[3];
64 std::memcpy(tmp, dac_bot + len, 3*sizeof(unsigned char));
65 std::memcpy(dac_bot + 3*sizeof(unsigned char), dac_bot, len);
66 std::memcpy(dac_bot, tmp, 3*sizeof(unsigned char));
67 }
68 }
69 }
70 driver_write_palette();
71 driver_delay(g_colors - g_dac_count - 1);
72}
