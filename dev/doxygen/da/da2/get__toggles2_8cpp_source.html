83 for (int i = 0; i < 3; i++)

85 values[++k].type = 's';

100 values[k].type = '*';
101
102 choices[++k] = "Color cycling from color (0 ... 254)";

150 changed = true;
151 }
152

158 if (g_potential.params[0] != 0.0)

160 changed = true;

164 {

167 end_disk();

171 g_disk_16_bit = false;
172 }
173 }
174 }
175
176 ++k;
177 g_user.distance_estimator_value = values[k].uval.Lval;
178 if (g_user.distance_estimator_value != old_user_distance_estimator_value)
179 {
180 changed = true;
181 }
182 ++k;
183 g_distance_estimator_width_factor = values[k].uval.ival;
184 if (g_user.distance_estimator_value && g_distance_estimator_width_factor != old_distance_estimator_width_factor)
185 {
186 changed = true;
187 }
188
189 for (int i = 0; i < 3; i++)
190 {
191 if (values[++k].uval.sval[0] == 'a' || values[k].uval.sval[0] == 'A')
192 {
193 g_inversion.params[i] = AUTO_INVERT;
194 }
195 else
196 {
197 g_inversion.params[i] = std::atof(values[k].uval.sval);
198 }
199 if (old_inversion[i] != g_inversion.params[i]
200 && (i == 0 || g_inversion.params[0] != 0.0))
201 {
202 changed = true;
203 }
204 }
205 g_inversion.invert = g_inversion.params[0] == 0.0 ? 0 : 3;
206 ++k;
207
208 g_color_cycle_range_lo = values[++k].uval.ival;
209 g_color_cycle_range_hi = values[++k].uval.ival;
210 if (g_color_cycle_range_lo < 0 || g_color_cycle_range_hi > 255 || g_color_cycle_range_lo > g_color_cycle_range_hi)
211 {
212 g_color_cycle_range_lo = old_rotate_lo;
213 g_color_cycle_range_hi = old_rotate_hi;
214 }
215
216 return changed ? 1 : 0;
217}
