33*/
34
35int get_toggles2()
36{
37 ChoiceBuilder<13> choices;
38 std::array<double, 3> old_potential_param;
39 std::array<double, 3> old_inversion;
40
41 // fill up the choices (and previous values) arrays
42 old_potential_param[0] = g_potential.params[0];
43 old_potential_param[1] = g_potential.params[1];
44 old_potential_param[2] = g_potential.params[2];
45
46 long old_user_distance_estimator_value = g_user.distance_estimator_value;
47 int old_distance_estimator_width_factor = g_distance_estimator_width_factor;
48
49 choices.int_number("Look for finite attractor (0=no,>0=yes,<0=phase)", g_attractor.enabled ? 1 : 0)
50 .int_number("Potential Max Color (0 means off)", static_cast<int>(old_potential_param[0]))
51 .double_number(" Slope", old_potential_param[1])
52 .int_number(" Bailout", static_cast<int>(old_potential_param[2]))
53 .yes_no(" 16 bit values", g_potential.store_16bit)
54 .long_number("Distance Estimator (0=off, <0=edge, >0=on):", old_user_distance_estimator_value)
55 .int_number(" width factor:", old_distance_estimator_width_factor);
56 const char *inversion_choices[] = {"Inversion radius or \"auto\" (0 means off)",
57 " center X coordinate or \"auto\"", " center Y coordinate or \"auto\""};
58 for (int i = 0; i < 3; i++)
59 {
60 old_inversion[i] = g_inversion.params[i];
61 if (g_inversion.params[i] == AUTO_INVERT)
62 {
63 choices.string(inversion_choices[i], "auto");
64 }
65 else
66 {
67 const std::string value{fmt::format("{:<1.15Lg}", g_inversion.params[i])};
68 choices.string(inversion_choices[i], value.c_str());
69 }
70 }
71 int old_rotate_lo = g_color_cycle_range_lo;
72 int old_rotate_hi = g_color_cycle_range_hi;
73 choices.comment(" (use fixed radius & center when zooming)")
74 .int_number("Color cycling from color (0 ... 254)", old_rotate_lo)
75 .int_number(" to color (1 ... 255)", old_rotate_hi);
76
77 const HelpLabels old_help_mode = g_help_mode;
78 g_help_mode = HelpLabels::HELP_Y_OPTIONS;
79 {
80 const int i = choices.prompt("Extended Options\n"
81 "(not all combinations make sense)");
82 g_help_mode = old_help_mode;
83 if (i < 0)
84 {
85 return -1;
86 }
87 }
88
89 // now check out the results (*hopefully* in the same order <grin>)
90 bool changed = false;
91
92 const bool attractor_enabled{choices.read_int_number() != 0};
93 if (attractor_enabled != g_attractor.enabled)
94 {
95 g_attractor.enabled = attractor_enabled;
96 changed = true;
97 }
98
99 g_potential.params[0] = choices.read_int_number();
100 if (g_potential.params[0] != old_potential_param[0])
101 {
102 changed = true;
103 }
104
105 g_potential.params[1] = choices.read_double_number();
106 if (g_potential.params[0] != 0.0 && g_potential.params[1] != old_potential_param[1])
107 {
108 changed = true;
109 }
110
111 g_potential.params[2] = choices.read_int_number();
112 if (g_potential.params[0] != 0.0 && g_potential.params[2] != old_potential_param[2])
113 {
114 changed = true;
115 }
116
117 const bool store_16bit{choices.read_yes_no()};
118 if (store_16bit != g_potential.store_16bit)
119 {
120 g_potential.store_16bit = store_16bit;
121 if (g_potential.store_16bit) // turned it on
122 {
123 if (g_potential.params[0] != 0.0)
124 {
125 changed = true;
126 }
127 }
128 else // turned it off
129 {
130 if (!driver_is_disk()) // ditch the disk video
131 {
132 end_disk();
133 }
134 else // keep disk video, but ditch the fraction part at end
135 {
136 g_disk_16_bit = false;
137 }
138 }
139 }
140
141 g_user.distance_estimator_value = choices.read_long_number();
142 if (g_user.distance_estimator_value != old_user_distance_estimator_value)
143 {
144 changed = true;
145 }
146 g_distance_estimator_width_factor = choices.read_int_number();
147 if (g_user.distance_estimator_value && g_distance_estimator_width_factor != old_distance_estimator_width_factor)
148 {
149 changed = true;
150 }
151
152 for (int i = 0; i < 3; i++)
153 {
154 const char *value{choices.read_string()};
155 if (value[0] == 'a' || value[0] == 'A')
156 {
157 g_inversion.params[i] = AUTO_INVERT;
158 }
159 else
160 {
161 g_inversion.params[i] = std::atof(value);
162 }
163 if (old_inversion[i] != g_inversion.params[i] && (i == 0 || g_inversion.params[0] != 0.0))
164 {
165 changed = true;
166 }
167 }
168 g_inversion.invert = g_inversion.params[0] == 0.0 ? 0 : 3;
169 choices.read_comment();
170
171 g_color_cycle_range_lo = choices.read_int_number();
172 g_color_cycle_range_hi = choices.read_int_number();
173 if (g_color_cycle_range_lo < 0 || g_color_cycle_range_hi > 255 || g_color_cycle_range_lo > g_color_cycle_range_hi)
174 {
175 g_color_cycle_range_lo = old_rotate_lo;
176 g_color_cycle_range_hi = old_rotate_hi;
177 }
178
179 return changed ? 1 : 0;
180}
