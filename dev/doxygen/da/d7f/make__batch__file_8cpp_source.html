284 y_multiple = builder.read_int_number();

616static int get_prec(const double a, const double b, const double c)
617{
618 constexpr double high_v = 1.0E20;
619 double diff = std::abs(a - b);
620 if (diff == 0.0)
621 {
622 diff = high_v;
623 }
624 double temp = std::abs(a - c);
625 if (temp == 0.0)
626 {
627 temp = high_v;
628 }
629 diff = std::min(temp, diff);
630 temp = std::abs(b - c);
631 if (temp == 0.0)
632 {
633 temp = high_v;
634 }
635 diff = std::min(temp, diff);
636 int digits = 7;
637 if (g_debug_flag >= DebugFlags::FORCE_PRECISION_0_DIGITS &&
638 g_debug_flag < DebugFlags::FORCE_PRECISION_20_DIGITS)
639 {
640 digits = +g_debug_flag - +DebugFlags::FORCE_PRECISION_0_DIGITS;
641 }
642 while (diff < 1.0 && digits <= DBL_DIG+1)
643 {
644 diff *= 10;
645 ++digits;
646 }
647 return digits;
648}

695 for (int j = 0; j < 3; ++j)

697 buf[j * 2 + 1] = hex_digits[(g_dac_box[cur_color][j] & 0xF0) >> 4];
698 buf[j * 2 + 2] = hex_digits[g_dac_box[cur_color][j] & 0x0F];
699 }
700 buf[7] = 0;
701 }
702 put_param(wb_data, buf);
703 if (++cur_color >= max_color) // quit if done last color
704 {
705 break;
706 }
707 if (g_debug_flag == DebugFlags::FORCE_LOSSLESS_COLORMAP) // lossless compression
708 {
709 continue;
710 }
711 /* Next a P Branderhorst special, a tricky scan for smooth-shaded
712 ranges which can be written as <nn> to compress .par file entry.
713 Method used is to check net change in each color value over
714 spans of 2 to 5 color numbers. First time for each span size
715 the value change is noted. After first time the change is
716 checked against noted change. First time it differs, a
717 difference of 1 is tolerated and noted as an alternate
718 acceptable change. When change is not one of the tolerated
719 values, loop exits. */
720 if (force)
721 {
722 --force;
723 continue;
724 }
725 int scan_color = cur_color;
726 int k;
727 while (scan_color < max_color)
728 {
729 // scan while same diff to next
730 int i = scan_color - cur_color;
731 i = std::min(i, 3); // check spans up to 4 steps
732 for (k = 0; k <= i; ++k)
733 {
734 int j;
735 for (j = 0; j < 3; ++j)
736 {
737 // check pattern of chg per color
738 if (g_debug_flag != DebugFlags::ALLOW_LARGE_COLORMAP_CHANGES &&
739 scan_color > cur_color + 4 && scan_color < max_color - 5)
740 {
741 if (std::abs(2 * g_dac_box[scan_color][j] - g_dac_box[scan_color - 5][j] -
742 g_dac_box[scan_color + 5][j]) >= 2)
743 {
744 break;
745 }
746 }
747 const int delta = static_cast<int>(g_dac_box[scan_color][j]) - static_cast<int>(g_dac_box[scan_color - k - 1][j]);
748 if (k == scan_color - cur_color)
749 {
750 diff1[k][j] = delta;
751 diff2[k][j] = delta;
752 }
753 else if (delta != diff1[k][j] && delta != diff2[k][j])
754 {
755 diff_mag = std::abs(delta - diff1[k][j]);
756 if (diff1[k][j] != diff2[k][j] || diff_mag != 1)
757 {
758 break;
759 }
760 diff2[k][j] = delta;
761 }
762 }
763 if (j < 3)
764 {
765 break; // must've exited from inner loop above
766 }
767 }
768 if (k <= i)
769 {
770 break; // must've exited from inner loop above
771 }
772 ++scan_color;
773 }
774 // now scanc-1 is next color which must be written explicitly
775 if (scan_color - cur_color > 2)
776 {
777 // good, we have a shaded range
778 if (scan_color != max_color)
779 {
780 if (diff_mag < 3)
781 {
782 // not a sharp slope change?
783 force = 2; // force more between ranges, to stop
784 --scan_color; // "drift" when load/store/load/store/
785 }
786 if (k)
787 {
788 // more of the same
789 force += k;
790 --scan_color;
791 }
792 }
793 if (--scan_color - cur_color > 1)
794 {
795 put_param(wb_data, "<%d>", scan_color - cur_color);
796 cur_color = scan_color;
797 }
798 else // changed our mind
799 {
800 force = 0;
801 }
802 }
803 }
804}
