180void Plot::set_dirty_region(const int x_min, const int y_min, const int x_max, const int y_max)
181{
182 RECT *r = &m_dirty_region;
183
184 _ASSERTE(x_min < x_max);
185 _ASSERTE(y_min < y_max);
186 _ASSERTE(r->left <= r->right && (r->top <= r->bottom));
187 if (r->left < 0)
188 {
189 r->left = x_min;
190 r->right = x_max;
191 r->top = y_min;
192 r->bottom = y_max;
193 m_dirty = true;
194 }
195 else
196 {
197 if (x_min < r->left)
198 {
199 r->left = x_min;
200 m_dirty = true;
201 }
202 if (x_max > r->right)
203 {
204 r->right = x_max;
205 m_dirty = true;
206 }
207 if (y_min < r->top)
208 {
209 r->top = y_min;
210 m_dirty = true;
211 }
212 if (y_max > r->bottom)
213 {
214 r->bottom = y_max;
215 m_dirty = true;
216 }
217 }
218}

220/* init_pixels
221 *
222 * Resize the pixel array to sxdots by sydots and initialize it to zero.
223 * Any existing pixel array is freed.
224 */
225void Plot::init_pixels()
226{
227 m_pixels.clear();
228 m_saved_pixels.clear();
229 m_width = g_screen_x_dots;
230 m_height = g_screen_y_dots;
231 m_row_len = m_width * sizeof(Byte);
232 m_row_len = (m_row_len + 3) / 4 *4;
233 m_pixels_len = m_row_len * m_height;
234 _ASSERTE(m_pixels_len > 0);
235 m_pixels.resize(m_pixels_len);
236 std::memset(m_pixels.data(), 0, m_pixels_len);
237 m_dirty = false;
238 {
239 constexpr RECT dirty_rect{-1, -1, -1, -1};
240 m_dirty_region = dirty_rect;
241 }
242 {
243 BITMAPINFOHEADER *h = &m_bmi.bmiHeader;
244
245 h->biSize = sizeof(m_bmi.bmiHeader);
246 h->biWidth = m_width;
247 h->biHeight = m_height;
248 h->biPlanes = 1;
249 h->biBitCount = 8;
250 h->biCompression = BI_RGB;
251 h->biSizeImage = 0;
252 h->biClrUsed = 256;
253 }
254}

256void Plot::on_paint(const HWND window)
257{
258 PAINTSTRUCT ps;
259 const HDC dc = BeginPaint(window, &ps);
260 const RECT *r = &ps.rcPaint;
261 const int width = r->right - r->left;
262 const int height = r->bottom - r->top;
263
264 _ASSERTE(width >= 0 && height >= 0);
265 if (width > 0 && height > 0)
266 {
267 const DWORD status = StretchDIBits(dc, 0, 0, m_width, m_height, 0, 0, m_width, m_height, m_pixels.data(),
268 &m_bmi, DIB_RGB_COLORS, SRCCOPY);
269 _ASSERTE(status != GDI_ERROR);
270 }
271 EndPaint(window, &ps);
272}

317static LRESULT CALLBACK plot_proc(const HWND window, const UINT message, const WPARAM wp, const LPARAM lp)
318{
319 _ASSERTE(s_plot != nullptr);
320 switch (message)
321 {
322 case WM_PAINT:
323 HANDLE_WM_PAINT(window, wp, lp, plot_on_paint);
324 break;
325
326 case WM_LBUTTONDOWN:
327 HANDLE_WM_LBUTTONDOWN(window, wp, lp, plot_on_primary_button_down);
328 break;
329
330 case WM_LBUTTONDBLCLK:
331 HANDLE_WM_LBUTTONDBLCLK(window, wp, lp, plot_on_primary_button_down);
332 break;
333
334 case WM_LBUTTONUP:
335 HANDLE_WM_LBUTTONUP(window, wp, lp, plot_on_primary_button_up);
336 break;
337
338 case WM_RBUTTONDOWN:
339 HANDLE_WM_RBUTTONDOWN(window, wp, lp, plot_on_secondary_button_down);
340 break;
341
342 case WM_RBUTTONDBLCLK:
343 HANDLE_WM_RBUTTONDBLCLK(window, wp, lp, plot_on_secondary_button_down);
344 break;
345
346 case WM_RBUTTONUP:
347 HANDLE_WM_RBUTTONUP(window, wp, lp, plot_on_secondary_button_up);
348 break;
349
350 case WM_MBUTTONDOWN:
351 HANDLE_WM_MBUTTONDOWN(window, wp, lp, plot_on_middle_button_down);
352 break;
353
354 case WM_MBUTTONDBLCLK:
355 HANDLE_WM_MBUTTONDBLCLK(window, wp, lp, plot_on_middle_button_down);
356 break;
357
358 case WM_MBUTTONUP:
359 HANDLE_WM_MBUTTONUP(window, wp, lp, plot_on_middle_button_up);
360 break;
361
362 case WM_MOUSEMOVE:
363 HANDLE_WM_MOUSEMOVE(window, wp, lp, plot_on_mouse_move);
364 break;
365
366 default:
367 return DefWindowProc(window, message, wp, lp);
368 }
369
370 return 0;
371}

373int Plot::init(const HINSTANCE instance, const LPCSTR title)
374{
375 WNDCLASS wc;
376
377 m_instance = instance;
378 m_title = title;
379
380 int result = GetClassInfoA(m_instance, WINDOW_CLASS, &wc);
381 if (!result)
382 {
383 wc.style = CS_DBLCLKS;
384 wc.lpfnWndProc = plot_proc;
385 wc.cbClsExtra = 0;
386 wc.cbWndExtra = 0;
387 wc.hInstance = m_instance;
388 wc.hIcon = nullptr;
389 wc.hCursor = LoadCursor(nullptr, IDC_ARROW);
390 wc.hbrBackground = static_cast<HBRUSH>(GetStockObject(BLACK_BRUSH));
391 wc.lpszMenuName = m_title.c_str();
392 wc.lpszClassName = WINDOW_CLASS;
393
394 result = RegisterClassA(&wc);
395 }
396
397 return result;
398}

565void Plot::display_string(int x, const int y, const int fg, const int bg, const char *text)
566{
567 while (*text)
568 {
569 for (int row = 0; row < 8; row++)
570 {
571 int x1 = x;
572 int col = 8;
573 const Byte pixel = FONT_8x8[row][static_cast<unsigned char>(*text)];
574 while (col-- > 0)
575 {
576 const int color = pixel & 1 << col ? fg : bg;
577 write_pixel(x1++, y + row, color);
578 }
579 }
580 x += 8;
581 text++;
582 }
583 flush();
584}
