236/*
237 * This routine gets an int16_t out of the buffer.
238 * It updates the buffer pointer accordingly.
239 */
240static void get_int16(std::int16_t *dst, unsigned char **src, const int dir)
241{
242 if (dir == 1)
243 {
244 *dst = static_cast<std::int16_t>((*src)[0] + (reinterpret_cast<char *>(*src)[1] << 8));
245 }
246 else
247 {
248 (*src)[0] = *dst & 0xff;
249 (*src)[1] = (*dst & 0xff00) >> 8;
250 }
251 *src += 2; // sizeof(std::uint16_t)
252}

254/*
255 * This routine gets a long out of the buffer.
256 * It updates the buffer pointer accordingly.
257 */
258static void get_int32(std::int32_t *dst, unsigned char **src, const int dir)
259{
260 if (dir == 1)
261 {
262 *dst = static_cast<unsigned long>((*src)[0]) +
263 (static_cast<unsigned long>((*src)[1]) << 8) +
264 (static_cast<unsigned long>((*src)[2]) << 16) +
265 (static_cast<long>(reinterpret_cast<char *>(*src)[3]) << 24);
266 }
267 else
268 {
269 (*src)[0] = *dst &0xff;
270 (*src)[1] = (*dst &0xff00) >> 8;
271 (*src)[2] = (*dst &0xff0000) >> 16;
272 (*src)[3] = (*dst &0xff000000) >> 24;
273 }
274 *src += 4; // sizeof(long) in MS_DOS
275}

289/*
290 * This routine gets a double out of the buffer, or puts a double into the
291 * buffer;
292 * It updates the buffer pointer accordingly.
293 */
294static void get_double(double *dst, unsigned char **src, const int dir)
295{
296 int e;
297 double f;
298 if (dir == 1)
299 {
300 int i;
301 for (i = 0; i < 8; i++)
302 {
303 if ((*src)[i] != 0)
304 {
305 break;
306 }
307 }
308 if (i == 8)
309 {
310 *dst = 0;
311 }
312 else
313 {
314 e = (((*src)[7]&0x7f) << 4) + (((*src)[6]&0xf0) >> 4) - 1023;
315 f = 1 + ((*src)[6]&0x0f)/P4 + (*src)[5]/P12 + (*src)[4]/P20 +
316 (*src)[3]/P28 + (*src)[2]/P36 + (*src)[1]/P44 + (*src)[0]/P52;
317 f *= std::pow(2., static_cast<double>(e));
318 if ((*src)[7]&0x80)
319 {
320 f = -f;
321 }
322 *dst = f;
323 }
324 }
325 else
326 {
327 if (*dst == 0)
328 {
329 std::memset(*src, 0, 8);
330 }
331 else
332 {
333 int s = 0;
334 f = *dst;
335 if (f < 0)
336 {
337 s = 0x80;
338 f = -f;
339 }
340 e = std::log(f)/std::log(2.);
341 f = f/std::pow(2., static_cast<double>(e)) - 1;
342 if (f < 0)
343 {
344 e--;
345 f = (f+1)*2-1;
346 }
347 else if (f >= 1)
348 {
349 e++;
350 f = (f+1)/2-1;
351 }
352 e += 1023;
353 (*src)[7] = s | (e & 0x7f0) >> 4;
354 f *= P4;
355 (*src)[6] = (e & 0x0f) << 4 | static_cast<int>(f) & 0x0f;
356 f = (f- static_cast<int>(f))*P8;
357 (*src)[5] = static_cast<int>(f) & 0xff;
358 f = (f- static_cast<int>(f))*P8;
359 (*src)[4] = static_cast<int>(f) & 0xff;
360 f = (f- static_cast<int>(f))*P8;
361 (*src)[3] = static_cast<int>(f) & 0xff;
362 f = (f- static_cast<int>(f))*P8;
363 (*src)[2] = static_cast<int>(f) & 0xff;
364 f = (f- static_cast<int>(f))*P8;
365 (*src)[1] = static_cast<int>(f) & 0xff;
366 f = (f- static_cast<int>(f))*P8;
367 (*src)[0] = static_cast<int>(f) & 0xff;
368 }
369 }
370 *src += 8; // sizeof(double) in MSDOS
371}

373/*
374 * This routine gets a float out of the buffer.
375 * It updates the buffer pointer accordingly.
376 */
377static void get_float(float *dst, unsigned char **src, const int dir)
378{
379 int e;
380 double f;
381 if (dir == 1)
382 {
383 int i;
384 for (i = 0; i < 4; i++)
385 {
386 if ((*src)[i] != 0)
387 {
388 break;
389 }
390 }
391 if (i == 4)
392 {
393 *dst = 0;
394 }
395 else
396 {
397 e = (((*src)[3] & 0x7f) << 1 | ((*src)[2] & 0x80) >> 7) - 127;
398 f = 1 + ((*src)[2]&0x7f)/P7 + (*src)[1]/P15 + (*src)[0]/P23;
399 f *= std::pow(2., static_cast<double>(e));
400 if ((*src)[3]&0x80)
401 {
402 f = -f;
403 }
404 *dst = f;
405 }
406 }
407 else
408 {
409 if (*dst == 0)
410 {
411 std::memset(*src, 0, 4);
412 }
413 else
414 {
415 int s = 0;
416 f = *dst;
417 if (f < 0)
418 {
419 s = 0x80;
420 f = -f;
421 }
422 e = std::log(f)/std::log(2.);
423 f = f/std::pow(2., static_cast<double>(e)) - 1;
424 if (f < 0)
425 {
426 e--;
427 f = (f+1)*2-1;
428 }
429 else if (f >= 1)
430 {
431 e++;
432 f = (f+1)/2-1;
433 }
434 e += 127;
435 (*src)[3] = s | (e & 0xf7) >> 1;
436 f *= P7;
437 (*src)[2] = (e & 0x01) << 7 | static_cast<int>(f) & 0x7f;
438 f = (f- static_cast<int>(f))*P8;
439 (*src)[1] = static_cast<int>(f) & 0xff;
440 f = (f- static_cast<int>(f))*P8;
441 (*src)[0] = static_cast<int>(f) & 0xff;
442 }
443 }
444 *src += 4; // sizeof(float) in MSDOS
445}
