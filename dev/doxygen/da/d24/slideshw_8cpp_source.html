165// this routine reads the file g_auto_name and returns keystrokes
166int slide_show()
167{
168 int out;
169 char buffer[81];
170 if (s_calc_wait)
171 {
172 if (g_calc_status == CalcStatus::IN_PROGRESS || g_busy) // restart timer - process not done
173 {
174 return 0; // wait for calc to finish before reading more keystrokes
175 }
176 s_calc_wait = false;
177 }
178 if (s_slide_show_file == nullptr) // open files first time through
179 {
180 if (start_slide_show() == SlidesMode::OFF)
181 {
182 stop_slide_show();
183 return 0;
184 }
185 }
186
187 const clock_t now = std::clock();
188 if (s_ticks) // if waiting, see if waited long enough
189 {
190 if (now - s_start_tick < s_ticks) // haven't waited long enough
191 {
192 return 0;
193 }
194 s_ticks = 0;
195 }
196 constexpr int SLOW_ADJUST_CHAR_COUNT = 15;
197 constexpr int SLOW_INITIAL_CHAR_COUNT = 5;
198 if (++s_slow_count <= SLOW_ADJUST_CHAR_COUNT)
199 {
200 s_start_tick = now;
201 s_ticks = CLOCKS_PER_SEC/10; // a slight delay so keystrokes are visible
202 if (s_slow_count > SLOW_INITIAL_CHAR_COUNT)
203 {
204 s_ticks = std::max(CLOCKS_PER_SEC/150, s_ticks/2);
205 }
206 driver_set_keyboard_timeout(1000 * s_ticks /CLOCKS_PER_SEC);
207 }
208 if (s_repeats > 0)
209 {
210 s_repeats--;
211 return s_last1;
212 }
213start:
214 if (s_quotes) // reading a quoted string
215 {
216 out = std::fgetc(s_slide_show_file);
217 if (out != '\"' && out != EOF)
218 {
219 s_last1 = out;
220 return out;
221 }
222 s_quotes = false;
223 }
224 // skip white space:
225 while ((out = std::fgetc(s_slide_show_file)) == ' ' || out == '\t' || out == '\n')
226 {
227 }
228 switch (out)
229 {
230 case EOF:
231 stop_slide_show();
232 return 0;
233 case '\"': // begin quoted string
234 s_quotes = true;
235 goto start;
236 case ';': // comment from here to end of line, skip it
237 while (out != '\n' && out != EOF)
238 {
239 out = std::fgetc(s_slide_show_file);
240 }
241 goto start;
242 case '*':
243 if (std::fscanf(s_slide_show_file, "%d", &s_repeats) != 1
244 || s_repeats <= 1
245 || s_repeats >= 256
246 || std::feof(s_slide_show_file))
247 {
248 slide_show_err("error in * argument");
249 s_repeats = 0;
250 s_last1 = 0;
251 }
252 s_repeats -= 2;
253 out = s_last1;
254 return out;
255 }
256
257 int i = 0;
258 while (true) // get a token
259 {
260 if (i < 80)
261 {
262 buffer[i++] = static_cast<char>(out);
263 }
264 out = std::fgetc(s_slide_show_file);
265 if (out == ' ' || out == '\t' || out == '\n' || out == EOF)
266 {
267 break;
268 }
269 }
270 buffer[i] = 0;
271 if (buffer[i-1] == ':')
272 {
273 goto start;
274 }
275 out = -12345;
276 if (std::isdigit(buffer[0])) // an arbitrary scan code number - use it
277 {
278 out = std::atoi(buffer);
279 }
280 else if (std::strcmp(buffer, "MESSAGE") == 0)
281 {
282 int secs;
283 if (std::fscanf(s_slide_show_file, "%d", &secs) != 1)
284 {
285 slide_show_err("MESSAGE needs argument");
286 }
287 else
288 {
289 char buf[41];
290 buf[40] = 0;
291 if (std::fgets(buf, std::size(buf), s_slide_show_file) == nullptr)
292 {
293 throw std::system_error(errno, std::system_category(), "slideshw failed fgets");
294 }
295 const int len = static_cast<int>(std::strlen(buf));
296 buf[len-1] = 0; // zap newline
297 message(secs, buf);
298 }
299 out = 0;
300 }
301 else if (std::strcmp(buffer, "GOTO") == 0)
302 {
303 if (std::fscanf(s_slide_show_file, "%s", buffer) != 1)
304 {
305 slide_show_err("GOTO needs target");
306 out = 0;
307 }
308 else
309 {
310 char line[80];
311 std::fseek(s_slide_show_file, 0, SEEK_SET);
312 std::strcat(buffer, ":");
313 int count;
314 do
315 {
316 count = std::fscanf(s_slide_show_file, "%s", line);
317 }
318 while (count == 1 && std::strcmp(line, buffer) != 0);
319 if (std::feof(s_slide_show_file))
320 {
321 slide_show_err("GOTO target not found");
322 return 0;
323 }
324 goto start;
325 }
326 }
327 else if ((i = get_scancode(buffer)) > 0)
328 {
329 out = i;
330 }
331 else if (std::strcmp("WAIT", buffer) == 0)
332 {
333 float f_ticks;
334 const int count = std::fscanf(s_slide_show_file, "%f", &f_ticks); // how many seconds to wait
335 driver_set_keyboard_timeout(static_cast<int>(f_ticks * 1000.f)); // timeout in ms
336 f_ticks *= CLOCKS_PER_SEC; // convert from seconds to ticks
337 if (count == 1)
338 {
339 s_ticks = static_cast<long>(f_ticks);
340 s_start_tick = now; // start timing
341 }
342 else
343 {
344 slide_show_err("WAIT needs argument");
345 }
346 out = 0;
347 s_slow_count = 0;
348 }
349 else if (std::strcmp("CALCWAIT", buffer) == 0) // wait for calc to finish
350 {
351 s_calc_wait = true;
352 out = 0;
353 s_slow_count = 0;
354 }
355 else if ((i = check_vid_mode_key_name(buffer)) != 0)
356 {
357 out = i;
358 }
359 if (out == -12345)
360 {
361 std::ostringstream msg;
362 msg << "Can't understand " << buffer;
363 slide_show_err(msg.str().c_str());
364 out = 0;
365 }
366 s_last1 = out;
367 return out;
368}

394void record_show(const int key)
395{
396 float dt = static_cast<float>(s_ticks); // save time of last call
397 s_ticks = std::clock(); // current time
398 if (s_slide_show_file == nullptr)
399 {
400 std::string path{get_save_path(WriteFile::KEY, g_auto_name.filename().string()).string()};
401 assert(!path.empty());
402 check_write_file(path, ".key");
403 s_slide_show_file = std::fopen(path.c_str(), "w");
404 if (s_slide_show_file == nullptr)
405 {
406 return;
407 }
408 }
409 dt = s_ticks-dt;
410 dt /= CLOCKS_PER_SEC; // dt now in seconds
411 if (dt > .5) // don't bother with less than half a second
412 {
413 if (s_quotes) // close quotes first
414 {
415 s_quotes = false;
416 std::fprintf(s_slide_show_file, "\"\n");
417 }
418 std::fprintf(s_slide_show_file, "WAIT %4.1f\n", dt);
419 }
420 if (key >= 32 && key < 128)
421 {
422 if (!s_quotes)
423 {
424 s_quotes = true;
425 std::fputc('\"', s_slide_show_file);
426 }
427 std::fputc(key, s_slide_show_file);
428 }
429 else
430 {
431 if (s_quotes) // not an ASCII character - turn off quotes
432 {
433 std::fprintf(s_slide_show_file, "\"\n");
434 s_quotes = false;
435 }
436 char mn[MAX_MNEMONIC]{};
437 get_mnemonic(key, mn);
438 if (*mn)
439 {
440 std::fprintf(s_slide_show_file, "%s", mn);
441 }
442 else if (check_vid_mode_key(key) >= 0)
443 {
444 std::fputs(vid_mode_key_name(key).c_str(), s_slide_show_file);
445 }
446 else // not ASCII and not FN key
447 {
448 std::fprintf(s_slide_show_file, "%4d", key);
449 }
450 std::fputc('\n', s_slide_show_file);
451 }
452}

454// suspend process # of seconds
455static void sleep_secs(const int secs)
456{
457 g_slides = SlidesMode::OFF;
458 const int iterations = secs * 100;
459 for (int i = 0; i < iterations; ++i)
460 {
461 if (driver_key_pressed() != 0)
462 {
463 // bailout if key hit
464 break;
465 }
466 // sleep 10ms per iteration
467 std::this_thread::sleep_for(std::chrono::milliseconds(10));
468 }
469 g_slides = SlidesMode::PLAY;
470}

480// handle_special_keys
481//
482// First, do some slideshow processing. Then handle F1 and TAB display.
483//
484// Because we want context-sensitive help to work everywhere, with the
485// help to display indicated by a non-zero value in help_mode, we need
486// to trap the F1 key at a very low level. The same is true of the
487// TAB display.
488//
489// What we do here is check for these keys and invoke their displays.
490// To avoid a recursive invoke of help(), a static is used to avoid
491// recursing on ourselves as help will invoke get key!
492//
493int handle_special_keys(int ch)
494{
495 if (g_slides == SlidesMode::PLAY)
496 {
497 if (ch == ID_KEY_ESC)
498 {
499 stop_slide_show();
500 ch = 0;
501 }
502 else if (!ch)
503 {
504 ch = slide_show();
505 }
506 }
507 else if (g_slides == SlidesMode::RECORD && ch)
508 {
509 record_show(ch);
510 }
511
512 static bool inside_help = false;
513 if (ID_KEY_F1 == ch && g_help_mode != HelpLabels::HELP_INDEX && !inside_help)
514 {
515 inside_help = true;
516 help::help();
517 inside_help = false;
518 ch = 0;
519 }
520 else if (ID_KEY_TAB == ch && g_tab_enabled)
521 {
522 ValueSaver saved_tab_mode(g_tab_enabled, false);
523 tab_display();
524 ch = 0;
525 }
526
527 return ch;
528}
