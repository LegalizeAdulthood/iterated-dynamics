47bool show_temp_msg(const char *msg)
48{
49 static size_t size = 0;
50 char buffer[41];
51
52 std::strncpy(buffer, msg, 40);
53 buffer[40] = 0; // ensure max message len of 40 chars
54 if (driver_is_disk()) // disk video, screen in text mode, easy
55 {
56 dvid_status(0, buffer);
57 return false;
58 }
59 if (g_first_init) // & cmdfiles hasn't finished 1st try
60 {
61 // TODO: handle this in an OS-specific way
62 std::printf("%s\n", buffer);
63 return false;
64 }
65
66 const int x_repeat = g_screen_x_dots >= 640 ? 2 : 1;
67 const int y_repeat = g_screen_y_dots >= 300 ? 2 : 1;
68 s_text_x_dots = static_cast<int>(std::strlen(buffer)) * x_repeat * 8;
69 s_text_y_dots = y_repeat * 8;
70
71 size = static_cast<long>(s_text_x_dots) * static_cast<long>(s_text_y_dots);
72 if (s_text_save.size() != size)
73 {
74 s_text_save.clear();
75 }
76 int save_screen_x_offset = g_logical_screen.x_offset;
77 int save_screen_y_offset = g_logical_screen.y_offset;
78 g_logical_screen.y_offset = 0;
79 g_logical_screen.x_offset = 0;
80 if (s_text_save.empty()) // only save screen first time called
81 {
82 s_text_save.resize(s_text_x_dots*s_text_y_dots);
83 for (int i = 0; i < s_text_y_dots; ++i)
84 {
85 read_span(i, 0, s_text_x_dots-1, &s_text_save[s_text_x_dots*i]);
86 }
87 }
88
89 find_special_colors(); // get g_color_dark & g_color_bright set
90 driver_display_string(0, 0, g_color_bright, g_color_dark, buffer);
91 g_logical_screen.x_offset = save_screen_x_offset;
92 g_logical_screen.y_offset = save_screen_y_offset;
93
94 return false;
95}

107 g_logical_screen.y_offset = 0;
108 g_logical_screen.x_offset = 0;
109 for (int i = 0; i < s_text_y_dots; ++i)
110 {
111 write_span(i, 0, s_text_x_dots-1, &s_text_save[s_text_x_dots*i]);
112 }
113 if (!g_using_jiim) // jiim frees memory with freetempmsg()
114 {
115 s_text_save.clear();
116 }
117 g_logical_screen.x_offset = save_screen_x_offset;
118 g_logical_screen.y_offset = save_screen_y_offset;
119 driver_flush();
120 }
