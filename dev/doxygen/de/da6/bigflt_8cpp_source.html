34/**********************************************************************/
35// strtobf() - converts a string into a bigfloat
36// r - pointer to a bigfloat
37// s - string in the floating point format [+-][dig].[dig]e[+-][dig]
38// note: the string may not be empty or have extra space.
39// It may use scientific notation.
40// USES: g_bf_tmp1
41
42BigFloat str_to_bf(BigFloat r, const char *s)
43{
44 Byte ones_byte;
45 bool sign_flag = false;
46 const char *l;
47 int power_ten = 0;
48
49 clear_bf(r);
50
51 if (s[0] == '+') // for + sign
52 {
53 s++;
54 }
55 else if (s[0] == '-') // for neg sign
56 {
57 sign_flag = true;
58 s++;
59 }
60
61 const char *d = std::strchr(s, '.');
62 const char *e = std::strchr(s, 'e');
63 if (e == nullptr)
64 {
65 e = std::strchr(s, 'E');
66 }
67 if (e != nullptr)
68 {
69 power_ten = std::atoi(e+1); // read in the e (x10^) part
70 l = e - 1; // just before e
71 }
72 else
73 {
74 l = s + std::strlen(s) - 1; // last digit
75 }
76
77 if (d != nullptr) // is there a decimal point?
78 {
79 while (*l >= '0' && *l <= '9') // while a digit
80 {
81 ones_byte = static_cast<Byte>(*l-- - '0');
82 int_to_bf(g_bf_tmp1, ones_byte);
83 unsafe_add_a_bf(r, g_bf_tmp1);
84 div_a_bf_int(r, 10);
85 }
86
87 if (*l-- == '.') // the digit was found
88 {
89 bool keep_looping = *l >= '0' && *l <= '9' && l >= s;
90 while (keep_looping) // while a digit
91 {
92 ones_byte = static_cast<Byte>(*l-- - '0');
93 int_to_bf(g_bf_tmp1, ones_byte);
94 unsafe_add_a_bf(r, g_bf_tmp1);
95 keep_looping = *l >= '0' && *l <= '9' && l >= s;
96 if (keep_looping)
97 {
98 div_a_bf_int(r, 10);
99 power_ten++; // increase the power of ten
100 }
101 }
102 }
103 }
104 else
105 {
106 bool keep_looping = *l >= '0' && *l <= '9' && l >= s;
107 while (keep_looping) // while a digit
108 {
109 ones_byte = static_cast<Byte>(*l-- - '0');
110 int_to_bf(g_bf_tmp1, ones_byte);
111 unsafe_add_a_bf(r, g_bf_tmp1);
112 keep_looping = *l >= '0' && *l <= '9' && l >= s;
113 if (keep_looping)
114 {
115 div_a_bf_int(r, 10);
116 power_ten++; // increase the power of ten
117 }
118 }
119 }
120
121 if (power_ten > 0)
122 {
123 for (; power_ten > 0; power_ten--)
124 {
125 mult_a_bf_int(r, 10);
126 }
127 }
128 else if (power_ten < 0)
129 {
130 for (; power_ten < 0; power_ten++)
131 {
132 div_a_bf_int(r, 10);
133 }
134 }
135 if (sign_flag)
136 {
137 neg_a_bf(r);
138 }
139
140 return r;
141}

226/*********************************************************************/
227// bn = floor(bf)
228// Converts a bigfloat to a bignumber (integer)
229// g_bf_length must be at least g_bn_length+2
230BigNum bf_to_bn(BigNum n, BigFloat f)
231{
232 const int f_exp = static_cast<S16>(BIG_ACCESS16(f + g_bf_length));
233 if (f_exp >= g_int_length)
234 {
235 // if it's too big, use max value
236 max_bn(n);
237 if (is_bf_neg(f))
238 {
239 neg_a_bn(n);
240 }
241 return n;
242 }
243
244 if (-f_exp > g_bn_length - g_int_length) // too small, return zero
245 {
246 clear_bn(n);
247 return n;
248 }
249
250 // already checked for over/underflow, this should be ok
251 const int move_bytes = g_bn_length - g_int_length + f_exp + 1;
252 std::memcpy(n, f + g_bf_length - move_bytes - 1, move_bytes);
253 const Byte hi_byte = *(f + g_bf_length - 1);
254 std::memset(n+move_bytes, hi_byte, g_bn_length-move_bytes); // sign extends
255 return n;
256}

258/*********************************************************************/
259// bf = bn
260// Converts a bignumber (integer) to a bigfloat
261// g_bf_length must be at least g_bn_length+2
262BigFloat bn_to_bf(BigFloat f, BigNum n)
263{
264 std::memcpy(f+g_bf_length-g_bn_length-1, n, g_bn_length);
265 std::memset(f, 0, g_bf_length - g_bn_length - 1);
266 *(f+g_bf_length-1) = static_cast<Byte>(is_bn_neg(n) ? 0xFF : 0x00); // sign extend
267 BIG_SET16(f+g_bf_length, static_cast<S16>(g_int_length - 1)); // exp
268 norm_bf(f);
269 return f;
270}

273// b = l
274// Converts a long to a bigfloat
275BigFloat int_to_bf(BigFloat r, const long value)
276{
277 clear_bf(r);
278 BIG_SET32(r+g_bf_length-4, static_cast<S32>(value));
279 BIG_SET16(r+g_bf_length, static_cast<S16>(2));
280 norm_bf(r);
281 return r;

284/*********************************************************************/
285// l = floor(b), floor rounds down
286// Converts a bigfloat to a long
287// note: a bf value of 2.999... will be return a value of 2, not 3
288long bf_to_int(BigFloat f)
289{
290 long result;
291
292 const int f_exp = static_cast<S16>(BIG_ACCESS16(f + g_bf_length));
293 if (f_exp > 3)
294 {
295 result = 0x7FFFFFFFL;
296 if (is_bf_neg(f))
297 {
298 result = -result;
299 }
300 return result;
301 }
302 result = BIG_ACCESS32(f+g_bf_length-5);
303 result >>= 8*(3-f_exp);
304 return result;
305}

337/********************************************************************/
338// r = 1/n
339// uses g_bf_tmp1 - g_bf_tmp2 - global temp bigfloats
340// SIDE-EFFECTS:
341// n ends up as |n|/256^exp Make copy first if necessary.
342BigFloat unsafe_inv_bf(BigFloat r, BigFloat n)
343{
344 bool sign_flag = false;
345
346 // use Newton's recursive method for zeroing in on 1/n : r=r(2-rn)
347
348 if (is_bf_neg(n))
349 {
350 // will be a lot easier to deal with just positives
351 sign_flag = true;
352 neg_a_bf(n);
353 }
354
355 const int f_exp = static_cast<S16>(BIG_ACCESS16(n + g_bf_length));
356 BIG_SET16(n+g_bf_length, static_cast<S16>(0)); // put within LDouble range
357
358 LDouble f = bf_to_float(n);
359 if (f == 0) // division by zero
360 {
361 max_bf(r);
362 return r;
363 }
364 f = 1/f; // approximate inverse
365
366 // With Newton's Method, there is no need to calculate all the digits
367 // every time. The precision approximately doubles each iteration.
368 // Save original values.
369 // orig_bftmp1 not needed here
370 int orig_bf_length = g_bf_length;
371 int orig_bn_length = g_bn_length;
372 int orig_padding = g_padding;
373 int orig_r_length = g_r_length;
374 int orig_shift_factor = g_shift_factor;
375 int orig_r_bf_length = g_r_bf_length;
376 BigFloat orig_r = r;
377 BigFloat orig_n = n;
378 // orig_bftmp1 = g_bf_tmp1;
379
380 // calculate new starting values
381 g_bn_length = g_int_length + static_cast<int>((LDBL_DIG / LOG10_256)) + 1; // round up
382 g_bn_length = std::min(g_bn_length, orig_bn_length);
383 calc_lengths();
384
385 // adjust pointers
386 r = orig_r + orig_bf_length - g_bf_length;
387 // g_bf_tmp1 = orig_bftmp1 + orig_bf_length - g_bf_length;
388
389 float_to_bf(r, f); // start with approximate inverse
390
391 for (int i = 0; i < 25; i++) // safety net, this shouldn't ever be needed
392 {
393 // adjust lengths
394 g_bn_length <<= 1; // double precision
395 g_bn_length = std::min(g_bn_length, orig_bn_length);
396 calc_lengths();
397 r = orig_r + orig_bf_length - g_bf_length;
398 n = orig_n + orig_bf_length - g_bf_length;
399 // g_bf_tmp1 = orig_bftmp1 + orig_bf_length - g_bf_length;
400
401 unsafe_mult_bf(g_bf_tmp1, r, n); // g_bf_tmp1=rn
402 int_to_bf(g_bf_tmp2, 1); // will be used as 1.0
403
404 // There seems to very little difficulty getting g_bf_tmp1 to be EXACTLY 1
405 if (g_bf_length == orig_bf_length && cmp_bf(g_bf_tmp1, g_bf_tmp2) == 0)
406 {
407 break;
408 }
409
410 int_to_bf(g_bf_tmp2, 2); // will be used as 2.0
411 unsafe_sub_a_bf(g_bf_tmp2, g_bf_tmp1); // g_bf_tmp2=2-rn
412 unsafe_mult_bf(g_bf_tmp1, r, g_bf_tmp2); // g_bf_tmp1=r(2-rn)
413 copy_bf(r, g_bf_tmp1); // r = g_bf_tmp1
414 }
415
416 // restore original values
417 g_bf_length = orig_bf_length;
418 g_bn_length = orig_bn_length;
419 g_padding = orig_padding;
420 g_r_length = orig_r_length;
421 g_shift_factor = orig_shift_factor;
422 g_r_bf_length = orig_r_bf_length;
423 r = orig_r;
424 // g_bf_tmp1 = orig_bftmp1;
425
426 if (sign_flag)
427 {
428 neg_a_bf(r);
429 }
430 int r_exp = static_cast<S16>(BIG_ACCESS16(r + g_bf_length));
431 r_exp -= f_exp;
432 BIG_SET16(r+g_bf_length, static_cast<S16>(r_exp)); // adjust result exponent
433 return r;
434}

436/********************************************************************/
437// r = n1/n2
438// r - result of length g_bf_length
439// uses g_bf_tmp1 - g_bf_tmp2 - global temp bigfloats
440// SIDE-EFFECTS:
441// n1, n2 end up as |n1|/256^x, |n2|/256^x
442// Make copies first if necessary.
443BigFloat unsafe_div_bf(BigFloat r, BigFloat n1, BigFloat n2)
444{
445 // first, check for valid data
446
447 const int a_exp = static_cast<S16>(BIG_ACCESS16(n1 + g_bf_length));
448 BIG_SET16(n1+g_bf_length, static_cast<S16>(0)); // put within LDouble range
449
450 LDouble a = bf_to_float(n1);
451 if (a == 0) // division into zero
452 {
453 clear_bf(r); // return 0
454 return r;
455 }
456
457 const int b_exp = static_cast<S16>(BIG_ACCESS16(n2 + g_bf_length));
458 BIG_SET16(n2+g_bf_length, static_cast<S16>(0)); // put within LDouble range
459
460 LDouble b = bf_to_float(n2);
461 if (b == 0) // division by zero
462 {
463 max_bf(r);
464 return r;
465 }
466
467 unsafe_inv_bf(r, n2);
468 unsafe_mult_bf(g_bf_tmp1, n1, r);
469 copy_bf(r, g_bf_tmp1); // r = g_bf_tmp1
470
471 int r_exp = static_cast<S16>(BIG_ACCESS16(r + g_bf_length));
472 r_exp += a_exp - b_exp;
473 BIG_SET16(r+g_bf_length, static_cast<S16>(r_exp)); // adjust result exponent
474
475 return r;
476}

478/********************************************************************/
479// sqrt(r)
480// uses g_bf_tmp1 - g_bf_tmp3 - global temp bigfloats
481// SIDE-EFFECTS:
482// n ends up as |n|
483BigFloat unsafe_sqrt_bf(BigFloat r, BigFloat n)
484{
485 int almost_match = 0;
486
487 // use Newton's recursive method for zeroing in on sqrt(n): r=.5(r+n/r)
488
489 if (is_bf_neg(n))
490 {
491 // sqrt of a neg, return 0
492 clear_bf(r);
493 return r;
494 }
495
496 LDouble f = bf_to_float(n);
497 if (f == 0) // division by zero will occur
498 {
499 clear_bf(r); // sqrt(0) = 0
500 return r;
501 }
502 f = std::sqrt(f); // approximate square root
503 // no need to check overflow
504
505 // With Newton's Method, there is no need to calculate all the digits
506 // every time. The precision approximately doubles each iteration.
507 // Save original values.
508 int orig_bf_length = g_bf_length;
509 int orig_bn_length = g_bn_length;
510 int orig_padding = g_padding;
511 int orig_r_length = g_r_length;
512 int orig_shift_factor = g_shift_factor;
513 int orig_r_bf_length = g_r_bf_length;
514 BigFloat orig_r = r;
515 BigFloat orig_n = n;
516
517 // calculate new starting values
518 g_bn_length = g_int_length + static_cast<int>((LDBL_DIG / LOG10_256)) + 1; // round up
519 g_bn_length = std::min(g_bn_length, orig_bn_length);
520 calc_lengths();
521
522 // adjust pointers
523 r = orig_r + orig_bf_length - g_bf_length;
524
525 float_to_bf(r, f); // start with approximate sqrt
526
527 for (int i = 0; i < 25; i++) // safety net, this shouldn't ever be needed
528 {
529 // adjust lengths
530 g_bn_length <<= 1; // double precision
531 g_bn_length = std::min(g_bn_length, orig_bn_length);
532 calc_lengths();
533 r = orig_r + orig_bf_length - g_bf_length;
534 n = orig_n + orig_bf_length - g_bf_length;
535
536 unsafe_div_bf(g_bf_tmp3, n, r);
537 unsafe_add_a_bf(r, g_bf_tmp3);
538 half_a_bf(r);
539 if (g_bf_length == orig_bf_length)
540 {
541 const int comp = std::abs(cmp_bf(r, g_bf_tmp3)); // if match or almost match
542 if (comp < 8)
543 {
544 if (comp < 4 // perfect or near perfect match
545 || almost_match == 1) // close enough for 2nd time
546 {
547 break;
548 }
549 // this is the first time they almost matched
550 almost_match++;
551 }
552 }
553 }
554
555 // restore original values
556 g_bf_length = orig_bf_length;
557 g_bn_length = orig_bn_length;
558 g_padding = orig_padding;
559 g_r_length = orig_r_length;
560 g_shift_factor = orig_shift_factor;
561 g_r_bf_length = orig_r_bf_length;
562 r = orig_r;
563
564 return r;
565}

567/********************************************************************/
568// exp(r)
569// uses g_bf_tmp1, g_bf_tmp2, g_bf_tmp3 - global temp bigfloats
570BigFloat exp_bf(BigFloat r, BigFloat n)
571{
572 U16 fact = 1;
573 S16 *test_exp = reinterpret_cast<S16 *>(g_bf_tmp2 + g_bf_length);
574 S16 *r_exp = reinterpret_cast<S16 *>(r + g_bf_length);
575
576 if (is_bf_zero(n))
577 {
578 int_to_bf(r, 1);
579 return r;
580 }
581
582 // use Taylor Series (very slow convergence)
583 int_to_bf(r, 1); // start with r=1.0
584 copy_bf(g_bf_tmp2, r);
585 while (true)
586 {
587 copy_bf(g_bf_tmp1, n);
588 unsafe_mult_bf(g_bf_tmp3, g_bf_tmp2, g_bf_tmp1);
589 unsafe_div_bf_int(g_bf_tmp2, g_bf_tmp3, fact);
590 if (BIG_ACCESS_S16(test_exp) < BIG_ACCESS_S16(r_exp)-(g_bf_length-2))
591 {
592 break; // too small to register
593 }
594 unsafe_add_a_bf(r, g_bf_tmp2);
595 fact++;
596 }
597
598 return r;
599}

601/********************************************************************/
602// ln(r)
603// uses g_bf_tmp1 - g_bf_tmp6 - global temp bigfloats
604// SIDE-EFFECTS:
605// n ends up as |n|
606BigFloat unsafe_ln_bf(BigFloat r, BigFloat n)
607{
608 int almost_match = 0;
609
610 // use Newton's recursive method for zeroing in on ln(n): r=r+n*exp(-r)-1
611
612 if (is_bf_neg(n) || is_bf_zero(n))
613 {
614 // error, return largest neg value
615 max_bf(r);
616 neg_a_bf(r);
617 return r;
618 }
619
620 LDouble f = bf_to_float(n);
621 f = std::log(f); // approximate ln(x)
622 // no need to check overflow
623 // appears to be ok, do ln
624
625 // With Newton's Method, there is no need to calculate all the digits
626 // every time. The precision approximately doubles each iteration.
627 // Save original values.
628 int orig_bf_length = g_bf_length;
629 int orig_bn_length = g_bn_length;
630 int orig_padding = g_padding;
631 int orig_r_length = g_r_length;
632 int orig_shift_factor = g_shift_factor;
633 int orig_r_bf_length = g_r_bf_length;
634 BigFloat orig_r = r;
635 BigFloat orig_n = n;
636 BigFloat orig_bf_tmp5 = g_bf_tmp5;
637
638 // calculate new starting values
639 g_bn_length = g_int_length + static_cast<int>((LDBL_DIG / LOG10_256)) + 1; // round up
640 g_bn_length = std::min(g_bn_length, orig_bn_length);
641 calc_lengths();
642
643 // adjust pointers
644 r = orig_r + orig_bf_length - g_bf_length;
645 g_bf_tmp5 = orig_bf_tmp5 + orig_bf_length - g_bf_length;
646
647 float_to_bf(r, f); // start with approximate ln
648 neg_a_bf(r); // -r
649 copy_bf(g_bf_tmp5, r); // -r
650
651 for (int i = 0; i < 25; i++) // safety net, this shouldn't ever be needed
652 {
653 // adjust lengths
654 g_bn_length <<= 1; // double precision
655 g_bn_length = std::min(g_bn_length, orig_bn_length);
656 calc_lengths();
657 r = orig_r + orig_bf_length - g_bf_length;
658 n = orig_n + orig_bf_length - g_bf_length;
659 g_bf_tmp5 = orig_bf_tmp5 + orig_bf_length - g_bf_length;
660
661 exp_bf(g_bf_tmp6, r); // exp(-r)
662 unsafe_mult_bf(g_bf_tmp2, g_bf_tmp6, n); // n*exp(-r)
663 int_to_bf(g_bf_tmp4, 1);
664 unsafe_sub_a_bf(g_bf_tmp2, g_bf_tmp4); // n*exp(-r) - 1
665 unsafe_sub_a_bf(r, g_bf_tmp2); // -r - (n*exp(-r) - 1)
666 if (g_bf_length == orig_bf_length)
667 {
668 const int comp = std::abs(cmp_bf(r, g_bf_tmp5));
669 if(comp < 8) // if match or almost match
670 {
671 if (comp < 4 // perfect or near perfect match
672 || almost_match == 1) // close enough for 2nd time
673 {
674 break;
675 }
676 // this is the first time they almost matched
677 almost_match++;
678 }
679 }
680 copy_bf(g_bf_tmp5, r); // -r
681 }
682
683 // restore original values
684 g_bf_length = orig_bf_length;
685 g_bn_length = orig_bn_length;
686 g_padding = orig_padding;
687 g_r_length = orig_r_length;
688 g_shift_factor = orig_shift_factor;
689 g_r_bf_length = orig_r_bf_length;
690 r = orig_r;
691 g_bf_tmp5 = orig_bf_tmp5;
692
693 neg_a_bf(r); // -(-r)
694 return r;
695}

697/********************************************************************/
698// sincos_bf(r)
699// uses g_bf_tmp1 - g_bf_tmp2 - global temp bigfloats
700// SIDE-EFFECTS:
701// n ends up as |n| mod (pi/4)
702BigFloat unsafe_sin_cos_bf(BigFloat s, BigFloat c, BigFloat n)
703{
704 U16 fact = 2;
705 S16 *test_exp = reinterpret_cast<S16 *>(g_bf_tmp1 + g_bf_length);
706 S16 *c_exp = reinterpret_cast<S16 *>(c + g_bf_length);
707 S16 *s_exp = reinterpret_cast<S16 *>(s + g_bf_length);
708
709#ifndef CALCULATING_BIG_PI
710 // assure range 0 <= x < pi/4
711
712 if (is_bf_zero(n))
713 {
714 clear_bf(s); // sin(0) = 0
715 int_to_bf(c, 1); // cos(0) = 1
716 return s;
717 }
718
719 bool sign_sin = false;
720 if (is_bf_neg(n))
721 {
722 sign_sin = !sign_sin; // sin(-x) = -sin(x), odd; cos(-x) = cos(x), even
723 neg_a_bf(n);
724 }
725 // n >= 0
726
727 double_bf(g_bf_tmp1, g_bf_pi); // 2*pi
728 // this could be done with remainders, but it would probably be slower
729 while (cmp_bf(n, g_bf_tmp1) >= 0) // while n >= 2*pi
730 {
731 copy_bf(g_bf_tmp2, g_bf_tmp1);
732 unsafe_sub_a_bf(n, g_bf_tmp2);
733 }
734 // 0 <= n < 2*pi
735
736 bool sign_cos = false;
737 copy_bf(g_bf_tmp1, g_bf_pi); // pi
738 if (cmp_bf(n, g_bf_tmp1) >= 0) // if n >= pi
739 {
740 unsafe_sub_a_bf(n, g_bf_tmp1);
741 sign_sin = !sign_sin;
742 sign_cos = !sign_cos;
743 }
744 // 0 <= n < pi
745
746 half_bf(g_bf_tmp1, g_bf_pi); // pi/2
747 if (cmp_bf(n, g_bf_tmp1) > 0) // if n > pi/2
748 {
749 copy_bf(g_bf_tmp2, g_bf_pi);
750 unsafe_sub_bf(n, g_bf_tmp2, n);
751 sign_cos = !sign_cos;
752 }
753 // 0 <= n < pi/2
754
755 bool switch_sin_cos = false;
756 half_bf(g_bf_tmp1, g_bf_pi); // pi/2
757 half_a_bf(g_bf_tmp1); // pi/4
758 if (cmp_bf(n, g_bf_tmp1) > 0) // if n > pi/4
759 {
760 copy_bf(g_bf_tmp2, n);
761 half_bf(g_bf_tmp1, g_bf_pi); // pi/2
762 unsafe_sub_bf(n, g_bf_tmp1, g_bf_tmp2); // pi/2 - n
763 switch_sin_cos = !switch_sin_cos;
764 }
765 // 0 <= n < pi/4
766
767 // this looks redundant, but n could now be zero when it wasn't before
768 if (is_bf_zero(n))
769 {
770 clear_bf(s); // sin(0) = 0
771 int_to_bf(c, 1); // cos(0) = 1
772 return s;
773 }
774
775 // at this point, the double angle trig identities could be used as many
776 // times as desired to reduce the range to pi/8, pi/16, etc... Each time
777 // the range is cut in half, the number of iterations required is reduced
778 // by "quite a bit." It's just a matter of testing to see what gives the
779 // optimal results.
780 // halves = g_bf_length / 10; // this is experimental
781 const int halves = 1;
782 for (int i = 0; i < halves; i++)
783 {
784 half_a_bf(n);
785 }
786#endif
787
788 // use Taylor Series (very slow convergence)
789 copy_bf(s, n); // start with s=n
790 int_to_bf(c, 1); // start with c=1
791 copy_bf(g_bf_tmp1, n); // the current x^n/n!
792 bool sin_done = false;
793 bool cos_done = false;
794 bool k = false;
795 do
796 {
797 // even terms for cosine
798 copy_bf(g_bf_tmp2, g_bf_tmp1);
799 unsafe_mult_bf(g_bf_tmp1, g_bf_tmp2, n);
800 div_a_bf_int(g_bf_tmp1, fact++);
801 if (!cos_done)
802 {
803 cos_done =
804 BIG_ACCESS_S16(test_exp) < BIG_ACCESS_S16(c_exp) - (g_bf_length - 2); // too small to register
805 if (!cos_done)
806 {
807 if (k) // alternate between adding and subtracting
808 {
809 unsafe_add_a_bf(c, g_bf_tmp1);
810 }
811 else
812 {
813 unsafe_sub_a_bf(c, g_bf_tmp1);
814 }
815 }
816 }
817
818 // odd terms for sine
819 copy_bf(g_bf_tmp2, g_bf_tmp1);
820 unsafe_mult_bf(g_bf_tmp1, g_bf_tmp2, n);
821 div_a_bf_int(g_bf_tmp1, fact++);
822 if (!sin_done)
823 {
824 sin_done =
825 BIG_ACCESS_S16(test_exp) < BIG_ACCESS_S16(s_exp) - (g_bf_length - 2); // too small to register
826 if (!sin_done)
827 {
828 if (k) // alternate between adding and subtracting
829 {
830 unsafe_add_a_bf(s, g_bf_tmp1);
831 }
832 else
833 {
834 unsafe_sub_a_bf(s, g_bf_tmp1);
835 }
836 }
837 }
838 k = !k; // toggle
839#if defined(CALCULATING_BIG_PI) && !defined(_WIN32)
840 std::printf("."); // lets you know it's doing something
841#endif
842 }
843 while (!cos_done || !sin_done);
844
845#ifndef CALCULATING_BIG_PI
846 // now need to undo what was done by cutting angles in half
847 for (int i = 0; i < halves; i++)
848 {
849 unsafe_mult_bf(g_bf_tmp2, s, c); // no need for safe mult
850 double_bf(s, g_bf_tmp2); // sin(2x) = 2*sin(x)*cos(x)
851 unsafe_square_bf(g_bf_tmp2, c);
852 double_a_bf(g_bf_tmp2);
853 int_to_bf(g_bf_tmp1, 1);
854 unsafe_sub_bf(c, g_bf_tmp2, g_bf_tmp1); // cos(2x) = 2*cos(x)*cos(x) - 1
855 }
856
857 if (switch_sin_cos)
858 {
859 copy_bf(g_bf_tmp1, s);
860 copy_bf(s, c);
861 copy_bf(c, g_bf_tmp1);
862 }
863 if (sign_sin)
864 {
865 neg_a_bf(s);
866 }
867 if (sign_cos)
868 {
869 neg_a_bf(c);
870 }
871#endif
872
873 return s; // return sine I guess
874}

876/********************************************************************/
877// atan(r)
878// uses g_bf_tmp1 - g_bf_tmp5 - global temp bigfloats
879// SIDE-EFFECTS:
880// n ends up as |n| or 1/|n|
881BigFloat unsafe_atan_bf(BigFloat r, BigFloat n)
882{
883 int almost_match = 0;
884 bool sign_flag = false;
885
886 // use Newton's recursive method for zeroing in on atan(n): r=r-cos(r)(sin(r)-n*cos(r))
887
888 if (is_bf_neg(n))
889 {
890 sign_flag = true;
891 neg_a_bf(n);
892 }
893
894 // If n is very large, atanl() won't give enough decimal places to be a
895 // good enough initial guess for Newton's Method. If it is larger than
896 // say, 1, atan(n) = pi/2 - acot(n) = pi/2 - atan(1/n).
897
898 LDouble f = bf_to_float(n);
899 const bool large_arg = f > 1.0;
900 if (large_arg)
901 {
902 unsafe_inv_bf(g_bf_tmp3, n);
903 copy_bf(n, g_bf_tmp3);
904 f = bf_to_float(n);
905 }
906
907 clear_bf(g_bf_tmp3); // not really necessary, but makes things more consistent
908
909 // With Newton's Method, there is no need to calculate all the digits
910 // every time. The precision approximately doubles each iteration.
911 // Save original values.
912 int orig_bf_length = g_bf_length;
913 int orig_bn_length = g_bn_length;
914 int orig_padding = g_padding;
915 int orig_r_length = g_r_length;
916 int orig_shift_factor = g_shift_factor;
917 int orig_r_bf_length = g_r_bf_length;
918 BigFloat orig_bf_pi = g_bf_pi;
919 BigFloat orig_r = r;
920 BigFloat orig_n = n;
921 BigFloat orig_bf_tmp3 = g_bf_tmp3;
922
923 // calculate new starting values
924 g_bn_length = g_int_length + static_cast<int>((LDBL_DIG / LOG10_256)) + 1; // round up
925 g_bn_length = std::min(g_bn_length, orig_bn_length);
926 calc_lengths();
927
928 // adjust pointers
929 r = orig_r + orig_bf_length - g_bf_length;
930 g_bf_pi = orig_bf_pi + orig_bf_length - g_bf_length;
931 g_bf_tmp3 = orig_bf_tmp3 + orig_bf_length - g_bf_length;
932
933 f = atanl(f); // approximate arctangent
934 // no need to check overflow
935
936 float_to_bf(r, f); // start with approximate atan
937 copy_bf(g_bf_tmp3, r);
938
939 for (int i = 0; i < 25; i++) // safety net, this shouldn't ever be needed
940 {
941 // adjust lengths
942 g_bn_length <<= 1; // double precision
943 g_bn_length = std::min(g_bn_length, orig_bn_length);
944 calc_lengths();
945 r = orig_r + orig_bf_length - g_bf_length;
946 n = orig_n + orig_bf_length - g_bf_length;
947 g_bf_pi = orig_bf_pi + orig_bf_length - g_bf_length;
948 g_bf_tmp3 = orig_bf_tmp3 + orig_bf_length - g_bf_length;
949
950#if defined(CALCULATING_BIG_PI) && !defined(_WIN32)
951 std::printf("\natan() loop #%i, g_bf_length=%i\nsincos() loops\n", i, g_bf_length);
952#endif
953 unsafe_sin_cos_bf(g_bf_tmp4, g_bf_tmp5, g_bf_tmp3); // sin(r), cos(r)
954 copy_bf(g_bf_tmp3, r); // restore g_bf_tmp3 from sincos_bf()
955 copy_bf(g_bf_tmp1, g_bf_tmp5);
956 unsafe_mult_bf(g_bf_tmp2, n, g_bf_tmp1); // n*cos(r)
957 unsafe_sub_a_bf(g_bf_tmp4, g_bf_tmp2); // sin(r) - n*cos(r)
958 unsafe_mult_bf(g_bf_tmp1, g_bf_tmp5, g_bf_tmp4); // cos(r) * (sin(r) - n*cos(r))
959 copy_bf(g_bf_tmp3, r);
960 unsafe_sub_a_bf(r, g_bf_tmp1); // r - cos(r) * (sin(r) - n*cos(r))
961#if defined(CALCULATING_BIG_PI) && !defined(_WIN32)
962 putchar('\n');
963 bf_hexdump(r);
964#endif
965 if (g_bf_length == orig_bf_length)
966 {
967 const int comp = std::abs(cmp_bf(r, g_bf_tmp3));
968 if (comp < 8) // if match or almost match
969 {
970#if defined(CALCULATING_BIG_PI) && !defined(_WIN32)
971 std::printf("atan() loop comp=%i\n", comp);
972#endif
973 if (comp < 4 // perfect or near perfect match
974 || almost_match == 1) // close enough for 2nd time
975 {
976 break;
977 }
978 // this is the first time they almost matched
979 almost_match++;
980 }
981#if defined(CALCULATING_BIG_PI) && !defined(_WIN32)
982 else
983 {
984 std::printf("atan() loop comp=%i\n", comp);
985 }
986#endif
987 }
988
989 copy_bf(g_bf_tmp3, r); // make a copy for later comparison
990 }
991
992 // restore original values
993 g_bf_length = orig_bf_length;
994 g_bn_length = orig_bn_length;
995 g_padding = orig_padding;
996 g_r_length = orig_r_length;
997 g_shift_factor = orig_shift_factor;
998 g_r_bf_length = orig_r_bf_length;
999 g_bf_pi = orig_bf_pi;
1000 r = orig_r;
1001 g_bf_tmp3 = orig_bf_tmp3;
1002
1003 if (large_arg)
1004 {
1005 half_bf(g_bf_tmp3, g_bf_pi); // pi/2
1006 sub_a_bf(g_bf_tmp3, r); // pi/2 - atan(1/n)
1007 copy_bf(r, g_bf_tmp3);
1008 }
1009
1010 if (sign_flag)
1011 {
1012 neg_a_bf(r);
1013 }
1014 return r;
1015}

1019// uses g_bf_tmp1 - g_bf_tmp6 - global temp bigfloats
1020BigFloat unsafe_atan2_bf(BigFloat r, BigFloat ny, BigFloat nx)
1021{

1026 {
1027 if (sign_x < 0)

1030 }
1031 else // sign_x >= 0 positive x axis, 0

1033 clear_bf(r);

1036 }
1037 if (sign_x == 0)
1038 {
1039 copy_bf(r, g_bf_pi); // y axis
1040 half_a_bf(r); // +90 deg
1041 if (sign_y < 0)
1042 {
1043 neg_a_bf(r); // -90 deg
1044 }
1045 return r;
1046 }

1050 neg_a_bf(ny);

1054 neg_a_bf(nx);
1055 }

1061 }
1062 if (sign_y < 0)
1063 {
1064 neg_a_bf(r);
1065 }
1066 return r;

1268// The following used to be in bigfltc.c
1269/********************************************************************/
1270// normalize big float
1271BigFloat norm_bf(BigFloat r)
1272{
1273 S16 *r_exp = reinterpret_cast<S16 *>(r + g_bf_length);
1274
1275 // check for overflow
1276 Byte hi_byte = r[g_bf_length - 1];
1277 if (hi_byte != 0x00 && hi_byte != 0xFF)
1278 {
1279 std::memmove(r, r+1, g_bf_length-1);
1280 r[g_bf_length-1] = static_cast<Byte>(hi_byte & 0x80 ? 0xFF : 0x00);
1281 BIG_SET_S16(r_exp, BIG_ACCESS_S16(r_exp) + static_cast<S16>(1)); // exp
1282 }
1283
1284 // check for underflow
1285 else
1286 {
1287 int scale;
1288 for (scale = 2; scale < g_bf_length && r[g_bf_length-scale] == hi_byte; scale++)
1289 {
1290 // do nothing
1291 }
1292 if (scale == g_bf_length && hi_byte == 0) // zero
1293 {
1294 BIG_SET_S16(r_exp, 0);
1295 }
1296 else
1297 {
1298 scale -= 2;
1299 if (scale > 0) // it did underflow
1300 {
1301 std::memmove(r+scale, r, g_bf_length-scale-1);
1302 std::memset(r, 0, scale);
1303 BIG_SET_S16(r_exp, BIG_ACCESS_S16(r_exp) - static_cast<S16>(scale)); // exp
1304 }
1305 }
1306 }
1307
1308 return r;
1309}

1320/******************************************************/
1321// adjust n1, n2 for before addition or subtraction
1322// by forcing exp's to match.
1323// returns the value of the adjusted exponents
1324S16 adjust_bf_add(BigFloat n1, BigFloat n2)
1325{
1326 int scale;
1327 int fill_byte;
1328 S16 r_exp;
1329
1330 // scale n1 or n2
1331 // compare exp's
1332 S16 *n1_exp = reinterpret_cast<S16 *>(n1 + g_bf_length);
1333 S16 *n2_exp = reinterpret_cast<S16 *>(n2 + g_bf_length);
1334 if (BIG_ACCESS_S16(n1_exp) > BIG_ACCESS_S16(n2_exp))
1335 {
1336 // scale n2
1337 scale = BIG_ACCESS_S16(n1_exp) - BIG_ACCESS_S16(n2_exp); // n1exp - n2exp
1338 if (scale < g_bf_length)
1339 {

1384/****************************************************************************/
1385// n1 != n2 ?
1386// RETURNS:
1387// if n1 == n2 returns 0
1388// if n1 > n2 returns a positive (bytes left to go when mismatch occurred)
1389// if n1 < n2 returns a negative (bytes left to go when mismatch occurred)
1390
1391int cmp_bf(BigFloat n1, BigFloat n2)
1392{
1393 // compare signs
1394 const int sign1 = sign_bf(n1);
1395 const int sign2 = sign_bf(n2);
1396 if (sign1 > sign2)
1397 {
1398 return g_bf_length;
1399 }
1400 if (sign1 < sign2)
1401 {
1402 return -g_bf_length;
1403 }
1404 // signs are the same
1405
1406 // compare exponents, using signed comparisons
1407 S16 *n1_exp = reinterpret_cast<S16 *>(n1 + g_bf_length);
1408 S16 *n2_exp = reinterpret_cast<S16 *>(n2 + g_bf_length);
1409 if (BIG_ACCESS_S16(n1_exp) > BIG_ACCESS_S16(n2_exp))
1410 {
1411 return sign1*g_bf_length;
1412 }
1413 if (BIG_ACCESS_S16(n1_exp) < BIG_ACCESS_S16(n2_exp))
1414 {
1415 return -sign1*g_bf_length;
1416 }
1417
1418 // To get to this point, the signs must match
1419 // so unsigned comparison is ok.
1420 // two bytes at a time
1421 for (int i = g_bf_length-2; i >= 0; i -= 2)
1422 {
1423 const U16 value1 = BIG_ACCESS16(n1 + i);
1424 const U16 value2 = BIG_ACCESS16(n2 + i);
1425 if (value1 > value2)
1426 {
1427 // now determine which of the two bytes was different
1428 if ((value1&0xFF00) > (value2&0xFF00)) // compare just high bytes
1429 {
1430 return i+2; // high byte was different
1431 }
1432
1433 return i+1; // low byte was different
1434 }
1435 if (value1 < value2)
1436 {
1437 // now determine which of the two bytes was different
1438 if ((value1&0xFF00) < (value2&0xFF00)) // compare just high bytes
1439 {
1440 return -(i+2); // high byte was different
1441 }
1442
1443 return -(i+1); // low byte was different
1444 }
1445 }
1446 return 0;
1447}

1929// extracts the mantissa and exponent of f
1930// finds m and n such that 1<=|m|<b and f = m*b^n
1931// n is stored in *exp_ptr and m is returned, sort of like frexp()
1932LDouble extract_value(LDouble f, LDouble b, int *exp_ptr)
1933{
1934 LDouble value[15];
1935
1936 if (b <= 0 || f == 0)
1937 {
1938 *exp_ptr = 0;
1939 return 0;
1940 }
1941
1942 const LDouble orig_b = b;
1943 const LDouble af = f >= 0 ? f : -f; // abs value
1944 LDouble ff = af > 1 ? af : 1 / af;
1945 int n = 0;
1946 unsigned power_two = 1;
1947 while (b < ff)
1948 {
1949 value[n] = b;
1950 n++;
1951 power_two <<= 1;
1952 b *= b;
1953 }
1954
1955 *exp_ptr = 0;
1956 for (; n > 0; n--)
1957 {
1958 power_two >>= 1;
1959 if (value[n-1] < ff)
1960 {
1961 ff /= value[n-1];
1962 *exp_ptr += power_two;
1963 }
1964 }
1965 if (f < 0)
1966 {
1967 ff = -ff;
1968 }
1969 if (af < 1)
1970 {
1971 ff = orig_b/ff;
1972 *exp_ptr = -*exp_ptr - 1;
1973 }
1974
1975 return ff;
1976}

1979// calculates and returns the value of f*b^n
1980// sort of like ldexp()
1981LDouble scale_value(LDouble f, LDouble b , const int n)
1982{
1983 LDouble total = 1;
1984
1985 if (b == 0 || f == 0)
1986 {
1987 return 0;
1988 }
1989
1990 if (n == 0)
1991 {
1992 return f;
1993 }
1994
1995 int an = std::abs(n);
1996
1997 while (an != 0)
1998 {
1999 if (an & 0x0001)
2000 {
2001 total *= b;
2002 }
2003 b *= b;
2004 an >>= 1;
2005 }
2006
2007 if (n > 0)
2008 {
2009 f *= total;
2010 }
2011 else // n < 0
2012 {
2013 f /= total;
2014 }
2015 return f;
2016}
