59int select_video_mode(const int current_mode)
60{
61 std::vector<int> attributes;
62
63 attributes.resize(g_video_table_len);
64 s_entry_nums.resize(g_video_table_len);
65 // init tables
66 std::fill(attributes.begin(), attributes.end(), 1);
67 std::iota(s_entry_nums.begin(), s_entry_nums.end(), 0);
68 std::sort(s_entry_nums.begin(), s_entry_nums.end(), ent_less);
69
70 // pick default mode
71 if (current_mode < 0)
72 {
73 g_video_entry.colors = 256;
74 }
75 else
76 {
77 g_video_entry = g_video_table[current_mode];
78 }
79 int i;
80 for (i = 0; i < g_video_table_len; ++i) // find default mode
81 {
82 if (g_video_entry.colors == g_video_table[s_entry_nums[i]].colors //
83 && (current_mode < 0 || g_video_entry == g_video_table[s_entry_nums[i]]))
84 {
85 break;
86 }
87 }
88 if (i >= g_video_table_len) // no match, default to first entry
89 {
90 i = 0;
91 }
92
93 {
94 ValueSaver saved_tab_mode{g_tab_enabled, false};
95 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_VIDEO_MODE};
96 s_modes_changed = false;
97 i = full_screen_choice(ChoiceFlags::HELP, "Select Video Mode",
98 "key...name..........xdot..ydot.colr.driver......comment......", nullptr, g_video_table_len,
99 nullptr, attributes.data(), 1, 16, 74, i, format_vid_table, nullptr, nullptr, check_mode_key);
100 }
101 if (i == -1)
102 {
103 // update id.cfg for new key assignments
104 if (s_modes_changed
105 && g_bad_config == ConfigStatus::OK
106 && stop_msg(StopMsgFlags::CANCEL | StopMsgFlags::NO_BUZZER | StopMsgFlags::INFO_ONLY,
107 "Save new function key assignments or cancel changes?") == 0)
108 {
109 update_id_cfg();
110 }
111 return -1;
112 }
113 // picked by function key or ENTER key
114 i = i < 0 ? -1 - i : s_entry_nums[i];
115 // the selected entry now in g_video_entry
116 g_video_entry = g_video_table[i];
117
118 // copy id.cfg table to resident table, note selected entry
119 int k = 0;
120 for (i = 0; i < g_video_table_len; ++i)
121 {
122 if (g_video_table[i].key > 0)
123 {
124 if (g_video_entry == g_video_table[i])
125 {
126 k = g_video_table[i].key;
127 }
128 }
129 }
130 int ret = k;
131 if (k == 0) // selected entry not a copied (assigned to key) one
132 {
133 g_video_table[MAX_VIDEO_MODES - 1] = g_video_entry;
134 ret = 1400; // special value for check_vidmode_key
135 }
136
137 // update id.cfg for new key assignments
138 if (s_modes_changed && g_bad_config == ConfigStatus::OK)
139 {
140 update_id_cfg();
141 }
142
143 return ret;
144}

146static int check_mode_key(const int key, const int choice)
147{
148 if (const int i = check_vid_mode_key(key); i >= 0)
149 {
150 return -1 - i;
151 }
152
153 if (key != '-' && key != '+')
154 {
155 return 0;
156 }
157
158 const int i = s_entry_nums[choice];
159 if (g_video_table[i].key != 0 && g_video_table[i].key < ID_KEY_SHF_F1)
160 {
161 return 0;
162 }
163
164 if (g_bad_config != ConfigStatus::OK)
165 {
166 stop_msg("Missing or bad id.cfg file. Can't reassign keys.");
167 return 0;
168 }
169
170 if (key == '-')
171 {
172 // unassign key?
173 if (g_video_table[i].key >= ID_KEY_SHF_F1)
174 {
175 g_video_table[i].key = 0;
176 s_modes_changed = true;
177 }
178 return 0;
179 }
180
181 // assign key?
182 int ret = 0;
183 if (const int j = get_a_key_no_help(); j >= ID_KEY_SHF_F1 && j <= ID_KEY_ALT_F10)
184 {
185 for (int k = 0; k < g_video_table_len; ++k)
186 {
187 if (g_video_table[k].key == j)
188 {
189 g_video_table[k].key = 0;
190 ret = -1; // force redisplay
191 }
192 }
193 g_video_table[i].key = j;
194 s_modes_changed = true;
195 }
196 return ret;
197}

199static bool ent_less(const int lhs, const int rhs)
200{
201 int i = g_video_table[lhs].key;
202 if (i == 0)
203 {
204 i = 9999;
205 }
206 int j = g_video_table[rhs].key;
207 if (j == 0)
208 {
209 j = 9999;
210 }
211 return i < j || i == j && lhs < rhs;
212}

214static void update_id_cfg()
215{
216 const fs::path save_path{get_save_path(WriteFile::ID_CONFIG, "id.cfg")};
217 assert(!save_path.empty());
218 if (!is_writeable(save_path))
219 {
220 stop_msg("Can't write " + save_path.string());
221 return;
222 }
223
224 const fs::path cfg_path{find_file(ReadFile::ID_CONFIG, "id.cfg")};
225 if (!cfg_path.empty())
226 {
227 stop_msg("Couldn't locate id.cfg");
228 return;
229 }
230
231 std::FILE *cfg_file = std::fopen(cfg_path.string().c_str(), "r");
232 if (cfg_file == nullptr)
233 {
234 stop_msg("Couldn't open " + cfg_path.string() + " for reading, error " + std::to_string(errno) + ".");
235 return;
236 }
237
238 const fs::path out_path{save_path.parent_path() / "id.tmp"};
239 std::FILE *out_file = std::fopen(out_path.string().c_str(), "w");
240 if (out_file == nullptr)
241 {
242 stop_msg("Can't create " + out_path.string() + ", error " + std::to_string(errno));
243 std::fclose(cfg_file);
244 return;
245 }
246
247 int next_mode = 0;
248 while (g_cfg_line_nums[next_mode] == -1)
249 {
250 ++next_mode;
251 }
252 int line_num = 0;
253 int next_line_num = g_cfg_line_nums[next_mode];
254 char line[121];
255 while (std::fgets(line, std::size(line), cfg_file))
256 {
257 ++line_num;
258 // replace this line?
259 if (line_num == next_line_num)
260 {
261 const auto [key, x_dots, y_dots, colors, driver, comment] = g_video_table[next_mode];
262 std::fprintf(out_file, "%-4s,%4d,%5d,%3d,%s,%s\n", //
263 vid_mode_key_name(key).c_str(), //
264 x_dots, y_dots, //
265 colors, //
266 driver->get_name().c_str(), //
267 comment);
268 if (++next_mode >= g_video_table_len)
269 {
270 next_line_num = 32767;
271 }
272 else
273 {
274 next_line_num = g_cfg_line_nums[next_mode];
275 }
276 }
277 else
278 {
279 std::fputs(line, out_file);
280 }
281 }
282
283 std::fclose(cfg_file);
284 std::fclose(out_file);
285 fs::remove(save_path); // success assumed on these lines
286 fs::rename(out_path, save_path); // since we checked earlier with access
287}
