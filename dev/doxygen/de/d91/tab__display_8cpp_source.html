70// Wrapping version of putstring for long numbers
71// row -- pointer to row variable, internally incremented if needed
72// col1 -- starting column
73// col2 -- last column
74// color -- attribute (same as for putstring)
75// maxrow -- max number of rows to write
76// returns false if success, true if hit maxrow before done
77static bool put_string_wrap(int *row, int col1, const int col2, const int color, char *str, const int max_row)
78{
79 int dec_pt;
80 bool done = false;
81 const int start_row = *row;
82 int length = static_cast<int>(std::strlen(str));
83 int padding = 3; // space between col1 and decimal.
84 // find decimal point
85 for (dec_pt = 0; dec_pt < length; dec_pt++)
86 {
87 if (str[dec_pt] == '.')
88 {
89 break;
90 }
91 }
92 if (dec_pt >= length)
93 {
94 dec_pt = 0;
95 }
96 if (dec_pt < padding)
97 {
98 padding -= dec_pt;
99 }
100 else
101 {
102 padding = 0;
103 }
104 col1 += padding;
105 dec_pt += col1+1; // column just past where decimal is
106 while (length > 0)
107 {
108 if (col2-col1 < length)
109 {
110 done = *row - start_row + 1 >= max_row;
111 char save1 = str[col2 - col1 + 1];
112 char save2 = str[col2 - col1 + 2];
113 if (done)
114 {
115 str[col2-col1+1] = '+';
116 }
117 else
118 {
119 str[col2-col1+1] = '\\';
120 }
121 str[col2-col1+2] = 0;
122 driver_put_string(*row, col1, color, str);
123 if (done)
124 {
125 break;
126 }
127 str[col2-col1+1] = save1;
128 str[col2-col1+2] = save2;
129 str += col2-col1;
130 (*row)++;
131 }
132 else
133 {
134 driver_put_string(*row, col1, color, str);
135 }
136 length -= col2-col1;
137 col1 = dec_pt; // align with decimal
138 }
139 return done;
140}

155static void write_row(const int row, const char *format, ...)
156{
157 char text[78]{};
158 std::va_list args;
159
160 va_start(args, format);
161 std::vsnprintf(text, std::size(text), format, args);
162 va_end(args);
163
164 driver_put_string(row, 2, C_GENERAL_HI, text);
