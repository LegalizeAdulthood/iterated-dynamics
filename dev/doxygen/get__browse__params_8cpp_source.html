25// get browse parameters, returns 3 if anything changes.
26int get_browse_params()
27{
28 ChoiceBuilder<10> choices;
29 int i;
30
31 bool old_auto_browse = g_browse.auto_browse;
32 bool old_browse_check_fractal_type = g_browse.check_fractal_type;
33 bool old_browse_check_params = g_browse.check_fractal_params;
34 bool old_double_caution = g_browse.confirm_delete;
35 int old_smallest_box_size_shown = g_browse.smallest_box;
36 double old_smallest_window_display_size = g_browse.smallest_window;
37 std::filesystem::path old_browse_mask = g_browse.mask;
38
39get_brws_restart:
40 choices.reset()
41 .yes_no("Autobrowsing? (y/n)", g_browse.auto_browse)
42 .yes_no("Ask about GIF video mode? (y/n)", g_ask_video)
43 .yes_no("Check fractal type? (y/n)", g_browse.check_fractal_type)
44 .yes_no("Check fractal parameters (y/n)", g_browse.check_fractal_params)
45 .yes_no("Confirm file deletes (y/n)", g_browse.confirm_delete)
46 .float_number("Smallest window to display (size in pixels)", g_browse.smallest_window)
47 .int_number("Smallest box size shown before crosshairs used (pix)", g_browse.smallest_box)
48 .string("Browse search filename mask ", g_browse.mask.string().c_str())
49 .comment("")
50 .comment("Press F4 to reset browse parameters to defaults.");
51
52 {
53 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_BROWSE_PARAMETERS};
54 i = choices.prompt("Browse ('L'ook) Mode Options", 16);
55 }
56 if (i < 0)
57 {
58 return 0;
59 }
60
61 if (i == ID_KEY_F4)
62 {
63 g_browse.smallest_window = 6;
64 g_browse.auto_browse = false;
65 g_ask_video = true;
66 g_browse.check_fractal_params = true;
67 g_browse.check_fractal_type = true;
68 g_browse.confirm_delete = true;
69 g_browse.smallest_box = 3;
70 g_browse.mask = "*.gif";
71 goto get_brws_restart;
72 }
73
74 g_browse.auto_browse = choices.read_yes_no();
75 g_ask_video = choices.read_yes_no();
76 g_browse.check_fractal_type = choices.read_yes_no();
77 g_browse.check_fractal_params = choices.read_yes_no();
78 g_browse.confirm_delete = choices.read_yes_no();
79 g_browse.smallest_window = choices.read_float_number();
80 g_browse.smallest_window = std::max(g_browse.smallest_window, 0.0);
81 g_browse.smallest_box = choices.read_int_number();
82 g_browse.smallest_box = std::max(g_browse.smallest_box, 1);
83 g_browse.smallest_box = std::min(g_browse.smallest_box, 10);
84 g_browse.mask = std::filesystem::path{choices.read_string()}.filename().string();
85
86 i = 0;
87 if (g_browse.auto_browse != old_auto_browse //
88 || g_browse.check_fractal_type != old_browse_check_fractal_type //
89 || g_browse.check_fractal_params != old_browse_check_params //
90 || g_browse.confirm_delete != old_double_caution //
91 || g_browse.smallest_window != old_smallest_window_display_size //
92 || g_browse.smallest_box != old_smallest_box_size_shown //
93 || g_browse.mask != old_browse_mask)
94 {
95 i = -3;
96 }
97
98 if (g_evolving != EvolutionModeFlags::NONE)
99 {
100 // can't browse
101 g_browse.auto_browse = false;
102 i = 0;
103 }
104
105 return i;
106}
