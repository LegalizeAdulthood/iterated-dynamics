18int one_or_two_pass()
19{
20 g_total_passes = 1;
21 if (g_std_calc_mode == CalcMode::TWO_PASS)
22 {
23 g_total_passes = 2;
24 }
25 if (g_std_calc_mode == CalcMode::TWO_PASS && g_work_pass == 0) // do 1st pass of two
26 {
27 if (standard_calc(1) == -1)
28 {
29 add_work_list(g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, g_col, g_row, 0, g_work_symmetry);
30 return -1;
31 }
32 if (g_num_work_list > 0) // worklist not empty, defer 2nd pass
33 {
34 add_work_list(
35 g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, g_start_pt.x, g_start_pt.y, 1, g_work_symmetry);
36 return 0;
37 }
38 g_work_pass = 1;
39 g_begin_pt.x = g_start_pt.x;
40 g_begin_pt.y = g_start_pt.y;
41 }
42 // second or only pass
43 if (standard_calc(2) == -1)
44 {
45 int i = g_stop_pt.y;
46 if (g_i_stop_pt.y != g_stop_pt.y) // must be due to symmetry
47 {
48 i -= g_row - g_i_start_pt.y;
49 }
50 add_work_list(g_start_pt.x, g_row, g_stop_pt.x, i, g_col, g_row, g_work_pass, g_work_symmetry);
51 return -1;
52 }
53
54 return 0;
55}

57static int standard_calc(const int pass_num)
58{
59 g_passes = Passes::SEQUENTIAL_SCAN;
60 g_current_pass = pass_num;
61 g_row = g_begin_pt.y;
62 g_col = g_begin_pt.x;
63
64 while (g_row <= g_i_stop_pt.y)
65 {
66 g_current_row = g_row;
67 g_reset_periodicity = true;
68 while (g_col <= g_i_stop_pt.x)
69 {
70 // on 2nd pass of two, skip even pts
71 if (g_quick_calc && !g_resuming)
72 {
73 g_color = get_color(g_col, g_row);
74 if (g_color != g_inside_color)
75 {
76 ++g_col;
77 continue;
78 }
79 }
80 if (pass_num == 1 || g_std_calc_mode == CalcMode::ONE_PASS || (g_row&1) != 0 || (g_col&1) != 0)
81 {
82 if (g_calc_type() == -1) // standard_fractal(), calcmand() or calcmandfp()
83 {
84 return -1; // interrupted
85 }
86 g_resuming = false; // reset so quick_calc works
87 g_reset_periodicity = false;
88 if (pass_num == 1) // first pass, copy pixel and bump col
89 {
90 if ((g_row&1) == 0 && g_row < g_i_stop_pt.y)
91 {
92 g_plot(g_col, g_row+1, g_color);
93 if ((g_col&1) == 0 && g_col < g_i_stop_pt.x)
94 {
95 g_plot(g_col+1, g_row+1, g_color);
96 }
97 }
98 if ((g_col&1) == 0 && g_col < g_i_stop_pt.x)
99 {
100 g_plot(++g_col, g_row, g_color);
101 }
102 }
103 }
104 ++g_col;
105 }
106 g_col = g_i_start_pt.x;
107 if (pass_num == 1 && (g_row&1) == 0)
108 {
109 ++g_row;
110 }
111 ++g_row;
112 }
113 return 0;
114}
