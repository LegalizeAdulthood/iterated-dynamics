45int TwoPass::run()
46{
47 g_total_passes = 2;
48 if (g_work_pass == 0) // do 1st pass of two
49 {
50 if (standard_calc(1, CalcMode::TWO_PASS) == -1)
51 {
52 if (calc_interrupted())
53 {
54 add_work_list(g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, m_resume_col, m_resume_row, 0,
55 g_work_symmetry);
56 }
57 return -1;
58 }
59 if (g_num_work_list > 0) // worklist not empty, defer 2nd pass
60 {
61 add_work_list(
62 g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, g_start_pt.x, g_start_pt.y, 1, g_work_symmetry);
63 return 0;
64 }
65 g_work_pass = 1;
66 g_begin_pt.x = g_start_pt.x;
67 g_begin_pt.y = g_start_pt.y;
68 }
69 // second or only pass
70 if (standard_calc(2, CalcMode::TWO_PASS) == -1)
71 {
72 if (calc_interrupted())
73 {
74 add_work_list(g_start_pt.x, m_resume_row, g_stop_pt.x, stop_row_for_resume(), m_resume_col, m_resume_row,
75 g_work_pass, g_work_symmetry);
76 }
77 return -1;
78 }
79
80 return 0;
81}

83int OneOrTwoPass::standard_calc(const int pass_num, const CalcMode calc_mode)
84{
85 g_passes = Passes::SEQUENTIAL_SCAN;
86 m_current_pass = pass_num;
87 g_current_pass = m_current_pass;
88 if (!m_standard_calc_active)
89 {
90 m_row = g_begin_pt.y;
91 m_col = g_begin_pt.x;
92 m_standard_calc_active = true;
93 }
94 g_row = m_row;
95 g_col = m_col;
96
97 while (m_row <= g_i_stop_pt.y)
98 {
99 g_current_row = m_row;
100 g_reset_periodicity = true;
101 while (m_col <= g_i_stop_pt.x)
102 {
103 g_row = m_row;
104 g_col = m_col;
105 // on 2nd pass of two, skip even pts
106 if (g_quick_calc && !g_resuming)
107 {
108 g_color = get_color(m_col, m_row);
109 if (g_color != g_inside_color)
110 {
111 ++m_col;
112 continue;
113 }
114 }
115 if (pass_num == 1 || calc_mode == CalcMode::ONE_PASS || (m_row & 1) != 0 || (m_col & 1) != 0)
116 {
117 int result{-1};
118 if (g_dispatch.calc_type() != lyapunov_type || !calc_interrupted())
119 {
120 result = calc_type(); // standard_fractal(), calcmand() or calcmandfp()
121 }
122 m_row = g_row;
123 m_col = g_col;
124 if (result == -1)
125 {
126 m_resume_row = m_row;
127 m_resume_col = m_col;
128 return -1; // interrupted
129 }
130 g_resuming = false; // reset so quick_calc works
131 g_reset_periodicity = false;
132 if (pass_num == 1) // first pass, copy pixel and bump col
133 {
134 if ((m_row & 1) == 0 && m_row < g_i_stop_pt.y)
135 {
136 g_plot(m_col, m_row + 1, g_color);
137 if ((m_col & 1) == 0 && m_col < g_i_stop_pt.x)
138 {
139 g_plot(m_col + 1, m_row + 1, g_color);
140 }
141 }
142 if ((m_col & 1) == 0 && m_col < g_i_stop_pt.x)
143 {
144 ++m_col;
145 g_col = m_col;
146 g_plot(m_col, m_row, g_color);
147 }
148 }
149 }
150 ++m_col;
151 if (StandardFractal *standard_fractal = active_standard_fractal();
152 standard_fractal != nullptr && standard_fractal->consume_standard_pixel_yield())
153 {
154 return -1;
155 }
156 }
157 m_col = g_i_start_pt.x;
158 if (pass_num == 1 && (m_row & 1) == 0)
159 {
160 ++m_row;
161 }
162 ++m_row;
163 }
164 g_row = m_row;
165 g_col = m_col;
166 m_standard_calc_active = false;
167 return 0;
168}

170int OneOrTwoPass::stop_row_for_resume() const
171{
172 int stop_row = g_stop_pt.y;
173 if (g_i_stop_pt.y != g_stop_pt.y) // must be due to symmetry
174 {
175 stop_row -= m_resume_row - g_i_start_pt.y;
176 }
177 return stop_row;
178}
