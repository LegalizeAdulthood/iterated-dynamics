12DComplex sqrt(const double x, const double y)
13{
14 DComplex result;
15
16 if (x == 0.0 && y == 0.0)
17 {
18 result.x = 0.0;
19 result.y = 0.0;
20 }

22 {
23 const double mag = std::sqrt(std::sqrt(x*x + y*y));
24 const double theta = std::atan2(y, x) / 2;

57// rz=Arcsin(z)=-i*Log{i*z+sqrt(1-z*z)}
58void asin(const DComplex &z, DComplex &rz)
59{
60 DComplex temp_z1;
61 DComplex temp_z2;
62
63 fpu_cmplx_mul(z, z, temp_z1);
64 temp_z1.x = 1 - temp_z1.x;
65 temp_z1.y = -temp_z1.y; // tempz1 = 1 - tempz1
66 temp_z1 = sqrt(temp_z1);
67
68 temp_z2.x = -z.y;
69 temp_z2.y = z.x; // tempz2 = i*z
70 temp_z1.x += temp_z2.x;
71 temp_z1.y += temp_z2.y; // tempz1 += tempz2
72 cmplx_log(temp_z1, temp_z1);
73 rz.x = temp_z1.y;
74 rz.y = -temp_z1.x; // rz = (-i)*tempz1
75} // end. Arcsinz

119void atanh(const DComplex &z, DComplex &rz)
120{
121 DComplex temp0;
122 DComplex temp1;
123 DComplex temp2;
124
125 if (z.x == 0.0)
126 {
127 rz.x = 0;
128 rz.y = std::atan(z.y);
129 }
130 else
131 {

133 {
134 }
135 else if (std::abs(z.x) < 1.0 && z.y == 0.0)

154// rz=Arctan(z)=i/2*Log{(1-i*z)/(1+i*z)}
155void atan(const DComplex &z, DComplex &rz)
156{
157 DComplex temp0;
158 DComplex temp1;
159 DComplex temp2;
160 DComplex temp3;
161 if (z.x == 0.0 && z.y == 0.0)
162 {
163 rz.y = 0;
164 rz.x = 0;
165 }
166 else if (z.x != 0.0 && z.y == 0.0)
167 {
168 rz.x = std::atan(z.x);
169 rz.y = 0;
170 }
171 else if (z.x == 0.0 && z.y != 0.0)
172 {
173 temp0.x = z.y;
174 temp0.y = 0.0;
175 atanh(temp0, temp0);
176 rz.x = -temp0.y;
177 rz.y = temp0.x; // i*temp0
178 }
179 else if (z.x != 0.0 && z.y != 0.0)
180 {
181
182 temp0.x = -z.y;
183 temp0.y = z.x; // i*z
184 temp1.x = 1 - temp0.x;
185 temp1.y = -temp0.y; // temp1 = 1 - temp0
186 temp2.x = 1 + temp0.x;
187 temp2.y = temp0.y; // temp2 = 1 + temp0
188
189 fpu_cmplx_div(temp1, temp2, temp3);
190 cmplx_log(temp3, temp3);
191 rz.x = -temp3.y*.5;
192 rz.y = .5*temp3.x; // .5*i*temp0
193 }
194} // end. Arctanz
