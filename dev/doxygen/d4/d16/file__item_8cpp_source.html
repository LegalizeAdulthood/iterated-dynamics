23static bool is_newline(const int c)
24{
25 return c == '\n' || c == '\r';

28// skip to next non-white space character and return it
29static int skip_white_space(std::FILE *infile, long *file_offset)
30{
31 int c;
32 do
33 {
34 c = std::getc(infile);
35 (*file_offset)++;
36 }
37 while (c == ' ' || c == '\t' || is_newline(c));
38 return c;
39}

41// skip to end of line
42static int skip_comment(std::FILE *infile, long *file_offset)
43{
44 int c;
45 do
46 {
47 c = std::getc(infile);
48 (*file_offset)++;
49 } while (!is_newline(c) && c != EOF);
50 return c;
51}

53static int scan_entries(std::FILE *infile, FileEntry *choices, const char *item_name)
54{
55 // returns the number of entries found; if a
56 // specific entry is being looked for, returns -1 if
57 // the entry is found, 0 otherwise.
58 char buf[101];
59 int exclude_entry;
60 long file_offset = -1;
61 int num_entries = 0;
62
63 while (true)
64 {
65 // scan the file for entry names
66top:
67 int c = skip_white_space(infile, &file_offset);
68 if (c == ';')
69 {
70 c = skip_comment(infile, &file_offset);
71 if (c == EOF)
72 {
73 break;
74 }
75 continue;
76 }
77 long temp_offset = file_offset;
78 const long name_offset = file_offset;
79 // next equiv roughly to fscanf(..,"%40[^* \n\r\t({\032]",buf)
80 int len = 0;
81 // allow spaces in entry names in next
82 while (c != ' ' && c != '\t' && c != '(' && c != ';' //
83 && c != '{' && !is_newline(c) && c != EOF)
84 {
85 if (len < 40)
86 {
87 buf[len++] = static_cast<char>(c);
88 }
89 c = std::getc(infile);
90 ++file_offset;
91 }
92 if (is_newline(c))
93 {
94 continue;
95 }
96 buf[len] = 0;
97 while (c != '{' && !is_newline(c) && c != EOF)
98 {
99 if (c == ';')
100 {
101 c = skip_comment(infile, &file_offset);
102 }
103 else
104 {
105 c = std::getc(infile);
106 ++file_offset;
107 }
108 }
109 if (is_newline(c))
110 {
111 continue;
112 }
113 if (c == '{')
114 {
115 while (c != '}' && c != EOF)
116 {
117 if (c == ';')
118 {
119 c = skip_comment(infile, &file_offset);
120 }
121 else
122 {
123 if (is_newline(c)) // reset temp_offset to
124 {
125 temp_offset = file_offset; // beginning of new line
126 }
127 c = std::getc(infile);
128 ++file_offset;
129 }
130 if (c == '{') //second '{' found
131 {
132 if (temp_offset == name_offset) // if on same line, skip line
133 {
134 skip_comment(infile, &file_offset);
135 goto top;
136 }
137 std::fseek(infile, temp_offset, SEEK_SET); // else, go back to
138 file_offset = temp_offset - 1; // beginning of line
139 goto top;
140 }
141 }
142 if (c != '}') // i.e. is EOF
143 {
144 break;
145 }
146
147 if (string_case_equal(buf, "frm:", 4) //
148 || string_case_equal(buf, "ifs:", 4) //
149 || string_case_equal(buf, "par:", 4))
150 {
151 exclude_entry = 4;
152 }
153 else if (string_case_equal(buf, "lsys:", 5))
154 {
155 exclude_entry = 5;
156 }
157 else
158 {
159 exclude_entry = 0;
160 }
161
162 buf[ITEM_NAME_LEN + exclude_entry] = 0;
163 if (item_name != nullptr) // looking for one entry
164 {
165 if (string_case_equal(buf, item_name))
166 {
167 std::fseek(infile, name_offset + static_cast<long>(exclude_entry), SEEK_SET);
168 return -1;
169 }
170 }
171 else // make a whole list of entries
172 {
173 if (buf[0] != 0 && !string_case_equal(buf, "comment") && !exclude_entry)
174 {
175 std::strcpy(choices[num_entries].name, buf);
176 choices[num_entries].point = name_offset;
177 if (++num_entries >= MAX_ENTRIES)
178 {
179 stop_msg(fmt::format("Too many entries in file, first {:d} used", MAX_ENTRIES));
180 break;
181 }
182 }
183 }
184 }
185 else if (c == EOF)
186 {
187 break;
188 }
189 }
190 return num_entries;
191}

203static bool check_path(const std::filesystem::path &path, std::FILE **infile, const std::string &item_name)
204{
205 if (std::FILE *f = std::fopen(path.string().c_str(), "rb"); f != nullptr)
206 {
207 if (search_for_entry(f, item_name))
208 {
209 *infile = f;
210 return true;
211 }
212
213 std::fclose(f);
214 *infile = nullptr;
215 }
216
217 return false;
218}

220bool find_file_item(
221 std::filesystem::path &path, const std::string &item_name, std::FILE **file_ptr, const ItemType item_type)
222{
223 std::FILE *infile = nullptr;
224 bool found = false;
225
226 if (!string_case_equal(path.string().c_str(), g_parameter_file.string().c_str()))
227 {
228 found = check_path(path, &infile, item_name);
229
230 if (!found && g_check_cur_dir)
231 {
232 const std::filesystem::path full_path{path.filename()};
233 found = check_path(full_path, &infile, item_name);
234 if (found)
235 {
236 path = full_path;
237 }
238 }
239 }
240
241 if (!found)
242 {
243 std::string par_search_name;
244 switch (item_type)
245 {
246 case ItemType::FORMULA:
247 par_search_name = "frm:" + item_name;
248 break;
249 case ItemType::L_SYSTEM:
250 par_search_name = "lsys:" + item_name;
251 break;
252 case ItemType::IFS:
253 par_search_name = "ifs:" + item_name;
254 break;
255 case ItemType::PAR_SET:
256 default:
257 par_search_name = item_name;
258 break;
259 }
260 found = check_path(g_parameter_file, &infile, par_search_name);
261 if (found)
262 {
263 path = g_parameter_file;
264 }
265 }
266
267 if (!found)
268 {
269 const auto read_file = [](ItemType type)
270 {
271 switch (type)
272 {
273 case ItemType::FORMULA:
274 return ReadFile::FORMULA;
275 case ItemType::L_SYSTEM:
276 return ReadFile::LSYSTEM;
277 case ItemType::IFS:
278 return ReadFile::IFS;
279 case ItemType::PAR_SET:
280 return ReadFile::PARAMETER;
281 }
282 throw std::runtime_error("Unknown ItemType " + std::to_string(static_cast<int>(type)));
283 };
284 const std::filesystem::path lib_path{find_file(read_file(item_type), path.filename())};
285 if (!lib_path.empty())
286 {
287 found = check_path(lib_path, &infile, item_name);
288 }
289 if (found)
290 {
291 path = lib_path;
292 }
293 }
294
295 if (!found)
296 {
297 stop_msg(fmt::format("'{:s}' file entry item not found", item_name));
298 return true;
299 }
300
301 // found file
302 if (file_ptr != nullptr)
303 {
304 *file_ptr = infile;
305 }
306 else if (infile != nullptr)
307 {
308 std::fclose(infile);
309 }
310 return false;
311}
