97/*----------------------------------------------------------------------
98*
99* initdacbox --
100*
101* Put something nice in the dac.
102*
103* The conditions are:
104* Colors 1 and 2 should be bright so ifs fractals show up.
105* Color 15 should be bright for lsystem.
106* Color 1 should be bright for bifurcation.
107* Colors 1, 2, 3 should be distinct for periodicity.
108* The color map should look good for mandelbrot.
109*
110* Results:
111* None.
112*
113* Side effects:
114* Loads the dac.
115*
116*----------------------------------------------------------------------
117*/
118static void init_dac_box()
119{
120 for (int i = 0; i < 256; i++)
121 {
122 g_dac_box[i][0] = (i >> 5)*8+7;
123 g_dac_box[i][1] = ((i + 16 & 28) >> 2)*8+7;
124 g_dac_box[i][2] = (i + 2 & 3)*16+15;
125 }
126 g_dac_box[0][0] = 0;
127 g_dac_box[0][1] = 0;
128 g_dac_box[0][2] = 0;
129 g_dac_box[1][0] = 255;
130 g_dac_box[1][1] = 255;
131 g_dac_box[1][2] = 255;
132 g_dac_box[2][0] = 190;
133 g_dac_box[2][1] = 255;
134 g_dac_box[2][2] = 255;
135}

142bool DiskDriver::init(int *argc, char **argv)
143{
144 if (!Win32BaseDriver::init(argc, argv))
145 {
146 return false;
147 }
148
149 init_dac_box();
150
151 // add default list of video modes
152 for (VideoInfo &mode : s_modes)
153 {
154 add_video_mode(this, &mode);
155 }
156
157 return true;
158}

160/* resize
161 *
162 * Check if we need resizing. If no, return false.
163 * If yes, resize internal buffers and return true.
164 */
165bool DiskDriver::resize()
166{
167 g_frame.resize(m_win_text.get_max_width(), m_win_text.get_max_height());
168 if (g_video_table[g_adapter].x_dots == m_width
169 && g_video_table[g_adapter].y_dots == m_height)
170 {
171 return false;
172 }
173
174 if (g_disk_flag)
175 {
176 end_disk();
177 }
178 start_disk();
179
180 return true;
181}

183/*----------------------------------------------------------------------
184 * read_palette
185 *
186 * Reads the current video palette into g_dac_box.
187 *
188 *
189 * Results:
190 * None.
191 *
192 * Side effects:
193 * Fills in g_dac_box.
194 *
195 *----------------------------------------------------------------------
196 */
197void DiskDriver::read_palette()
198{
199 ODS("DiskDriver::read_palette");
200 if (!g_got_real_dac)
201 {
202 return;
203 }
204 for (int i = 0; i < 256; i++)
205 {
206 g_dac_box[i][0] = m_clut[i][0];
207 g_dac_box[i][1] = m_clut[i][1];
208 g_dac_box[i][2] = m_clut[i][2];
209 }
210}
