108bool DiskDriver::init(std::vector<std::string> &args)
109{
110 if (!Win32BaseDriver::init(args))
111 {
112 return false;
113 }
114
115 // add default list of video modes
116 for (VideoInfo &mode : s_modes)
117 {
118 add_video_mode(this, &mode);
119 }
120
121 return true;
122}

124/* resize
125 *
126 * Check if we need resizing. If no, return false.
127 * If yes, resize internal buffers and return true.
128 */
129bool DiskDriver::resize()
130{
131 g_frame.resize(m_win_text.get_max_width(), m_win_text.get_max_height());
132 if (g_video_table[g_adapter].x_dots == m_width
133 && g_video_table[g_adapter].y_dots == m_height)
134 {
135 return false;
136 }
137
138 if (g_disk_flag)
139 {
140 end_disk();
141 }
142 start_disk();
143
144 return true;
145}

147/*----------------------------------------------------------------------
148 * read_palette
149 *
150 * Reads the current video palette into g_dac_box.
151 *
152 *
153 * Results:
154 * None.
155 *
156 * Side effects:
157 * Fills in g_dac_box.
158 *
159 *----------------------------------------------------------------------
160 */
161void DiskDriver::read_palette()
162{
163 ODS("DiskDriver::read_palette");
164 if (!g_got_real_dac)
165 {
166 return;
167 }
168 for (int i = 0; i < 256; i++)
169 {
170 g_dac_box[i][0] = m_clut[i][0];
171 g_dac_box[i][1] = m_clut[i][1];
172 g_dac_box[i][2] = m_clut[i][2];
173 }
174}
175
