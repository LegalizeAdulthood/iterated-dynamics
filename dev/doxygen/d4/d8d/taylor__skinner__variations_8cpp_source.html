23inline void cmplx_sqr_old(DComplex &out)
24{
25 out.y = (g_old_z.x + g_old_z.x) * g_old_z.y;
26 out.x = g_temp_sqr_x - g_temp_sqr_y;
27}

29// generalization of Scott and Skinner types
30static int trig_plus_sqr_fractal()
31{
32 // { z=pixel: z=(p1,p2)*trig(z)+(p3,p4)*sqr(z), |z|<BAILOUT }
33 cmplx_trig0(g_old_z, g_tmp_z); // tmp = trig(old)
34 cmplx_mult(g_param_z1, g_tmp_z, g_new_z); // new = parm*trig(old)
35 cmplx_sqr_old(g_tmp_z); // tmp = sqr(old)
36 DComplex tmp2; //
37 cmplx_mult(g_param_z2, g_tmp_z, tmp2); // tmp = parm2*sqr(old)
38 g_new_z += tmp2; // new = parm*trig(old)+parm2*sqr(old)
39 return g_bailout_float();
40}

51static int skinner_trig_sub_sqr_fractal()
52{
53 // { z=pixel: z=sin(z)-sqr(z), |z|<BAILOUT }
54 cmplx_trig0(g_old_z, g_new_z); // new = trig(old)
55 cmplx_sqr_old(g_tmp_z); // old = sqr(old)
56 g_new_z -= g_tmp_z; // new = trig(old)-sqr(old)
57 return g_bailout_float();

132static bool fn_plus_fn_sym() // set symmetry matrix for fn+fn type
133{

385int spider_orbit()
386{
387 // Spider(XAXIS) { c=z=pixel: z=z*z+c; c=c/2+z, |z|<=4 }
388 g_new_z.x = g_temp_sqr_x - g_temp_sqr_y + g_tmp_z.x;
