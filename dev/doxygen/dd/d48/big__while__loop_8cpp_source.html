163void ZoomMouseNotification::move(const int x, const int y, int /*key_flags*/)

175bool ZoomMouseNotification::process_zoom_primary(MainContext &context)
176{
177 if (g_zoom_box_width == 0.0)
178 {
179 zoom_box_in(context);
180 draw_box(true);
181 return true;
182 }
183 if (m_position_updated)
184 {
185 const int delta_y = m_y - m_down_y;
186 const int amt{static_cast<int>(g_logical_screen.y_size_dots * 0.02)};
187 const int steps{delta_y / amt};
188 if (steps != 0)
189 {
190 if (steps > 0 && g_zoom_box_width >= 0.95 && g_zoom_box_height >= 0.95)
191 {
192 g_zoom_box_width = 0.0;
193 }
194 else
195 {
196 m_down_y = m_y;
197 resize_box(steps);
198 }
199 draw_box(true);
200 }
201 return false;
202 }
203 return false;
204}
205
206bool ZoomMouseNotification::process_zoom_secondary(MainContext &/*context*/)
207{
208 if (g_zoom_box_width == 0.0)
209 {
210 return false;
211 }
212 if (m_position_updated)
213 {
214 const int delta_x = m_x - m_down_x;
215 const int amt{static_cast<int>(g_logical_screen.x_size_dots * 0.02)};
216 const int steps{delta_x / amt};
217 if (steps != 0)
218 {
219 m_down_x = m_x;
220 g_zoom_box_rotation += steps;
221 draw_box(true);
222 }
223 return false;
224 }
225
226 return false;
227}
228
229bool ZoomMouseNotification::process_zoom_middle(MainContext &/*context*/)
230{
231 if (g_zoom_box_width == 0.0)
232 {
233 return false;
234 }
235 if (m_position_updated)
236 {
237 const int delta_y = m_y - m_down_y;
238 m_down_y = m_y;
239 change_box(0, 2 * delta_y);
240 draw_box(true);
241 return false;
242 }
243
244 return false;
245}

247bool ZoomMouseNotification::process_zoom(MainContext &context)
248{
249 if (m_zoom_in_pending)
250 {
251 request_zoom_in(context);
252 m_zoom_in_pending = false;
253 return true;
254 }
255 if (m_zoom_out_pending)
256 {
257 request_zoom_out(context);
258 m_zoom_out_pending = false;
259 return true;
260 }
261 if (m_primary_down)
262 {
263 return process_zoom_primary(context);
264 }
265 if (m_secondary_down)
266 {
267 return process_zoom_secondary(context);
268 }
269 if (m_middle_down)
270 {
271 return process_zoom_middle(context);
272 }
273 if (m_position_updated)
274 {
275 if (g_box_count)
276 {
277 g_zoom_box_x = static_cast<double>(m_x) / g_logical_screen.x_size_dots - g_zoom_box_width / 2.0;
278 g_zoom_box_y = static_cast<double>(m_y) / g_logical_screen.y_size_dots - g_zoom_box_height / 2.0;
279 draw_box(true);
280 }
281 m_position_updated = false;
282 return false;
283 }
284 return false;
285}

296MainState big_while_loop(MainContext &context)
297{
298 int i = 0; // temporary loop counters
299 MainState mms_value;
300 const auto mouse{std::make_shared<ZoomMouseNotification>()};
301 MouseSubscription subscription{mouse};
302
303 driver_check_memory();
304 context.from_mandel = false; // if julia entered from mandel
305 if (context.resume)
306 {
307 goto resumeloop;
308 }
309
310 while (true) // eternal loop
311 {
312 driver_check_memory();
313 if (g_calc_status != CalcStatus::RESUMABLE || g_show_file == ShowFile::LOAD_IMAGE)
314 {
315 g_video_entry = g_video_table[g_adapter];
316 g_logical_screen.x_dots = g_video_entry.x_dots; // # dots across the screen
317 g_logical_screen.y_dots = g_video_entry.y_dots; // # dots down the screen
318 g_colors = g_video_entry.colors; // # colors available
319 g_screen_x_dots = g_logical_screen.x_dots;
320 g_screen_y_dots = g_logical_screen.y_dots;
321 g_logical_screen.y_offset = 0;
322 g_logical_screen.x_offset = 0;
323 g_color_cycle_range_hi = g_color_cycle_range_hi < g_colors ? g_color_cycle_range_hi : g_colors - 1;
324
325 std::memcpy(g_old_dac_box, g_dac_box, 256*3); // save the DAC
326
327 if (g_overlay_3d && g_init_batch == BatchMode::NONE)
328 {
329 driver_unstack_screen(); // restore old graphics image
330 g_overlay_3d = false;
331 }
332 else
333 {
334 driver_set_video_mode(g_video_entry); // switch video modes
335 // switching video modes may have changed drivers or disk flag...
336 if (!g_good_mode)
337 {
338 if (!driver_is_disk())
339 {
340 stop_msg("That video mode is not available with your adapter.");
341 }
342 g_ask_video = true;
343 g_init_mode = -1;
344 driver_set_for_text(); // switch to text mode
345 return MainState::RESTORE_START;
346 }
347
348 if (g_virtual_screens && (g_logical_screen.x_dots > g_screen_x_dots || g_logical_screen.y_dots > g_screen_y_dots))
349 {
350#define MSG_XY1 "Can't set virtual line that long, width cut down."
351#define MSG_XY2 "Not enough video memory for that many lines, height cut down."
352 if (g_logical_screen.x_dots > g_screen_x_dots && g_logical_screen.y_dots > g_screen_y_dots)
353 {
354 stop_msg(MSG_XY1 "\n" MSG_XY2);
355 }
356 else if (g_logical_screen.y_dots > g_screen_y_dots)
357 {
358 stop_msg(MSG_XY2);
359 }
360 else
361 {
362 stop_msg(MSG_XY1);
363 }
364#undef MSG_XY1
365#undef MSG_XY2
366 }
367 g_logical_screen.x_dots = g_screen_x_dots;
368 g_logical_screen.y_dots = g_screen_y_dots;
369 g_video_entry.x_dots = g_logical_screen.x_dots;
370 g_video_entry.y_dots = g_logical_screen.y_dots;
371 }
372
373 if (g_save_dac != SaveDAC::NO || g_colors_preloaded)
374 {
375 std::memcpy(g_dac_box, g_old_dac_box, 256*3); // restore the DAC
376 refresh_dac();
377 g_colors_preloaded = false;
378 }
379 else
380 {
381 // reset DAC to defaults, which setvideomode has done for us
382 if (g_map_specified)
383 {
384 // but there's a map=, so load that
385 for (int j = 0; j < 256; ++j)
386 {
387 g_dac_box[j][0] = g_map_clut[j][0];
388 g_dac_box[j][1] = g_map_clut[j][1];
389 g_dac_box[j][2] = g_map_clut[j][2];
390 }
391 refresh_dac();
392 }
393 else if ((driver_is_disk() && g_colors == 256) || !g_colors)
394 {
395 // disk video, setvideomode via bios didn't get it right, so:
396 validate_luts("default"); // read the default palette file
397 }
398 g_color_state = ColorState::DEFAULT_MAP;
399 }
400 if (g_viewport.enabled)
401 {
402 // bypass for VESA virtual screen
403 const double f_temp{g_viewport.final_aspect_ratio *
404 (static_cast<double>(g_screen_y_dots) / static_cast<double>(g_screen_x_dots) / g_screen_aspect)};
405 g_logical_screen.x_dots = g_viewport.x_dots;
406 if (g_logical_screen.x_dots != 0)
407 {
408 // xdots specified
409 g_logical_screen.y_dots = g_viewport.y_dots;
410 if (g_logical_screen.y_dots == 0) // calc ydots?
411 {
412 g_logical_screen.y_dots = iround(g_logical_screen.x_dots * f_temp);
413 }
414 }
415 else if (g_viewport.final_aspect_ratio <= g_screen_aspect)
416 {
417 g_logical_screen.x_dots = iround(static_cast<double>(g_screen_x_dots) / g_viewport.reduction);
418 g_logical_screen.y_dots = iround(g_logical_screen.x_dots * f_temp);
419 }
420 else
421 {
422 g_logical_screen.y_dots = iround(static_cast<double>(g_screen_y_dots) / g_viewport.reduction);
423 g_logical_screen.x_dots = iround(g_logical_screen.y_dots / f_temp);
424 }
425 if (g_logical_screen.x_dots > g_screen_x_dots || g_logical_screen.y_dots > g_screen_y_dots)
426 {
427 stop_msg("View window too large; using full screen.");
428 g_viewport.enabled = false;
429 g_viewport.x_dots = g_screen_x_dots;
430 g_logical_screen.x_dots = g_viewport.x_dots;
431 g_viewport.y_dots = g_screen_y_dots;
432 g_logical_screen.y_dots = g_viewport.y_dots;
433 }
434 // changed test to 1, so a 2x2 window will work with the sound feature
435 else if ((g_logical_screen.x_dots <= 1 || g_logical_screen.y_dots <= 1) &&
436 !bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
437 {
438 // so ssg works
439 // but no check if in evolve mode to allow lots of small views
440 stop_msg("View window too small; using full screen.");
441 g_viewport.enabled = false;
442 g_logical_screen.x_dots = g_screen_x_dots;
443 g_logical_screen.y_dots = g_screen_y_dots;
444 }
445 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP) &&
446 bit_set(g_cur_fractal_specific->flags, FractalFlags::INF_CALC))
447 {
448 stop_msg("Fractal doesn't terminate! switching off evolution.");
449 g_evolving ^= EvolutionModeFlags::FIELD_MAP;
450 g_viewport.enabled = false;
451 g_logical_screen.x_dots = g_screen_x_dots;
452 g_logical_screen.y_dots = g_screen_y_dots;
453 }
454 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
455 {
456 const int grout = bit_set(g_evolving, EvolutionModeFlags::NO_GROUT) ? 0 : 1;
457 g_logical_screen.x_dots = g_screen_x_dots / g_evolve_image_grid_size - grout;
458 // trim to multiple of 4 for SSG
459 g_logical_screen.x_dots = g_logical_screen.x_dots - g_logical_screen.x_dots % 4;
460 g_logical_screen.y_dots = g_screen_y_dots / g_evolve_image_grid_size - grout;
461 g_logical_screen.y_dots = g_logical_screen.y_dots - g_logical_screen.y_dots % 4;
462 }
463 else
464 {
465 g_logical_screen.x_offset = (g_screen_x_dots - g_logical_screen.x_dots) / 2;
466 g_logical_screen.y_offset = (g_screen_y_dots - g_logical_screen.y_dots) / 3;
467 }
468 }
469 g_logical_screen.x_size_dots = g_logical_screen.x_dots - 1; // convert just once now
470 g_logical_screen.y_size_dots = g_logical_screen.y_dots - 1;
471 }
472 // assume we save next time (except jb)
473 if (g_save_dac == SaveDAC::NO)
474 {
475 g_save_dac = SaveDAC::NEXT_TIME;
476 }
477 else
478 {
479 g_save_dac = SaveDAC::YES;
480 }
481
482 if (g_show_file == ShowFile::LOAD_IMAGE)
483 {
484 // loading an image
485 g_out_line_cleanup = nullptr; // outln routine can set this
486 if (g_display_3d != Display3DMode::NONE) // set up 3D decoding
487 {
488 g_out_line = call_line3d;
489 }
490 else if (g_compare_gif) // debug 50
491 {
492 g_out_line = cmp_line;
493 }
494 else if (g_potential.store_16bit)
495 {
496 // .pot format input file
497 if (pot_start_disk() < 0)
498 {
499 // pot file failed?
500 g_show_file = ShowFile::IMAGE_LOADED;
501 g_potential.flag = false;
502 g_potential.store_16bit = false;
503 g_init_mode = -1;
504 g_calc_status = CalcStatus::RESUMABLE; // "resume" without 16-bit
505 driver_set_for_text();
506 get_fract_type();
507 return MainState::IMAGE_START;
508 }
509 g_out_line = pot_line;
510 }
511 else if ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) > SOUNDFLAG_BEEP &&
512 g_evolving == EvolutionModeFlags::NONE) // regular gif/fra input file
513 {
514 g_out_line = sound_line; // sound decoding
515 }
516 else
517 {
518 g_out_line = out_line; // regular decoding
519 }
520 i = funny_glasses_call(gif_view);
521 if (g_out_line_cleanup) // cleanup routine defined?
522 {
523 g_out_line_cleanup();
524 }
525 if (i == 0)
526 {
527 driver_buzzer(Buzzer::COMPLETE);
528 }
529 else
530 {
531 g_calc_status = CalcStatus::NO_FRACTAL;
532 if (driver_key_pressed())
533 {
534 driver_buzzer(Buzzer::INTERRUPT);
535 while (driver_key_pressed())
536 {
537 driver_get_key();
538 }
539 text_temp_msg("*** load incomplete ***");
540 }
541 }
542 }
543
544 // for these cases disable zooming
545 g_zoom_enabled = !driver_is_disk() && !bit_set(g_cur_fractal_specific->flags, FractalFlags::NO_ZOOM);
546 if (g_evolving == EvolutionModeFlags::NONE)
547 {
548 calc_frac_init();
549 }
550 driver_schedule_alarm(1);
551
552 g_save_image_region = g_image_region; // save 3 corners for zoom.c ref points
553
554 if (g_bf_math != BFMathType::NONE)
555 {
556 copy_bf(g_bf_save_x_min, g_bf_x_min);
557 copy_bf(g_bf_save_x_max, g_bf_x_max);
558 copy_bf(g_bf_save_y_min, g_bf_y_min);
559 copy_bf(g_bf_save_y_max, g_bf_y_max);
560 copy_bf(g_bf_save_x_3rd, g_bf_x_3rd);
561 copy_bf(g_bf_save_y_3rd, g_bf_y_3rd);
562 }
563 save_history_info();
564
565 if (g_show_file == ShowFile::LOAD_IMAGE)
566 {
567 // image has been loaded
568 g_show_file = ShowFile::IMAGE_LOADED;
569 if (g_init_batch == BatchMode::NORMAL && g_calc_status == CalcStatus::RESUMABLE)
570 {
571 g_init_batch = BatchMode::FINISH_CALC_BEFORE_SAVE;
572 }
573 if (g_loaded_3d) // 'r' of image created with '3'
574 {
575 g_display_3d = Display3DMode::YES; // so set flag for 'b' command
576 }
577 }
578 else
579 {
580 // draw an image
581 if (g_save_time_interval != 0 // autosave and resumable?
582 && bit_clear(g_cur_fractal_specific->flags, FractalFlags::NO_RESUME))
583 {
584 start_save_timer();
585 g_finish_row = -1;
586 }
587 g_browse.browsing = false; // regenerate image, turn off browsing
588 g_browse.stack.clear(); // reset filename stack
589 g_browse.name.clear();
590 if (g_viewport.enabled && bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP) &&
591 g_calc_status != CalcStatus::COMPLETED)
592 {
593 // generate a set of images with varied parameters on each one
594 int count;
595 GeneBase gene[NUM_GENES];
596 copy_genes_from_bank(gene);
597 if (g_have_evolve_info && g_calc_status == CalcStatus::RESUMABLE)
598 {
599 g_evolve_x_parameter_range = g_evolve_info.x_parameter_range;
600 g_evolve_y_parameter_range = g_evolve_info.y_parameter_range;
601 g_evolve_new_x_parameter_offset = g_evolve_info.x_parameter_offset;
602 g_evolve_x_parameter_offset = g_evolve_new_x_parameter_offset;
603 g_evolve_new_y_parameter_offset = g_evolve_info.y_parameter_offset;
604 g_evolve_y_parameter_offset = g_evolve_new_y_parameter_offset;
605 g_evolve_new_discrete_x_parameter_offset = static_cast<char>(g_evolve_info.discrete_x_parameter_offset);
606 g_evolve_discrete_x_parameter_offset = g_evolve_new_discrete_x_parameter_offset;
607 g_evolve_new_discrete_y_parameter_offset = static_cast<char>(g_evolve_info.discrete_y_parameter_offset);
608 g_evolve_discrete_y_parameter_offset = g_evolve_new_discrete_y_parameter_offset;
609 g_evolve_param_grid_x = g_evolve_info.px;
610 g_evolve_param_grid_y = g_evolve_info.py;
611 g_logical_screen.x_offset = g_evolve_info.screen_x_offset;
612 g_logical_screen.y_offset = g_evolve_info.screen_y_offset;
613 g_logical_screen.x_dots = g_evolve_info.x_dots;
614 g_logical_screen.y_dots = g_evolve_info.y_dots;
615 g_evolve_image_grid_size = g_evolve_info.image_grid_size;
616 g_evolve_this_generation_random_seed = g_evolve_info.this_generation_random_seed;
617 g_evolve_max_random_mutation = g_evolve_info.max_random_mutation;
618 g_evolving = static_cast<EvolutionModeFlags>(g_evolve_info.evolving);
619 g_viewport.enabled = g_evolving != EvolutionModeFlags::NONE;
620 count = g_evolve_info.count;
621 g_have_evolve_info = false;
622 }
623 else
624 {
625 // not resuming, start from the beginning
626 int mid = g_evolve_image_grid_size / 2;
627 if (g_evolve_param_grid_x != mid || g_evolve_param_grid_y != mid)
628 {
629 g_evolve_this_generation_random_seed = static_cast<unsigned int>(std::clock()); // time for new set
630 }
631 save_param_history();
632 count = 0;
633 g_evolve_max_random_mutation = g_evolve_max_random_mutation * g_evolve_mutation_reduction_factor;
634 g_evolve_x_parameter_offset = g_evolve_new_x_parameter_offset;

843 }
844 goodbye(); // done, exit

850 {
851 mms_value = evolver_menu_switch(context);

935 }

937}
