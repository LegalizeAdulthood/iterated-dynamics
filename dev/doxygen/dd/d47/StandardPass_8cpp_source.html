50StandardPassStatus StandardPass::status() const
51{
52 StandardPassStatus status;
53
54 switch (m_state.index())
55 {
56 case ONE_PASS:
57 case TWO_PASS:
58 status.title = fmt::format("{:d} Pass Mode", g_total_passes);
59 break;
60
61 case SYNCHRONOUS_ORBIT:
62 status.title = "Synchronous Orbits";
63 break;
64
65 case BOUNDARY_TRACE:
66 status.title = "Boundary Tracing";
67 status.detail = fmt::format("Working on block (y, x) [{:d}, {:d}]...[{:d}, {:d}], at [{:d}, {:d}]",
68 g_start_pt.y, g_start_pt.x, g_stop_pt.y, g_stop_pt.x, g_current_row, g_current_column);
69 break;
70
71 case SOLID_GUESS:
72 status.title = "Solid Guessing";
73 break;
74
75 case DIFFUSION:
76 status.title = "Diffusion";
77 break;
78
79 case TESSERAL:
80 status.title = "Tesseral";
81 status.detail = fmt::format("Working on block (y, x) [{:d}, {:d}]...[{:d}, {:d}], at [{:d}, {:d}]",
82 g_start_pt.y, g_start_pt.x, g_stop_pt.y, g_stop_pt.x, g_current_row, g_current_column);
83 break;
84
85 case ORBIT:
86 status.title = "Orbits";
87 break;
88
89 default:
90 return status;
91 }
92
93 if (g_user.std_calc_mode == CalcMode::THREE_PASS)
94 {
95 status.title += " (threepass)";
96 }
97
98 if (!status.detail.empty() || m_state.index() == DIFFUSION)
99 {
100 if (m_state.index() == DIFFUSION)
101 {
102 const Diffusion &diffusion{std::get<Diffusion>(m_state)};
103 if (diffusion.limit() != 0)
104 {
105 status.progress_percent =
106 100.0F * static_cast<float>(diffusion.counter()) / static_cast<float>(diffusion.limit());
107 status.detail = fmt::format("{:2.2f}% done, counter at {:d} of {:d} ({:d} bits)",
108 status.progress_percent, diffusion.counter(), diffusion.limit(), diffusion.bits());
109 }
110 }
111 return status;
112 }
113
114 status.detail = fmt::format(
115 "Working on block (y, x) [{:d}, {:d}]...[{:d}, {:d}], ", g_start_pt.y, g_start_pt.x, g_stop_pt.y, g_stop_pt.x);
116 if (g_total_passes > 1)
117 {
118 status.detail += fmt::format("pass {:d} of {:d}, ", g_current_pass, g_total_passes);
119 }
120 status.detail += fmt::format("at row {:d} col {:d}", g_current_row, g_col);
121 return status;
122}

124bool StandardPass::iterate()
125{
126 switch (m_state.index())
127 {
128 case SYNCHRONOUS_ORBIT:
129 return std::get<SOI>(m_state).iterate();
130
131 case BOUNDARY_TRACE:
132 return std::get<BoundaryTrace>(m_state).iterate();
133
134 case SOLID_GUESS:
135 if (g_calc_status != CalcStatus::COMPLETED)
136 {
137 return std::get<SolidGuess>(m_state).iterate();
138 }
139 return true;
140
141 case DIFFUSION:
142 return std::get<Diffusion>(m_state).iterate();
143
144 case TESSERAL:
145 {
146 Tesseral &tesseral{std::get<Tesseral>(m_state)};
147 return tesseral.iterate() && tesseral.done();
148 }
149
150 case ORBIT:
151 return std::get<StickyOrbits>(m_state).iterate();
152
153 case ONE_PASS:
154 return std::get<OnePass>(m_state).iterate();
155
156 case TWO_PASS:
157 return std::get<TwoPass>(m_state).iterate();
158
159 default:
160 return true;
161 }
162}

164void StandardPass::select(const CalcMode calc_mode)
165{
166 if (calc_mode == this->calc_mode())
167 {
168 return;
169 }
170
171 switch (calc_mode)
172 {
173 case CalcMode::ONE_PASS:
174 m_state.emplace<OnePass>();
175 break;
176
177 case CalcMode::TWO_PASS:
178 m_state.emplace<TwoPass>();
179 break;
180
181 case CalcMode::SYNCHRONOUS_ORBIT:
182 m_state.emplace<SOI>();
183 break;
184
185 case CalcMode::BOUNDARY_TRACE:
186 m_state.emplace<BoundaryTrace>();
187 break;
188
189 case CalcMode::SOLID_GUESS:
190 m_state.emplace<SolidGuess>();
191 break;
192
193 case CalcMode::DIFFUSION:
194 m_state.emplace<Diffusion>();
195 break;
196
197 case CalcMode::TESSERAL:
198 m_state.emplace<Tesseral>();
199 break;
200
201 case CalcMode::ORBIT:
202 m_state.emplace<StickyOrbits>();
203 break;
204
205 default:
206 reset();
207 break;
208 }
209}
