89// This routine sets up the sequence for forcing the Rate parameter
90// to vary between the two values. It fills the array lyaRxy[] and
91// sets lyaLength to the length of the sequence.
92//
93// The sequence is coded in the bit pattern in an integer.
94// Briefly, the sequence starts with an A the leading zero bits
95// are ignored and the remaining bit sequence is decoded. The
96// sequence ends with a B. Not all possible sequences can be
97// represented in this manner, but every possible sequence is
98// either represented as itself, as a rotation of one of the
99// representable sequences, or as the inverse of a representable
100// sequence (swapping 0s and 1s in the array.) Sequences that
101// are the rotation and/or inverses of another sequence will generate
102// the same lyapunov exponents.
103//
104// A few examples follow:
105// number sequence
106// 0 ab
107// 1 aab
108// 2 aabb
109// 3 aaab
110// 4 aabbb
111// 5 aabab
112// 6 aaabb (this is a duplicate of 4, a rotated inverse)
113// 7 aaaab
114// 8 aabbbb etc.
115//
116bool lyapunov_per_image()
117{
118 s_filter_cycles = static_cast<long>(g_params[2]);
119 if (s_filter_cycles == 0)
120 {
121 s_filter_cycles = g_max_iterations/2;
122 }
123 s_lya_seed_ok = g_params[1] > 0 && g_params[1] <= 1 && g_debug_flag != DebugFlags::FORCE_STANDARD_FRACTAL;
124 s_lya_length = 1;
125
126 const long i = static_cast<long>(g_params[0]);
127 s_lya_rxy[0] = 1;
128 int t;
129 for (t = 31; t >= 0; t--)
130 {
131 if (i & 1 << t)
132 {
133 break;
134 }
135 }
136 for (; t >= 0; t--)
137 {
138 s_lya_rxy[s_lya_length++] = (i & 1 << t) != 0;
139 }
140 s_lya_rxy[s_lya_length++] = 0;
141 if (g_inside_method < ColorMethod::COLOR)
142 {
143 stop_msg("Sorry, inside options other than inside=nnn are not supported by the lyapunov");
144 g_inside_method = ColorMethod::COLOR;
145 g_inside_color = 1;
146 }
147 if (g_user.std_calc_mode == CalcMode::ORBIT)
148 {
149 // Oops,lyapunov type
150 g_user.std_calc_mode = CalcMode::ONE_PASS; // doesn't use new & breaks orbits
151 g_std_calc_mode = CalcMode::ONE_PASS;
152 }
153 return true;
154}

156static int lyapunov_cycles(const long filter_cycles, const double a, const double b)
157{
158 int color;
159 double temp;
160 // e10=22026.4657948 e-10=0.0000453999297625
161
162 double total = 1.0;
163 int ln_adjust = 0;
164 long i;
165 for (i = 0; i < filter_cycles; i++)
166 {
167 for (int count = 0; count < s_lya_length; count++)
168 {
169 g_rate = s_lya_rxy[count] ? a : b;
170 if (g_cur_fractal_specific->orbit_calc())
171 {
172 g_overflow = true;
173 goto jump_out;
174 }
175 }
176 }
177 for (i = 0; i < g_max_iterations/2; i++)
178 {
179 for (int count = 0; count < s_lya_length; count++)
180 {
181 g_rate = s_lya_rxy[count] ? a : b;
182 if (g_cur_fractal_specific->orbit_calc())
183 {
184 g_overflow = true;
185 goto jump_out;
186 }
187 temp = std::abs(g_rate-2.0*g_rate*g_population);
188 total *= temp;
189 if (total == 0)
190 {
191 g_overflow = true;
192 goto jump_out;
193 }
194 }
195 while (total > 22026.4657948)
196 {
197 total *= 0.0000453999297625;
198 ln_adjust += 10;
199 }
200 while (total < 0.0000453999297625)
201 {
202 total *= 22026.4657948;
203 ln_adjust -= 10;
204 }
205 }
206
207jump_out:
208 if (g_overflow || total <= 0 || (temp = std::log(total) + ln_adjust) > 0)
209 {
210 color = 0;
211 }
212 else
213 {
214 double lyap;
215 if (g_log_map_flag)
216 {
217 lyap = -temp/(static_cast<double>(s_lya_length) *i);
218 }
219 else
220 {
221 lyap = 1 - std::exp(temp/(static_cast<double>(s_lya_length) *i));
222 }
223 color = 1 + static_cast<int>(lyap * (g_colors - 1));
224 }
225 return color;
226}
