47static int orbit_draw_line()
48{
49 int final; // final row or column number
50 int g; // used to test for new row or column
51 int inc1; // G increment when row or column doesn't change
52 int inc2; // G increment when row or column changes
53 const int dx = g_i_stop_pt.x - g_i_start_pt.x; // find vector components
54 const int dy = g_i_stop_pt.y - g_i_start_pt.y;
55 bool pos_slope = dx > 0; // is slope positive?
56 if (dy < 0)
57 {
58 pos_slope = !pos_slope;
59 }
60 if (std::abs(dx) > std::abs(dy)) // shallow line case
61 {
62 if (dx > 0) // determine start point and last column
63 {
64 g_col = g_begin_pt.x;

188static int orbit_draw_function()
189{
190 double x_ctr;
191 double y_ctr;
192 LDouble magnification; // LDouble not really needed here, but used to match function parameters
193 double x_mag_factor;
194 double rotation;
195 double skew;
196 constexpr double FACTOR = PI / 180.0;
197 const double x_factor = g_logical_screen.x_dots / 2.0;
198 const double y_factor = g_logical_screen.y_dots / 2.0;
199
200 int angle = g_begin_pt.x; // save angle in x parameter
201
202 cvt_center_mag(x_ctr, y_ctr, magnification, x_mag_factor, rotation, skew);
203 if (rotation <= 0)
204 {
205 rotation += 360;
206 }
207
208 while (angle < rotation)
209 {
210 const double theta = static_cast<double>(angle) * FACTOR;
211 g_col = static_cast<int>(x_factor + (x_ctr + x_mag_factor * std::cos(theta)));
212 g_row = static_cast<int>(y_factor + (y_ctr + x_mag_factor * std::sin(theta)));
213 if (plot_orbits2d() == -1)
214 {
215 add_work_list(angle, 0, 0, 0, 0, 0, 0, g_work_symmetry);
216 return -1; // interrupted
217 }
218 angle++;
219 }
220 return 0;
221}
