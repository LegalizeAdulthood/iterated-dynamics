141 j = 1;

143
144 if (i == ID_KEY_F2)

156
157 if (i == ID_KEY_F6)

159 if (get_corners() > 0)
160 {
161 ret = 1;
162 }
163 if (j)
164 {
165 ret = 1;
166 }
167 goto pass_option_restart;
168 }
169
170 return j + ret;
171}
