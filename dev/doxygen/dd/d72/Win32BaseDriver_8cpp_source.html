44static void flush_output()
45{
46 static time_t start = 0;
47 static long ticks_per_second = 0;
48 static long last = 0;
49 static long frames_per_second = 10;
50
51 if (!ticks_per_second)
52 {
53 if (!start)
54 {
55 std::time(&start);
56 last = read_ticker();
57 }
58 else
59 {
60 const std::time_t now = std::time(nullptr);
61 const long now_ticks = read_ticker();
62 if (now > start)
63 {
64 ticks_per_second = (now_ticks - last)/ static_cast<long>(now - start);
65 }
66 }
67 }
68 else
69 {
70 long now = read_ticker();
71 if ((now - last)*frames_per_second > ticks_per_second)
72 {
73 driver_flush();
74 g_frame.pump_messages(false);
75 last = now;
76 }
77 }
78}

101/* key_pressed
102 *
103 * Return 0 if no key has been pressed, or the FIK value if it has.
104 * driver_get_key() must still be called to eat the key; this routine
105 * only peeks ahead.
106 *
107 * When a keystroke has been found by the underlying wintext_xxx
108 * message pump, stash it in the one key buffer for later use by
109 * get_key.
110 */
111int Win32BaseDriver::key_pressed()
112{
113 if (m_key_buffer)
114 {
115 return m_key_buffer;
116 }
117 if (auto_save_needed())
118 {
119 unget_key(ID_KEY_FAKE_AUTOSAVE);
120 assert(m_key_buffer == ID_KEY_FAKE_AUTOSAVE);
121 return m_key_buffer;
122 }
123 flush_output();
124 const int ch = handle_special_keys(g_frame.get_key_press(false));
125 if (m_key_buffer)
126 {
127 return m_key_buffer;
128 }
129 m_key_buffer = ch;
130
131 return ch;
132}

134/* unget_key
135 *
136 * Unread a key! The key buffer is only one character deep, so we
137 * assert if it's already full. This should never happen in real life :-).
138 */
139void Win32BaseDriver::unget_key(const int key)
140{
141 assert(0 == m_key_buffer);
142 m_key_buffer = key;
143}
144
145/* get_key
146 *
147 * Get a keystroke, blocking if necessary. First, check the key buffer
148 * and if that's empty ask the wintext window to pump a keystroke for us.
149 * If we get it, pass it off to handle tab and help displays. If those
150 * displays ate the key, then get another one.
151 */
152int Win32BaseDriver::get_key()
153{
154 int ch;
155
156 do
157 {
158 if (auto_save_needed())
159 {
160 return ID_KEY_FAKE_AUTOSAVE;
161 }
162 if (m_key_buffer)
163 {
164 ch = m_key_buffer;
165 m_key_buffer = 0;
166 }
167 else
168 {
169 ch = handle_special_keys(g_frame.get_key_press(true));
170 }
171 }
172 while (ch == 0);
173
174 return ch;
175}

279* Implement stack and unstack window functions by using multiple curses

281*/
282void Win32BaseDriver::stack_screen()
283{
284 // set for text mode if this is the first screen stacked
285 if (m_saved_screens.empty())
286 {
287 driver_set_for_text();
288 }
289 m_saved_cursor.push_back(g_text_row * 80 + g_text_col);
290 m_saved_screens.push_back(m_win_text.get_screen());
291 driver_set_clear();
292}
293
294void Win32BaseDriver::unstack_screen()
295{
296 _ASSERTE(!m_saved_cursor.empty());
297 const int packed{m_saved_cursor.back()};
298 m_saved_cursor.pop_back();
299 g_text_row = packed / 80;
300 g_text_col = packed % 80;
301 if (!m_saved_screens.empty())
302 {
303 // unstack
304 m_win_text.set_screen(m_saved_screens.back());
305 m_saved_screens.pop_back();
306 move_cursor(-1, -1);
307 }
308 // unstacking the last saved screen reverts to graphics display
309 if (m_saved_screens.empty())
310 {
311 set_for_graphics();
312 }
313}

406int Win32BaseDriver::wait_key_pressed(const bool timeout)
407{
408 int count = 10;
409 while (!key_pressed())
410 {
411 Sleep(25);
412 if (timeout)
413 {
414 // timeout early if zooming
415 if (count == 0 || g_zoom_box_width != 0.0)
416 {
417 break;
418 }
419 --count;
420 }
421 }
422
423 return key_pressed();
424}

465bool Win32BaseDriver::get_filename(
466 const char *hdg, const char *type_desc, const char *type_wildcard, std::string &result_filename)
467{
468 CurrentPathSaver saved_current;
469 const std::filesystem::path current{std::filesystem::current_path()};
470 std::filesystem::path path{result_filename};
471 if (path.has_filename())
472 {
473 if (path.is_relative())
474 {
475 path = std::filesystem::current_path() / path;
476 }
477 }
478 else
479 {
480 path.clear();
481 }
482 OPENFILENAMEA info{};
483 info.lStructSize = sizeof(OPENFILENAMEA);
484 info.hwndOwner = g_frame.get_window();
485 std::string filter{type_desc};
486 filter.append(" Files (");
487 filter.append(type_wildcard);
488 filter.append(")");
489 filter.append(1, 0);
490 filter.append(type_wildcard);
491 filter.append(1, 0);
492 filter.append("All files (*.*)");
493 filter.append(1, 0);
494 filter.append("*.*");
495 filter.append(2, 0);
496 info.lpstrFilter = filter.data();
497 info.nFilterIndex = 1;
498 char filename[ID_FILE_MAX_PATH]{};
499 std::strcpy(filename, path.string().c_str());
500 info.lpstrFile = filename;
501 info.nMaxFile = ID_FILE_MAX_PATH;
502 info.lpstrFileTitle = nullptr;
503 info.nMaxFileTitle = 0;
504 info.lpstrInitialDir = nullptr;
505 info.lpstrTitle = hdg;
506 info.Flags =
507 OFN_DONTADDTORECENT | OFN_LONGNAMES | OFN_NONETWORKBUTTON | OFN_FILEMUSTEXIST | OFN_PATHMUSTEXIST;
508 if (GetOpenFileNameA(&info))
509 {
510 result_filename = info.lpstrFile;
511 return false;
512 }
513 if (const DWORD last_error{GetLastError()}; last_error != ERROR_SUCCESS)
514 {
515 debug_text(("GetOpenFileNameA failed: " + std::to_string(last_error)).c_str());
516 }
517 return true;
518}
