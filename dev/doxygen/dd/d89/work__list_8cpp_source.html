106void tidy_work_list()
107{
108 if (g_num_work_list < 2)
109 {
110 return; // nothing to do
111 }
112
113 {
114 int i;
115 while ((i = combine_work_list()) != 0)
116 {
117 // merged two, delete the gone one
118 while (++i < g_num_work_list)
119 {
120 g_work_list[i-1] = g_work_list[i];
121 }
122 --g_num_work_list;
123 }
124 }
125 for (int i = 0; i < g_num_work_list; ++i)
126 {
127 for (int j = i+1; j < g_num_work_list; ++j)
128 {
129 if (g_work_list[j].pass < g_work_list[i].pass
130 || (g_work_list[j].pass == g_work_list[i].pass
131 && (g_work_list[j].start.y < g_work_list[i].start.y
132 || (g_work_list[j].start.y == g_work_list[i].start.y
133 && g_work_list[j].start.x < g_work_list[i].start.x))))
134 {
135 // dumb sort, swap 2 entries to correct order

137 g_work_list[i] = g_work_list[j];
