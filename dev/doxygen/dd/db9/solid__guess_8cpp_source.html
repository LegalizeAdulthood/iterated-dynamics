31int ssg_block_size()
32{
33 // blocksize 4 if <300 rows, 8 if 300-599, 16 if 600-1199, 32 if >=1200
34 int block_size = 4;
35 int i = 300;
36 while (i <= g_logical_screen.y_dots)
37 {
38 block_size += block_size;
39 i += i;
40 }
41 // increase blocksize if prefix array not big enough
42 while (block_size*(MAX_X_BLK-2) < g_logical_screen.x_dots || block_size*(MAX_Y_BLK-2)*16 < g_logical_screen.y_dots)
43 {
44 block_size += block_size;
45 }
46 return block_size;
47}

110// Timothy Wegner invented this solid guessing idea and implemented it in
111// more or less the overall framework you see here. Tim added this note
112// now in a possibly vain attempt to secure his place in history, because
113// Pieter Branderhorst has totally rewritten this routine, incorporating
114// a *MUCH* more sophisticated algorithm. His revised code is not only
115// faster, but is also more accurate. Harrumph!
116int SolidGuess::scan()
117{
118 if (g_work_pass == 0) // otherwise first pass already done
119 {
120 // first pass, calc every blocksize**2 pixel, quarter result & paint it
121 g_current_pass = 1;
122 if (g_i_start_pt.y <= g_start_pt.y) // first time for this window, init it
123 {
124 g_current_row = 0;
125 std::memset(&m_prefix[1][0][0], 0, MAX_X_BLK*MAX_Y_BLK*2); // noskip flags off
126 g_reset_periodicity = true;
127 g_row = g_i_start_pt.y;
128 for (g_col = g_i_start_pt.x; g_col <= g_i_stop_pt.x; g_col += m_max_block)
129 {
130 // calc top row
131 if (g_calc_type() == -1)
132 {
133 add_work_list(g_start_pt.x, g_start_pt.y, //
134 g_stop_pt.x, g_stop_pt.y, //
135 g_begin_pt.x, g_begin_pt.y, //
136 0, g_work_symmetry);
137 return 0;
138 }
139 g_reset_periodicity = false;
140 }
141 }
142 else
143 {
144 std::memset(&m_prefix[1][0][0], -1, MAX_X_BLK*MAX_Y_BLK*2); // noskip flags on
145 }
146 for (int y = g_i_start_pt.y; y <= g_i_stop_pt.y; y += m_block_size)
147 {
148 g_current_row = y;
149 int i = 0;
150 if (y+m_block_size <= g_i_stop_pt.y)
151 {
152 // calc the row below
153 g_row = y+m_block_size;
154 g_reset_periodicity = true;
155 for (g_col = g_i_start_pt.x; g_col <= g_i_stop_pt.x; g_col += m_max_block)
156 {
157 i = g_calc_type();
158 if (i == -1)
159 {
160 break;
161 }
162 g_reset_periodicity = false;
163 }
164 }
165 g_reset_periodicity = false;
166 if (i == -1 || guess_row(true, y, m_block_size)) // interrupted?
167 {
168 y = std::max(y, g_start_pt.y);
169 add_work_list(g_start_pt.x, g_start_pt.y, //
170 g_stop_pt.x, g_stop_pt.y, //
171 g_start_pt.x, y, //
172 0, g_work_symmetry);
173 return 0;
174 }
175 }
176
177 if (g_num_work_list) // work list not empty, just do 1st pass
178 {
179 add_work_list(g_start_pt.x, g_start_pt.y, //
180 g_stop_pt.x, g_stop_pt.y, //
181 g_start_pt.x, g_start_pt.y, //
182 1, g_work_symmetry);
183 return 0;
184 }
185 ++g_work_pass;
186 g_i_start_pt.y = g_start_pt.y & -1 - (m_max_block - 1);
187
188 // calculate skip flags for skippable blocks
189 const int x_lim = (g_i_stop_pt.x + m_max_block) / m_max_block + 1;
190 const int y_lim = ((g_i_stop_pt.y + m_max_block) / m_max_block + 15) / 16 + 1;
191 if (!m_right_guess) // no right edge guessing, zap border
192 {
193 for (int y = 0; y <= y_lim; ++y)
194 {
195 m_prefix[1][y][x_lim] = 0xffff;
196 }
197 }
198 if (!m_bottom_guess) // no bottom edge guessing, zap border
199 {
200 int i = (g_i_stop_pt.y + m_max_block) / m_max_block + 1;
201 const int y = i/16+1;
202 i = 1 << (i&15);
203 for (int x = 0; x <= x_lim; ++x)
204 {
205 m_prefix[1][y][x] |= i;
206 }
207 }
208 // set each bit in tprefix[0] to OR of it & surrounding 8 in tprefix[1]
209 for (int y = 0; ++y < y_lim;)
210 {
211 unsigned int *pfx_p0 = &m_prefix[0][y][0];
212 const unsigned int *pfx_p1 = &m_prefix[1][y][0];
213 for (int x = 0; ++x < x_lim;)
214 {
215 ++pfx_p1;
216 const unsigned int u = *(pfx_p1 - 1) | *pfx_p1 | *(pfx_p1 + 1);
217 *++pfx_p0 = u| u >> 1 | u << 1
218 |
219 (*(pfx_p1 - (MAX_X_BLK + 1)) | *(pfx_p1 - MAX_X_BLK) | *(pfx_p1 - (MAX_X_BLK - 1))) >>
220 15
221 |
222 (*(pfx_p1 + (MAX_X_BLK - 1)) | *(pfx_p1 + MAX_X_BLK) | *(pfx_p1 + (MAX_X_BLK + 1))) << 15;
223 }
224 }
225 }
226 else // first pass already done
227 {
228 std::memset(&m_prefix[0][0][0], -1, MAX_X_BLK*MAX_Y_BLK*2); // noskip flags on
229 }
230 if (g_three_pass)
231 {
232 return 0;
233 }
234
235 // remaining pass(es), halve blocksize & quarter each blocksize**2
236 int i = g_work_pass;
237 while (--i > 0) // allow for already done passes
238 {
239 m_block_size /= 2;
240 }
241 g_reset_periodicity = false;
242 while ((m_block_size = m_block_size >> 1) >= 2)
243 {
244 if (g_stop_pass > 0 && g_work_pass >= g_stop_pass)
245 {
246 return 0;
247 }
248 g_current_pass = g_work_pass + 1;
249 for (int y = g_i_start_pt.y; y <= g_i_stop_pt.y; y += m_block_size)
250 {
251 g_current_row = y;
252 if (guess_row(false, y, m_block_size))
253 {
254 y = std::max(y, g_start_pt.y);
255 add_work_list(g_start_pt.x, g_start_pt.y, //
256 g_stop_pt.x, g_stop_pt.y, //
257 g_start_pt.x, y, //
258 g_work_pass, g_work_symmetry);
259 return 0;
260 }
261 }
262 ++g_work_pass;
263 if (g_num_work_list // work list not empty, do one pass at a time
264 && m_block_size > 2) // if 2, we just did last pass
265 {
266 add_work_list(g_start_pt.x, g_start_pt.y, //
267 g_stop_pt.x, g_stop_pt.y, //
268 g_start_pt.x, g_start_pt.y, //
269 g_work_pass, g_work_symmetry);
270 return 0;
271 }
272 g_i_start_pt.y = g_start_pt.y & -1 - (m_max_block - 1);
273 }
274
275 return 0;
276}

285bool SolidGuess::guess_row(bool first_pass, int y, int block_size)
286{
287 int j;
288 int x_plus_half;
289 int x_plus_block;
290 int y_less_block;
291 int y_less_half;
292 int y_plus_half;
293 int y_plus_block;
294 int c21; // cxy is the color of pixel at (x,y)
295 int c31; // where c22 is the top left corner of
296 int c41; // the block being handled in current
297 int c12; // iteration
298 int c22;
299 int c32;
300 int c42;
301 int c13;
302 int c23;
303 int c33;
304 int c24;
305 int c44;
306 int guessed23;
307 int guessed32;
308 int guessed33;
309 int guessed12;
310 int guessed13;
311 int prev11;
312 int fix21;
313 int fix31;
314 unsigned int *pfx_ptr;
315 unsigned int pfx_mask;
316
317 c42 = 0; // just for warning
318 c41 = 0;
319 c44 = 0;
320
321 m_half_block = block_size / 2;
322 {
323 const int i = y / m_max_block;
324 pfx_ptr = &m_prefix[first_pass ? 1 : 0][(i >> 4) + 1][g_i_start_pt.x / m_max_block];
325 pfx_mask = 1 << (i & 15);
326 }
327 y_less_half = y - m_half_block;
328 y_less_block = y - block_size; // constants, for speed
329 y_plus_half = y + m_half_block;
330 y_plus_block = y + block_size;
331 prev11 = -1;
332 c22 = get_color(g_i_start_pt.x, y);
333 c13 = c22;
334 c12 = c22;
335 c24 = c22;
336 c21 = get_color(g_i_start_pt.x, y > 0 ?y_less_half:0);
337 c31 = c21;
338 if (y_plus_block <= g_i_stop_pt.y)
339 {
340 c24 = get_color(g_i_start_pt.x, y_plus_block);
341 }
342 else if (!m_bottom_guess)
343 {
344 c24 = -1;
345 }
346 guessed13 = 0;
347 guessed12 = 0;
348
349 for (int x = g_i_start_pt.x; x <= g_i_stop_pt.x;) // increment at end, or when doing continue
350 {
351 if ((x& m_max_block - 1) == 0) // time for skip flag stuff
352 {
353 ++pfx_ptr;
354 if (!first_pass && (*pfx_ptr&pfx_mask) == 0) // check for fast skip
355 {
356 x += m_max_block;
357 c13 = c22;
358 c12 = c22;
359 c24 = c22;
360 c21 = c22;
361 c31 = c22;
362 prev11 = c22;
363 guessed13 = 0;
364 guessed12 = 0;
365 continue;
366 }
367 }
368
369 if (first_pass) // 1st pass, paint topleft corner
370 {
371 plot_block(0, x, y, c22);
372 }
373 // setup variables
374 x_plus_half = x + m_half_block;
375 x_plus_block = x_plus_half + m_half_block;
376 if (x_plus_half > g_i_stop_pt.x)
377 {
378 if (!m_right_guess)
379 {
380 c31 = -1;
381 }
382 }
383 else if (y > 0)
384 {
385 c31 = get_color(x_plus_half, y_less_half);
386 }
387 if (x_plus_block <= g_i_stop_pt.x)
388 {
389 if (y_plus_block <= g_i_stop_pt.y)
390 {
391 c44 = get_color(x_plus_block, y_plus_block);
392 }
393 c41 = get_color(x_plus_block, y > 0 ?y_less_half:0);
394 c42 = get_color(x_plus_block, y);
395 }
396 else if (!m_right_guess)
397 {
398 c44 = -1;
399 c42 = -1;
400 c41 = -1;
401 }
402 if (y_plus_block > g_i_stop_pt.y)
403 {
404 c44 = m_bottom_guess ? c42 : -1;
405 }
406
407 // guess or calc the remaining 3 quarters of current block
408 guessed33 = 1;
409 guessed32 = 1;
410 guessed23 = 1;
411 c33 = c22;
412 c32 = c22;
413 c23 = c22;
414 if (y_plus_half > g_i_stop_pt.y)
415 {
416 if (!m_bottom_guess)
417 {
418 c33 = -1;
419 c23 = -1;
420 }
421 guessed33 = -1;
422 guessed23 = -1;
423 guessed13 = 0;
424 }
425 if (x_plus_half > g_i_stop_pt.x)
426 {
427 if (!m_right_guess)
428 {
429 c33 = -1;
430 c32 = -1;
431 }
432 guessed33 = -1;
433 guessed32 = -1;
434 }
435 while (true) // go around till none of 23,32,33 change anymore
436 {
437 if (guessed33 > 0
438 && (c33 != c44 || c33 != c42 || c33 != c24 || c33 != c32 || c33 != c23))
439 {
440 c33 = calc_a_dot(x_plus_half, y_plus_half);
441 if (c33 == -1)
442 {
443 return true;
444 }
445 guessed33 = 0;
446 }
447 if (guessed32 > 0
448 && (c32 != c33 || c32 != c42 || c32 != c31 || c32 != c21 || c32 != c41 || c32 != c23))
449 {
450 c32 = calc_a_dot(x_plus_half, y);
451 if (c32 == -1)
452 {
453 return true;
454 }
455 guessed32 = 0;
456 continue;
457 }
458 if (guessed23 > 0
459 && (c23 != c33 || c23 != c24 || c23 != c13 || c23 != c12 || c23 != c32))
460 {
461 c23 = calc_a_dot(x, y_plus_half);
462 if (c23 == -1)
463 {
464 return true;
465 }
466 guessed23 = 0;
467 continue;
468 }
469 break;
470 }
471
472 if (first_pass) // note whether any of block's contents were calculated
473 {
474 if (guessed23 == 0 || guessed32 == 0 || guessed33 == 0)
475 {
476 *pfx_ptr |= pfx_mask;
477 }
478 }
479
480 if (m_half_block > 1)
481 {
482 // not last pass, check if something to display
483 if (first_pass) // display guessed corners, fill in block
484 {
485 if (m_guess_plot)
486 {
487 if (guessed23 > 0)
488 {
489 g_plot(x, y_plus_half, c23);
490 }
491 if (guessed32 > 0)
492 {
493 g_plot(x_plus_half, y, c32);
494 }
495 if (guessed33 > 0)
496 {
497 g_plot(x_plus_half, y_plus_half, c33);
498 }
499 }
500 plot_block(1, x, y_plus_half, c23);
501 plot_block(0, x_plus_half, y, c32);
502 plot_block(1, x_plus_half, y_plus_half, c33);
503 }
504 else // repaint changed blocks
505 {
506 if (c23 != c22)
507 {
508 plot_block(-1, x, y_plus_half, c23);
509 }
510 if (c32 != c22)
511 {
512 plot_block(-1, x_plus_half, y, c32);
513 }
514 if (c33 != c22)
515 {
516 plot_block(-1, x_plus_half, y_plus_half, c33);
517 }
518 }
519 }
520
521 // check if some calcs in this block mean earlier guesses need fixing
522 fix21 = (c22 != c12 || c22 != c32) && c21 == c22 && c21 == c31 && c21 == prev11 && y > 0 &&
523 (x == g_i_start_pt.x || c21 == get_color(x - m_half_block, y_less_block)) &&
524 (x_plus_half > g_i_stop_pt.x || c21 == get_color(x_plus_half, y_less_block)) &&
525 c21 == get_color(x, y_less_block);
526 fix31 = c22 != c32 && c31 == c22 && c31 == c42 && c31 == c21 && c31 == c41 && y > 0 &&
527 x_plus_half <= g_i_stop_pt.x && c31 == get_color(x_plus_half, y_less_block) &&
528 (x_plus_block > g_i_stop_pt.x || c31 == get_color(x_plus_block, y_less_block)) &&
529 c31 == get_color(x, y_less_block);
530 prev11 = c31; // for next time around
531 if (fix21)
532 {
533 c21 = calc_a_dot(x, y_less_half);
534 if (c21 == -1)
535 {
536 return true;
537 }
538 if (m_half_block > 1 && c21 != c22)
539 {
540 plot_block(-1, x, y_less_half, c21);
541 }
542 }
543 if (fix31)
544 {
545 c31 = calc_a_dot(x_plus_half, y_less_half);
546 if (c31 == -1)
547 {
548 return true;
549 }
550 if (m_half_block > 1 && c31 != c22)
551 {
552 plot_block(-1, x_plus_half, y_less_half, c31);
553 }
554 }
555 if (c23 != c22)
556 {
557 if (guessed12)
558 {
559 c12 = calc_a_dot(x - m_half_block, y);
560 if (c12 == -1)
561 {
562 return true;
563 }
564 if (m_half_block > 1 && c12 != c22)
565 {
566 plot_block(-1, x-m_half_block, y, c12);
567 }
568 }
569 if (guessed13)
570 {
571 c13 = calc_a_dot(x - m_half_block, y_plus_half);
572 if (c13 == -1)
573 {
574 return true;
575 }
576 if (m_half_block > 1 && c13 != c22)
577 {
578 plot_block(-1, x-m_half_block, y_plus_half, c13);
579 }
580 }
581 }
582 c22 = c42;
583 c24 = c44;
584 c13 = c33;
585 c21 = c41;
586 c31 = c41;
587 c12 = c32;
588 guessed12 = guessed32;
589 guessed13 = guessed33;
590 x += block_size;
591 } // end x loop
592
593 if (!first_pass || m_guess_plot)
594 {
595 return false;
596 }
597
598 // paint rows the fast way
599 for (int i = 0; i < m_half_block; ++i)
600 {
601 j = y+i;
602 if (j <= g_i_stop_pt.y)
603 {
604 write_span(j, g_start_pt.x, g_i_stop_pt.x, &m_stack[g_start_pt.x]);
605 }
606 j = y+i+m_half_block;
607 if (j <= g_i_stop_pt.y)
608 {
609 write_span(j, g_start_pt.x, g_i_stop_pt.x, &m_stack[g_start_pt.x+OLD_MAX_PIXELS]);
610 }
611 if (driver_key_pressed())
612 {
613 return true;
614 }
615 }
616 if (g_plot != g_put_color) // symmetry, just vertical & origin the fast way
617 {
618 if (g_plot == sym_plot2j) // origin sym, reverse lines
619 {
620 int color;
621 for (int i = (g_i_stop_pt.x+g_start_pt.x+1)/2; --i >= g_start_pt.x;)
622 {
623 color = m_stack[i];
624 j = g_i_stop_pt.x - (i - g_start_pt.x);
625 m_stack[i] = m_stack[j];
626 m_stack[j] = static_cast<Byte>(color);
627 j += OLD_MAX_PIXELS;
628 color = m_stack[i + OLD_MAX_PIXELS];
629 m_stack[i + OLD_MAX_PIXELS] = m_stack[j];
630 m_stack[j] = static_cast<Byte>(color);
631 }
632 }
633 for (int i = 0; i < m_half_block; ++i)
634 {
635 j = g_stop_pt.y-(y+i-g_start_pt.y);
636 if (j > g_i_stop_pt.y && j < g_logical_screen.y_dots)
637 {
638 write_span(j, g_start_pt.x, g_i_stop_pt.x, &m_stack[g_start_pt.x]);
639 }
640 j = g_stop_pt.y-(y+i+m_half_block-g_start_pt.y);
641 if (j > g_i_stop_pt.y && j < g_logical_screen.y_dots)
642 {
643 write_span(j, g_start_pt.x, g_i_stop_pt.x, &m_stack[g_start_pt.x+OLD_MAX_PIXELS]);
644 }
645 if (driver_key_pressed())
646 {
647 return true;
648 }
649 }
650 }
651 return false;
652}

661void SolidGuess::plot_block(const int build_row, const int x, int y, const int color)
662{
663 int x_lim = x + m_half_block;
664 if (x_lim > g_i_stop_pt.x)
665 {
666 x_lim = g_i_stop_pt.x+1;
667 }
668 if (build_row >= 0 && !m_guess_plot) // save it for later put_line
669 {
670 if (build_row == 0)
671 {
672 fill_d_stack(x, x_lim, static_cast<Byte>(color));
673 }
674 else
675 {
