42static void set_plasma_palette()
43{
44 if (g_map_specified || g_colors_preloaded)
45 {
46 return; // map= specified
47 }
48
49 static constexpr Byte Red[3] = { 255, 0, 0 };
50 static constexpr Byte Green[3] = { 0, 255, 0 };
51 static constexpr Byte Blue[3] = { 0, 0, 255 };
52
53 g_dac_box[0][0] = 0;
54 g_dac_box[0][1] = 0;
55 g_dac_box[0][2] = 0;
56 for (int i = 1; i <= 85; i++)
57 {
58 g_dac_box[i][0] = static_cast<Byte>((i * Green[0] + (86 - i) * Blue[0]) / 85);
59 g_dac_box[i][1] = static_cast<Byte>((i * Green[1] + (86 - i) * Blue[1]) / 85);
60 g_dac_box[i][2] = static_cast<Byte>((i * Green[2] + (86 - i) * Blue[2]) / 85);
61
62 g_dac_box[i+85][0] = static_cast<Byte>((i * Red[0] + (86 - i) * Green[0]) / 85);
63 g_dac_box[i+85][1] = static_cast<Byte>((i * Red[1] + (86 - i) * Green[1]) / 85);
64 g_dac_box[i+85][2] = static_cast<Byte>((i * Red[2] + (86 - i) * Green[2]) / 85);
65 g_dac_box[i+170][0] = static_cast<Byte>((i * Blue[0] + (86 - i) * Red[0]) / 85);
66 g_dac_box[i+170][1] = static_cast<Byte>((i * Blue[1] + (86 - i) * Red[1]) / 85);
67 g_dac_box[i+170][2] = static_cast<Byte>((i * Blue[2] + (86 - i) * Red[2]) / 85);
68 }
69 refresh_dac();
70}

74// returns a random 16 bit value that is never 0
75static U16 rand16()
76{
77 U16 value = static_cast<U16>(RAND15());
78 value <<= 1;
79 value = static_cast<U16>(value + (RAND15() & 1));
80 value = std::max<U16>(value, 1U);
81 return value;
82}

84static void put_pot(const int x, const int y, U16 color)
85{
86 color = std::max<U16>(color, 1U);
87 g_put_color(x, y, color >> 8 ? color >> 8 : 1); // don't write 0
88 /* we don't write this if driver_diskp() because the above putcolor
89 was already a "writedisk" in that case */
90 if (!driver_is_disk())
91 {
92 disk_write_pixel(x+g_logical_screen.x_offset, y+g_logical_screen.y_offset, color >> 8); // upper 8 bits
93 }
94 disk_write_pixel(x+g_logical_screen.x_offset, y+g_screen_y_dots+g_logical_screen.y_offset, color&255); // lower 8 bits
95}

119{
120 U16 color = static_cast<U16>(disk_read_pixel(x + g_logical_screen.x_offset, y + g_logical_screen.y_offset));

278U16 Plasma::adjust(const int xa, const int ya, const int x, const int y, const int xb, const int yb, const int scale)
279{
280 S32 pseudorandom = m_i_param_x * (RAND15() - 16383);
281 pseudorandom = pseudorandom * scale;
282 pseudorandom = pseudorandom >> m_shift_value;
283 pseudorandom = ((static_cast<S32>(m_get_pix(xa, ya)) + static_cast<S32>(m_get_pix(xb, yb)) +1) >> 1)+pseudorandom;
284 if (m_max_plasma == 0)
285 {
286 if (pseudorandom >= m_p_colors)
287 {
288 pseudorandom = m_p_colors-1;
289 }
290 }
291 else if (pseudorandom >= static_cast<S32>(m_max_plasma))
292 {
293 pseudorandom = m_max_plasma;
294 }
295 pseudorandom = std::max<S32>(pseudorandom, 1);
296 g_plot(x, y, static_cast<U16>(pseudorandom));
297 return static_cast<U16>(pseudorandom);
298}

300void Plasma::subdivide()
301{
302 const Subdivision sd{m_subdivs.back()};
303 m_subdivs.pop_back();
304 if (sd.x2 - sd.x1 < 2 && sd.y2 - sd.y1 < 2)
305 {
306 return;
307 }
308 const int level{sd.level + 1};
309 const int scale = static_cast<int>(320L >> level);
310
311 const int x = (sd.x1 + sd.x2) >> 1;
312 const int y = (sd.y1 + sd.y2) >> 1;
313 S32 v = m_get_pix(x, sd.y1);
314 if (v == 0)
315 {
316 v = adjust(sd.x1, sd.y1, x, sd.y1, sd.x2, sd.y1, scale);
317 }
318 S32 i = v;
319 v = m_get_pix(sd.x2, y);
320 if (v == 0)
321 {
322 v = adjust(sd.x2, sd.y1, sd.x2, y, sd.x2, sd.y2, scale);
323 }
324 i += v;
325 v = m_get_pix(x, sd.y2);
326 if (v == 0)
327 {
328 v = adjust(sd.x1, sd.y2, x, sd.y2, sd.x2, sd.y2, scale);
329 }
330 i += v;
331 v = m_get_pix(sd.x1, y);
332 if (v == 0)
333 {
334 v = adjust(sd.x1, sd.y1, sd.x1, y, sd.x1, sd.y2, scale);
335 }
336 i += v;
337
338 if (m_get_pix(x, y) == 0)
339 {
340 g_plot(x, y, static_cast<U16>((i + 2) >> 2));
341 }
342
343 m_subdivs.push_back(Subdivision{sd.x1, y, x, sd.y2, level});
344 m_subdivs.push_back(Subdivision{x, y, sd.x2, sd.y2, level});
345 m_subdivs.push_back(Subdivision{x, sd.y1, sd.x2, y, level});
346 m_subdivs.push_back(Subdivision{sd.x1, sd.y1, x, y, level});
347}

349void Plasma::subdivide_new(const int x1, const int y1, const int x2, const int y2, const int level)
350{
351 m_scale = static_cast<int>(320L >> level);
352 m_sub_y.top = 2;
353 m_sub_y.value[0] = y2;
354 int ny = y2;
355 m_sub_y.value[2] = y1;
356 int ny1 = y1;
357 m_sub_y.level[0] = 0;
358 m_sub_y.level[1] = 1;
359 m_sub_y.level[2] = 0;
360 m_sub_y.value[1] = (ny1 + ny) >> 1;
361 int y = m_sub_y.value[1];
362
363 while (m_sub_y.top >= 1)
364 {
365 while (m_sub_y.level[m_sub_y.top-1] < level)
366 {
367 // 1. Create new entry at top of the stack
368 // 2. Copy old top value to new top value.
369 // This is largest y value.
370 // 3. Smallest y is now old mid-point
371 // 4. Set new mid-point recursion level
372 // 5. New mid-point value is average
373 // of largest and smallest
374
375 m_sub_y.top++;
376 m_sub_y.value[m_sub_y.top] = m_sub_y.value[m_sub_y.top-1];
377 ny1 = m_sub_y.value[m_sub_y.top];
378 ny = m_sub_y.value[m_sub_y.top-2];
379 m_sub_y.level[m_sub_y.top] = m_sub_y.level[m_sub_y.top-1];
380 m_sub_y.value[m_sub_y.top-1] = (ny1 + ny) >> 1;
381 y = m_sub_y.value[m_sub_y.top-1];
382 m_sub_y.level[m_sub_y.top-1] = std::max(m_sub_y.level[m_sub_y.top], m_sub_y.level[m_sub_y.top - 2]) + 1;
383 }
384 m_sub_x.top = 2;
385 m_sub_x.value[0] = x2;
386 int nx = x2;
387 m_sub_x.value[2] = x1;
388 int nx1 = x1;
389 m_sub_x.level[2] = 0;
390 m_sub_x.level[0] = 0;
391 m_sub_x.level[1] = 1;
392 m_sub_x.value[1] = (nx1 + nx) >> 1;
393 int x = m_sub_x.value[1];
394
395 while (m_sub_x.top >= 1)
396 {
397 while (m_sub_x.level[m_sub_x.top-1] < level)
398 {
399 m_sub_x.top++; // move the top ofthe stack up 1
400 m_sub_x.value[m_sub_x.top] = m_sub_x.value[m_sub_x.top-1];
401 nx1 = m_sub_x.value[m_sub_x.top];
402 nx = m_sub_x.value[m_sub_x.top-2];
403 m_sub_x.level[m_sub_x.top] = m_sub_x.level[m_sub_x.top-1];
404 m_sub_x.value[m_sub_x.top-1] = (nx1 + nx) >> 1;
405 x = m_sub_x.value[m_sub_x.top-1];
406 m_sub_x.level[m_sub_x.top-1] = std::max(m_sub_x.level[m_sub_x.top], m_sub_x.level[m_sub_x.top - 2]) + 1;
407 }
408
409 S32 i = m_get_pix(nx, y);
410 if (i == 0)
411 {
412 i = adjust(nx, ny1, nx, y, nx, ny, m_scale);
413 }
414 S32 v = i;
415 i = m_get_pix(x, ny);
416 if (i == 0)
417 {
418 i = adjust(nx1, ny, x, ny, nx, ny, m_scale);
419 }
420 v += i;
421 if (m_get_pix(x, y) == 0)
422 {
423 i = m_get_pix(x, ny1);
424 if (i == 0)
425 {
426 i = adjust(nx1, ny1, x, ny1, nx, ny1, m_scale);
427 }
428 v += i;
429 i = m_get_pix(nx1, y);
430 if (i == 0)
431 {
432 i = adjust(nx1, ny1, nx1, y, nx1, ny, m_scale);
433 }
434 v += i;
435 g_plot(x, y, static_cast<U16>((v + 2) >> 2));
436 }
437
438 if (m_sub_x.level[m_sub_x.top-1] == level)
439 {
440 m_sub_x.top = m_sub_x.top - 2;
441 }
442 }
443
444 if (m_sub_y.level[m_sub_y.top-1] == level)
445 {
446 m_sub_y.top = m_sub_y.top - 2;
447 }
448 }
449}

451void Plasma::iterate()
452{
453 if (m_algo == Algorithm::OLD)
454 {
455 subdivide();
456 m_done = m_subdivs.empty();
457 }
458 else
459 {
460 subdivide_new(0, 0, g_logical_screen.x_dots - 1, g_logical_screen.y_dots - 1, m_level);
461 if (!m_done)
462 {
463 m_k *= 2;
464 if (m_k > static_cast<int>(std::max(g_logical_screen.x_dots - 1, g_logical_screen.y_dots - 1)))
465 {
466 m_done = true;
467 }
468 m_level++;
469 }
470 }
