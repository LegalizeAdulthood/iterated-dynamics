182/*
183 Conversion of complex plane to screen coordinates for rotating zoom box.
184 Assume there is an affine transformation mapping complex zoom parallelogram
185 to rectangular screen. We know this map must map parallelogram corners to
186 screen corners, so we have the following equations:
187
188 a*xxmin+b*yymax+e == 0 (upper left)
189 c*xxmin+d*yymax+f == 0
190
191 a*xx3rd+b*yy3rd+e == 0 (lower left)
192 c*xx3rd+d*yy3rd+f == ydots-1
193
194 a*xxmax+b*yymin+e == xdots-1 (lower right)
195 c*xxmax+d*yymin+f == ydots-1
196
197 First we must solve for a,b,c,d,e,f - (which we do once per image),
198 then we just apply the transformation to each orbit value.
199*/
200
201/*
202 Thanks to Sylvie Gallet for the following. The original code for
203 setup_convert_to_screen() solved for coefficients of the
204 complex-plane-to-screen transformation using a very straight-forward
205 application of determinants to solve a set of simulataneous
206 equations. The procedure was simple and general, but inefficient.
207 The inefficiecy wasn't hurting anything because the routine was called
208 only once per image, but it seemed positively sinful to use it
209 because the code that follows is SO much more compact, at the
210 expense of being less general. Here are Sylvie's notes. I have further
211 optimized the code a slight bit.
212 Tim Wegner
213 July, 1996
214 Sylvie's notes, slightly edited follow:
215
216 You don't need 3x3 determinants to solve these sets of equations because
217 the unknowns e and f have the same coefficient: 1.
218
219 First set of 3 equations:
220 a*xxmin+b*yymax+e == 0
221 a*xx3rd+b*yy3rd+e == 0
222 a*xxmax+b*yymin+e == xdots-1
223 To make things easy to read, I just replace xxmin, xxmax, xx3rd by x1,
224 x2, x3 (ditto for yy...) and xdots-1 by xd.
225
226 a*x1 + b*y2 + e == 0 (1)
227 a*x3 + b*y3 + e == 0 (2)
228 a*x2 + b*y1 + e == xd (3)
229
230 I subtract (1) to (2) and (3):
231 a*x1 + b*y2 + e == 0 (1)
232 a*(x3-x1) + b*(y3-y2) == 0 (2)-(1)
233 a*(x2-x1) + b*(y1-y2) == xd (3)-(1)
234
235 I just have to calculate a 2x2 determinant:
236 det == (x3-x1)*(y1-y2) - (y3-y2)*(x2-x1)
237
238 And the solution is:
239 a = -xd*(y3-y2)/det
240 b = xd*(x3-x1)/det
241 e = - a*x1 - b*y2
242
243The same technique can be applied to the second set of equations:
244
245 c*xxmin+d*yymax+f == 0
246 c*xx3rd+d*yy3rd+f == ydots-1
247 c*xxmax+d*yymin+f == ydots-1
248
249 c*x1 + d*y2 + f == 0 (1)
250 c*x3 + d*y3 + f == yd (2)
251 c*x2 + d*y1 + f == yd (3)
252
253 c*x1 + d*y2 + f == 0 (1)
254 c*(x3-x2) + d*(y3-y1) == 0 (2)-(3)
255 c*(x2-x1) + d*(y1-y2) == yd (3)-(1)
256
257 det == (x3-x2)*(y1-y2) - (y3-y1)*(x2-x1)
258
259 c = -yd*(y3-y1)/det
260 d = yd*(x3-x2))det
261 f = - c*x1 - d*y2
262
263 - Sylvie
264*/
265
266bool setup_convert_to_screen(Affine *scrn_cnvt)
267{
268 double det = //
269 g_image_region.width3() * (g_image_region.m_min.y - g_image_region.m_max.y) +
270 (g_image_region.m_max.y - g_image_region.m_3rd.y) * g_image_region.width();
271 if (det == 0)
272 {
273 return true;
274 }
275 const double xd = g_logical_screen.x_size_dots / det;
276 scrn_cnvt->a = xd*(g_image_region.m_max.y-g_image_region.m_3rd.y);
277 scrn_cnvt->b = xd*(g_image_region.width3());
278 scrn_cnvt->e = -scrn_cnvt->a*g_image_region.m_min.x - scrn_cnvt->b*g_image_region.m_max.y;
279
280 det = //
281 (g_image_region.m_3rd.x - g_image_region.m_max.x) * (g_image_region.m_min.y - g_image_region.m_max.y) +
282 (g_image_region.m_min.y - g_image_region.m_3rd.y) * g_image_region.width();
283 if (det == 0)
284 {
285 return true;
286 }
287 const double yd = g_logical_screen.y_size_dots / det;
288 scrn_cnvt->c = yd*(g_image_region.m_min.y-g_image_region.m_3rd.y);
289 scrn_cnvt->d = yd*(g_image_region.m_3rd.x-g_image_region.m_max.x);
290 scrn_cnvt->f = -scrn_cnvt->c*g_image_region.m_min.x - scrn_cnvt->d*g_image_region.m_max.y;
291 return false;
292}

327 s_projection = 1; // plot x and z
328 }

330 s_init_orbit[0] = 1; // initial conditions
331 s_init_orbit[1] = 1;
332 s_init_orbit[2] = 1;
333 if (g_fractal_type == FractalType::GINGERBREAD)
334 {
335 s_init_orbit[0] = g_params[0]; // initial conditions
336 s_init_orbit[1] = g_params[1];
337 }
338
339 if (g_fractal_type == FractalType::ICON || g_fractal_type == FractalType::ICON_3D)

361 s_init_orbit[0] = 0.01; // initial conditions

408 }

410 else if (g_fractal_type == FractalType::INVERSE_JULIA)
411 {
412 s_cx = g_params[0];
413 s_cy = g_params[1];
414
415 s_max_hits = static_cast<int>(g_params[2]);
416 s_run_length = static_cast<int>(g_params[3]);
417 if (s_max_hits <= 0)
418 {
419 s_max_hits = 1;
420 }
421 else if (s_max_hits >= g_colors)
422 {
423 s_max_hits = g_colors - 1;
424 }
425 g_params[2] = s_max_hits;
426

428
429 // find fixed points: guaranteed to be in the set
430 DComplex sqrt = complex_sqrt_float(1 - 4 * s_cx, -4 * s_cy);
431 switch (g_major_method)
432 {
433 case Major::BREADTH_FIRST:
434 if (!init_queue(32*1024UL))
435 {
436 // can't get queue memory: fall back to random walk
437 fallback_to_random_walk();
438 goto random_walk;
439 }
440 enqueue_float(static_cast<float>((1 + sqrt.x) / 2), static_cast<float>(sqrt.y / 2));
441 enqueue_float(static_cast<float>((1 - sqrt.x) / 2), static_cast<float>(-sqrt.y / 2));
442 break;
443 case Major::DEPTH_FIRST: // depth first (choose direction)
444 if (!init_queue(32*1024UL))
445 {
446 // can't get queue memory: fall back to random walk
447 fallback_to_random_walk();
448 goto random_walk;
449 }
450 switch (g_inverse_julia_minor_method)
451 {
452 case Minor::LEFT_FIRST:
453 push_float(static_cast<float>((1 + sqrt.x) / 2), static_cast<float>(sqrt.y / 2));
454 push_float(static_cast<float>((1 - sqrt.x) / 2), static_cast<float>(-sqrt.y / 2));
455 break;
456 case Minor::RIGHT_FIRST:
457 push_float(static_cast<float>((1 - sqrt.x) / 2), static_cast<float>(-sqrt.y / 2));
458 push_float(static_cast<float>((1 + sqrt.x) / 2), static_cast<float>(sqrt.y / 2));
459 break;
460 }
461 break;
462 case Major::RANDOM_WALK:
463random_walk:
464 s_init_orbit[0] = 1 + sqrt.x / 2;
465 g_new_z.x = s_init_orbit[0];
466 s_init_orbit[1] = sqrt.y / 2;
467 g_new_z.y = s_init_orbit[1];
468 break;
469 case Major::RANDOM_RUN: // random run, choose intervals
470 g_major_method = Major::RANDOM_RUN;
471 s_init_orbit[0] = 1 + sqrt.x / 2;
472 g_new_z.x = s_init_orbit[0];
473 s_init_orbit[1] = sqrt.y / 2;
474 g_new_z.y = s_init_orbit[1];
475 break;
476 }
477 }
478 else
479 {
480 s_dt = g_params[0];
481 s_a = g_params[1];
482 s_b = g_params[2];
483 s_c = g_params[3];
484 }
485
486 // precalculations for speed
487 s_adt = s_a*s_dt;
488 s_bdt = s_b*s_dt;
489 s_cdt = s_c*s_dt;
490
491 return true;
492}

498int inverse_julia_orbit()
499{
500 static int random_dir = 0;
501 static int random_len = 0;
502
503 /*
504 * First, compute new point
505 */
506 switch (g_major_method)
507 {
508 case Major::BREADTH_FIRST:
509 if (queue_empty())
510 {
511 return -1;
512 }
513 g_new_z = dequeue_float();
514 break;
515 case Major::DEPTH_FIRST:
516 if (queue_empty())
517 {
518 return -1;
519 }
520 g_new_z = pop_float();
521 break;
522 case Major::RANDOM_WALK:
523 break;
524 case Major::RANDOM_RUN:
525 break;
526 }
527
528 /*
529 * Next, find its pixel position
530 */
531 const int new_col = static_cast<int>(s_cvt.a * g_new_z.x + s_cvt.b * g_new_z.y + s_cvt.e);
532 const int new_row = static_cast<int>(s_cvt.c * g_new_z.x + s_cvt.d * g_new_z.y + s_cvt.f);
533
534 /*
535 * Now find the next point(s), and flip a coin to choose one.
536 */
537
538 g_new_z = complex_sqrt_float(g_new_z.x - s_cx, g_new_z.y - s_cy);
539 const int left_right = random(2) ? 1 : -1;
540
541 if (new_col < 1 || new_col >= g_logical_screen.x_dots || new_row < 1 || new_row >= g_logical_screen.y_dots)
542 {
543 /*
544 * MIIM must skip points that are off the screen boundary,
545 * since it cannot read their color.
546 */
547 switch (g_major_method)
548 {
549 case Major::BREADTH_FIRST:
550 enqueue_float(
551 static_cast<float>(left_right * g_new_z.x), static_cast<float>(left_right * g_new_z.y));
552 return 1;
553 case Major::DEPTH_FIRST:
554 push_float(
555 static_cast<float>(left_right * g_new_z.x), static_cast<float>(left_right * g_new_z.y));
556 return 1;
557 case Major::RANDOM_RUN:
558 case Major::RANDOM_WALK:
559 break;
560 }
561 }
562
563 /*
564 * Read the pixel's color:
565 * For MIIM, if color >= mxhits, discard the point
566 * else put the point's children onto the queue
567 */
568 const int color = get_color(new_col, new_row);
569 switch (g_major_method)
570 {
571 case Major::BREADTH_FIRST:
572 if (color < s_max_hits)
573 {
574 g_put_color(new_col, new_row, color+1);
575 enqueue_float(static_cast<float>(g_new_z.x), static_cast<float>(g_new_z.y));
576 enqueue_float(static_cast<float>(-g_new_z.x), static_cast<float>(-g_new_z.y));
577 }
578 break;
579 case Major::DEPTH_FIRST:
580 if (color < s_max_hits)
581 {
582 g_put_color(new_col, new_row, color+1);
583 if (g_inverse_julia_minor_method == Minor::LEFT_FIRST)
584 {
585 if (queue_full_almost())
586 {
587 push_float(static_cast<float>(-g_new_z.x), static_cast<float>(-g_new_z.y));
588 }
589 else
590 {
591 push_float(static_cast<float>(g_new_z.x), static_cast<float>(g_new_z.y));
592 push_float(static_cast<float>(-g_new_z.x), static_cast<float>(-g_new_z.y));
593 }
594 }
595 else
596 {
597 if (queue_full_almost())
598 {
599 push_float(static_cast<float>(g_new_z.x), static_cast<float>(g_new_z.y));
600 }
601 else
602 {
603 push_float(static_cast<float>(-g_new_z.x), static_cast<float>(-g_new_z.y));
604 push_float(static_cast<float>(g_new_z.x), static_cast<float>(g_new_z.y));
605 }
606 }
607 }
608 break;
609 case Major::RANDOM_RUN:
610 if (random_len-- == 0)
611 {
612 random_len = random(s_run_length);
613 random_dir = random(3);
614 }
615 switch (random_dir)
616 {
617 case 0: // left
618 break;
619 case 1: // right
620 g_new_z.x = -g_new_z.x;
621 g_new_z.y = -g_new_z.y;
622 break;
623 case 2: // random direction
624 g_new_z.x = left_right * g_new_z.x;
625 g_new_z.y = left_right * g_new_z.y;
626 break;
627 }
628 if (color < g_colors-1)
629 {
630 g_put_color(new_col, new_row, color+1);
631 }
632 break;
633 case Major::RANDOM_WALK:
634 if (color < g_colors-1)
635 {
636 g_put_color(new_col, new_row, color+1);
637 }
638 g_new_z.x = left_right * g_new_z.x;
639 g_new_z.y = left_right * g_new_z.y;
640 break;
641 }
642 return 1;
643}

645int lorenz3d1_orbit(double *x, double *y, double *z)
646{
647 s_xdt = *x *s_dt;
648 s_ydt = *y *s_dt;
649 s_zdt = *z *s_dt;
650
651 // 1-lobe Lorenz
652 const double norm = std::sqrt(*x * *x + *y * *y);
653 s_dx = (-s_adt-s_dt)* *x + (s_adt-s_bdt)* *y + (s_dt-s_adt)*norm + s_ydt* *z;
654 s_dy = (s_bdt-s_adt)* *x - (s_adt+s_dt)* *y + (s_bdt+s_adt)*norm - s_xdt* *z -
655 norm*s_zdt;
656 s_dz = s_ydt / 2 - s_cdt* *z;
657
658 *x += s_dx;
659 *y += s_dy;
660 *z += s_dz;
661 return 0;
662}

694 (s_bdt+s_adt-s_zdt)*(*x * *x - *y * *y))/(3*norm);
695 s_dz = (3*s_xdt* *x * *y -s_ydt* *y * *y)/2 - s_cdt* *z;

698 *y += s_dy;

873int icon_orbit(double *x, double *y, double *z)
874{
875 const double old_x = *x;
876 const double old_y = *y;
877
878 const double z_z_bar = old_x * old_x + old_y * old_y;
879 double z_real = old_x;
880 double z_imag = old_y;
881
882 for (int i = 1; i <= DEGREE - 2; i++)
883 {
884 const double za = z_real * old_x - z_imag * old_y;
885 const double zb = z_imag * old_x + z_real * old_y;
886 z_real = za;
887 z_imag = zb;
888 }
889 const double zn = old_x * z_real - old_y * z_imag;
890 const double p = LAMBDA + ALPHA * z_z_bar + BETA * zn;
891 *x = p * old_x + GAMMA * z_real - OMEGA * old_y;
892 *y = p * old_y - GAMMA * z_imag + OMEGA * old_x;
893
894 *z = z_z_bar;
895 return 0;
896}

1845static bool float_view_transf3d(ViewTransform3D *inf)
1846{
1847 if (g_color_iter == 1) // initialize on first call
1848 {
1849 for (int i = 0; i < 3; i++)
1850 {
1851 inf->min_vals[i] = 100000.0; // impossible value
1852 inf->max_vals[i] = -100000.0;
1853 }
1854 setup_matrix(inf->double_mat);
1855 if (s_real_time)
1856 {
1857 setup_matrix(inf->double_mat1);
1858 }
1859 }
1860
1861 // 3D VIEWING TRANSFORM
1862 vec_mat_mul(inf->orbit, inf->double_mat, inf->view_vect);
1863 if (s_real_time)
1864 {
1865 vec_mat_mul(inf->orbit, inf->double_mat1, inf->view_vect1);
1866 }
1867
1868 if (g_color_iter <= s_waste) // waste this many points to find minz and maxz
1869 {
1870 // find minz and maxz
1871 for (int i = 0; i < 3; i++)
1872 {
1873 if (const double tmp = inf->view_vect[i]; tmp < inf->min_vals[i])
1874 {
1875 inf->min_vals[i] = tmp;
1876 }
1877 else if (tmp > inf->max_vals[i])
1878 {
1879 inf->max_vals[i] = tmp;
1880 }
1881 }
1882 if (g_color_iter == s_waste) // time to work it out
1883 {
1884 g_view[0] = 0; // center on origin
1885 g_view[1] = 0;
1886 /* z value of user's eye - should be more negative than extreme
1887 negative part of image */
1888 g_view[2] = (inf->min_vals[2]-inf->max_vals[2])* static_cast<double>(g_viewer_z) /100.0;
1889
1890 // center image on origin
1891 double tmp_x = (-inf->min_vals[0]-inf->max_vals[0])/2.0; // center x
1892 double tmp_y = (-inf->min_vals[1]-inf->max_vals[1])/2.0; // center y
1893
1894 // apply perspective shift
1895 const DComplex size{g_image_region.size()};
1896 tmp_x += static_cast<double>(g_x_shift) * size.x / g_logical_screen.x_dots;
1897 tmp_y += static_cast<double>(g_y_shift) * size.y / g_logical_screen.y_dots;
1898 double tmp_z = -inf->max_vals[2];
1899 trans(tmp_x, tmp_y, tmp_z, inf->double_mat);
1900
1901 if (s_real_time)
1902 {
1903 // center image on origin
1904 tmp_x = (-inf->min_vals[0]-inf->max_vals[0])/2.0; // center x
1905 tmp_y = (-inf->min_vals[1]-inf->max_vals[1])/2.0; // center y
1906
1907 tmp_x += static_cast<double>(g_x_shift1) * size.x / g_logical_screen.x_dots;
1908 tmp_y += static_cast<double>(g_y_shift1) * size.y / g_logical_screen.y_dots;
1909 tmp_z = -inf->max_vals[2];
1910 trans(tmp_x, tmp_y, tmp_z, inf->double_mat1);
1911 }
1912 }
1913 return false;
1914 }
1915
1916 // apply perspective if requested
1917 if (g_viewer_z)
1918 {
1919 perspective(inf->view_vect);
1920 if (s_real_time)
1921 {
1922 perspective(inf->view_vect1);
1923 }
1924 }
