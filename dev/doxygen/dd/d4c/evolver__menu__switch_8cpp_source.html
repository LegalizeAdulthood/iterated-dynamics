40static MainState save_evolver_image(MainContext &/*context*/)
41{
42 if (driver_is_disk() && g_disk_targa)
43 {
44 // disk video and targa, nothing to save
45 return MainState::NOTHING;
46 }
47
48 GeneBase gene[NUM_GENES];
49 copy_genes_from_bank(gene);
50 {
51 ValueSaver saved_logical_screen_x_offset{g_logical_screen.x_offset, 0};
52 ValueSaver saved_logical_screen_y_offset{g_logical_screen.y_offset, 0};
53 ValueSaver saved_logical_screen_x_dots{g_logical_screen.x_dots, g_screen_x_dots};
54 ValueSaver saved_logical_screen_y_dots{g_logical_screen.y_dots, g_screen_y_dots};
55 {
56 ValueSaver saved_evolve_param_grid_x{g_evolve_param_grid_x, g_evolve_image_grid_size / 2};
57 ValueSaver saved_evolve_param_grid_y{g_evolve_param_grid_y, g_evolve_image_grid_size / 2};
58 restore_param_history();
59 fiddle_params(gene, 0);
60 draw_param_box(1);
61 save_image(g_save_filename);
62 }
63 restore_param_history();
64 fiddle_params(gene, unspiral_map());
65 }
66 copy_genes_to_bank(gene);
67 return MainState::CONTINUE;
68}

70static MainState prompt_evolver_options(MainContext &context)
71{
72 clear_zoom_box();
73 if (g_from_text)
74 {
75 g_from_text = false;
76 }
77 else
78 {
79 driver_stack_screen();
80 }
81 int i;
82 if (context.key == 'x')
83 {
84 i = get_toggles();
85 }
86 else if (context.key == 'y')
87 {
88 i = get_toggles2();
89 }
90 else if (context.key == 'p')
91 {
92 i = passes_options();
93 }
94 else if (context.key == 'z')
95 {
96 i = get_fract_params(true);
97 }
98 else if (context.key == ID_KEY_CTL_E || context.key == ID_KEY_SPACE)
99 {
100 i = get_evolve_params();
101 }
102 else
103 {
104 i = get_cmd_string();
105 }
106 driver_unstack_screen();
107 if (g_evolving != EvolutionModeFlags::NONE && g_true_color)
108 {
109 g_true_color = false; // truecolor doesn't play well with the evolver
110 }
111 if (i > 0)
112 {
113 // time to redraw?
114 save_param_history();
115 context.more_keys = false;
116 g_calc_status = CalcStatus::PARAMS_CHANGED;
117 }
118 return MainState::NOTHING;
119}

131static MainState move_evolver_selection(MainContext &context)
132{
133 if (!g_box_count)
134 {
135 return move_zoom_box(context);
136 }
137
138 // if no zoombox, scroll by arrows
139 // borrow ctrl cursor keys for moving selection box in evolver mode
140 GeneBase gene[NUM_GENES];
141 copy_genes_from_bank(gene);
142 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
143 {
144 if (context.key == ID_KEY_CTL_LEFT_ARROW)
145 {
146 g_evolve_param_grid_x--;
147 }
148 if (context.key == ID_KEY_CTL_RIGHT_ARROW)
149 {
150 g_evolve_param_grid_x++;
151 }
152 if (context.key == ID_KEY_CTL_UP_ARROW)
153 {
154 g_evolve_param_grid_y--;
155 }
156 if (context.key == ID_KEY_CTL_DOWN_ARROW)
157 {
158 g_evolve_param_grid_y++;
159 }
160 if (g_evolve_param_grid_x < 0)
161 {
162 g_evolve_param_grid_x = g_evolve_image_grid_size - 1;
163 }
164 if (g_evolve_param_grid_x > g_evolve_image_grid_size - 1)
165 {
166 g_evolve_param_grid_x = 0;
167 }
168 if (g_evolve_param_grid_y < 0)
169 {
170 g_evolve_param_grid_y = g_evolve_image_grid_size - 1;
171 }
172 if (g_evolve_param_grid_y > g_evolve_image_grid_size - 1)
173 {
174 g_evolve_param_grid_y = 0;
175 }
176 const int grout = bit_set(g_evolving, EvolutionModeFlags::NO_GROUT) ? 0 : 1;
177 g_logical_screen.x_offset = g_evolve_param_grid_x * static_cast<int>(g_logical_screen.x_size_dots + 1 + grout);
178 g_logical_screen.y_offset = g_evolve_param_grid_y * static_cast<int>(g_logical_screen.y_size_dots + 1 + grout);
179
180 restore_param_history();
181 fiddle_params(gene, unspiral_map()); // change all parameters
182 // to values appropriate to the image selected
183 set_evolve_ranges();
184 change_box(0, 0);
185 draw_param_box(0);
186 }
187 copy_genes_to_bank(gene);
188 return MainState::NOTHING;
189}

215static MainState evolver_zoom_in(MainContext &/*context*/)
216{
217 if (g_zoom_enabled)
218 {
219 if (g_zoom_box_width == 0)
220 {
221 // start zoombox
222 g_zoom_box_height = 1;
223 g_zoom_box_width = 1;
224 g_zoom_box_rotation = 0;
225 g_zoom_box_skew = 0.0;
226 g_zoom_box_x = 0.0;
227 g_zoom_box_y = 0.0;
228 find_special_colors();
229 g_box_color = g_color_bright;
230 if (bit_set(g_evolving, EvolutionModeFlags::FIELD_MAP))
231 {
232 // set screen view params back (previously changed to allow full screen saves in viewwindow
233 // mode)
234 const int grout = bit_set(g_evolving, EvolutionModeFlags::NO_GROUT) ? 0 : 1;
235 g_logical_screen.x_offset =
236 g_evolve_param_grid_x * static_cast<int>(g_logical_screen.x_size_dots + 1 + grout);
237 g_logical_screen.y_offset =
238 g_evolve_param_grid_y * static_cast<int>(g_logical_screen.y_size_dots + 1 + grout);
239 setup_param_box();
240 draw_param_box(0);
241 }
242 move_box(0.0, 0.0); // force scrolling
243 }
244 else
245 {
246 resize_box(0 - key_count(ID_KEY_PAGE_UP));
247 }
248 }
249 return MainState::NOTHING;

342static MainState request_mutation_level(MainContext &context)
343{
344 int mutation_level;
345 if (context.key >= ID_KEY_ALT_1 && context.key <= ID_KEY_ALT_7)
346 {
347 mutation_level = context.key - ID_KEY_ALT_1 + 1;
348 }
349 else if (context.key >= '1' && context.key <= '7')
350 {
351 mutation_level = context.key - '1' + 1;
352 }
353 else
354 {
355 throw std::runtime_error(
356 "Bad mutation level character (" + std::to_string(context.key) + ") to request_mutation_level");
357 }
358 set_mutation_level(mutation_level);
359 restore_param_history();
360 context.more_keys = false;
361 g_calc_status = CalcStatus::PARAMS_CHANGED;
362 return MainState::NOTHING;
363}

445MainState evolver_menu_switch(MainContext &context)
446{
447 assert(std::is_sorted(std::begin(HANDLERS), std::end(HANDLERS)));
448 if (const auto it = std::lower_bound(std::cbegin(HANDLERS), std::cend(HANDLERS), context.key);
449 it != std::cend(HANDLERS) && it->key == context.key)
450 {
451 return it->handler(context);
452 }
453
454 if (context.key == ID_KEY_DELETE)
455 {
456 // select video mode from list
457 request_video_mode(context.key);
458 }
459
460 // other (maybe a valid Fn key)
461 return requested_video_fn(context);
462}
