39/*
40 get_view_params() is called whenever the 'v' key
41 is pressed. Return codes are:
42 -1 routine was ESCAPEd - no need to re-generate the image.
43 0 minor variable changed. No need to re-generate the image.
44 1 View changed. Re-generate the image.
45*/
46
47int get_view_params()
48{
49 std::array<const char *, 16> choices;
50 std::array<FullScreenValues, 25> values;
51 int i;
52 int x_max;
53 int y_max;
54 char dim1[50];
55 char dim2[50];
56
57 driver_get_max_screen(x_max, y_max);
58
59 const bool old_view_window = g_viewport.enabled;
60 float old_view_reduction = g_viewport.reduction;
61 float old_aspect_ratio = g_viewport.final_aspect_ratio;
62 int old_view_x_dots = g_viewport.x_dots;
63 int old_view_y_dots = g_viewport.y_dots;
64 int old_screen_x_dots = g_screen_x_dots;
65 int old_screen_y_dots = g_screen_y_dots;
66
67get_view_restart:
68 // fill up the previous values arrays
69 int k = -1;
70
71 if (!driver_is_disk())
72 {
73 choices[++k] = "Preview display? (no for full screen)";
74 values[k].type = 'y';
75 values[k].uval.ch.val = g_viewport.enabled ? 1 : 0;
76
77 choices[++k] = "Auto window size reduction factor";
78 values[k].type = 'f';
79 values[k].uval.dval = g_viewport.reduction;
80
81 choices[++k] = "Final media overall aspect ratio, y/x";
82 values[k].type = 'f';
83 values[k].uval.dval = g_viewport.final_aspect_ratio;
84
85 choices[++k] = "Crop starting coordinates to new aspect ratio?";
86 values[k].type = 'y';
87 values[k].uval.ch.val = g_viewport.crop ? 1 : 0;
88
89 choices[++k] = "Explicit size x pixels (0 for auto size)";
90 values[k].type = 'i';
91 values[k].uval.ival = g_viewport.x_dots;
92
93 choices[++k] = " y pixels (0 to base on aspect ratio)";
94 values[k].type = 'i';
95 values[k].uval.ival = g_viewport.y_dots;
96 }
97
98 choices[++k] = "";
99 values[k].type = '*';
100
101 choices[++k] = "Virtual screen total x pixels";
102 values[k].type = 'i';
103 values[k].uval.ival = g_screen_x_dots;
104
105 choices[++k] = driver_is_disk() ?
106 " y pixels" :
107 " y pixels (0: by aspect ratio)";
108 values[k].type = 'i';
109 values[k].uval.ival = g_screen_y_dots;
110
111 choices[++k] = "Keep aspect? (cuts both x & y when either too big)";
112 values[k].type = 'y';
113 values[k].uval.ch.val = g_viewport.keep_aspect_ratio ? 1 : 0;
114
115 {
116 static const char *scroll_types[] = {"fixed", "relaxed"};
117 choices[++k] = "Zoombox scrolling (f[ixed], r[elaxed])";
118 values[k].type = 'l';
119 values[k].uval.ch.vlen = 7;
120 values[k].uval.ch.list_len = std::size(scroll_types);
121 values[k].uval.ch.list = scroll_types;
122 values[k].uval.ch.val = g_viewport.z_scroll ? 1 : 0;
123 }
124
125 choices[++k] = "";
126 values[k].type = '*';
127
128 *fmt::format_to(dim1, "Video memory limits: (for y = {:4d}) x <= {:d}", y_max, x_max).out = '\0';
129 choices[++k] = dim1;
130 values[k].type = '*';
131
132 *fmt::format_to(dim2, " (for x = {:4d}) y <= {:d}", x_max, y_max).out = '\0';
133 choices[++k] = dim2;
134 values[k].type = '*';
135
136 choices[++k] = "";
137 values[k].type = '*';
138
139 if (!driver_is_disk())
140 {
141 choices[++k] = "Press F4 to reset view parameters to defaults.";
142 values[k].type = '*';
143 }
144
145 {
146 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_VIEW};
147 i = full_screen_prompt("View Window Options", k+1, choices.data(), values.data(), 16, nullptr);
148 }
149 if (i < 0)
150 {
151 return -1;
152 }
153
154 if (i == ID_KEY_F4 && !driver_is_disk())
155 {
156 g_viewport.enabled = false;
157 g_viewport.x_dots = 0;
158 g_viewport.y_dots = 0;
159 g_viewport.reduction = 4.2F;
160 g_viewport.crop = true;
161 g_viewport.final_aspect_ratio = g_screen_aspect;
162 g_screen_x_dots = old_screen_x_dots;
163 g_screen_y_dots = old_screen_y_dots;
164 g_viewport.keep_aspect_ratio = true;
165 g_viewport.z_scroll = true;
166 goto get_view_restart;
167 }
168
169 // now check out the results (*hopefully* in the same order <grin>)
170 k = -1;
171
172 if (!driver_is_disk())
173 {
174 g_viewport.enabled = values[++k].uval.ch.val != 0;
175 g_viewport.reduction = static_cast<float>(values[++k].uval.dval);
176 g_viewport.final_aspect_ratio = static_cast<float>(values[++k].uval.dval);
177 g_viewport.crop = values[++k].uval.ch.val != 0;
178 g_viewport.x_dots = values[++k].uval.ival;
179 g_viewport.y_dots = values[++k].uval.ival;
180 }
181
182 ++k;
183
184 g_screen_x_dots = values[++k].uval.ival;
185 g_screen_y_dots = values[++k].uval.ival;
186 g_viewport.keep_aspect_ratio = values[++k].uval.ch.val != 0;
187 g_viewport.z_scroll = values[++k].uval.ch.val != 0;
188
189 if (x_max != -1 && g_screen_x_dots > x_max)
190 {
191 g_screen_x_dots = x_max;
192 }
193 g_screen_x_dots = std::max(g_screen_x_dots, 2);
194 if (g_screen_y_dots == 0) // auto by aspect ratio request
195 {
196 if (g_viewport.final_aspect_ratio == 0.0)
197 {
198 g_viewport.final_aspect_ratio = g_viewport.enabled && g_viewport.x_dots != 0 && g_viewport.y_dots != 0 ?
199 static_cast<float>(g_viewport.y_dots) / static_cast<float>(g_viewport.x_dots) : old_aspect_ratio;
200 }
201 g_screen_y_dots = static_cast<int>(std::lround(g_viewport.final_aspect_ratio * g_screen_x_dots));
202 }
203 if (y_max != -1 && g_screen_y_dots > y_max)
204 {
205 g_screen_y_dots = y_max;
206 }
207 g_screen_y_dots = std::max(g_screen_y_dots, 2);
208
209 if (driver_is_disk())
210 {
211 g_video_entry.x_dots = g_screen_x_dots;
212 g_video_entry.y_dots = g_screen_y_dots;
213 g_video_table[g_adapter] = g_video_entry;
214 if (g_viewport.final_aspect_ratio == 0.0)
215 {
216 g_viewport.final_aspect_ratio = static_cast<float>(g_screen_y_dots) / static_cast<float>(g_screen_x_dots);
217 }
218 }
219
220 if (g_viewport.x_dots != 0 && g_viewport.y_dots != 0 && g_viewport.enabled && g_viewport.final_aspect_ratio == 0.0)
221 {
222 g_viewport.final_aspect_ratio = static_cast<float>(g_viewport.y_dots) / static_cast<float>(g_viewport.x_dots);
223 }
224 else if (g_viewport.final_aspect_ratio == 0.0 && (g_viewport.x_dots == 0 || g_viewport.y_dots == 0))
225 {
226 g_viewport.final_aspect_ratio = old_aspect_ratio;
227 }
228
229 if (g_viewport.final_aspect_ratio != old_aspect_ratio && g_viewport.crop)
230 {
231 aspect_ratio_crop(old_aspect_ratio, g_viewport.final_aspect_ratio);
232 }
233
234 return g_viewport.enabled != old_view_window || g_screen_x_dots != old_screen_x_dots ||
235 g_screen_y_dots != old_screen_y_dots ||
236 (g_viewport.enabled &&
237 (g_viewport.reduction != old_view_reduction || g_viewport.final_aspect_ratio != old_aspect_ratio ||
238 g_viewport.x_dots != old_view_x_dots || (g_viewport.y_dots != old_view_y_dots && g_viewport.x_dots))) ? 1 : 0;
239}
