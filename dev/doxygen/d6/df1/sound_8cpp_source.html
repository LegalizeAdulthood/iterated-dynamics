46// open sound file
47bool sound_open()
48{
49 std::string sound_name{"sound001.txt"};
50 if ((g_orbit_save_flags & OSF_MIDI) != 0 && s_snd_fp == nullptr)
51 {
52 const std::string path{get_checked_save_path(WriteFile::SOUND, sound_name).string()};
53 assert(!path.empty());
54 s_snd_fp = std::fopen(path.c_str(), "w");
55 if (s_snd_fp == nullptr)
56 {
57 stop_msg(fmt::format("Can't open {:s} for writing.", sound_name));
58 }
59 }
60 return s_snd_fp != nullptr;
61}

63/* This routine plays a tone in the speaker and optionally writes a file

69 // cppcheck-suppress leakNoVarFunctionCall

72 std::fprintf(s_snd_fp, "%-d\n", tone);

77 // driver_key_pressed calls driver_sound_off() if TAB or F1 pressed

81 if (driver_sound_on(tone))

83 wait_until(g_orbit_delay);
