42// open sound file
43bool sound_open()
44{
45 std::string sound_name{"sound001.txt"};
46 if ((g_orbit_save_flags & OSF_MIDI) != 0 && s_snd_fp == nullptr)
47 {
48 std::string path{get_save_path(WriteFile::SOUND, sound_name).string()};
49 assert(!path.empty());
50 check_write_file(path, ".txt");
51 s_snd_fp = std::fopen(path.c_str(), "w");
52 if (s_snd_fp == nullptr)
53 {
54 stop_msg(fmt::format("Can't open {:s} for writing.", sound_name));
55 }
56 }
57 return s_snd_fp != nullptr;
58}

60/* This routine plays a tone in the speaker and optionally writes a file

66 // cppcheck-suppress leakNoVarFunctionCall

76 // if (20 < tone && tone < 15000) better limits?
77 // if (10 < tone && tone < 5000) better limits?
78 if (driver_sound_on(tone))

80 wait_until(g_orbit_delay);

82 }
83}
