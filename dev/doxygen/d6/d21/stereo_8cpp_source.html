106/*
107 The getdepth() function allows using the grayscale value of the color
108 as DEPTH, rather than the color number. Maybe I got a little too
109 sophisticated trying to avoid a divide, so the comment tells what all
110 the multiplies and shifts are trying to do. The result should be from
111 0 to 255.
112*/
113
114#define DAC (*((DACBox) (s_data->save_dac)))

116static int get_depth(const int xd, const int yd)
117{
118 int pal = get_color(xd, yd);
119 if (g_gray_flag)
120 {
121 // effectively (30*R + 59*G + 11*B)/100 scaled 0 to 255
122 pal = static_cast<int>(DAC[pal][0]) * 77 + static_cast<int>(DAC[pal][1]) * 151 +
123 static_cast<int>(DAC[pal][2]) * 28;
124 pal >>= 6;
125 }
126 return pal;
127}

133static bool get_min_max()
134{
135 MIN_C = g_colors;
136 MAX_C = 0;
137 for (int yd = 0; yd < g_logical_screen.y_dots; yd++)
138 {
139 if (driver_key_pressed())
140 {
141 return true;
142 }
143 if (yd == 20)
144 {
145 show_temp_msg("Getting min and max");
146 }
147 for (int xd = 0; xd < g_logical_screen.x_dots; xd++)
148 {
149 int depth = get_depth(xd, yd);
150 MIN_C = std::min(depth, MIN_C);
151 MAX_C = std::max(depth, MAX_C);
152 }
153 }
154 clear_temp_msg();
155 return false;
156}

158static void toggle_bars(bool *bars, const int bar_width, const int *colour)
159{
160 find_special_colors();
161 int ct = 0;
162 for (int i = X_CENTER; i < X_CENTER + bar_width; i++)
163 {
164 for (int j = Y_CENTER; j < Y_CENTER + BAR_HEIGHT; j++)
165 {
166 if (*bars)
167 {
168 g_put_color(i + static_cast<int>(AVG), j, g_color_bright);
169 g_put_color(i - static_cast<int>(AVG), j, g_color_bright);
170 }
171 else
172 {
173 g_put_color(i + static_cast<int>(AVG), j, colour[ct++]);
174 g_put_color(i - static_cast<int>(AVG), j, colour[ct++]);
175 }
176 }
177 }
178 *bars = !*bars;
179}

181int out_line_stereo(Byte *pixels, const int line_len)
182{
183 if (Y >= g_logical_screen.y_dots)
184 {
185 return 1;
186 }
187
188 std::vector<int> same;
189 same.resize(g_logical_screen.x_dots);
190 for (int x = 0; x < g_logical_screen.x_dots; ++x)
191 {
192 same[x] = x;
193 }
194 std::vector<int> colour;
195 colour.resize(g_logical_screen.x_dots);
196 for (int x = 0; x < g_logical_screen.x_dots; ++x)
197 {
198 if (REVERSE)
199 {
200 SEP = GROUND - static_cast<int>(DEPTH * (get_depth(x, Y) - MIN_C) / MAX_CC);
201 }
202 else
203 {
204 SEP = GROUND - static_cast<int>(DEPTH * (MAX_CC - (get_depth(x, Y) - MIN_C)) / MAX_CC);
205 }
206 SEP = static_cast<int>(SEP * 10.0 / WIDTH); // adjust for media WIDTH
207
208 // get average value under calibration bars
209 if (X1 <= x && x <= X2 && Y1 <= Y && Y <= Y2)
210 {
211 AVG += SEP;
212 AVG_CT++;
213 }
214 int i = x - (SEP + (SEP & Y & 1)) / 2;
215 int j = i + SEP;
216 if (0 <= i && j < g_logical_screen.x_dots)
217 {
218 // there are cases where next never terminates so we time out
219 int ct = 0;
220 for (int s = same[i]; s != i && s != j && ct++ < g_logical_screen.x_dots; s = same[i])
221 {
222 if (s > j)
223 {
224 same[i] = j;
225 i = j;
226 j = s;
227 }
228 else
229 {
230 i = s;
231 }
232 }
233 same[i] = j;
234 }
235 }
236 for (int x = g_logical_screen.x_dots - 1; x >= 0; x--)
237 {
238 if (same[x] == x)
239 {
240 colour[x] = static_cast<int>(pixels[x % line_len]);
241 }
242 else
243 {
244 colour[x] = colour[same[x]];
245 }
246 g_put_color(x, Y, colour[x]);
247 }
248 Y++;
249 return 0;
250}

285static bool convert_stereo_image(const bool interactive)
286{
287 // TODO: replace this stack variable with static data s_data
288 StereoData v;
289 bool bars;
290 std::vector<int> colour;
291 colour.resize(g_logical_screen.x_dots);
292
293 if (interactive)
294 {
295 g_save_rds_params = false;
296 }
297 s_data = &v; // set static vars to stack structure
298
299 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_RDS_KEYS};
300 StereoSaveRestore saved_stereo{v};
301
302 if (g_logical_screen.x_dots > OLD_MAX_PIXELS)
303 {
304 stop_msg("Stereo not allowed with resolution > 2048 pixels wide");
305 driver_buzzer(Buzzer::INTERRUPT);
306 return true;
307 }
308
309 // empirically determined adjustment to make WIDTH scale correctly
310 WIDTH = g_auto_stereo_width * .67;
311 WIDTH = std::max(WIDTH, 1.0);
312 GROUND = g_logical_screen.x_dots / 8;
313 if (g_auto_stereo_depth < 0)
314 {
315 REVERSE = 1;
316 }
317 else
318 {
319 REVERSE = 0;
320 }
321 {
322 ValueSaver saved_busy{g_busy, true};
323
324 DEPTH = static_cast<long>(g_logical_screen.x_dots) * static_cast<long>(g_auto_stereo_depth) / 4000L;
325 DEPTH = std::abs(DEPTH) + 1;
326 if (get_min_max())
327 {
328 driver_buzzer(Buzzer::INTERRUPT);
329 return true;
330 }
331 MAX_CC = MAX_C - MIN_C + 1;
332 AVG_CT = 0L;
333 AVG = AVG_CT;
334 BAR_HEIGHT = 1 + g_logical_screen.y_dots / 20;
335 X_CENTER = g_logical_screen.x_dots / 2;
336 if (g_calibrate > CalibrationBars::MIDDLE)
337 {
338 Y_CENTER = BAR_HEIGHT / 2;
339 }
340 else
341 {
342 Y_CENTER = g_logical_screen.y_dots / 2;
343 }
344
345 // box to average for calibration bars
346 X1 = X_CENTER - g_logical_screen.x_dots / 16;
347 X2 = X_CENTER + g_logical_screen.x_dots / 16;
348 Y1 = Y_CENTER - BAR_HEIGHT / 2;
349 Y2 = Y_CENTER + BAR_HEIGHT / 2;
350
351 Y = 0;
352 if (g_use_stereo_texture)
353 {
354 g_out_line = out_line_stereo;
355 while (Y < g_logical_screen.y_dots)
356 {
357 if (gif_view())
358 {
359 return true;
360 }
361 }
362 }
363 else
364 {
365 set_random_seed();
366 std::vector<Byte> buf;
367 buf.resize(g_logical_screen.x_dots);
368 while (Y < g_logical_screen.y_dots)
369 {
370 if (driver_key_pressed())
371 {
372 return true;
373 }
374 random_dot_line(buf.data(), g_logical_screen.x_dots);
375 out_line_stereo(buf.data(), g_logical_screen.x_dots);
376 }
377 }
378 }
379
380 find_special_colors();
381 AVG /= AVG_CT;
382 AVG /= 2;
383 int ct = 0;
384 const int bar_width = 1 + g_logical_screen.x_dots / 200;
385 for (int i = X_CENTER; i < X_CENTER + bar_width; i++)
386 {
387 for (int j = Y_CENTER; j < Y_CENTER + BAR_HEIGHT; j++)
388 {
389 colour[ct++] = get_color(i + static_cast<int>(AVG), j);
390 colour[ct++] = get_color(i - static_cast<int>(AVG), j);
391 }
392 }
393 bars = g_calibrate != CalibrationBars::NONE;
394 toggle_bars(&bars, bar_width, colour.data());
395 if (interactive)
396 {
397 bool done = false;
398 while (!done)
399 {
400 driver_wait_key_pressed(false);
401 switch (int key = driver_get_key(); key)
402 {
403 case ID_KEY_ENTER: // toggle bars
404 case ID_KEY_SPACE:
405 toggle_bars(&bars, bar_width, colour.data());
406 break;
407 case 'c':
408 case '+':
409 case '-':
410 rotate(key == 'c' ? SpinDirection::NONE
411 : key == '+' ? SpinDirection::FORWARD
412 : SpinDirection::BACKWARD);
413 break;
414 case 's':
415 case 'S':
416 save_image(g_save_filename);
417 break;
418 default:
419 g_save_rds_params = key == 'b' || key == 'B';
420 if (key == ID_KEY_ESC) // if ESC avoid returning to menu
421 {
422 key = 255;
423 }
424 driver_unget_key(key);
425 driver_buzzer(Buzzer::COMPLETE);
426 done = true;
427 break;
428 }
429 }
430 }
431 else if (save_image(g_save_filename) != 0)
432 {
433 return true;
434 }
435
436 return false;
437}
