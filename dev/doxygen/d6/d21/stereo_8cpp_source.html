103/*
104 The getdepth() function allows using the grayscale value of the color
105 as DEPTH, rather than the color number. Maybe I got a little too
106 sophisticated trying to avoid a divide, so the comment tells what all
107 the multiplies and shifts are trying to do. The result should be from
108 0 to 255.
109*/
110
111#define DAC (*((DACBox)(s_data->save_dac)))

113static int get_depth(const int xd, const int yd)
114{
115 int pal = get_color(xd, yd);
116 if (g_gray_flag)
117 {
118 // effectively (30*R + 59*G + 11*B)/100 scaled 0 to 255
119 pal = static_cast<int>(DAC[pal][0]) * 77 + static_cast<int>(DAC[pal][1]) * 151 + static_cast<int>(DAC[pal][2]) * 28;
120 pal >>= 6;
121 }
122 return pal;
123}

125/*
126 Get min and max DEPTH value in picture

129static bool get_min_max()
130{
131 MIN_C = g_colors;
132 MAX_C = 0;
133 for (int yd = 0; yd < g_logical_screen.y_dots; yd++)
134 {
135 if (driver_key_pressed())
136 {
137 return true;
138 }
139 if (yd == 20)
140 {
141 show_temp_msg("Getting min and max");
142 }
143 for (int xd = 0; xd < g_logical_screen.x_dots; xd++)
144 {
145 int depth = get_depth(xd, yd);
146 MIN_C = std::min(depth, MIN_C);
147 MAX_C = std::max(depth, MAX_C);
148 }
149 }
150 clear_temp_msg();
151 return false;
152}

154static void toggle_bars(bool *bars, const int bar_width, const int *colour)
155{
156 find_special_colors();
157 int ct = 0;
158 for (int i = X_CENTER; i < X_CENTER + bar_width; i++)
159 {
160 for (int j = Y_CENTER; j < Y_CENTER + BAR_HEIGHT; j++)
161 {
162 if (*bars)
163 {
164 g_put_color(i + static_cast<int>(AVG), j , g_color_bright);
165 g_put_color(i - static_cast<int>(AVG), j , g_color_bright);
166 }
167 else
168 {
169 g_put_color(i + static_cast<int>(AVG), j, colour[ct++]);
170 g_put_color(i - static_cast<int>(AVG), j, colour[ct++]);
171 }
172 }
173 }
174 *bars = !*bars;
175}

177int out_line_stereo(Byte *pixels, const int line_len)
178{
179 if (Y >= g_logical_screen.y_dots)
180 {
181 return 1;
182 }
183
184 std::vector<int> same;
185 same.resize(g_logical_screen.x_dots);
186 for (int x = 0; x < g_logical_screen.x_dots; ++x)
187 {
188 same[x] = x;
189 }
190 std::vector<int> colour;
191 colour.resize(g_logical_screen.x_dots);
192 for (int x = 0; x < g_logical_screen.x_dots; ++x)
193 {
194 if (REVERSE)
195 {
196 SEP = GROUND - static_cast<int>((DEPTH * (get_depth(x, Y) - MIN_C) / MAX_CC));
197 }
198 else
199 {
200 SEP = GROUND - static_cast<int>((DEPTH * (MAX_CC - (get_depth(x, Y) - MIN_C)) / MAX_CC));
201 }
202 SEP = static_cast<int>((SEP * 10.0 / WIDTH)); // adjust for media WIDTH
203
204 // get average value under calibration bars
205 if (X1 <= x && x <= X2 && Y1 <= Y && Y <= Y2)
206 {
207 AVG += SEP;
208 AVG_CT++;
209 }
210 int i = x - (SEP + (SEP & Y & 1)) / 2;
211 int j = i + SEP;
212 if (0 <= i && j < g_logical_screen.x_dots)
213 {
214 // there are cases where next never terminates so we time out
215 int ct = 0;
216 for (int s = same[i]; s != i && s != j && ct++ < g_logical_screen.x_dots; s = same[i])
217 {
218 if (s > j)
219 {
220 same[i] = j;
221 i = j;
222 j = s;
223 }
224 else
225 {
226 i = s;
227 }
228 }
229 same[i] = j;
230 }
231 }
232 for (int x = g_logical_screen.x_dots - 1; x >= 0; x--)
233 {
234 if (same[x] == x)
235 {
236 colour[x] = static_cast<int>(pixels[x % line_len]);
237 }
238 else
239 {
240 colour[x] = colour[same[x]];
241 }
242 g_put_color(x, Y, colour[x]);
243 }
244 Y++;
245 return 0;
246}

248/**************************************************************************
249 Convert current image into Auto Stereo Picture
250**************************************************************************/
251
252bool auto_stereo_convert()
253{
254 // TODO: replace this stack variable with static data s_data
255 StereoData v;
256 Byte save_dac_box[256*3];
257 bool ret = false;
258 bool bars;
259 std::time_t now;
260 std::vector<int> colour;
261 colour.resize(g_logical_screen.x_dots);
262
263 s_data = &v; // set static vars to stack structure
264 s_data->save_dac = save_dac_box;
265
266 // Use the current time to randomize the random number sequence.
267 std::time(&now);
268 std::srand(static_cast<unsigned int>(now));
269
270 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_RDS_KEYS};
271 driver_save_graphics(); // save graphics image
272 std::memcpy(save_dac_box, g_dac_box, 256 * 3); // save colors
273
274 if (g_logical_screen.x_dots > OLD_MAX_PIXELS)
275 {
276 stop_msg("Stereo not allowed with resolution > 2048 pixels wide");
277 driver_buzzer(Buzzer::INTERRUPT);
278 ret = true;
279 goto exit_stereo;
280 }
281
282 // empircally determined adjustment to make WIDTH scale correctly
283 WIDTH = g_auto_stereo_width*.67;
284 WIDTH = std::max(WIDTH, 1.0);
285 GROUND = g_logical_screen.x_dots / 8;
286 if (g_auto_stereo_depth < 0)
287 {
288 REVERSE = 1;
289 }
290 else
291 {
292 REVERSE = 0;
293 }
294 DEPTH = static_cast<long>(g_logical_screen.x_dots) * static_cast<long>(g_auto_stereo_depth) / 4000L;
295 DEPTH = std::abs(DEPTH) + 1;
296 if (get_min_max())
297 {
298 driver_buzzer(Buzzer::INTERRUPT);
299 ret = true;
300 goto exit_stereo;
301 }
302 MAX_CC = MAX_C - MIN_C + 1;
303 AVG_CT = 0L;
304 AVG = AVG_CT;
305 BAR_HEIGHT = 1 + g_logical_screen.y_dots / 20;
306 X_CENTER = g_logical_screen.x_dots/2;
307 if (g_calibrate > CalibrationBars::MIDDLE)
308 {
309 Y_CENTER = BAR_HEIGHT/2;
310 }
311 else
312 {
313 Y_CENTER = g_logical_screen.y_dots/2;
314 }
315
316 // box to average for calibration bars
317 X1 = X_CENTER - g_logical_screen.x_dots/16;
318 X2 = X_CENTER + g_logical_screen.x_dots/16;
319 Y1 = Y_CENTER - BAR_HEIGHT/2;
320 Y2 = Y_CENTER + BAR_HEIGHT/2;
321
322 Y = 0;
323 if (g_image_map)
324 {
325 g_out_line = out_line_stereo;
326 while (Y < g_logical_screen.y_dots)
327 {
328 if (gif_view())
329 {
330 ret = true;
331 goto exit_stereo;
332 }
333 }
334 }
335 else
336 {
337 std::vector<Byte> buf;
338 buf.resize(g_logical_screen.x_dots);
339 while (Y < g_logical_screen.y_dots)
340 {
341 if (driver_key_pressed())
342 {
343 ret = true;
344 goto exit_stereo;
345 }
346 for (int i = 0; i < g_logical_screen.x_dots; i++)
347 {
348 buf[i] = static_cast<unsigned char>(std::rand() % g_colors);
349 }
350 out_line_stereo(buf.data(), g_logical_screen.x_dots);
351 }
352 }
353
354 find_special_colors();
355 AVG /= AVG_CT;
356 AVG /= 2;
357 {
358 bool done = false;
359 int ct = 0;
360 const int bar_width = 1 + g_logical_screen.x_dots / 200;
361 for (int i = X_CENTER; i < X_CENTER + bar_width; i++)
362 {
363 for (int j = Y_CENTER; j < Y_CENTER + BAR_HEIGHT; j++)
364 {
365 colour[ct++] = get_color(i + static_cast<int>(AVG), j);
366 colour[ct++] = get_color(i - static_cast<int>(AVG), j);
367 }
368 }
369 bars = g_calibrate != CalibrationBars::NONE;
370 toggle_bars(&bars, bar_width, colour.data());
371 while (!done)
372 {
373 driver_wait_key_pressed(false);
374 switch (int key = driver_get_key(); key)
375 {
376 case ID_KEY_ENTER: // toggle bars
377 case ID_KEY_SPACE:
378 toggle_bars(&bars, bar_width, colour.data());
379 break;
380 case 'c':
381 case '+':
382 case '-':
383 rotate(key == 'c' ? SpinDirection::NONE
384 : key == '+' ? SpinDirection::FORWARD
385 : SpinDirection::BACKWARD);
386 break;
387 case 's':
388 case 'S':
389 save_image(g_save_filename);
390 break;
391 default:
392 if (key == ID_KEY_ESC) // if ESC avoid returning to menu
393 {
394 key = 255;
395 }
396 driver_unget_key(key);
397 driver_buzzer(Buzzer::COMPLETE);
398 done = true;
399 break;
400 }
401 }
402 }
403
404exit_stereo:
405 driver_restore_graphics();
406 std::memcpy(g_dac_box, save_dac_box, 256 * 3);
407 refresh_dac();
408 return ret;
409}
