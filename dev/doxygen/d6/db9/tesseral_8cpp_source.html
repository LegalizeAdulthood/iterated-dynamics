56Tesseral::Tesseral() :
57 m_guess_plot(g_plot != g_put_color && g_plot != sym_plot2),
58 m_tp(&m_stack[0])
59{
60 m_tp->x1 = g_i_start_pt.x; // set up initial box
61 m_tp->x2 = g_i_stop_pt.x;
62 m_tp->y1 = g_i_start_pt.y;
63 m_tp->y2 = g_i_stop_pt.y;
64
65 if (g_work_pass == 0) // not resuming
66 {
67 m_tp->top = tess_row(g_i_start_pt.x, g_i_stop_pt.x, g_i_start_pt.y); // Do top row
68 m_tp->bot = tess_row(g_i_start_pt.x, g_i_stop_pt.x, g_i_stop_pt.y); // Do bottom row
69 m_tp->lft = tess_col(g_i_start_pt.x, g_i_start_pt.y+1, g_i_stop_pt.y-1); // Do left column
70 m_tp->rgt = tess_col(g_i_stop_pt.x, g_i_start_pt.y+1, g_i_stop_pt.y-1); // Do right column
71 }
72 else // resuming, rebuild work stack
73 {
74 m_tp->rgt = -2;
75 m_tp->lft = -2;
76 m_tp->bot = -2;
77 m_tp->top = -2;
78 const int cur_y = g_begin_pt.y & 0xfff;
79 int y_size = 1;
80 int i = static_cast<unsigned>(g_begin_pt.y) >> 12;
81 while (--i >= 0)
82 {
83 y_size <<= 1;
84 }
85 const int cur_x = g_work_pass & 0xfff;
86 int x_size = 1;
87 i = static_cast<unsigned>(g_work_pass) >> 12;
88 while (--i >= 0)
89 {
90 x_size <<= 1;
91 }
92 while (true)
93 {
94 Tess *tp2 = m_tp;
95 if (m_tp->x2 - m_tp->x1 > m_tp->y2 - m_tp->y1)
96 {
97 // next divide down middle
98 if (m_tp->x1 == cur_x && m_tp->x2 - m_tp->x1 - 2 < x_size)
99 {
100 break;
101 }
102 const int mid = (m_tp->x1 + m_tp->x2) >> 1; // Find mid-point
103 if (mid > cur_x)
104 {
105 // stack right part
106 *++m_tp = *tp2; // copy current box
107 m_tp->x2 = mid;
108 }
109 tp2->x1 = mid;
110 }
111 else
112 {
113 // next divide across
114 if (m_tp->y1 == cur_y && m_tp->y2 - m_tp->y1 - 2 < y_size)
115 {
116 break;
117 }
118 const int mid = (m_tp->y1 + m_tp->y2) >> 1; // Find mid-point
119 if (mid > cur_y)
120 {
121 // stack bottom part
122 *++m_tp = *tp2; // copy current box
123 m_tp->y2 = mid;
124 }
125 tp2->y1 = mid;
126 }
127 }
128 }
129
130 g_passes = Passes::TESSERAL; // for tab_display
131}

145bool Tesseral::split_needed()
146{
147 if (m_tp->top == -1 || m_tp->bot == -1 || m_tp->lft == -1 || m_tp->rgt == -1)
148 {
149 return true;
150 }
151 // for any edge whose color is unknown, set it
152 if (m_tp->top == -2)
153 {
154 m_tp->top = tess_check_row(m_tp->x1, m_tp->x2, m_tp->y1);
155 }
156 if (m_tp->top == -1)
157 {
158 return true;
159 }
160 if (m_tp->bot == -2)
161 {
162 m_tp->bot = tess_check_row(m_tp->x1, m_tp->x2, m_tp->y2);
163 }
164 if (m_tp->bot != m_tp->top)
165 {
166 return true;
167 }
168 if (m_tp->lft == -2)
169 {
170 m_tp->lft = tess_check_col(m_tp->x1, m_tp->y1, m_tp->y2);
171 }
172 if (m_tp->lft != m_tp->top)
173 {
174 return true;
175 }
176 if (m_tp->rgt == -2)
177 {
178 m_tp->rgt = tess_check_col(m_tp->x2, m_tp->y1, m_tp->y2);
179 }
180 if (m_tp->rgt != m_tp->top)
181 {
182 return true;
183 }
184
185 if (m_tp->x2 - m_tp->x1 > m_tp->y2 - m_tp->y1)
186 {
187 // divide down the middle
188 const int mid = (m_tp->x1 + m_tp->x2) >> 1; // Find mid-point
189 const int mid_color = tess_col(mid, m_tp->y1 + 1, m_tp->y2 - 1); // Do mid column
190 if (mid_color != m_tp->top)
191 {
192 return true;
193 }
194 }
195 else
196 {
197 // divide across the middle
198 const int mid = (m_tp->y1 + m_tp->y2) >> 1; // Find mid-point
199 const int mid_color = tess_row(m_tp->x1 + 1, m_tp->x2 - 1, mid); // Do mid row
200 if (mid_color != m_tp->top)
201 {
202 return true;
203 }
204 }
205
206 return false;
207}

209void Tesseral::fill_box()
210{
211 // all 4 edges are the same color, fill in
212 if (g_fill_color != 0)
213 {
214 if (g_fill_color > 0)
215 {
216 m_tp->top = g_fill_color % g_colors;
217 }
218 if (const int j = m_tp->x2 - m_tp->x1 - 1; m_guess_plot || j < 2)
219 {
220 for (g_col = m_tp->x1 + 1; g_col < m_tp->x2; g_col++)
221 {
222 for (g_row = m_tp->y1 + 1; g_row < m_tp->y2; g_row++)
223 {
224 g_plot(g_col, g_row, m_tp->top);
225 }
226 }
227 }
228 else
229 {
230 // use write_span for speed
231 const std::vector pixels(j, static_cast<Byte>(m_tp->top));
232 for (g_row = m_tp->y1 + 1; g_row < m_tp->y2; g_row++)
233 {
234 write_span(g_row, m_tp->x1 + 1, m_tp->x2 - 1, pixels.data());
235 if (g_plot != g_put_color) // symmetry
236 {
237 if (const int k = g_stop_pt.y - (g_row - g_start_pt.y);
238 k > g_i_stop_pt.y && k < g_logical_screen.y_dots)
239 {
240 write_span(k, m_tp->x1 + 1, m_tp->x2 - 1, pixels.data());
241 }
242 }
243 }
244 }
245 }
246 --m_tp;
247}

249bool Tesseral::split_box()
250{
251 // box not surrounded by same color, subdivide
252 int mid;
253 int mid_color;
254 Tess *tp2;
255 if (m_tp->x2 - m_tp->x1 > m_tp->y2 - m_tp->y1)
256 {
257 // divide down the middle
258 mid = (m_tp->x1 + m_tp->x2) >> 1; // Find mid-point
259 mid_color = tess_col(mid, m_tp->y1 + 1, m_tp->y2 - 1); // Do mid column
260 if (mid_color == -3)
261 {
262 return true;
263 }
264 if (m_tp->x2 - mid > 1)
265 {
266 // right part >= 1 column
267 if (m_tp->top == -1)
268 {
269 m_tp->top = -2;
270 }
271 if (m_tp->bot == -1)
272 {
273 m_tp->bot = -2;
274 }
275 tp2 = m_tp;
276 if (mid - m_tp->x1 > 1)
277 {
278 // left part >= 1 col, stack right
279 *++m_tp = *tp2; // copy current box
280 m_tp->x2 = mid;
281 m_tp->rgt = mid_color;
282 }
283 tp2->x1 = mid;
284 tp2->lft = mid_color;
285 }
286 else
287 {
288 --m_tp;
289 }
290 }
291 else
292 {
293 // divide across the middle
294 mid = (m_tp->y1 + m_tp->y2) >> 1; // Find mid-point
295 mid_color = tess_row(m_tp->x1 + 1, m_tp->x2 - 1, mid); // Do mid row
296 if (mid_color == -3)
297 {
298 return true;
299 }
300 if (m_tp->y2 - mid > 1)
301 {
302 // bottom part >= 1 column
303 if (m_tp->lft == -1)
304 {
305 m_tp->lft = -2;
306 }
307 if (m_tp->rgt == -1)
308 {
309 m_tp->rgt = -2;
310 }
311 tp2 = m_tp;
312 if (mid - m_tp->y1 > 1)
313 {
314 // top also >= 1 col, stack bottom
315 *++m_tp = *tp2; // copy current box
316 m_tp->y2 = mid;
317 m_tp->bot = mid_color;
318 }
319 tp2->y1 = mid;
320 tp2->top = mid_color;
321 }
322 else
323 {
324 --m_tp;
325 }
326 }
327 return false;
328}

330void Tesseral::suspend()
331{
332 // didn't complete
333 int y_size = 1;
334 int x_size = 1;
335 int i = 2;
336 while (m_tp->x2 - m_tp->x1 - 2 >= i)
337 {
338 i <<= 1;
339 ++x_size;
340 }
341 i = 2;
342 while (m_tp->y2 - m_tp->y1 - 2 >= i)
343 {
344 i <<= 1;
345 ++y_size;
346 }
347 add_work_list(g_start_pt.x, g_start_pt.y, g_stop_pt.x, g_stop_pt.y, g_start_pt.x,
348 (y_size << 12) + m_tp->y1, (x_size << 12) + m_tp->x1, g_work_symmetry);
349}

351int tesseral()
352{
353 Tesseral tess;
354
355 while (!tess.done())
356 {
357 tess.next_box();
358
359 if (tess.split_needed())
360 {
361 if (tess.split_box())
362 {
363 break;
364 }
365 }
366 else
367 {
368 tess.fill_box();
369 if (check_key())
370 {
371 break;
372 }
373 }
374 }
375
376 if (!tess.done())
377 {
378 tess.suspend();
379 return -1;
380 }
381
382 return 0;
383}

412static int tess_col(const int x, const int y1, const int y2)
413{
414 g_col = x;
415 g_row = y1;
416 g_reset_periodicity = true;
417 int col_color = g_calc_type();
418 // cppcheck-suppress redundantAssignment
419 g_reset_periodicity = false;
420 while (++g_row <= y2)
421 {
422 // generate the column
423 const int i = g_calc_type();
424 if (i < 0)
425 {
426 return -3;
