21/* This routine copies the current screen to by flipping the x-axis, y-axis,
22 or both. Refuses to work if calculation in progress or if fractal
23 non-resumable. Clears zoombox if any. Resets corners so resulting fractal
24 is still valid. */
25MainState flip_image(MainContext &context)
26{
27 int temp_dot;
28
29 // fractal must be rotate-able and be finished
30 if (bit_set(g_cur_fractal_specific->flags, FractalFlags::NO_ROTATE) //
31 || g_calc_status == CalcStatus::IN_PROGRESS //
32 || g_calc_status == CalcStatus::RESUMABLE)
33 {
34 return MainState::NOTHING;
35 }
36 if (g_bf_math != BFMathType::NONE)
37 {
38 clear_zoom_box(); // clear, don't copy, the zoombox
39 }
40 const int x_half = g_logical_screen.x_dots / 2;
41 const int y_half = g_logical_screen.y_dots / 2;
42 switch (context.key)
43 {
44 case ID_KEY_CTL_X: // control-X - reverse X-axis
45 for (int i = 0; i < x_half; i++)
46 {
47 if (driver_key_pressed())
48 {
49 break;
50 }
51 for (int j = 0; j < g_logical_screen.y_dots; j++)
52 {
53 temp_dot = get_color(i, j);
54 g_put_color(i, j, get_color(g_logical_screen.x_dots-1-i, j));
55 g_put_color(g_logical_screen.x_dots-1-i, j, temp_dot);
56 }
57 }
58 g_save_image_region.m_min.x = g_image_region.m_max.x + g_image_region.m_min.x - g_image_region.m_3rd.x;
59 g_save_image_region.m_min.y = g_image_region.m_3rd.y;
60 g_save_image_region.m_max.x = g_image_region.m_3rd.x;
61 g_save_image_region.m_max.y = g_image_region.m_max.y + g_image_region.m_min.y - g_image_region.m_3rd.y;
62 g_save_image_region.m_3rd.x = g_image_region.m_max.x;
63 g_save_image_region.m_3rd.y = g_image_region.m_min.y;
64 if (g_bf_math != BFMathType::NONE)
65 {
66 add_bf(g_bf_save_x_min, g_bf_x_max, g_bf_x_min); // sxmin = xxmax + xxmin - xx3rd;
67 sub_a_bf(g_bf_save_x_min, g_bf_x_3rd);
68 add_bf(g_bf_save_y_max, g_bf_y_max, g_bf_y_min); // symax = yymax + yymin - yy3rd;
69 sub_a_bf(g_bf_save_y_max, g_bf_y_3rd);
70 copy_bf(g_bf_save_x_max, g_bf_x_3rd); // sxmax = xx3rd;
71 copy_bf(g_bf_save_y_min, g_bf_y_3rd); // symin = yy3rd;
72 copy_bf(g_bf_save_x_3rd, g_bf_x_max); // sx3rd = xxmax;
73 copy_bf(g_bf_save_y_3rd, g_bf_y_min); // sy3rd = yymin;
74 }
75 break;
76 case ID_KEY_CTL_Y: // control-Y - reverse Y-aXis
77 for (int j = 0; j < y_half; j++)
78 {
79 if (driver_key_pressed())
80 {
81 break;
82 }
83 for (int i = 0; i < g_logical_screen.x_dots; i++)
84 {
85 temp_dot = get_color(i, j);
86 g_put_color(i, j, get_color(i, g_logical_screen.y_dots-1-j));
87 g_put_color(i, g_logical_screen.y_dots-1-j, temp_dot);
88 }
89 }
90 g_save_image_region.m_min.x = g_image_region.m_3rd.x;
91 g_save_image_region.m_min.y = g_image_region.m_max.y + g_image_region.m_min.y - g_image_region.m_3rd.y;
92 g_save_image_region.m_max.x = g_image_region.m_max.x + g_image_region.m_min.x - g_image_region.m_3rd.x;
93 g_save_image_region.m_max.y = g_image_region.m_3rd.y;
94 g_save_image_region.m_3rd.x = g_image_region.m_min.x;
95 g_save_image_region.m_3rd.y = g_image_region.m_max.y;
96 if (g_bf_math != BFMathType::NONE)
97 {
98 copy_bf(g_bf_save_x_min, g_bf_x_3rd); // sxmin = xx3rd;
99 copy_bf(g_bf_save_y_max, g_bf_y_3rd); // symax = yy3rd;
100 add_bf(g_bf_save_x_max, g_bf_x_max, g_bf_x_min); // sxmax = xxmax + xxmin - xx3rd;
101 sub_a_bf(g_bf_save_x_max, g_bf_x_3rd);
102 add_bf(g_bf_save_y_min, g_bf_y_max, g_bf_y_min); // symin = yymax + yymin - yy3rd;
103 sub_a_bf(g_bf_save_y_min, g_bf_y_3rd);
104 copy_bf(g_bf_save_x_3rd, g_bf_x_min); // sx3rd = xxmin;
105 copy_bf(g_bf_save_y_3rd, g_bf_y_max); // sy3rd = yymax;
106 }
107 break;
108 case ID_KEY_CTL_Z: // control-Z - reverse X and Y aXis
109 for (int i = 0; i < x_half; i++)
110 {
111 if (driver_key_pressed())
112 {
113 break;
114 }
115 for (int j = 0; j < g_logical_screen.y_dots; j++)
116 {
117 temp_dot = get_color(i, j);
118 g_put_color(i, j, get_color(g_logical_screen.x_dots-1-i, g_logical_screen.y_dots-1-j));
119 g_put_color(g_logical_screen.x_dots-1-i, g_logical_screen.y_dots-1-j, temp_dot);
120 }
121 }
122 g_save_image_region.m_min = g_image_region.m_max;
123 g_save_image_region.m_max = g_image_region.m_min;
124 g_save_image_region.m_3rd = g_image_region.m_max + g_image_region.m_min - g_image_region.m_3rd;
125 if (g_bf_math != BFMathType::NONE)
126 {
127 copy_bf(g_bf_save_x_min, g_bf_x_max); // sxmin = xxmax;
128 copy_bf(g_bf_save_y_max, g_bf_y_min); // symax = yymin;
129 copy_bf(g_bf_save_x_max, g_bf_x_min); // sxmax = xxmin;
130 copy_bf(g_bf_save_y_min, g_bf_y_max); // symin = yymax;
131 add_bf(g_bf_save_x_3rd, g_bf_x_max, g_bf_x_min); // sx3rd = xxmax + xxmin - xx3rd;
132 sub_a_bf(g_bf_save_x_3rd, g_bf_x_3rd);
133 add_bf(g_bf_save_y_3rd, g_bf_y_max, g_bf_y_min); // sy3rd = yymax + yymin - yy3rd;
134 sub_a_bf(g_bf_save_y_3rd, g_bf_y_3rd);
135 }
136 break;
137 }
138 reset_zoom_corners();
139 g_calc_status = CalcStatus::PARAMS_CHANGED;
140 return MainState::NOTHING;
141}
