22int lambda_orbit()
23{
24 // variation of classical Mandelbrot/Julia
25 // note that fast >= 287 equiv in fracsuba.asm must be kept in step
26
27 g_temp_sqr_x = g_old_z.x - g_temp_sqr_x + g_temp_sqr_y;
28 g_temp_sqr_y = -(g_old_z.y * g_old_z.x);
29 g_temp_sqr_y += g_temp_sqr_y + g_old_z.y;
30
31 g_new_z.x = g_float_param->x * g_temp_sqr_x - g_float_param->y * g_temp_sqr_y;
32 g_new_z.y = g_float_param->x * g_temp_sqr_y + g_float_param->y * g_temp_sqr_x;
33 return g_bailout_float();
34}

36int lambda_trig_orbit()
37{
38 if (std::abs(g_old_z.x) >= g_magnitude_limit2 || std::abs(g_old_z.y) >= g_magnitude_limit2)
39 {
40 return 1;
41 }
42 cmplx_trig0(g_old_z, g_tmp_z); // tmp = trig(old)
43 cmplx_mult(*g_float_param, g_tmp_z, g_new_z); // new = longparm*trig(old)
44 g_old_z = g_new_z;
45 return 0;
46}

74static int lambda_exponent_fractal()
75{
76 // found this in "Science of Fractal Images"
77 if (std::abs(g_old_z.y) >= 1.0e3)
78 {
79 return 1;
80 }
81 if (std::abs(g_old_z.x) >= 8)
82 {
83 return 1;
84 }
85 double sin_y;
86 double cos_y;
87 sin_cos(g_old_z.y, &sin_y, &cos_y);
88
89 if (g_old_z.x >= g_magnitude_limit && cos_y >= 0.0)
90 {
91 return 1;
92 }
93 const double tmp_exp = std::exp(g_old_z.x);
94 g_tmp_z.x = tmp_exp*cos_y;
95 g_tmp_z.y = tmp_exp*sin_y;
96
97 //multiply by lamda
98 g_new_z.x = g_float_param->x*g_tmp_z.x - g_float_param->y*g_tmp_z.y;
99 g_new_z.y = g_float_param->y*g_tmp_z.x + g_float_param->x*g_tmp_z.y;
100 g_old_z = g_new_z;
101 return 0;
102}

104bool lambda_trig_per_image()
105{
106 g_cur_fractal_specific->orbit_calc = lambda_trig_orbit;
107 switch (g_trig_index[0])
108 {
109 case TrigFn::SIN:
110 case TrigFn::COSXX:
111 case TrigFn::COS:
112 g_symmetry = SymmetryType::PI_SYM;
113 g_cur_fractal_specific->orbit_calc = lambda_trig_fractal1;
114 break;
115 case TrigFn::SINH:
116 case TrigFn::COSH:
117 g_symmetry = SymmetryType::ORIGIN;
118 g_cur_fractal_specific->orbit_calc = lambda_trig_fractal2;
119 break;
120 case TrigFn::SQR:
121 g_symmetry = SymmetryType::ORIGIN;
122 break;
123 case TrigFn::EXP:
124 g_cur_fractal_specific->orbit_calc = lambda_exponent_fractal;
125 g_symmetry = SymmetryType::NONE;
126 break;
127 case TrigFn::LOG:
128 g_symmetry = SymmetryType::NONE;
129 break;
130 default: // default for additional functions
131 g_symmetry = SymmetryType::ORIGIN;
132 break;
133 }
134 get_julia_attractor(0.0, 0.0); // an attractor?
135 return julia_per_image();
136}
