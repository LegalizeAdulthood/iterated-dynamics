40/* int stopmsg(flags,message) displays message and waits for a key:
41 message should be a max of 9 lines with \n's separating them;
42 no leading or trailing \n's in message;
43 no line longer than 76 chars for best appearance;
44 flag options:
45 &1 if already in text display mode, stackscreen is not called
46 and message is displayed at (12,0) instead of (4,0)
47 &2 if continue/cancel indication is to be returned;
48 when not set, "Any key to continue..." is displayed
49 when set, "Escape to cancel, any other key to continue..."
50 true is returned for cancel, false for continue
51 &4 set to suppress buzzer
52 &8 for parser - use a fixed pitch font
53 &16 for info only message (green box instead of red in DOS vsn)
54 */
55bool stop_msg(const StopMsgFlags flags, const std::string &msg)
56{
57 int top_row;
58 static bool batch_mode = false;
59 if (g_debug_flag != DebugFlags::NONE || g_init_batch >= BatchMode::NORMAL)
60 {
61 if (std::FILE *fp = dir_fopen(g_working_dir, "stopmsg.txt", g_init_batch == BatchMode::NONE ? "w" : "a"))
62 {
63 std::fprintf(fp, "%s\n", msg.c_str());
64 std::fclose(fp);
65 }
66 }
67 if (g_first_init)
68 {
69 // & cmdfiles hasn't finished 1st try
70 driver_set_for_text();
71 driver_buzzer(Buzzer::PROBLEM);
72 driver_put_string(0, 0, 15, "*** Error during startup:");
73 driver_put_string(2, 0, 15, msg);
74 driver_move_cursor(8, 0);
75 sleep_ms(10000);
76 close_drivers();
77 std::exit(1);
78 }
79 if (g_init_batch >= BatchMode::NORMAL || batch_mode)
80 {
81 // in batch mode
82 g_init_batch = BatchMode::BAILOUT_INTERRUPTED_TRY_SAVE; // used to set errorlevel
83 batch_mode = true; // fixes *second* stopmsg in batch mode bug
84 return true;
85 }
86 ValueSaver saved_look_at_mouse{g_look_at_mouse, mouse_look_key(ID_KEY_ENTER)};
87 if (bit_set(flags, StopMsgFlags::NO_STACK))
88 {
89 blank_rows(top_row = 12, 10, 7);
90 }
91 else
92 {
93 driver_stack_screen();
94 top_row = 4;
95 driver_move_cursor(4, 0);
96 }
97 g_text_col_base = 2; // left margin is 2
98 driver_put_string(top_row, 0, 7, msg);
99 if (bit_set(flags, StopMsgFlags::CANCEL))
100 {
101 driver_put_string(g_text_row+2, 0, 7, "Escape to cancel, any other key to continue...");
102 }
103 else
104 {
105 driver_put_string(g_text_row+2, 0, 7, "Any key to continue...");
106 }
107 g_text_col_base = 0; // back to full line
108 const int color = bit_set(flags, StopMsgFlags::INFO_ONLY) ? C_STOP_INFO : C_STOP_ERR;
109 driver_set_attr(top_row, 0, color, (g_text_row+1-top_row)*80);
110 driver_hide_text_cursor(); // cursor off
111 if (!bit_set(flags, StopMsgFlags::NO_BUZZER))
112 {
113 driver_buzzer(bit_set(flags, StopMsgFlags::INFO_ONLY) ? Buzzer::COMPLETE
114 : Buzzer::PROBLEM);
115 }
116 while (driver_key_pressed()) // flush any keyahead
117 {
118 driver_get_key();
119 }
120 bool ret = false;
121 if (g_debug_flag != DebugFlags::SHOW_FORMULA_INFO_AFTER_COMPILE)
122 {
123 if (get_a_key_no_help() == ID_KEY_ESC)
124 {
125 ret = true;
126 }
127 }
128 if (bit_set(flags, StopMsgFlags::NO_STACK))
129 {
130 blank_rows(top_row, 10, 7);
131 }
132 else
133 {
134 driver_unstack_screen();
135 }
136 return ret;
137}
