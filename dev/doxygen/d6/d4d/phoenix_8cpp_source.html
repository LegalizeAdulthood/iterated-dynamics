23int phoenix_orbit()
24{
25 // z(n+1) = z(n)^2 + p + qy(n), y(n+1) = z(n)
26 g_tmp_z.x = g_old_z.x * g_old_z.y;
27 g_new_z.x = g_temp_sqr_x - g_temp_sqr_y + g_float_param->x + g_float_param->y * s_tmp2.x;
28 g_new_z.y = g_tmp_z.x + g_tmp_z.x + g_float_param->y * s_tmp2.y;
29 s_tmp2 = g_old_z; // set tmp2 to Y value
30 return g_bailout_float();
31}

38 g_new_z.y = g_tmp_z.x + g_tmp_z.x + g_float_param->y + g_param_z2.x * s_tmp2.y + g_param_z2.y * s_tmp2.x;
39 s_tmp2 = g_old_z; // set tmp2 to Y value

43int phoenix_plus_fractal()
44{
45 // z(n+1) = z(n)^(degree-1) * (z(n) + p) + qy(n), y(n+1) = z(n)
46 DComplex old_plus;
47 DComplex new_minus;
48 old_plus = g_old_z;
49 g_tmp_z = g_old_z;
50 for (int i = 1; i < g_degree; i++)
51 {
52 // degree >= 2, degree=degree-1 in setup
53 fpu_cmplx_mul(&g_old_z, &g_tmp_z, &g_tmp_z); // = old^(degree-1)
54 }
55 old_plus.x += g_float_param->x;
56 fpu_cmplx_mul(&g_tmp_z, &old_plus, &new_minus);
57 g_new_z.x = new_minus.x + g_float_param->y * s_tmp2.x;
58 g_new_z.y = new_minus.y + g_float_param->y * s_tmp2.y;
59 s_tmp2 = g_old_z; // set tmp2 to Y value
60 return g_bailout_float();
61}

126int phoenix_per_pixel()
127{
128 if (g_inversion.invert != 0)
129 {
130 invertz2(&g_old_z);
131 }
132 else
133 {
134 g_old_z.x = dx_pixel();
135 g_old_z.y = dy_pixel();
136 }
137 g_temp_sqr_x = sqr(g_old_z.x); // precalculated value
138 g_temp_sqr_y = sqr(g_old_z.y);
139 s_tmp2.x = 0; // use tmp2 as the complex Y value
140 s_tmp2.y = 0;
141 return 0;
142}

174bool phoenix_per_image()
175{
176 g_float_param = &g_param_z1;
177 g_degree = static_cast<int>(g_param_z2.x);
178 if (g_degree < 2 && g_degree > -3)
179 {
180 g_degree = 0;
181 }
182 g_params[2] = static_cast<double>(g_degree);
183 if (g_degree == 0)
184 {
185 g_cur_fractal_specific->orbit_calc = phoenix_orbit;
186 }
187 if (g_degree >= 2)
188 {
189 g_degree = g_degree - 1;
190 g_cur_fractal_specific->orbit_calc = phoenix_plus_fractal;
191 }
192 if (g_degree <= -3)
193 {
194 g_degree = std::abs(g_degree) - 2;
195 g_cur_fractal_specific->orbit_calc = phoenix_minus_fractal;
196 }
197
198 return true;
