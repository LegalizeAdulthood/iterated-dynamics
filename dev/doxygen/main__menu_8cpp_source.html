53static int menu_check_key(const int key, int /*choice*/)
54{
55 int test_key = key >= 'A' && key <= 'Z' ? key+('a'-'A') : key;
56 if (test_key == '2')
57 {
58 test_key = '@';
59 }
60 if (std::strchr("#@2txyzgvir3dj", test_key)
61 || test_key == ID_KEY_INSERT || test_key == ID_KEY_CTL_B
62 || test_key == ID_KEY_ESC || test_key == ID_KEY_DELETE
63 || test_key == ID_KEY_CTL_F)
64 {
65 return -test_key;
66 }
67 if (s_full_menu)
68 {
69 if (std::strchr("\\sobkrh", test_key)
70 || test_key == ID_KEY_TAB || test_key == ID_KEY_CTL_A
71 || test_key == ID_KEY_CTL_E || test_key == ID_KEY_BACKSPACE
72 || test_key == ID_KEY_CTL_S || test_key == ID_KEY_CTL_U) // Ctrl+A, E, H, S, U
73 {
74 return -test_key;
75 }
76 if (test_key == ' ')
77 {
78 if (has_julia_toggle())
79 {
80 return -test_key;
81 }
82 }
83 if (g_got_real_dac && g_colors >= 16)
84 {
85 if (std::strchr("c+-", test_key))
86 {
87 return -test_key;
88 }
89 if (g_colors > 16 && (test_key == 'a' || test_key == 'e'))
90 {
91 return -test_key;
92 }
93 }
94 // Alt+A and Alt+S
95 if (test_key == ID_KEY_ALT_A || test_key == ID_KEY_ALT_S)
96 {
97 return -test_key;
98 }
99 }
100 if (check_vid_mode_key(test_key) >= 0)
101 {
102 return -test_key;
103 }
104 return 0;
105}

243int MainMenu::prompt()
244{
245 int i = driver_key_pressed() ? driver_get_key() : 0;
246 if (menu_check_key(i, 0) == 0)
247 {
248 g_help_mode = HelpLabels::HELP_MAIN; // switch help modes
249 m_next_left += 2;
250 if (m_next_left < m_next_right)
251 {
252 m_next_left = m_next_right + 1;
253 }
254 i = full_screen_choice(ChoiceFlags::MENU | ChoiceFlags::CRUNCH, "MAIN MENU", nullptr, nullptr, m_next_left,
255 m_choices, m_attributes, 2, m_next_left / 2, 29, 0, nullptr, nullptr, nullptr, menu_check_key);
256 if (i == -1) // escape
257 {
258 i = ID_KEY_ESC;
259 }
260 else if (i < 0)
261 {
262 i = 0 - i;
263 }
264 else // user selected a choice
265 {
266 i = m_choice_key[i];
267 if (i == -ID_KEY_CTL_ENTER)
268 {
269 g_help_mode = HelpLabels::HELP_ZOOM;
270 help::help();
271 i = 0;
272 }
273 }
274 }
275 return i;
276}

280static bool exit_prompt()
281{
282 help_title();
283 driver_set_attr(1, 0, C_GENERAL_MED, 24*80);
284 for (int j = 9; j <= 11; ++j)
285 {
286 driver_set_attr(j, 18, C_GENERAL_INPUT, 40);
287 }
288 put_string_center(10, 18, 40, C_GENERAL_INPUT, "Exit from " ID_PROGRAM_NAME " (y/n)? y");
289 driver_hide_text_cursor();
290 int i;
291 while ((i = driver_get_key()) != 'y' && i != 'Y' && i != ID_KEY_ENTER)
292 {
293 if (i == 'n' || i == 'N')
294 {
295 return false;
296 }
297 }
298 return true;
299}

301int main_menu(const bool full_menu)
302{
303 ValueSaver saved_tab_mode{g_tab_enabled};
304
305 bool prompt{true};
306 int key{};
307 while (prompt)
308 {
309 s_full_menu = full_menu;
310 g_tab_enabled = false;
311 MainMenu menu(full_menu);
312
313 key = menu.prompt();
314 if (key == ID_KEY_ESC) // escape from menu exits
315 {
316 if (exit_prompt())
317 {
318 goodbye();
319 }
320 }
321 else
322 {
323 prompt = false;
324 }
325 }
326 if (key == ID_KEY_TAB)
327 {
328 tab_display();
329 key = 0;
330 }
331 if (key == ID_KEY_ENTER || key == ID_KEY_ENTER_2)
332 {
333 key = 0; // don't trigger new calc
334 }
335 return key;
336}
