31/*
32convert corners to center/mag
33Rotation angles indicate how much the IMAGE has been rotated, not the
34zoom box. Same goes for the skew angles
35*/
36void cvt_center_mag(double &ctr_x, double &ctr_y, LDouble &mag, double &x_mag_factor, double &rot, double &skew)
37{
38 // simple normal case first
39 if (g_image_region.m_3rd.x == g_image_region.m_min.x && g_image_region.m_3rd.y == g_image_region.m_min.y)
40 {
41 // no rotation or skewing, but stretching is allowed
42 const DComplex size{g_image_region.size()};
43 ctr_x = g_image_region.center_x();
44 ctr_y = g_image_region.center_y();
45 mag = 2.0/size.y;
46 x_mag_factor = size.y / (DEFAULT_ASPECT * size.x);
47 rot = 0.0;
48 skew = 0.0;
49 }
50 else
51 {
52 const auto mag_squared = [](double x, double y) { return sqr(x) + sqr(y); };
53 // set up triangle ABC, having sides abc
54 // side a = bottom, b = left, c = diagonal not containing (x3rd,y3rd)
55 const double c2 = mag_squared(g_image_region.width(), g_image_region.height());
56
57 const double x2 = g_image_region.m_max.x - g_image_region.m_3rd.x;
58 const double y2 = g_image_region.m_min.y - g_image_region.m_3rd.y;
59 const double a2 = mag_squared(x2, y2);
60 const double a = std::sqrt(a2);
61 rot = -rad_to_deg(std::atan2(y2, x2)); // negative for image rotation
62
63 const double x3 = g_image_region.m_min.x - g_image_region.m_3rd.x;
64 const double y3 = g_image_region.m_max.y - g_image_region.m_3rd.y;
65 const double b2 = mag_squared(x3, y3);
66 const double b = std::sqrt(b2);
67
68 const double tmp_a = std::acos((a2+b2-c2)/(2*a*b)); // save tmpa for later use
69 skew = 90.0 - rad_to_deg(tmp_a);
70
71 ctr_x = g_image_region.center_x();
72 ctr_y = g_image_region.center_y();
73
74 const double height = b * std::sin(tmp_a);
75
76 mag = 2.0/height; // 1/(h/2)
77 x_mag_factor = height / (DEFAULT_ASPECT * a);
78
79 // if vector_a cross vector_b is negative
80 // then adjust for left-hand coordinate system
81 if (x2*y3 - x3*y2 < 0 //
82 && g_debug_flag != DebugFlags::ALLOW_NEGATIVE_CROSS_PRODUCT)
83 {
84 skew = -skew;
85 x_mag_factor = -x_mag_factor;
86 mag = -mag;
87 }
88 }
89 // just to make par file look nicer
90 if (mag < 0)
91 {
92 mag = -mag;
93 rot += 180;
94 }
95#ifndef NDEBUG
96 {
97 const double t_x_min = g_image_region.m_min.x;
98 const double t_x_max = g_image_region.m_max.x;
99 const double t_x_3rd = g_image_region.m_3rd.x;
100 const double t_y_min = g_image_region.m_min.y;
101 const double t_y_max = g_image_region.m_max.y;
102 const double t_y_3rd = g_image_region.m_3rd.y;
103 cvt_corners(ctr_x, ctr_y, mag, x_mag_factor, rot, skew);
104 const double error = sqr(t_x_min - g_image_region.m_min.x) //
105 + sqr(t_x_max - g_image_region.m_max.x) //
106 + sqr(t_x_3rd - g_image_region.m_3rd.x) //
107 + sqr(t_y_min - g_image_region.m_min.y) //
108 + sqr(t_y_max - g_image_region.m_max.y) //
109 + sqr(t_y_3rd - g_image_region.m_3rd.y); //
110 if (error > .001)
111 {
112 show_corners_dbl("cvtcentermag problem");
113 }
114 g_image_region.m_min.x = t_x_min;
115 g_image_region.m_max.x = t_x_max;
116 g_image_region.m_3rd.x = t_x_3rd;
117 g_image_region.m_min.y = t_y_min;
118 g_image_region.m_max.y = t_y_max;
119 g_image_region.m_3rd.y = t_y_3rd;
120 }
121#endif
122}

124// convert corners to center/mag using bf
125void cvt_center_mag_bf(BigFloat ctr_x, BigFloat ctr_y, LDouble &mag, double &x_mag_factor, double &rot, double &skew)
126{
127 // needs to be LDouble or won't work past 307 (-DBL_MIN_10_EXP) or so digits
128 BigStackSaver saved;
129
130 // simple normal case first
131 // if (g_image_region.m_3rd.x == g_image_region.m_min.x && g_image_region.m_3rd.y == g_image_region.m_min.y)
132 if (!cmp_bf(g_bf_x_3rd, g_bf_x_min) && !cmp_bf(g_bf_y_3rd, g_bf_y_min))
133 {
134 // no rotation or skewing, but stretching is allowed
135 const BigFloat width_bf = alloc_stack(g_bf_length + 2);
136 const BigFloat height_bf = alloc_stack(g_bf_length+2);
137 // width = x_max - x_min;
138 sub_bf(width_bf, g_bf_x_max, g_bf_x_min);
139 const LDouble width = bf_to_float(width_bf);
140 // height = y_max - y_min;
141 sub_bf(height_bf, g_bf_y_max, g_bf_y_min);
142 const LDouble height = bf_to_float(height_bf);
143 // *ctr_x = (x_min + x_max)/2;
144 add_bf(ctr_x, g_bf_x_min, g_bf_x_max);
145 half_a_bf(ctr_x);
146 // *ctr_y = (y_min + y_max)/2;
147 add_bf(ctr_y, g_bf_y_min, g_bf_y_max);
148 half_a_bf(ctr_y);
149 mag = 2/height;
150 x_mag_factor = static_cast<double>(height / (DEFAULT_ASPECT * width));
151 rot = 0.0;
152 skew = 0.0;
153 }
154 else
155 {
156 const BigFloat tmp_x_bf = alloc_stack(g_bf_length + 2);
157 const BigFloat tmp_y_bf = alloc_stack(g_bf_length+2);
158
159 // set up triangle ABC, having sides abc
160 // side a = bottom, b = left, c = diagonal not containing (x3rd,y3rd)
161 // IMPORTANT: convert from bf AFTER subtracting
162
163 // tmpx = x_max - x_min;
164 sub_bf(tmp_x_bf, g_bf_x_max, g_bf_x_min);
165 LDouble tmp_x1 = bf_to_float(tmp_x_bf);
166 // tmpy = y_max - y_min;
167 sub_bf(tmp_y_bf, g_bf_y_max, g_bf_y_min);
168 LDouble tmp_y1 = bf_to_float(tmp_y_bf);
169 const LDouble c2 = tmp_x1*tmp_x1 + tmp_y1*tmp_y1;
170
171 // tmpx = x_max - x_3rd;
172 sub_bf(tmp_x_bf, g_bf_x_max, g_bf_x_3rd);
173 tmp_x1 = bf_to_float(tmp_x_bf);
174
175 // tmpy = y_min - y_3rd;
176 sub_bf(tmp_y_bf, g_bf_y_min, g_bf_y_3rd);
177 tmp_y1 = bf_to_float(tmp_y_bf);
178 const LDouble a2 = tmp_x1 * tmp_x1 + tmp_y1 * tmp_y1;
179 const LDouble a = std::sqrt(a2);
180
181 // divide tmpx and tmpy by |tmpx| so that double version of atan2() can be used
182 // atan2() only depends on the ratio, this puts it in double's range
183 const int sign_x = sign(tmp_x1);
184 LDouble tmp_y = 0.0;
185 if (sign_x)
186 {
187 tmp_y = tmp_y1/tmp_x1 * sign_x; // tmpy = tmpy / |tmpx|
188 }
189 rot = -rad_to_deg(std::atan2(static_cast<double>(tmp_y), sign_x)); // negative for image rotation
190
191 // tmpx = x_min - x_3rd;
192 sub_bf(tmp_x_bf, g_bf_x_min, g_bf_x_3rd);
193 const LDouble tmp_x2 = bf_to_float(tmp_x_bf);
194 // tmpy = y_max - y_3rd;
195 sub_bf(tmp_y_bf, g_bf_y_max, g_bf_y_3rd);
196 const LDouble tmp_y2 = bf_to_float(tmp_y_bf);
197 const LDouble b2 = tmp_x2 * tmp_x2 + tmp_y2 * tmp_y2;
198 const LDouble b = std::sqrt(b2);
199
200 const double tmp_a = std::acos(static_cast<double>((a2 + b2 - c2) / (2 * a * b))); // save tmpa for later use
201 skew = 90 - rad_to_deg(tmp_a);
202
203 // these are the only two variables that must use big precision
204 // *ctr_x = (x_min + x_max)/2;
205 add_bf(ctr_x, g_bf_x_min, g_bf_x_max);
206 half_a_bf(ctr_x);
207 // *ctr_y = (y_min + y_max)/2;
208 add_bf(ctr_y, g_bf_y_min, g_bf_y_max);
209 half_a_bf(ctr_y);
210
211 const LDouble height = b * std::sin(tmp_a);
212 mag = 2/height; // 1/(h/2)
213 x_mag_factor = static_cast<double>(height / (DEFAULT_ASPECT * a));
214
215 // if vector_a cross vector_b is negative
216 // then adjust for left-hand coordinate system
217 if (tmp_x1*tmp_y2 - tmp_x2*tmp_y1 < 0
218 && g_debug_flag != DebugFlags::ALLOW_NEGATIVE_CROSS_PRODUCT)
219 {
220 skew = -skew;
221 x_mag_factor = -x_mag_factor;
222 mag = -mag;
223 }
224 }
225 if (mag < 0)
226 {
227 mag = -mag;
228 rot += 180;
229 }
230}
