195// rotate-the-palette routine
196void rotate(SpinDirection direction)
197{
198 int from_red = 0;
199 int from_blue = 0;
200 int from_green = 0;
201 int to_red = 0;
202 int to_blue = 0;
203 int to_green = 0;
204
205 static int f_steps[] = {2, 4, 8, 12, 16, 24, 32, 40, 54, 100}; // (for Fkeys)
206
207 if (!g_got_real_dac // ??? no DAC to rotate!
208 || g_colors < 16)
209 {
210 // strange things happen in 2x modes
211 driver_buzzer(Buzzer::PROBLEM);
212 return;
213 }
214
215 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_CYCLING};
216
217 s_paused = false; // not paused
218 int f_key = 0; // no random coloring
219 int step = 1;
220 int old_step = 1; // single-step
221 int f_step = 1;
222 int incr = 999; // ready to randomize
223 std::srand(static_cast<unsigned>(std::time(nullptr))); // randomize things
224
225 if (direction == SpinDirection::NONE)
226 {
227 // firing up in paused mode?
228 pause_rotate(); // then force a pause
229 direction = SpinDirection::FORWARD; // and set a rotate direction
230 }
231
232 const int rotate_max = g_color_cycle_range_hi < g_colors ? g_color_cycle_range_hi : g_colors - 1;
233 const int rotate_size = rotate_max - g_color_cycle_range_lo + 1;
234 int last = rotate_max; // last box that was filled
235 int next = g_color_cycle_range_lo; // next box to be filled
236 if (direction < SpinDirection::NONE)
237 {
238 last = g_color_cycle_range_lo;
239 next = rotate_max;
240 }
241
242 bool more = true;
243 while (more)
244 {
245 if (driver_is_disk())
246 {
247 if (!s_paused)
248 {
249 pause_rotate();
250 }
251 }
252 else
253 {
254 while (!driver_key_pressed())
255 {
256 // rotate until key hit, at least once so step=oldstep ok
257 if (f_key > 0)
258 {
259 // randomizing is on
260 for (int i_step = 0; i_step < step; i_step++)
261 {
262 int j_step = next + i_step * +direction;
263 while (j_step < g_color_cycle_range_lo)
264 {
265 j_step += rotate_size;
266 }
267 while (j_step > rotate_max)
268 {
269 j_step -= rotate_size;
270 }
271 if (++incr > f_step)
272 {
273 // time to randomize
274 incr = 1;
275 f_step = ((f_steps[f_key-1]* (RAND15() >> 8)) >> 6) + 1;
276 from_red = g_dac_box[last][0];
277 from_green = g_dac_box[last][1];
278 from_blue = g_dac_box[last][2];
279 to_red = RAND15() >> 9;
280 to_green = RAND15() >> 9;
281 to_blue = RAND15() >> 9;
282 }
283 g_dac_box[j_step][0] = static_cast<Byte>(from_red + (to_red - from_red) * incr / f_step);
284 g_dac_box[j_step][1] = static_cast<Byte>(from_green + (to_green - from_green) * incr / f_step);
285 g_dac_box[j_step][2] = static_cast<Byte>(from_blue + (to_blue - from_blue) * incr / f_step);
286 }
287 }
288 if (step >= rotate_size)
289 {
290 step = old_step;
291 }
292 spin_dac(direction, step);
293 }
294 }
295 if (step >= rotate_size)
296 {
297 step = old_step;
298 }
299 const int key = driver_get_key();
300 if (s_paused
301 && key != ' ' && key != 'c' && key != ID_KEY_HOME && key != 'C')
302 {
303 s_paused = false; // clear paused condition
304 }
305 switch (key)
306 {
307 case '+': // '+' means rotate forward
308 case ID_KEY_RIGHT_ARROW: // RightArrow = rotate fwd
309 f_key = 0;
310 direction = SpinDirection::FORWARD;
311 last = rotate_max;
312 next = g_color_cycle_range_lo;
313 incr = 999;
314 break;
315 case '-': // '-' means rotate backward
316 case ID_KEY_LEFT_ARROW: // LeftArrow = rotate bkwd
317 f_key = 0;
318 direction = SpinDirection::BACKWARD;
319 last = g_color_cycle_range_lo;
320 next = rotate_max;
321 incr = 999;
322 break;
323 case ID_KEY_UP_ARROW: // UpArrow means speed up
324 if (++g_dac_count >= g_colors)
325 {
326 --g_dac_count;
327 }
328 break;
329 case ID_KEY_DOWN_ARROW: // DownArrow means slow down
330 if (g_dac_count > 1)
331 {
332 g_dac_count--;
333 }
334 break;
335 case '1':
336 case '2':
337 case '3':
338 case '4':
339 case '5':
340 case '6':
341 case '7':
342 case '8':
343 case '9':
344 step = key - '0'; // change step-size
345 step = std::min(step, rotate_size);
346 break;
347 case ID_KEY_F1: // ID_KEY_F1 - ID_KEY_F10:
348 case ID_KEY_F2: // select a shading factor
349 case ID_KEY_F3:
350 case ID_KEY_F4:
351 case ID_KEY_F5:
352 case ID_KEY_F6:
353 case ID_KEY_F7:
354 case ID_KEY_F8:
355 case ID_KEY_F9:
356 case ID_KEY_F10:
357 f_key = key - (ID_KEY_F1 - 1);
358 f_step = 1;
359 incr = 999;
360 break;
361 case ID_KEY_ENTER: // enter key: randomize all colors
362 case ID_KEY_ENTER_2: // also the Numeric-Keypad Enter
363 f_key = RAND15()/3277 + 1;
364 f_step = 1;
365 incr = 999;
366 old_step = step;
367 step = rotate_size;
368 break;
369 case 'r':
370 change_palette_channel(CH_RED, -1);
371 s_paused = false;
372 break;
373 case 'g':
374 change_palette_channel(CH_GREEN, -1);
375 s_paused = false;
376 break;
377 case 'b':
378 change_palette_channel(CH_BLUE, -1);
379 s_paused = false;
380 break;
381 case 'R':
382 change_palette_channel(CH_RED, 1);
383 s_paused = false;
384 break;
385 case 'G':
386 change_palette_channel(CH_GREEN, 1);
387 s_paused = false;
388 break;
389 case 'B':
390 change_palette_channel(CH_BLUE, 1);
391 s_paused = false;
392 break;
393 case ' ': // use the spacebar as a "pause" toggle
394 case 'c': // for completeness' sake, the 'c' too
395 case 'C':
396 pause_rotate();
397 break;
398 case '>': // single-step
399 case '.':
400 case '<':
401 case ',':
402 if (key == '>' || key == '.')
403 {
404 direction = SpinDirection::BACKWARD;
405 last = g_color_cycle_range_lo;
406 next = rotate_max;
407 incr = 999;
408 }
409 else
410 {
411 direction = SpinDirection::FORWARD;
412 last = rotate_max;
413 next = g_color_cycle_range_lo;
414 incr = 999;
415 }
416 f_key = 0;
417 spin_dac(direction, 1);
418 if (! s_paused)
419 {
420 pause_rotate();
421 }
422 break;
423 case 'd': // load colors from "default.map"
424 case 'D':
425 if (validate_luts("default"))
426 {
427 break;
428 }
429 f_key = 0; // disable random generation
430 pause_rotate(); // update palette and pause
431 break;
432 case 'a': // load colors from "altern.map"
433 case 'A':
434 if (validate_luts("altern"))
435 {
436 break;
437 }
438 f_key = 0; // disable random generation
439 pause_rotate(); // update palette and pause
440 break;
441 case 'l': // load colors from a specified map
442 case 'L':
443 load_palette();
444 f_key = 0; // disable random generation
445 pause_rotate(); // update palette and pause
446 break;
447 case 's': // save the palette
448 case 'S':
449 save_palette();
450 f_key = 0; // disable random generation
451 pause_rotate(); // update palette and pause
452 break;
453 case ID_KEY_ESC: // escape
454 more = false; // time to bail out
455 break;
456 case ID_KEY_HOME: // restore palette
457 std::memcpy(g_dac_box, g_old_dac_box, 256 * 3);
458 pause_rotate(); // pause
459 break;
460 default: // maybe a new palette
461 f_key = 0; // disable random generation
462 set_palette_range(key);
463 pause_rotate();
464 break;
465 }
466 }
467}

469static void pause_rotate() // pause-the-rotate routine
470{
471 if (s_paused) // if already paused, just clear
472 {
473 s_paused = false;
474 return;
475 }
476
477 // else set border, wait for a key
478 int old_dac_count = g_dac_count; // saved dac-count value goes here
479 Byte old_dac0 = g_dac_box[0][0];
480 Byte old_dac1 = g_dac_box[0][1];
481 Byte old_dac2 = g_dac_box[0][2];
482 g_dac_count = 256;
483 g_dac_box[0][0] = 48;
484 g_dac_box[0][1] = 48;
485 g_dac_box[0][2] = 48;
486 refresh_dac(); // show white border
487 if (driver_is_disk())
488 {
489 dvid_status(100, " Paused in \"color cycling\" mode ");
490 }
491 driver_wait_key_pressed(false); // wait for any key
492
493 if (driver_is_disk())
494 {
495 dvid_status(0, "");
496 }
497 g_dac_box[0][0] = old_dac0;
498 g_dac_box[0][1] = old_dac1;
499 g_dac_box[0][2] = old_dac2;
500 refresh_dac(); // show black border
501 g_dac_count = old_dac_count;
502 s_paused = true;
503}

505static void set_palette(Byte start[3], Byte finish[3])
506{
507 g_dac_box[0][0] = 0;
508 g_dac_box[0][1] = 0;
509 g_dac_box[0][2] = 0;
510 for (int i = 1; i <= 255; i++) // fill the palette
511 {
512 for (int j = 0; j < 3; j++)
513 {
514 g_dac_box[i][j] = static_cast<Byte>((i * start[j] + (256 - i) * finish[j]) / 255);
515 }
516 }
517}

550void save_palette()
551{
552 driver_stack_screen();
553 char filename[256]{};
554 ValueSaver saved_help_mode{g_help_mode, HelpLabels::HELP_COLORMAP};
555 int i = field_prompt("Name of map file to write", nullptr, filename, 60, nullptr);
556 driver_unstack_screen();
557 if (i != -1 && filename[0])
558 {
559 if (std::strchr(filename, '.') == nullptr)
560 {
561 std::strcat(filename, ".map");
562 }
563 const std::filesystem::path pal_name{get_save_path(WriteFile::MAP, filename)};
564 assert(!pal_name.empty());
565 if (std::FILE *dac_file = std::fopen(pal_name.string().c_str(), "w"); dac_file == nullptr)
566 {
567 driver_buzzer(Buzzer::PROBLEM);
568 }
569 else
570 {
571 for (i = 0; i < g_colors; i++)
572 {
573 std::fprintf(dac_file, "%3d %3d %3d\n", g_dac_box[i][0], g_dac_box[i][1], g_dac_box[i][2]);
574 }
575 std::memcpy(g_old_dac_box, g_dac_box, 256*3);
576 g_color_state = ColorState::MAP_FILE;
577 g_last_map_name = pal_name.filename().string();
578 std::fclose(dac_file);
579 }
580 }
