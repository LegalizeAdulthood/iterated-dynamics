41/* timer function:
42 timer(timer_type::ENGINE,(*fractal)()) fractal engine
43 timer(timer_type::DECODER,nullptr,int width) decoder
44 timer(timer_type::ENCODER) encoder
45 */
46static int timer(const TimerType type, int (*fn)(), ...)
47{
48 std::va_list arg_marker; // variable arg list
49 std::FILE *fp = nullptr;
50 int out = 0;
51 int i;
52
53 va_start(arg_marker, fn);
54
55 bool do_bench = g_timer_flag; // record time?
56 if (type == TimerType::ENCODER) // encoder, record time only if debug flag set
57 {
58 do_bench = g_debug_flag == DebugFlags::BENCHMARK_ENCODER;
59 }
60 if (do_bench)
61 {
62 fp = std::fopen(get_save_path(WriteFile::ROOT, "id-bench.txt").string().c_str(), "a");
63 }
64 g_engine_timer_start = std::clock();
65 switch (type)
66 {
67 case TimerType::ENGINE:
68 out = fn();
69 break;
70 case TimerType::DECODER:
71 i = va_arg(arg_marker, int);
72 out = static_cast<int>(decoder(static_cast<short>(i)));
73 break;
74 case TimerType::ENCODER:
75 out = encoder();
76 break;
77 }
78 // next assumes CLOCKS_PER_SEC is 10^n, n>=2
79 g_timer_interval = (std::clock() - g_engine_timer_start) / (CLOCKS_PER_SEC / 100);
80
81 if (do_bench)
82 {
83 std::time_t now;
84 std::time(&now);
85 char *text = std::ctime(&now);
86 text[24] = 0; // clobber newline in time string
87 switch (type)
88 {
89 case TimerType::DECODER:
90 fmt::print(fp, "decode ");
91 break;
92 case TimerType::ENCODER:
93 fmt::print(fp, "encode ");
94 break;
95 default:
96 break;
97 }
98 fmt::print(fp, "{:s} type={:s} resolution = {:d}x{:d} maxiter={:d}", //
99 text, //
100 g_cur_fractal_specific->name, //
101 g_logical_screen.x_dots, g_logical_screen.y_dots, //
102 g_max_iterations);
103 fmt::print(fp, " time= {:d}.{:02d} secs\n", g_timer_interval / 100, g_timer_interval % 100);
104 std::fclose(fp);
105 }
106 return out;
107}
