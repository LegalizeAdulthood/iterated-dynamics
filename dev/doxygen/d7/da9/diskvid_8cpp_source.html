150int common_start_disk(const long new_row_size, const long new_col_size, const int colors)
151{
152 if (g_disk_flag)
153 {
154 end_disk();
155 }
156 if (driver_is_disk()) // otherwise, real screen also in use, don't hit it
157 {
158 help_title();
159 driver_set_attr(1, 0, C_DVID_BKGRD, 24*80); // init rest to background
160 for (int i = 0; i < BOX_DEPTH; ++i)
161 {
162 driver_set_attr(BOX_ROW+i, BOX_COL, C_DVID_LO, BOX_WIDTH); // init box
163 }
164 driver_put_string(BOX_ROW+2, BOX_COL+4, C_DVID_HI, "'Disk-Video' mode");
165 driver_put_string(BOX_ROW + 4, BOX_COL + 4, C_DVID_LO,
166 fmt::format("Screen resolution: {:d} x {:d}", g_screen_x_dots, g_screen_y_dots));
167 if (g_disk_targa)
168 {
169 driver_put_string(-1, -1, C_DVID_LO, " 24 bit Targa");
170 }
171 else
172 {
173 driver_put_string(-1, -1, C_DVID_LO, " Colors: ");
174 driver_put_string(-1, -1, C_DVID_LO, std::to_string(colors));
175 }
176 driver_put_string(BOX_ROW+8, BOX_COL+4, C_DVID_LO, "Save name: " + g_save_filename.string());
177 driver_put_string(BOX_ROW+10, BOX_COL+4, C_DVID_LO, "Status:");
178 dvid_status(0, "clearing the 'screen'");
179 }
180 s_high_offset = -1;
181 s_seek_offset = s_high_offset;
182 s_cur_offset = s_seek_offset;
183 s_cur_row = -1;
184 if (g_disk_targa)
185 {
186 s_pixel_shift = 0;
187 }
188 else
189 {
190 s_pixel_shift = 3;
191 int i = 2;
192 while (i < colors)
193 {
194 i *= i;
195 --s_pixel_shift;
196 }
197 }
198 s_time_to_display = g_bf_math != BFMathType::NONE ? 10 : 1000; // time-to-g_driver-status counter
199
200 constexpr unsigned int CACHE_SIZE{CACHE_MAX};
201 long long_tmp = static_cast<long>(CACHE_SIZE) << 10;
202 s_cache_start = static_cast<Cache *>(malloc(long_tmp));
203 s_cache_lru = s_cache_start;
204 s_cache_end = s_cache_lru + long_tmp/sizeof(*s_cache_start);
205 s_mem_buf.resize(BLOCK_LEN);
206 if (s_cache_start == nullptr)
207 {
208 stop_msg("*** insufficient free memory for cache buffers ***");
209 return -1;
210 }
211 if (driver_is_disk())
212 {
213 driver_put_string(BOX_ROW + 6, BOX_COL + 4, C_DVID_LO, fmt::format("Cache size: {:d}K", CACHE_SIZE));
214 }
215
216 // preset cache to all invalid entries so we don't need free list logic
217 for (unsigned int &elem : s_hash_ptr)
218 {
219 elem = 0xffff; // 0xffff marks the end of a hash chain
220 }
221 long_tmp = 100000000L;
222 for (Cache *ptr1 = s_cache_start; ptr1 < s_cache_end; ++ptr1)
223 {
224 ptr1->dirty = false;
225 ptr1->lru = false;
226 long_tmp += BLOCK_LEN;
227 unsigned int *fwd_link = &s_hash_ptr[static_cast<unsigned short>(long_tmp) >> BLOCK_SHIFT & HASH_SIZE - 1];
228 ptr1->offset = long_tmp;
229 ptr1->hash_link = *fwd_link;
230 *fwd_link = static_cast<int>(reinterpret_cast<char *>(ptr1) - reinterpret_cast<char *>(s_cache_start));
231 }
232
233 long memory_size = new_col_size *new_row_size + s_header_length;
234 {
235 const int i = static_cast<short>(memory_size) & BLOCK_LEN - 1;
236 if (i != 0)
237 {
238 memory_size += BLOCK_LEN - i;
239 }
240 }
241 memory_size >>= s_pixel_shift;
242 memory_size >>= BLOCK_SHIFT;
243 g_disk_flag = true;
244 s_row_size = static_cast<unsigned int>(new_row_size);
245 s_col_size = static_cast<unsigned int>(new_col_size);
246
247 if (g_disk_targa)
248 {
249 // Retrieve the header information first
250 std::fseek(s_fp, 0L, SEEK_SET);
251 for (int i = 0; i < s_header_length; i++)
252 {
253 s_mem_buf[i] = static_cast<Byte>(std::fgetc(s_fp));
254 }
255 std::fclose(s_fp);
256 s_dv_handle = memory_alloc(BLOCK_LEN, memory_size, MemoryLocation::DISK);
257 }
258 else
259 {
260 s_dv_handle = memory_alloc(BLOCK_LEN, memory_size, MemoryLocation::MEMORY);
261 }
262 if (!s_dv_handle)
263 {
264 stop_msg("*** insufficient free memory/disk space ***");
265 g_good_mode = false;
266 s_row_size = 0;
267 return -1;
268 }
269
270 if (driver_is_disk())
271 {
272 driver_put_string(BOX_ROW+2, BOX_COL+23, C_DVID_LO,
273 memory_type(s_dv_handle) == MemoryLocation::DISK ? "Using your Disk Drive" : "Using your memory");
274 }
275
276 s_mem_buf_ptr = s_mem_buf.data();
277
278 if (g_disk_targa)
279 {
280 // Put header information in the file
281 s_dv_handle.from_memory(s_mem_buf.data(), static_cast<U16>(s_header_length), 1L, 0);
282 }
283 else
284 {
285 for (long offset = 0; offset < memory_size; offset++)
286 {
287 s_dv_handle.set(0, BLOCK_LEN, 1L, offset);
288 if (driver_key_pressed()) // user interrupt
289 {
290 // esc to cancel, else continue
291 if (stop_msg(StopMsgFlags::CANCEL, "Disk Video initialization interrupted:\n"))
292 {
293 end_disk();
294 g_good_mode = false;
295 return -2; // -1 == failed, -2 == cancel
296 }
297 }
298 }
299 }
300
301 if (driver_is_disk())
302 {
303 dvid_status(0, "");
304 }
305 return 0;
306}

308void end_disk()
309{
310 if (s_fp != nullptr)
311 {
312 if (g_disk_targa) // flush the cache
313 {
314 for (s_cache_lru = s_cache_start; s_cache_lru < s_cache_end; ++s_cache_lru)
315 {
316 if (s_cache_lru->dirty)
317 {
318 write_cache_lru();
319 }
320 }
321 }
322 std::fclose(s_fp);
323 s_fp = nullptr;
324 }
325
326 if (s_dv_handle)
327 {
328 memory_release(s_dv_handle);
329 s_dv_handle = MemoryHandle{};
330 }
331 if (s_cache_start != nullptr)
332 {
333 free(s_cache_start);
334 s_cache_start = nullptr;
335 }
336 s_mem_buf.clear();
337 g_disk_flag = false;
338 s_row_size = 0;
339 g_disk_16_bit = false;
340}

342int disk_read_pixel(const int col, int row)
343{
344 if (--s_time_to_display < 0) // time to g_driver status?
345 {
346 if (driver_is_disk())
347 {
348 dvid_status(0,
349 fmt::format(" reading line {:4d}",
350 row >= g_screen_y_dots ? row - g_screen_y_dots : row)); // adjust when potfile
351 }
352 if (g_bf_math != BFMathType::NONE)
353 {
354 s_time_to_display = 10; // time-to-g_driver-status counter
355 }
356 else
357 {
358 s_time_to_display = 1000; // time-to-g_driver-status counter
359 }
360 }
361 if (row != s_cur_row) // try to avoid ghastly code generated for multiply
362 {
363 if (row >= s_col_size) // while we're at it avoid this test if not needed
364 {
365 return 0;
366 }
367 s_cur_row = row;
368 s_cur_row_base = static_cast<long>(s_cur_row) * s_row_size;
369 }
370 if (col >= s_row_size)
371 {
372 return 0;
373 }
374 const long offset = s_cur_row_base + col;
375 const int col_index = static_cast<short>(offset) & BLOCK_LEN - 1; // offset within cache entry
376 if (s_cur_offset != (offset & 0L - BLOCK_LEN)) // same entry as last ref?
377 {
378 find_load_cache(offset & 0L - BLOCK_LEN);
379 }
380 return s_cur_cache->pixel[col_index];
381}

383bool from_mem_disk(const long offset, const int size, void *dest)
384{
385 const int col_index = static_cast<int>(offset & BLOCK_LEN - 1);
386 if (col_index + size > BLOCK_LEN) // access violates a
387 {
388 return false; // cache boundary
389 }
390 if (s_cur_offset != (offset & 0L - BLOCK_LEN)) // same entry as last ref?
391 {
392 find_load_cache(offset & 0L - BLOCK_LEN);
393 }
394 std::memcpy(dest, &s_cur_cache->pixel[col_index], size);
395 s_cur_cache->dirty = false;
396 return true;
397}

407void disk_write_pixel(const int col, int row, const int color)
408{
409 if (--s_time_to_display < 0) // time to display status?
410 {
411 if (driver_is_disk())
412 {
413 dvid_status(0,
414 fmt::format(" writing line {:4d}",
415 row >= g_screen_y_dots ? row - g_screen_y_dots : row)); // adjust when potfile
416 }
417 s_time_to_display = 1000;
418 }
419 if (row != static_cast<unsigned int>(s_cur_row)) // try to avoid ghastly code generated for multiply
420 {
421 if (row >= s_col_size) // while we're at it avoid this test if not needed
422 {
423 return;
424 }
425 s_cur_row = row;
426 s_cur_row_base = static_cast<long>(s_cur_row) *s_row_size;
427 }
428 if (col >= s_row_size)
429 {
430 return;
431 }
432 const long offset = s_cur_row_base + col;
433 const int col_index = static_cast<short>(offset) & BLOCK_LEN - 1;
434 if (s_cur_offset != (offset & 0L - BLOCK_LEN)) // same entry as last ref?
435 {
436 find_load_cache(offset & 0L - BLOCK_LEN);
437 }
438 if (s_cur_cache->pixel[col_index] != (color & 0xff))
439 {
440 s_cur_cache->pixel[col_index] = static_cast<Byte>(color);
441 s_cur_cache->dirty = true;
442 }
443}

446{
447 const int col_index = static_cast<int>(offset & BLOCK_LEN - 1);
448
449 if (col_index + size > BLOCK_LEN) // access violates a

460 s_cur_cache->dirty = true;
461 return true;

471static void find_load_cache(const long offset) // used by read/write
472{
473 s_cur_offset = offset; // note this for next reference
474 // check if required entry is in cache - lookup by hash
475 unsigned int tbl_offset = s_hash_ptr[static_cast<unsigned short>(offset) >> BLOCK_SHIFT & HASH_SIZE - 1];
476 while (tbl_offset != 0xffff) // follow the hash chain
477 {
478 s_cur_cache = reinterpret_cast<Cache *>(reinterpret_cast<char *>(s_cache_start) + tbl_offset);
479 if (s_cur_cache->offset == offset) // great, it is in the cache
480 {
481 s_cur_cache->lru = true;
482 return;
483 }
484 tbl_offset = s_cur_cache->hash_link;
485 }
486 // must load the cache entry from backing store
487 while (true) // look around for something not recently used
488 {
489 if (++s_cache_lru >= s_cache_end)
490 {
491 s_cache_lru = s_cache_start;
492 }
493 if (!s_cache_lru->lru)
494 {
495 break;
496 }
497 s_cache_lru->lru = false;
498 }
499 if (s_cache_lru->dirty) // must write this block before reusing it
500 {
501 write_cache_lru();
502 }
503 // remove block at cache_lru from its hash chain
504 unsigned int *fwd_link =
505 &s_hash_ptr[(static_cast<unsigned short>(s_cache_lru->offset) >> BLOCK_SHIFT & HASH_SIZE - 1)];
506 tbl_offset = static_cast<int>(reinterpret_cast<char *>(s_cache_lru) - reinterpret_cast<char *>(s_cache_start));
507 while (*fwd_link != tbl_offset)
508 {
509 fwd_link = &reinterpret_cast<Cache *>(reinterpret_cast<char *>(s_cache_start) + *fwd_link)->hash_link;
510 }
511 *fwd_link = s_cache_lru->hash_link;
512 // load block
513 s_cache_lru->dirty = false;
514 s_cache_lru->lru = true;
515 s_cache_lru->offset = offset;
516 Byte *pixel_ptr = &s_cache_lru->pixel[0];
517 if (offset > s_high_offset) // never been this high before, just clear it
518 {
519 s_high_offset = offset;
520 std::memset(pixel_ptr, 0, BLOCK_LEN);
521 }
522 else
523 {
524 if (offset != s_seek_offset)
525 {
526 mem_seek(offset >> s_pixel_shift);
527 }
528 s_seek_offset = offset + BLOCK_LEN;
529 switch (s_pixel_shift)
530 {
531 case 0:
532 for (int i = 0; i < BLOCK_LEN; ++i)
533 {
534 *pixel_ptr++ = mem_getc();
535 }
536 break;
537
538 case 1:
539 for (int i = 0; i < BLOCK_LEN/2; ++i)
540 {
541 const Byte tmp_char = mem_getc();
542 *pixel_ptr++ = static_cast<Byte>(tmp_char >> 4);
543 *pixel_ptr++ = static_cast<Byte>(tmp_char & 15);
544 }
545 break;
546 case 2:
547 for (int i = 0; i < BLOCK_LEN/4; ++i)
548 {
549 const Byte tmp_char = mem_getc();
550 for (int j = 6; j >= 0; j -= 2)
551 {
552 *pixel_ptr++ = static_cast<Byte>(tmp_char >> j & 3);
553 }
554 }
555 break;
556 case 3:
557 for (int i = 0; i < BLOCK_LEN/8; ++i)
558 {
559 const Byte tmp_char = mem_getc();
560 for (int j = 7; j >= 0; --j)
561 {
562 *pixel_ptr++ = static_cast<Byte>(tmp_char >> j & 1);
563 }
564 }
565 break;
566 }
567 }
568 // add new block to its hash chain
569 fwd_link = &s_hash_ptr[(static_cast<unsigned short>(offset) >> BLOCK_SHIFT & HASH_SIZE - 1)];
570 s_cache_lru->hash_link = *fwd_link;
571 *fwd_link = static_cast<int>(reinterpret_cast<char *>(s_cache_lru) - reinterpret_cast<char *>(s_cache_start));
572 s_cur_cache = s_cache_lru;
573}

575// lookup for write_cache_lru
576static Cache *find_cache(const long offset)
577{
578 unsigned int tbl_offset = s_hash_ptr[static_cast<unsigned short>(offset) >> BLOCK_SHIFT & HASH_SIZE - 1];
579 while (tbl_offset != 0xffff)
580 {
581 Cache *ptr1 = reinterpret_cast<Cache *>(reinterpret_cast<char *>(s_cache_start) + tbl_offset);
582 if (ptr1->offset == offset)
583 {
584 return ptr1;
585 }
586 tbl_offset = ptr1->hash_link;
587 }
588 return nullptr;
589}

591enum
592{
593 WRITE_GAP = 4 // 1 for no gaps
594};
595
596static void write_cache_lru()
597{
598 Byte tmp_char = 0;
599 // scan back to also write any preceding dirty blocks, skipping small gaps
600 Cache *ptr1 = s_cache_lru;
601 long offset = ptr1->offset;
602 int i = 0;
603 while (++i <= WRITE_GAP)
604 {
605 offset -= BLOCK_LEN;
606 Cache *ptr2 = find_cache(offset);
607 if (ptr2 != nullptr && ptr2->dirty)
608 {
609 ptr1 = ptr2;
610 i = 0;
611 }
612 }
613 // write all consecutive dirty blocks (often whole cache in 1pass modes)
614 // keep going past small gaps
615
616write_seek:
617 mem_seek(ptr1->offset >> s_pixel_shift);
618
619write_stuff:
620 const Byte *pixel_ptr = &ptr1->pixel[0];
621 switch (s_pixel_shift)
622 {
623 case 0:
624 for (int j = 0; j < BLOCK_LEN; ++j)
625 {
626 mem_putc(*pixel_ptr++);
627 }
628 break;
629 case 1:
630 for (int j = 0; j < BLOCK_LEN/2; ++j)
631 {
632 tmp_char = static_cast<Byte>(*pixel_ptr++ << 4);
633 tmp_char = static_cast<Byte>(tmp_char + *pixel_ptr++);
634 mem_putc(tmp_char);
635 }
636 break;
637 case 2:
638 for (int j = 0; j < BLOCK_LEN/4; ++j)
639 {
640 for (int k = 6; k >= 0; k -= 2)
641 {
642 tmp_char = static_cast<Byte>((tmp_char << 2) + *pixel_ptr++);
643 }
644 mem_putc(tmp_char);
645 }
646 break;
647 case 3:
648 for (int j = 0; j < BLOCK_LEN/8; ++j)
649 {
650 // clang-format off
651 mem_putc(static_cast<Byte>(((((((*pixel_ptr << 1
652 | *(pixel_ptr + 1)) << 1
653 | *(pixel_ptr + 2)) << 1
654 | *(pixel_ptr + 3)) << 1
655 | *(pixel_ptr + 4)) << 1
656 | *(pixel_ptr + 5)) << 1
657 | *(pixel_ptr + 6)) << 1
658 | *(pixel_ptr + 7)));
659 // clang-format on
660 pixel_ptr += 8;
661 }
662 break;
663 }
664 ptr1->dirty = false;
665 offset = ptr1->offset + BLOCK_LEN;
666 ptr1 = find_cache(offset);
667 if (ptr1 != nullptr && ptr1->dirty)
668 {
669 goto write_stuff;
670 }
671 i = 1;
672 while (++i <= WRITE_GAP)
673 {
674 offset += BLOCK_LEN;
675 ptr1 = find_cache(offset);
676 if (ptr1 != nullptr && ptr1->dirty)
677 {
678 goto write_seek;
679 }
680 }
681 s_seek_offset = -1; // force a seek before next read
682}

684// Seek, mem_getc, mem_putc routines follow.
685// Note that the calling logic always separates mem_getc and mem_putc
686// sequences with a seek between them. A mem_getc is never followed by
687// a mem_putc nor v.v. without a seek between them.
688//
689static void mem_seek(long offset) // mem seek
690{
691 offset += s_header_length;
692 s_mem_offset = offset >> BLOCK_SHIFT;
693 if (s_mem_offset != s_old_mem_offset)
694 {
695 s_dv_handle.from_memory(s_mem_buf.data(), BLOCK_LEN, 1L, s_old_mem_offset);
696 s_dv_handle.to_memory(s_mem_buf.data(), BLOCK_LEN, 1L, s_mem_offset);
697 }
698 s_old_mem_offset = s_mem_offset;
699 s_mem_buf_ptr = s_mem_buf.data() + (offset & BLOCK_LEN - 1);
700}
701
702static Byte mem_getc() // memory get_char
703{
704 if (s_mem_buf_ptr - s_mem_buf.data() >= BLOCK_LEN)
705 {
