27// merge existing full path with new one
28// attempt to detect if file or directory
29
30// copies the proposed new filename to the fullpath variable
31// does not copy directories for PAR files
32// (modes AT_AFTER_STARTUP and AT_CMD_LINE_SET_NAME)
33// attempts to extract directory and test for existence
34// (modes AT_CMD_LINE and SSTOOLS_INI)
35int merge_path_names(char *old_full_path, const char *new_filename, const CmdFile mode)
36{
37 char buff[ID_FILE_MAX_PATH];
38 std::strcpy(buff, fs::path(new_filename).make_preferred().string().c_str());
39
40 // no dot or slash so assume a file
41 bool is_a_file = std::strchr(buff, '.') == nullptr
42 && std::strchr(buff, SLASH_CH) == nullptr;
43 bool is_a_dir = is_a_directory(buff);
44 if (is_a_dir)
45 {
46 fix_dir_name(buff);
47 }
48
49 // if dot, slash, NUL, it's the current directory, set up full path
50 if (buff[0] == '.' && buff[1] == SLASH_CH && buff[2] == 0)
51 {
52 char temp_drive[ID_FILE_MAX_PATH];
53 expand_dir_name(buff, temp_drive);
54 std::strcat(temp_drive, buff);
55 std::strcpy(buff, temp_drive);
56 is_a_dir = true;
57 }
58
59 // if dot, slash, its relative to the current directory, set up full path
60 if (buff[0] == '.' && buff[1] == SLASH_CH)
61 {
62 bool test_dir = false;
63 char temp_drive[ID_FILE_MAX_PATH];
64 if (std::strrchr(buff, '.') == buff)
65 {
66 test_dir = true; // only one '.' assume it's a directory
67 }
68 expand_dir_name(buff, temp_drive);
69 std::strcat(temp_drive, buff);
70 std::strcpy(buff, temp_drive);
71 if (!test_dir)
72 {
73 buff[std::strlen(buff) - 1] = 0; // get rid of slash added by expand_dirname
74 }
75 }
76
77 // check existence
78 if (!is_a_dir || is_a_file)
79 {
80 if (fr_find_first(buff))
81 {
82 if (g_dta.attribute & SUB_DIR) // exists and is dir
83 {
84 fix_dir_name(buff); // add trailing slash
85 is_a_dir = true;
86 is_a_file = false;
87 }
88 else
89 {
90 is_a_file = true;
91 }
92 }
93 }
94
95 char drive[ID_FILE_MAX_DRIVE];
96 char dir[ID_FILE_MAX_DIR];
97 char fname[ID_FILE_MAX_FNAME];
98 char ext[ID_FILE_MAX_EXT];
99 split_path(buff, drive, dir, fname, ext);
100
101 char drive1[ID_FILE_MAX_DRIVE];
102 char dir1[ID_FILE_MAX_DIR];
103 char fname1[ID_FILE_MAX_FNAME];
104 char ext1[ID_FILE_MAX_EXT];
105 split_path(old_full_path, drive1, dir1, fname1, ext1);
106
107 const bool get_path = mode == CmdFile::AT_CMD_LINE || mode == CmdFile::SSTOOLS_INI;
108 if (get_path)
109 {
110 if (static_cast<int>(std::strlen(drive)) != 0)
111 {
112 std::strcpy(drive1, drive);
113 }
114 if (static_cast<int>(std::strlen(dir)) != 0)
115 {
116 std::strcpy(dir1, dir);
117 }
118 }
119 if (static_cast<int>(std::strlen(fname)) != 0)
120 {
121 std::strcpy(fname1, fname);
122 }
123 if (static_cast<int>(std::strlen(ext)) != 0)
124 {
125 std::strcpy(ext1, ext);
126 }
127 bool is_a_dir_error = false;
128 if (!is_a_dir && !is_a_file && get_path)
129 {
130 make_drive_dir(old_full_path, drive1, dir1);
131 int len = static_cast<int>(std::strlen(old_full_path));
132 if (len > 0)
133 {
134 // strip trailing slash
135 const char save = old_full_path[len - 1];
136 if (save == SLASH_CH)
137 {
138 old_full_path[len-1] = 0;
139 }
140 if (!fs::exists(old_full_path))
141 {
142 is_a_dir_error = true;
143 }
144 old_full_path[len-1] = save;
145 }
146 }
147 make_path(old_full_path, drive1, dir1, fname1, ext1);
148 return is_a_dir_error ? -1 : is_a_dir ? 1 : 0;
149}
