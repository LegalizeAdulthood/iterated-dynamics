40//
41// Generate a gaussian distributed number.
42// The right half of the distribution is folded onto the lower half.
43// That is, the curve slopes up to the peak and then drops to 0.
44// The larger slope is, the smaller the standard deviation.
45// The values vary from 0+offset to range+offset, with the peak
46// at range+offset.
47// To make this more complicated, you only have a
48// 1 in distribution*(1-probability/range*concentration)+1 chance of getting a
49// Gaussian; otherwise you just get offset.
50//
51static int gaussian_number(const int probability, const int range)
52{
53 long p = divide(static_cast<long>(probability) << 16, static_cast<long>(range) << 16, 16);
54 p = multiply(p, s_concentration, 16);
55 p = multiply(static_cast<long>(s_distribution) << 16, p, 16);
56 if (!(RAND15() % (s_distribution - static_cast<int>(p >> 16) + 1)))
57 {
58 long accum = 0;
59 for (int n = 0; n < s_slope; n++)
60 {
61 accum += RAND15();
62 }
63 accum /= s_slope;
64 int r = static_cast<int>(multiply(static_cast<long>(range) << 15, accum, 15) >> 14);
65 r = r - range;
66 if (r < 0)
67 {
68 r = -r;
69 }
70 return range - r + s_offset;
71 }
72 return s_offset;
73}

75int star_field()
76{
77 g_busy = true;
78 s_star_field_values[0] = std::max(s_star_field_values[0], 1.0);
79 s_star_field_values[0] = std::min(s_star_field_values[0], 100.0);
80 s_star_field_values[1] = std::max(s_star_field_values[1], 1.0);
81 s_star_field_values[1] = std::min(s_star_field_values[1], 100.0);
82 s_star_field_values[2] = std::max(s_star_field_values[2], 1.0);
83 s_star_field_values[2] = std::min(s_star_field_values[2], 100.0);
84
85 s_distribution = static_cast<int>(s_star_field_values[0]);
86 s_concentration = static_cast<long>(s_star_field_values[1] / 100.0 * (1L << 16));
87 s_slope = static_cast<int>(s_star_field_values[2]);
88
89 if (validate_luts(ALTERN_MAP_NAME))
90 {
91 stop_msg("Unable to load ALTERN.MAP");
92 g_busy = false;
93 return -1;
94 }
95 refresh_dac();
96 for (g_row = 0; g_row < g_logical_screen.y_dots; g_row++)
97 {
98 for (g_col = 0; g_col < g_logical_screen.x_dots; g_col++)
99 {
100 if (driver_key_pressed())
101 {
102 driver_buzzer(Buzzer::INTERRUPT);
103 g_busy = false;
104 return 1;
105 }
106 int c = get_color(g_col, g_row);
107 if (c == g_inside_color)
108 {
109 c = g_colors-1;
110 }
111 g_put_color(g_col, g_row, gaussian_number(c, g_colors));
112 }
113 }
114 driver_buzzer(Buzzer::COMPLETE);
115 g_busy = false;
116 return 0;
117}
