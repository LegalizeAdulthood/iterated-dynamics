33// bitmask values for FractalSpecific flags
34enum class FractalFlags
35{
36 // clang-format off
37 NONE = 0x00'00'00, // no flags
38 NO_ZOOM = 0x00'00'01, // zoom box not allowed at all
39 NO_GUESS = 0x00'00'02, // solid guessing not allowed
40 NO_TRACE = 0x00'00'04, // boundary tracing not allowed
41 NO_ROTATE = 0x00'00'08, // zoom box rotate/stretch not allowed
42 NO_RESUME = 0x00'00'10, // can't interrupt and resume
43 INF_CALC = 0x00'00'20, // this type calculates forever
44 TRIG1 = 0x00'00'30, // number of trig functions in formula
45 TRIG2 = 0x00'00'40, //
46 TRIG3 = 0x00'00'80, //
47 TRIG4 = 0x00'01'00, //
48 PARAMS_3D = 0x00'04'00, // uses 3d parameters
49 OK_JB = 0x00'08'00, // works with Julibrot
50 MORE = 0x00'10'00, // more than 4 params
51 BAIL_TEST = 0x00'20'00, // can use different bailout tests
52 BF_MATH = 0x00'40'00, // supports arbitrary precision
53 LD_MATH = 0x00'80'00, // supports long double
54 PERTURB = 0x01'00'00, // supports perturbation

56};
