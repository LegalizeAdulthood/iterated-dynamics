39// bitmask values for FractalSpecific flags
40enum class FractalFlags
41{
42 // clang-format off
43 NONE = 0x00'00'00, // no flags
44 NO_ZOOM = 0x00'00'01, // zoom box not allowed at all
45 NO_GUESS = 0x00'00'02, // solid guessing not allowed
46 NO_TRACE = 0x00'00'04, // boundary tracing not allowed
47 NO_ROTATE = 0x00'00'08, // zoom box rotate/stretch not allowed
48 NO_RESUME = 0x00'00'10, // can't interrupt and resume
49 INF_CALC = 0x00'00'20, // this type calculates forever
50 TRIG1 = 0x00'00'30, // number of trig functions in formula
51 TRIG2 = 0x00'00'40, //
52 TRIG3 = 0x00'00'80, //
53 TRIG4 = 0x00'01'00, //
54 PARAMS_3D = 0x00'04'00, // uses 3d parameters
55 OK_JB = 0x00'08'00, // works with Julibrot
56 MORE = 0x00'10'00, // more than 4 params
57 BAIL_TEST = 0x00'20'00, // can use different bailout tests
58 BF_MATH = 0x00'40'00, // supports arbitrary precision
59 LD_MATH = 0x00'80'00, // supports long double
60 PERTURB = 0x01'00'00, // supports perturbation
61 // clang-format on
62};
