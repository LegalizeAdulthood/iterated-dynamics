43int formula_per_pixel()
44{

47 return 1;

51
52 g_overflow = false;
53 g_runtime.jump_index = 0;
54 g_runtime.op_index = 0;
55 g_runtime.set_random = false;
56 if (g_formula.uses_rand && !g_runtime.randomized)
57 {
58 random_seed();
59 }
60 g_runtime.store_index = 0;
61 g_runtime.load_index = 0;

63 g_arg2 = g_runtime.stack.data();
64 g_arg2--;
65
66 g_formula.vars[10].a.d.x = static_cast<double>(g_col);
67 g_formula.vars[10].a.d.y = static_cast<double>(g_row);
68
69 if (g_row + g_col & 1)
70 {
71 g_formula.vars[9].a.d.x = 1.0;
72 }
73 else

84 }
85 else
86 {

88 g_formula.vars[0].a.d.y = dy_pixel();

93 g_formula.last_init_op = g_formula.op_count;

103 g_runtime.init_op_index = g_runtime.op_index;

110int formula_orbit()
111{
112 if (g_formula_name.empty() || g_overflow)
113 {
114 return 1;
115 }
116
117 g_runtime.orbit_begin();
118
119 g_runtime.load_index = g_runtime.init_load_index;
120 g_runtime.store_index = g_runtime.init_store_index;
121 g_runtime.op_index = g_runtime.init_op_index;
122 g_runtime.jump_index = g_runtime.init_jump_index;
123 // Set the random number
124 if (g_runtime.set_random || g_runtime.randomized)
125 {
126 d_random();
127 }
128
129 g_arg1 = g_runtime.stack.data();
130 g_arg2 = g_runtime.stack.data();
131 --g_arg2;
132 while (g_runtime.op_index < static_cast<int>(g_formula.op_count))
133 {
134 g_formula.fns[g_runtime.op_index]();
135 g_runtime.op_index++;
136 }
137
138 g_new_z = g_formula.vars[3].a.d;
139 g_old_z = g_new_z;
140
141 g_runtime.orbit_end();
142
143 return g_arg1->d.x == 0.0;
144}
