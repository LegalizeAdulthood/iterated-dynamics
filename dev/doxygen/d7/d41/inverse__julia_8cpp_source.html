262bool process_inverse_julia_keyboard(
263 InverseJuliaKeyboardContext &context, CrossHairCursor &cursor, const bool wait_for_key)
264{
265 if (wait_for_key)
266 {
267 cursor.wait_key();
268 }
269
270 context.begin_key_batch();
271 bool key_pressed = false;
272 while (driver_key_pressed())
273 {
274 cursor.wait_key();
275 key_pressed = true;
276 dispatch_keyboard_key(driver_get_key());
277 if (context.leaving_now())
278 {
279 break;
280 }
281 }
282 return key_pressed;
283}

290int inverse_julia_fractal_type()
291{
292 int color = 0;
293
294 if (g_resuming) // can't resume
295 {
296 return -1;
297 }
298
299 while (color >= 0) // generate points
300 {
301 if (check_key())
302 {
303 free_queue();
304 return -1;
305 }
306 color = orbit_calc();
307 g_old_z = g_new_z;
308 }
309 free_queue();
310 return 0;
311}
