42static void add_gif_extension(GifFileType *gif, const char *name, unsigned char *bytes, const int length)
43{
44 GifAddExtensionBlock(&gif->ExtensionBlockCount, &gif->ExtensionBlocks, APPLICATION_EXT_FUNC_CODE, //
45 11, reinterpret_cast<unsigned char *>(const_cast<char *>(name)));
46 unsigned int count{static_cast<unsigned int>(length)};
47 unsigned int offset{};
48 while (count > 0)
49 {
50 const unsigned int chunk{count > 255 ? 255 : count};
51 GifAddExtensionBlock(&gif->ExtensionBlockCount, &gif->ExtensionBlocks, CONTINUE_EXT_FUNC_CODE, //
52 chunk, bytes + offset);
53 count -= chunk;
54 offset += chunk;
55 }
56}

189template <int Size>
190ExtensionDeserializer<Size>::ExtensionDeserializer(const GifFileType *gif, const char *name)
191{
192 int count{Size};
193 for (int i = 0; i < gif->ExtensionBlockCount; ++i)
194 {
195 const ExtensionBlock &block{gif->ExtensionBlocks[i]};
196 if (is_fractint_extension(block, name))
197 {
198 int ext{i + 1};
199 int offset{};
200 while (count > 0 //
201 && ext < gif->ExtensionBlockCount //
202 && gif->ExtensionBlocks[ext].Function == CONTINUE_EXT_FUNC_CODE) //
203 {
204 const ExtensionBlock &src = gif->ExtensionBlocks[ext];
205 std::copy_n(src.Bytes, src.ByteCount, m_bytes.begin() + offset);
206 offset += src.ByteCount;
207 count -= src.ByteCount;
208 ++ext;
209 }
210 if (count != 0)
211 {
212 throw std::runtime_error("Expected " + std::to_string(Size) + " bytes, got overflow of " +
213 std::to_string(-count) + " bytes");
214 }
215 break;
216 }
217 }
218 if (count != 0)
219 {
220 throw std::runtime_error(
221 "Expected " + std::to_string(Size) + " bytes, got only " + std::to_string(Size - count) + " bytes");
222 }
223}

701std::vector<int> get_ranges_info(const GifFileType *gif)
702{
703 std::vector<Byte> bytes;
704 for (int i = 0; i < gif->ExtensionBlockCount; ++i)
705 {
706 const ExtensionBlock &block{gif->ExtensionBlocks[i]};
707 if (is_fractint_extension(block, "fractint004"))
708 {
709 int ext{i + 1};
710 while (ext < gif->ExtensionBlockCount //
711 && gif->ExtensionBlocks[ext].Function == CONTINUE_EXT_FUNC_CODE) //
712 {
713 const ExtensionBlock &src = gif->ExtensionBlocks[ext];
714 std::copy_n(src.Bytes, src.ByteCount, std::back_inserter(bytes));
715 ++ext;
716 }
717 break;
718 }
719 }
720
721 if (bytes.size() % 2 != 0)
722 {
723 throw std::runtime_error("Odd number of bytes in ranges extension block");
724 }
725
726 std::vector<int> result;
727 result.resize(bytes.size() / 2);
728 const Byte *src{bytes.data()};
729 for (int &i : result)
730 {
731 i = extract_int16(src);
732 src += 2;
733 }
734 return result;
735}
