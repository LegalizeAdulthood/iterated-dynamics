38int perturbation_mode_index(const PerturbationMode mode)
39{
40 switch (mode)
41 {
42 case PerturbationMode::AUTO:
43 return 0;
44
45 case PerturbationMode::YES:
46 return 1;
47
48 case PerturbationMode::NO:
49 return 2;
50 }
51 return 0;
52}

54PerturbationMode perturbation_mode_from_index(const int index)
55{
56 switch (index)
57 {
58 default:
59 case 0:
60 return PerturbationMode::AUTO;
61
62 case 1:
63 return PerturbationMode::YES;
64
65 case 2:
66 return PerturbationMode::NO;

72/*
73 get_toggles() is called whenever the 'x' key is pressed.
74 This routine prompts for several options,
75 sets the appropriate variables, and returns the following code
76 to the calling routine:
77
78 -1 routine was ESCAPEd - no need to re-generate the image.
79 0 nothing changed, or minor variable such as "overwrite=".
80 No need to re-generate the image.
81 >0 major variable changed (such as "inside="). Re-generate
82 the image.
83
84 Finally, remember to insert variables in the list *and* check
