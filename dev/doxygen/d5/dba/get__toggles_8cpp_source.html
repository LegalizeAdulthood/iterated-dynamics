36/*
37 get_toggles() is called whenever the 'x' key is pressed.
38 This routine prompts for several options,
39 sets the appropriate variables, and returns the following code
40 to the calling routine:
41
42 -1 routine was ESCAPEd - no need to re-generate the image.
43 0 nothing changed, or minor variable such as "overwrite=".
44 No need to re-generate the image.
45 >0 major variable changed (such as "inside="). Re-generate
46 the image.
47
48 Finally, remember to insert variables in the list *and* check
49 for them in the same order!!!
50*/

330 if (g_sound_flag != old_sound_flag &&
331 ((g_sound_flag & SOUNDFLAG_ORBIT_MASK) > SOUNDFLAG_BEEP ||
332 (old_sound_flag & SOUNDFLAG_ORBIT_MASK) > SOUNDFLAG_BEEP))

334 j++;

336
337 if (g_iteration_ranges.empty())

352 if (g_user.biomorph_value >= g_colors)

355 }
356 if (g_user.biomorph_value != old_biomorph)
357 {
358 j++;
359 }

362 if (g_decomp[0] != old_decomp)
363 {
364 j++;
365 }
366
367 const std::string new_fill_color{id::algos::ascii_to_lower_copy(choices.read_string())};
368 if (new_fill_color.compare(0, 4, "norm") == 0)

371 }
372 else
373 {
374 g_fill_color = std::atoi(new_fill_color.c_str());
375 }
376 if (g_fill_color < 0)
377 {
378 g_fill_color = -1;
379 }
380 if (g_fill_color >= g_colors)
381 {
382 g_fill_color = g_fill_color % g_colors + g_fill_color / g_colors;
383 }
384 if (g_fill_color != old_fill_color)
385 {
386 j++;
387 }
388
389 g_close_proximity = choices.read_double_number();
390 if (g_close_proximity != old_close_proximity)
391 {
392 j++;
393 }

395 return j;
