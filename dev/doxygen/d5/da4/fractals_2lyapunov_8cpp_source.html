77 {

79 }
80 else if (g_color >= g_colors)
81 {
82 g_color = g_colors - 1;
83 }
84 g_plot(g_col, g_row, g_color);

115// This routine sets up the sequence for forcing the Rate parameter
116// to vary between the two values. It fills the array lyaRxy[] and
117// sets lyaLength to the length of the sequence.
118//
119// The sequence is coded in the bit pattern in an integer.
120// Briefly, the sequence starts with an A the leading zero bits
121// are ignored and the remaining bit sequence is decoded. The
122// sequence ends with a B. Not all possible sequences can be
123// represented in this manner, but every possible sequence is
124// either represented as itself, as a rotation of one of the
125// representable sequences, or as the inverse of a representable
126// sequence (swapping 0s and 1s in the array.) Sequences that
127// are the rotation and/or inverses of another sequence will generate
128// the same lyapunov exponents.
129//
130// A few examples follow:
131// number sequence
132// 0 ab
133// 1 aab
134// 2 aabb
135// 3 aaab
136// 4 aabbb
137// 5 aabab
138// 6 aaabb (this is a duplicate of 4, a rotated inverse)
139// 7 aaaab
140// 8 aabbbb etc.
141//
142LyapunovSequence build_lyapunov_sequence(long order)
143{
144 LyapunovSequence result{};
145 result.length = 1;
146 if (g_version < parse_legacy_version(1732))
147 {
148 order &= 0x0FFFFL;
149 }
150 result.rxy[0] = 1;
151 int t;
152 for (t = 31; t >= 0; t--)
153 {
154 if (order & 1L << t)
155 {
156 break;
157 }
158 }
159 for (; t >= 0; t--)
160 {
161 result.rxy[result.length++] = (order & 1L << t) != 0;
162 }
163 result.rxy[result.length++] = 0;
164 if (g_version < parse_legacy_version(1732))
165 {
166 for (t = result.length; t >= 0; t--)
167 {
168 result.rxy[t] = !result.rxy[t];
169 }
170 }
171 return result;
172}

174bool lyapunov_per_image()
175{
176 s_filter_cycles = static_cast<long>(g_params[2]);
177 if (s_filter_cycles == 0)
178 {
179 s_filter_cycles = g_max_iterations / 2;
180 }
181 if (g_params[1] == 0 || g_params[1] == 1)
182 {
183 set_random_seed();
184 }
185 s_lya_seed_ok = g_params[1] > 0 && g_params[1] <= 1 && g_debug_flag != DebugFlags::FORCE_STANDARD_FRACTAL;
186
187 const LyapunovSequence sequence{build_lyapunov_sequence(static_cast<long>(g_params[0]))};
188 s_lya_length = sequence.length;
189 std::copy(sequence.rxy.begin(), sequence.rxy.end(), s_lya_rxy);
190 if (g_version < parse_legacy_version(1731))
191 {
192 g_user.std_calc_mode = CalcMode::ONE_PASS;
193 g_std_calc_mode = CalcMode::ONE_PASS;
194 if (g_inside_method == ColorMethod::COLOR && g_inside_color == 1)
195 {
196 g_inside_color = 0;
197 }
198 }
199 if (g_inside_method < ColorMethod::COLOR)
200 {
201 stop_msg("Sorry, inside options other than inside=nnn are not supported by the lyapunov");
202 g_inside_method = ColorMethod::COLOR;
203 g_inside_color = 1;
204 }
205 if (g_user.std_calc_mode == CalcMode::ORBIT)
206 {
207 // Oops,lyapunov type
208 g_user.std_calc_mode = CalcMode::ONE_PASS; // doesn't use new & breaks orbits
209 g_std_calc_mode = CalcMode::ONE_PASS;
210 }
211 return true;
212}

214static int lyapunov_cycles(const long filter_cycles, const double a, const double b)
215{
216 int color;
217 double temp;
218 // e10=22026.4657948 e-10=0.0000453999297625
219
220 double total = 1.0;
221 int ln_adjust = 0;
222 long i;
223 for (i = 0; i < filter_cycles; i++)
224 {
225 for (int count = 0; count < s_lya_length; count++)
226 {
227 g_rate = s_lya_rxy[count] ? a : b;
228 if (orbit_calc())
229 {
230 g_overflow = true;
231 goto jump_out;
232 }
233 }
234 }
235 for (i = 0; i < g_max_iterations / 2; i++)
236 {
237 for (int count = 0; count < s_lya_length; count++)
238 {
239 g_rate = s_lya_rxy[count] ? a : b;
240 if (orbit_calc())
241 {
242 g_overflow = true;
243 goto jump_out;
244 }
245 temp = std::abs(g_rate - 2.0 * g_rate * g_population);
246 total *= temp;
247 if (total == 0)
248 {
249 g_overflow = true;
250 goto jump_out;
251 }
252 }
253 while (total > 22026.4657948)
254 {
255 total *= 0.0000453999297625;
256 ln_adjust += 10;
257 }
258 while (total < 0.0000453999297625)
259 {
260 total *= 22026.4657948;
261 ln_adjust -= 10;
262 }
263 }
264
265jump_out:
266 if (g_overflow || total <= 0 || (temp = std::log(total) + ln_adjust) > 0)
267 {
268 color = 0;
269 }
270 else
271 {
272 double lyap;
273 if (g_log_map_flag)
274 {
275 lyap = -temp / (static_cast<double>(s_lya_length) * i);
276 }
277 else
278 {
279 lyap = 1 - std::exp(temp / (static_cast<double>(s_lya_length) * i));
280 }
281 color = 1 + static_cast<int>(lyap * (g_colors - 1));
282 }
283 return color;
284}
