44// take one step in the direction of going_to
45static void step_col_row()
46{
47 switch (s_going_to)
48 {
49 case Direction::NORTH:
50 g_col = s_trail_col;
51 g_row = s_trail_row - 1;
52 break;
53 case Direction::EAST:
54 g_col = s_trail_col + 1;
55 g_row = s_trail_row;
56 break;
57 case Direction::SOUTH:
58 g_col = s_trail_col;
59 g_row = s_trail_row + 1;
60 break;
61 case Direction::WEST:
62 g_col = s_trail_col - 1;
63 g_row = s_trail_row;
64 break;
65 }

68int boundary_trace()
69{
70 if ((g_inside_method == ColorMethod::COLOR && g_inside_color == 0) ||
71 (g_outside_method == ColorMethod::COLOR && g_outside_color == 0))
72 {
73 stop_msg("Boundary tracing cannot be used with inside=0 or outside=0");
74 return -1;
75 }
76 if (g_colors < 16)
77 {
78 stop_msg("Boundary tracing cannot be used with < 16 colors");
79 return -1;
80 }
81
82 int last_fill_color_used = -1;
83 g_passes = Passes::BOUNDARY_TRACE;
84 int max_put_line_length = 0; // reset max_putline_length
85 for (int cur_row = g_i_start_pt.y; cur_row <= g_i_stop_pt.y; cur_row++)
86 {
87 g_reset_periodicity = true; // reset for a new row
88 g_color = BK_COLOR;
89 for (int cur_col = g_i_start_pt.x; cur_col <= g_i_stop_pt.x; cur_col++)
90 {
91 if (get_color(cur_col, cur_row) != BK_COLOR)
92 {
93 continue;
94 }
95
96 int trail_color = g_color;
97 g_row = cur_row;
98 g_col = cur_col;
99 if (g_calc_type() == -1) // g_color, g_row, g_col are global
100 {
101 if (g_show_dot != BK_COLOR) // remove show dot pixel
102 {
103 g_plot(g_col, g_row, BK_COLOR);
104 }
105 if (g_i_stop_pt.y != g_stop_pt.y)
106 {
107 g_i_stop_pt.y = g_stop_pt.y - (cur_row - g_start_pt.y); // allow for sym
108 }
109 add_work_list(
110 g_start_pt.x, cur_row, g_stop_pt.x, g_i_stop_pt.y, cur_col, cur_row, 0, g_work_symmetry);
111 return -1;
112 }
113 g_reset_periodicity = false; // normal periodicity checking
114
115 // This next line may cause a few more pixels to be calculated,
116 // but at the savings of quite a bit of overhead
117 if (g_color != trail_color)
118 {
119 continue;
120 }
121
122 // sweep clockwise to trace outline
123 s_trail_row = cur_row;
124 s_trail_col = cur_col;
125 trail_color = g_color;
126 const int fill_color_used = g_fill_color > 0 ? g_fill_color : trail_color;
127 Direction coming_from = Direction::WEST;
128 s_going_to = Direction::EAST;
129 unsigned int matches_found = 0;
130 bool continue_loop = true;
131 do
132 {
133 step_col_row();
134 if (g_row >= cur_row //
135 && g_col >= g_i_start_pt.x //
136 && g_col <= g_i_stop_pt.x //
137 && g_row <= g_i_stop_pt.y)
138 {
139 g_color = get_color(g_col, g_row);
140 // g_color, g_row, g_col are global for g_calc_type()
141 if (g_color == BK_COLOR && g_calc_type()== -1)
142 {
143 if (g_show_dot != BK_COLOR) // remove show dot pixel
144 {
145 g_plot(g_col, g_row, BK_COLOR);
146 }
147 if (g_i_stop_pt.y != g_stop_pt.y)
148 {
149 g_i_stop_pt.y = g_stop_pt.y - (cur_row - g_start_pt.y); // allow for sym
150 }
151 add_work_list(
152 g_start_pt.x, cur_row, g_stop_pt.x, g_i_stop_pt.y, cur_col, cur_row, 0, g_work_symmetry);
153 return -1;
154 }
155 if (g_color == trail_color)
156 {
157 if (matches_found < 4) // to keep it from overflowing
158 {
159 matches_found++;
160 }
161 s_trail_row = g_row;
162 s_trail_col = g_col;
163 s_going_to = advance(s_going_to, -1);
164 coming_from = advance(s_going_to, -1);
165 }
166 else
167 {
168 s_going_to = advance(s_going_to, 1);
169 continue_loop = s_going_to != coming_from || matches_found > 0;
170 }
171 }
172 else
173 {
174 s_going_to = advance(s_going_to, 1);
175 continue_loop = s_going_to != coming_from || matches_found > 0;
176 }
177 } while (continue_loop && (g_col != cur_col || g_row != cur_row));
178
179 if (matches_found <= 3)
180 {
181 // no hole
182 g_color = BK_COLOR;
183 g_reset_periodicity = true;
184 continue;
185 }
186
187 // Fill in region by looping around again, filling lines to the left
188 // whenever going_to is South or West
189 s_trail_row = cur_row;
190 s_trail_col = cur_col;
191 coming_from = Direction::WEST;
192 s_going_to = Direction::EAST;
193 do
194 {
195 bool match_found = false;
196 do
197 {
198 step_col_row();
199 if (g_row >= cur_row //
200 && g_col >= g_i_start_pt.x //
201 && g_col <= g_i_stop_pt.x //
202 && g_row <= g_i_stop_pt.y //
203 && get_color(g_col, g_row) == trail_color) // getcolor() must be last
204 {
205 if (s_going_to == Direction::SOUTH ||
206 (s_going_to == Direction::WEST && coming_from != Direction::EAST))
207 {
208 // fill a row, but only once
209 int right = g_col;
210 while (--right >= g_i_start_pt.x && (g_color = get_color(right,g_row)) == trail_color)
211 {
212 // do nothing
213 }
214 if (g_color == BK_COLOR) // check last color
215 {
216 int left = right;
217 while (get_color(--left,g_row) == BK_COLOR)
218 {
219 // Should NOT be possible for left < g_i_start_pt.x
220 // do nothing
221 }
222 left++; // one pixel too far
223 if (right == left) // only one hole
224 {
225 g_plot(left,g_row,fill_color_used);
226 }
227 else
228 {
229 // fill the line to the left
230 if (const int length = right - left + 1;
231 fill_color_used != last_fill_color_used ||
232 length > max_put_line_length)
233 {
234 // only reset dstack if necessary
235 std::memset(s_stack, fill_color_used, length);
236 last_fill_color_used = fill_color_used;
237 max_put_line_length = length;
238 }
239 sym_fill_line(g_row, left, right, s_stack);
240 }
241 } // end of fill line
242 }
243 s_trail_row = g_row;
244 s_trail_col = g_col;
245 s_going_to = advance(s_going_to, -1);
246 coming_from = advance(s_going_to, -1);
247 match_found = true;
248 }
249 else
250 {
251 s_going_to = advance(s_going_to, 1);
252 }
253 } while (!match_found && s_going_to != coming_from);
254
255 if (!match_found)
256 {
257 // next one has to be a match
258 step_col_row();
259 s_trail_row = g_row;
260 s_trail_col = g_col;
261 s_going_to = advance(s_going_to, -1);
262 coming_from = advance(s_going_to, -1);
263 }
264 } while (s_trail_col != cur_col || s_trail_row != cur_row);
265 g_reset_periodicity = true; // reset after a trace/fill
266 g_color = BK_COLOR;
267 }
268 }
269 return 0;
270}
