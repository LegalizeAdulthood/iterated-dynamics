67static bool look(MainContext &context)
68{
69 const HelpLabels old_help_mode = g_help_mode;
70 g_help_mode = HelpLabels::HELP_BROWSE;
71 switch (file_get_window())
72 {
73 case ID_KEY_ENTER:
74 case ID_KEY_ENTER_2:
75 g_show_file = ShowFile::LOAD_IMAGE; // trigger load
76 g_browse.browsing = true; // but don't ask for the file name as it's just been selected
77 g_browse.stack.push_back(g_browse.name);
78 merge_path_names(g_read_filename, g_browse.name.c_str(), CmdFile::AT_AFTER_STARTUP);
79 if (g_ask_video)
80 {
81 driver_stack_screen(); // save graphics image
82 context.stacked = true;
83 }
84 return true; // hop off and do it!!
85
86 case '\\':
87 if (!g_browse.stack.empty())
88 {
89 // go back one file if somewhere to go (i.e. browsing)
90 g_browse.name = g_browse.stack.back();
91 g_browse.stack.pop_back();
92 merge_path_names(g_read_filename, g_browse.name.c_str(), CmdFile::AT_AFTER_STARTUP);
93 g_browse.browsing = true;
94 g_show_file = ShowFile::LOAD_IMAGE;
95 if (g_ask_video)
96 {
97 driver_stack_screen();// save graphics image
98 context.stacked = true;
99 }
100 return true;
101 }
102 // otherwise fall through and turn off browsing
103
104 case ID_KEY_ESC:
105 case 'l': // turn it off
106 case 'L':
107 g_browse.browsing = false;
108 g_help_mode = old_help_mode;
109 break;
110
111 case 's':
112 g_browse.browsing = false;
113 g_help_mode = old_help_mode;
114 save_image(g_save_filename);
115 break;
116
117 default: // or no files found, leave the state of browsing alone
118 break;
119 }
120
121 return false;
122}

124static void toggle_mandelbrot_julia(MainContext &context)
125{
126 static ImageRegion s_j; // "Julia mode" entry point
127
128 if (g_fractal_type == FractalType::FORMULA)
129 {
130 if (g_is_mandelbrot)
131 {
132 get_fractal_specific(g_fractal_type)->to_julia = g_fractal_type;
133 get_fractal_specific(g_fractal_type)->to_mandel = FractalType::NO_FRACTAL;
134 g_is_mandelbrot = false;
135 }
136 else
137 {
138 get_fractal_specific(g_fractal_type)->to_julia = FractalType::NO_FRACTAL;
139 get_fractal_specific(g_fractal_type)->to_mandel = g_fractal_type;
140 g_is_mandelbrot = true;
141 }
142 }
143 if (g_cur_fractal_specific->to_julia != FractalType::NO_FRACTAL //
144 && g_params[0] == 0.0 && g_params[1] == 0.0)
145 {
146 // switch to corresponding Julia set
147 g_has_inverse = g_fractal_type == FractalType::MANDEL && g_bf_math == BFMathType::NONE;
148 clear_zoom_box();
149 jiim(JIIMType::JIIM);
150 // flush keyboard buffer
151 if (const int key = driver_get_key(); key != ID_KEY_SPACE)
152 {
153 driver_unget_key(key);
154 return;
155 }
156 set_fractal_type(g_cur_fractal_specific->to_julia);
157 if (g_julia_c.x == JULIA_C_NOT_SET || g_julia_c.y == JULIA_C_NOT_SET)
158 {
159 g_params[0] = (g_image_region.m_max.x + g_image_region.m_min.x) / 2;
160 g_params[1] = (g_image_region.m_max.y + g_image_region.m_min.y) / 2;
161 }
162 else
163 {
164 g_params[0] = g_julia_c.x;
165 g_params[1] = g_julia_c.y;
166 g_julia_c.y = JULIA_C_NOT_SET;
167 g_julia_c.x = JULIA_C_NOT_SET;
168 }
169 s_j = g_save_image_region;
170 context.from_mandel = true;
171 g_image_region.m_min.x = g_cur_fractal_specific->x_min;
172 g_image_region.m_max.x = g_cur_fractal_specific->x_max;
173 g_image_region.m_min.y = g_cur_fractal_specific->y_min;
174 g_image_region.m_max.y = g_cur_fractal_specific->y_max;
175 g_image_region.m_3rd.x = g_image_region.m_min.x;
176 g_image_region.m_3rd.y = g_image_region.m_min.y;
177 if (g_user.distance_estimator_value == 0 && g_user.biomorph_value != -1)
178 {
179 g_image_region.m_min.x *= 3.0;
180 g_image_region.m_max.x *= 3.0;
181 g_image_region.m_min.y *= 3.0;
182 g_image_region.m_max.y *= 3.0;
183 g_image_region.m_3rd.x *= 3.0;
184 g_image_region.m_3rd.y *= 3.0;
185 }
186 g_zoom_enabled = true;
187 g_calc_status = CalcStatus::PARAMS_CHANGED;
188 context.more_keys = false;
189 }
190 else if (g_cur_fractal_specific->to_mandel != FractalType::NO_FRACTAL)
191 {
192 // switch to corresponding Mandel set
193 set_fractal_type(g_cur_fractal_specific->to_mandel);
194 if (context.from_mandel)
195 {
196 g_image_region = s_j;
197 }
198 else
199 {
200 g_image_region.m_3rd.x = g_cur_fractal_specific->x_min;
201 g_image_region.m_min.x = g_image_region.m_3rd.x;
202 g_image_region.m_max.x = g_cur_fractal_specific->x_max;
203 g_image_region.m_3rd.y = g_cur_fractal_specific->y_min;
204 g_image_region.m_min.y = g_image_region.m_3rd.y;
205 g_image_region.m_max.y = g_cur_fractal_specific->y_max;
206 }
207 g_save_c.x = g_params[0];
208 g_save_c.y = g_params[1];
209 g_params[0] = 0;
210 g_params[1] = 0;
211 g_zoom_enabled = true;
212 g_calc_status = CalcStatus::PARAMS_CHANGED;
213 context.more_keys = false;
214 }
215 else
216 {
217 driver_buzzer(Buzzer::PROBLEM); // can't switch
218 }
219}

221static MainState prompt_options(MainContext &context)
222{
223 const long old_max_iterations = g_max_iterations;
224 clear_zoom_box();
225 if (g_from_text)
226 {
227 g_from_text = false;
228 }
229 else
230 {
231 driver_stack_screen();
232 }
233 int i;
234 if (context.key == 'x')
235 {
236 i = get_toggles();
237 }
238 else if (context.key == 'y')
239 {
240 i = get_toggles2();
241 }
242 else if (context.key == 'p')
243 {
244 i = passes_options();
245 }
246 else if (context.key == 'z')
247 {
248 i = get_fract_params(true);
249 }
250 else if (context.key == 'v')
251 {
252 i = get_view_params(); // get the parameters
253 }
254 else if (context.key == ID_KEY_CTL_B)
255 {
256 i = get_browse_params();
257 }
258 else if (context.key == ID_KEY_CTL_E)
259 {
260 i = get_evolve_params();
261 if (i > 0)
262 {
263 g_start_show_orbit = false;
264 g_sound_flag &= ~(SOUNDFLAG_X | SOUNDFLAG_Y | SOUNDFLAG_Z); // turn off only x,y,z
265 g_log_map_auto_calculate = false; // turn it off
266 }
267 }
268 else if (context.key == ID_KEY_CTL_F)
269 {
270 i = get_sound_params();
271 }
272 else
273 {
274 i = get_cmd_string();
275 }
276 driver_unstack_screen();
277 if (g_evolving != EvolutionModeFlags::NONE && g_true_color)
278 {
279 g_true_color = false; // truecolor doesn't play well with the evolver
280 }
281 if (g_max_iterations > old_max_iterations //
282 && g_inside_method >= ColorMethod::COLOR //
283 && g_calc_status == CalcStatus::COMPLETED //
284 && g_cur_fractal_specific->calc_type == standard_fractal_type //
285 && !g_log_map_flag //
286 && !g_true_color // recalc not yet implemented with truecolor
287 && (g_user.std_calc_mode != CalcMode::TESSERAL || g_fill_color <= -1) // tesseral with fill doesn't work
288 && g_user.std_calc_mode != CalcMode::ORBIT //
289 && i == 1 // nothing else changed
290 && g_outside_method != ColorMethod::ATAN)
291 {
292 g_quick_calc = true;
293 g_old_std_calc_mode = g_user.std_calc_mode;
294 g_user.std_calc_mode = CalcMode::ONE_PASS;
295 context.more_keys = false;
296 g_calc_status = CalcStatus::RESUMABLE;
297 }
298 else if (i > 0)
299 {
300 // time to redraw?
301 g_quick_calc = false;
302 save_param_history();
303 context.more_keys = false;
304 g_calc_status = CalcStatus::PARAMS_CHANGED;
305 }
306 return MainState::NOTHING;
307}

309static MainState begin_ant(MainContext &/*context*/)
310{
311 clear_zoom_box();
312 double old_param[MAX_PARAMS];
313 FractalType old_type = g_fractal_type;
314 for (int j = 0; j < MAX_PARAMS; ++j)
315 {
316 old_param[j] = g_params[j];
317 }
318 if (g_fractal_type != FractalType::ANT)
319 {
320 set_fractal_type(FractalType::ANT);
321 load_params(g_fractal_type);
322 }
323 if (!g_from_text)
324 {
325 driver_stack_screen();
326 }
327 g_from_text = false;
328 const int err = get_fract_params(true);
329 driver_unstack_screen();
330 if (err >= 0 && ant_type() >= 0)
331 {
332 g_calc_status = CalcStatus::PARAMS_CHANGED;
333 }
334 set_fractal_type(old_type);
335 for (int j = 0; j < MAX_PARAMS; ++j)
336 {
337 g_params[j] = old_param[j];
338 }
339 return err >= 0 ? MainState::CONTINUE : MainState::NOTHING;
340}

387static MainState inverse_julia_toggle(MainContext &context)
388{
389 // if the inverse types proliferate, something more elegant will be needed
390 if (g_fractal_type == FractalType::JULIA || g_fractal_type == FractalType::INVERSE_JULIA)
391 {
392 static FractalType old_type = FractalType::NO_FRACTAL;
393 if (g_fractal_type == FractalType::JULIA)
394 {
395 old_type = g_fractal_type;
396 set_fractal_type(FractalType::INVERSE_JULIA);
397 }
398 else if (g_fractal_type == FractalType::INVERSE_JULIA)
399 {
400 set_fractal_type(old_type != FractalType::NO_FRACTAL ? old_type : FractalType::JULIA);
401 }
402 // TODO: is it really necessary to reset g_cur_fractal_specific?
403 assert(g_cur_fractal_specific == get_fractal_specific(g_fractal_type));
404 g_cur_fractal_specific = get_fractal_specific(g_fractal_type);
405 g_zoom_enabled = true;
406 g_calc_status = CalcStatus::PARAMS_CHANGED;
407 context.more_keys = false;
408 }
409 else
410 {
411 driver_buzzer(Buzzer::PROBLEM);
412 }
413 return MainState::NOTHING;

520static MainState execute_commands(MainContext &context)
521{
522 driver_stack_screen();
523 int i = +get_commands();
524 if (g_init_mode != -1)
525 {
526 // video= was specified
527 g_adapter = g_init_mode;
528 g_init_mode = -1;
529 i |= +CmdArgFlags::FRACTAL_PARAM;
530 g_save_dac = SaveDAC::NO;
531 }
532 else if (g_colors_preloaded)
533 {
534 // colors= was specified
535 refresh_dac();
536 g_colors_preloaded = false;
537 }
538 else if (i & +CmdArgFlags::RESET) // reset was specified
539 {
540 g_save_dac = SaveDAC::NO;
541 }
542 if (i & +CmdArgFlags::YES_3D)
543 {
544 // 3d = was specified
545 context.key = '3';
546 driver_unstack_screen();
547 // pretend '3' was keyed
548 return restore_from_3d(context);
549 }
550 if (i & +CmdArgFlags::FRACTAL_PARAM)
551 {
552 // fractal parameter changed
553 driver_discard_screen();
554 context.more_keys = false;
555 g_calc_status = CalcStatus::PARAMS_CHANGED;
556 }
557 else
558 {
559 driver_unstack_screen();
560 }
561 return MainState::NOTHING;
562}
