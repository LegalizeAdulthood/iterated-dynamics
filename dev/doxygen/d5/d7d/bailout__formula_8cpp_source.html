22static void get_sqr_mag()
23{
24 g_temp_sqr_x = sqr(g_new_z.x);
25 g_temp_sqr_y = sqr(g_new_z.y);
26 g_magnitude = g_temp_sqr_x + g_temp_sqr_y;
27}
28
29static int fp_mod_bailout()
30{
31 get_sqr_mag();
32 if (g_magnitude >= g_magnitude_limit)
33 {
34 return 1;
35 }
36 g_old_z = g_new_z;
37 return 0;
38}

116{

360void set_bailout_formula(const Bailout test)
361{
362 switch (test)
363 {
364 case Bailout::MOD:
365 g_bailout_float = fp_mod_bailout;
366 g_bailout_bignum = bn_mod_bailout;
367 g_bailout_bigfloat = bf_mod_bailout;
368 break;
369
370 case Bailout::REAL:
371 g_bailout_float = fp_real_bailout;
372 g_bailout_bignum = bn_real_bailout;
373 g_bailout_bigfloat = bf_real_bailout;
374 break;
375
376 case Bailout::IMAG:
377 g_bailout_float = fp_imag_bailout;
378 g_bailout_bignum = bn_imag_bailout;
379 g_bailout_bigfloat = bf_imag_bailout;
380 break;
381
382 case Bailout::OR:
383 g_bailout_float = fp_or_bailout;
384 g_bailout_bignum = bn_or_bailout;
385 g_bailout_bigfloat = bf_or_bailout;
386 break;
387
388 case Bailout::AND:
389 g_bailout_float = fp_and_bailout;
390 g_bailout_bignum = bn_and_bailout;
391 g_bailout_bigfloat = bf_and_bailout;
392 break;
393
394 case Bailout::MANH:
395 g_bailout_float = fp_manh_bailout;
396 g_bailout_bignum = bn_manh_bailout;
397 g_bailout_bigfloat = bf_manh_bailout;
398 break;
399
400 case Bailout::MANR:
401 g_bailout_float = fp_manr_bailout;
402 g_bailout_bignum = bn_manr_bailout;
403 g_bailout_bigfloat = bf_manr_bailout;
404 break;
405 }
406}
