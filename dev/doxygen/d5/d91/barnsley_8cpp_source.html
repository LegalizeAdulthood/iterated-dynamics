14int barnsley1_orbit()
15{
16 // Barnsley's Mandelbrot type M1 from "Fractals Everywhere" by Michael Barnsley, p. 322
17
18 // calculate intermediate products
19 const double old_x_init_x = g_old_z.x * g_float_param->x;
20 const double old_y_init_y = g_old_z.y * g_float_param->y;
21 const double old_x_init_y = g_old_z.x * g_float_param->y;
22 const double old_y_init_x = g_old_z.y * g_float_param->x;
23 // orbit calculation
24 if (g_old_z.x >= 0)
25 {
26 g_new_z.x = old_x_init_x - g_float_param->x - old_y_init_y;
27 g_new_z.y = old_y_init_x - g_float_param->y + old_x_init_y;
28 }
29 else
30 {
31 g_new_z.x = old_x_init_x + g_float_param->x - old_y_init_y;
32 g_new_z.y = old_y_init_x + g_float_param->y + old_x_init_y;
33 }
34 return g_bailout_float();
35}

62int barnsley3_orbit()
63{
64 // An unnamed Mandelbrot/Julia function from "Fractals Everywhere" by Michael Barnsley, p. 292,
65 // example 4.1
66
67 // calculate intermediate products
68 const double old_x_init_x = g_old_z.x * g_old_z.x;
69 const double old_y_init_y = g_old_z.y * g_old_z.y;
70 const double old_x_init_y = g_old_z.x * g_old_z.y;
71
72 // orbit calculation
73 if (g_old_z.x > 0)
74 {
75 g_new_z.x = old_x_init_x - old_y_init_y - 1.0;
76 g_new_z.y = old_x_init_y * 2;
77 }
78 else
79 {
80 g_new_z.x = old_x_init_x - old_y_init_y -1.0 + g_float_param->x * g_old_z.x;
81 g_new_z.y = old_x_init_y * 2;
82
83 /* This term added by Tim Wegner to make dependent on the
84 imaginary part of the parameter. (Otherwise Mandelbrot
85 is uninteresting.) */
86 g_new_z.y += g_float_param->y * g_old_z.x;
87 }
88 return g_bailout_float();
89}
