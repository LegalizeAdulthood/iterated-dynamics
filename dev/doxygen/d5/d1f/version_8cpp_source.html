23std::string to_string(const Version &value)
24{
25 if (value.legacy)
26 {
27 return fmt::format("{:d}.{:02d}", value.major, value.minor);
28 }
29
30 std::string text = fmt::format("{:d}.{:d}", value.major, value.minor);
31 if (value.patch || value.tweak)
32 {
33 text += fmt::format(".{:d}", value.patch);
34 }
35 if (value.tweak)
36 {
37 text += fmt::format(".{:d}", value.tweak);
38 }
39 return text;
40}

74bool operator<(const Version &lhs, const Version &rhs)
75{
76 if (lhs.legacy)
77 {
78 if (!rhs.legacy)
79 {
80 return true;
81 }
82 }
83 else if (rhs.legacy)
84 {
85 return false;
86 }
87 // now both lhs and rhs are either both legacy or both modern
88 if (lhs.major < rhs.major)
89 {
90 return true;
91 }
92 if (lhs.major == rhs.major)
93 {
94 if (lhs.minor < rhs.minor)
95 {
96 return true;
97 }
98 if (lhs.minor == rhs.minor)
99 {
100 if (lhs.patch < rhs.patch)
101 {
102 return true;
103 }
104 if (lhs.patch == rhs.patch && lhs.tweak < rhs.tweak)
105 {
106 return true;
107 }
108 }
109 }
110
111 return false;
112}
