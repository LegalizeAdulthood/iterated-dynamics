69/*
70 * DECODERLINEWIDTH is the width of the pixel buffer used by the decoder. A
71 * larger buffer gives better performance. However, this buffer does not
72 * have to be a whole line width, although historically it has
73 * been: images were decoded line by line and a whole line written to the
74 * screen at once. The requirement to have a whole line buffered at once
75 * has now been relaxed in order to support larger images. The one exception
76 * to this is in the case where the image is being decoded to a smaller size.
77 * The skipxdots and skipydots logic assumes that the buffer holds one line.
78 */
79
80constexpr int DECODER_LINE_WIDTH{MAX_PIXELS};

106int gif_view()
107{
108 Byte buffer[16];
109 unsigned top;
110 unsigned left;
111 std::filesystem::path input_file;
112 Byte byte_buf[257]; // for decoder
113
114 // using stack for decoder byte buf rather than static mem
115 set_byte_buff(byte_buf);
116
117 int status = 0;
118
119 // initialize the col and row count for write-lines
120 g_row_count = 0;
121 s_col_count = 0;
122
123 // Open the file
124 if (g_out_line == out_line_stereo)
125 {
126 input_file = g_stereo_map_filename;
127 }
128 else
129 {
130 input_file = g_read_filename;
131 }
132 if (!input_file.has_extension())
133 {
134 input_file.replace_extension(DEFAULT_FRACTAL_TYPE);
135 }
136 input_file = find_file(ReadFile::IMAGE, input_file.string());
137 if (input_file.empty())
138 {
139 input_file.replace_extension(ALTERNATE_FRACTAL_TYPE);
140 input_file = find_file(ReadFile::IMAGE, input_file.string());
141 }
142 if (input_file.empty())
143 {
144 return -1;
145 }
146 s_fp_in = std::fopen(input_file.string().c_str(), "rb");
147 if (s_fp_in == nullptr)
148 {
149 return -1;
150 }
151
152 // Get the screen description
153 for (int i = 0; i < 13; i++)
154 {
155 const int tmp = get_byte();
156 buffer[i] = static_cast<Byte>(tmp);
157 if (tmp < 0)
158 {
159 close_file();
160 return -1;
161 }
162 }
163
164 if (std::strncmp(reinterpret_cast<char *>(buffer), "GIF87a", 3) != 0 // use updated GIF specs
165 || buffer[3] < '0' || buffer[3] > '9'
166 || buffer[4] < '0' || buffer[4] > '9'
167 || buffer[5] < 'A' || buffer[5] > 'z')
168 {
169 close_file();
170 return -1;
171 }
172
173 unsigned width = buffer[6] | buffer[7] << 8;
174 g_height = buffer[8] | buffer[9] << 8;
175 int planes = (buffer[10] & 0x0F) + 1;
176 s_gif_view_image_width = width;
177
178 if ((buffer[10] & 0x80) == 0) // color map (better be!)
179 {
180 close_file();
181 return -1;
182 }
183 g_num_colors = 1 << planes;
184
185 if (g_dither_flag && g_num_colors > 2 && g_colors == 2 && g_out_line == out_line)
186 {
187 g_out_line = out_line_dither;
188 }
189
190 for (int i = 0; i < static_cast<int>(g_num_colors); i++)
191 {
192 for (int j = 0; j < 3; j++)
193 {
194 const int k = get_byte();
195 if (k < 0)
196 {
197 close_file();
198 return -1;
199 }
200 if ((g_display_3d == Display3DMode::NONE || !glasses_alternating_or_superimpose()) && g_read_color)
201 {
202 g_dac_box[i][j] = static_cast<Byte>(k);
203 }
204 }
205 }
206 g_color_state = ColorState::UNKNOWN_MAP; // colors aren't default and not a known .map file
207
208 // don't read if glasses
209 if (g_display_3d != Display3DMode::NONE && g_map_set && !glasses_alternating_or_superimpose())
210 {
211 validate_luts(g_map_name); // read the palette file
212 refresh_dac();
213 }
214 if (g_dac_box[0][0] != 255)
215 {
216 refresh_dac();
217 }
218 if (driver_is_disk())
219 {
220 // disk-video
221 dvid_status(1, "restoring " + std::filesystem::path(input_file).filename().string());
222 }
223 g_read_color = true;
224
225 // Now display one or more GIF objects
226 bool finished = false;
227 while (!finished)
228 {
229 switch (get_byte())
230 {
231 case ';':
232 // End of the GIF dataset
233 finished = true;
234 status = 0;
235 break;
236
237 case '!': // GIF Extension Block
238 get_byte(); // read (and ignore) the ID
239 {
240 int i;
241 while ((i = get_byte()) > 0) // get the data length
242 {
243 for (int j = 0; j < i; j++)
244 {
245 get_byte(); // flush the data
246 }
247 }
248 }
249 break;
250 case ',':
251 /*
252 * Start of an image object. Read the image description.
253 */
254
255 for (int i = 0; i < 9; i++)
256 {
257 const int tmp = get_byte();
258 buffer[i] = static_cast<Byte>(tmp);
259 if (tmp < 0)
260 {
261 status = -1;
262 break;
263 }
264 }
265 if (status < 0)
266 {
267 finished = true;
268 break;
269 }
270
271 left = buffer[0] | buffer[1] << 8;
272 top = buffer[2] | buffer[3] << 8;
273 width = buffer[4] | buffer[5] << 8;
274 g_height = buffer[6] | buffer[7] << 8;
275
276 // adjustments for handling MIGs
277 s_gif_view_image_top = top;
278 if (g_skip_x_dots > 0)
279 {
280 s_gif_view_image_top /= g_skip_y_dots + 1;
281 }
282 s_gif_view_image_left = left;
283 if (g_skip_y_dots > 0)
284 {
285 s_gif_view_image_left /= g_skip_x_dots + 1;
286 }
287 if (g_out_line == out_line)
288 {
289 // what about continuous potential????
290 if (width != s_gif_view_image_width || top != 0)
291 {
292 // we're using normal decoding and we have a MIG
293 g_out_line = out_line_migs;
294 }
295 else if (width > DECODER_LINE_WIDTH && g_skip_x_dots == 0)
296 {
297 g_out_line = out_line_too_wide;
298 }
299 }
300
301 if (g_potential.store_16bit)
302 {
303 width >>= 1;
304 }
305
306 // Skip local color palette
307 if ((buffer[8] & 0x80) == 0x80)
308 {
309 // local map?
310 // make this local
311 planes = (buffer[8] & 0x0F) + 1;
312 const int num_colors = 1 << planes;
313 // skip local map
314 for (int i = 0; i < num_colors; i++)
315 {
316 for (int j = 0; j < 3; j++)
317 {
318 if (get_byte() < 0)
319 {
320 close_file();
321 return -1;
322 }
323 }
324 }
325 }
326
327 // initialize the row count for write-lines
328 g_row_count = 0;
329
330 if (g_calc_status == CalcStatus::IN_PROGRESS) // should never be so, but make sure
331 {
332 g_calc_status = CalcStatus::PARAMS_CHANGED;
333 }
334 g_busy = true; // for slideshow CALCWAIT
335 /*
336 * Call decoder(width) via timer.
337 * Width is limited to DECODERLINE_WIDTH.
338 */
339 if (g_skip_x_dots == 0)
340 {
341 width = std::min(width, static_cast<unsigned>(DECODER_LINE_WIDTH));
342 }
343 status = decoder_timer(width);
344 g_busy = false; // for slideshow CALCWAIT
345 if (g_calc_status == CalcStatus::IN_PROGRESS) // e.g., set by line3d
346 {
347 g_calc_time = g_timer_interval; // note how long it took
348 g_calc_status = CalcStatus::COMPLETED; // complete
349 }
350 // Hey! the decoder doesn't read the last (0-length) block!!
351 if (get_byte() != 0)
352 {
353 status = -1;
354 finished = true;
355 }
356 break;
357 default:
358 status = -1;
359 finished = true;
360 break;
361 }
362 }
363 close_file();
364 if (driver_is_disk())
365 {
366 // disk-video
367 dvid_status(0, "Restore completed");
368 dvid_status(1, "");
369 }
370
371 return status;
372}

382static int out_line_migs(Byte *pixels, const int line_len)
383{
384 const int row = s_gif_view_image_top + g_row_count;
385 const int start_col = s_gif_view_image_left;
386 const int stop_col = start_col + line_len - 1;
387 write_span(row, start_col, stop_col, pixels);
388 g_row_count++;
389
390 return 0;
391}

393static int out_line_dither(Byte *pixels, const int line_len)
394{
395 int err;
396 s_dither_buf.resize(line_len + 1);
397 std::fill(s_dither_buf.begin(), s_dither_buf.end(), 0);
398
399 int next_err = (std::rand() & 0x1f) - 16;
400 for (int i = 0; i < line_len; i++)
401 {
402 // TODO: 6-bit color
403 int brt = (g_dac_box[pixels[i]][0] * 5 + g_dac_box[pixels[i]][1] * 9 + g_dac_box[pixels[i]][2] * 2) >>
404 4; // brightness from 0 to 63
405 brt += next_err;
406 if (brt > 32)
407 {
408 pixels[i] = 1;
409 err = brt-63;
410 }
411 else
412 {
413 pixels[i] = 0;
414 err = brt;
415 }
416 next_err = s_dither_buf[i+1]+err/3;
417 s_dither_buf[i] = static_cast<char>(err / 3);
418 s_dither_buf[i+1] = static_cast<char>(err / 3);
419 }
420 return out_line(pixels, line_len);
421}

424
425static int out_line_too_wide(Byte *pixels, int line_len)
426{
427 const int width = g_logical_screen.x_dots;
428 while (line_len > 0)
429 {
430 int extra = s_col_count + line_len - width;
431 if (extra > 0) // line wraps
432 {
433 write_span(g_row_count, s_col_count, width-1, pixels);
434 pixels += width-s_col_count;
435 line_len -= width-s_col_count;
436 s_col_count = width;
437 }
438 else
439 {
440 write_span(g_row_count, s_col_count, s_col_count+line_len-1, pixels);
441 s_col_count += line_len;
442 line_len = 0;
443 }
444 if (s_col_count >= width)
445 {
446 s_col_count = 0;
447 g_row_count++;
448 }
449 }
450 return 0;
451}

453static bool put_sound_line(const int row, const int col_start, const int col_stop, const Byte *pixels)
454{
455 for (int col = col_start; col <= col_stop; col++)
456 {
457 g_put_color(col, row, *pixels);

467int sound_line(Byte *pixels, int line_len)
468{
469 const int width = g_logical_screen.x_dots;
470 while (line_len > 0)
471 {
472 int extra = s_col_count + line_len - width;
473 if (extra > 0) // line wraps
474 {
475 if (put_sound_line(g_row_count, s_col_count, width-1, pixels))
476 {
477 break;
478 }
479 pixels += width-s_col_count;
480 line_len -= width-s_col_count;
481 s_col_count = width;
482 }
483 else
484 {
485 if (put_sound_line(g_row_count, s_col_count, s_col_count+line_len-1, pixels))
486 {
487 break;
488 }
489 s_col_count += line_len;
490 line_len = 0;
491 }
492 if (s_col_count >= width)
493 {
494 s_col_count = 0;
495 g_row_count++;
496 }
497 }
498 driver_mute();
499
500 return 0;
501}

503int pot_line(Byte *pixels, const int line_len)
504{
505 if (g_row_count == 0)
506 {
507 if (pot_start_disk() < 0)
508 {
509 return -1;
510 }
511 }
512 int save_row_count = g_row_count;
513 g_row_count >>= 1;
514 int row = g_row_count;
515 if ((save_row_count & 1) != 0) // odd line
516 {
517 row += g_logical_screen.y_dots;
518 }
519 else if (!driver_is_disk()) // even line - display the line too
520 {
521 out_line(pixels, line_len);
522 }
523 for (int col = 0; col < g_logical_screen.x_dots; ++col)
524 {
525 disk_write_pixel(col+g_logical_screen.x_offset, row+g_logical_screen.y_offset, *(pixels+col));
526 }
527 g_row_count = save_row_count + 1;
528 return 0;
529}
