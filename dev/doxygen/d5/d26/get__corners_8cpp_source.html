36/*
37 See if double value was changed by input screen. Problem is that the
38 conversion from double to string and back can make small changes
39 in the value, so will it twill test as "different" even though it
40 is not
41*/
42static int cmp_dbl(double old, const double new_val)
43{
44 char buf[81];
45
46 // change the old value with the same torture the new value had
47 double_to_string(buf, old); // convert "old" to string
48
49 old = std::atof(buf); // convert back
50 return std::abs(old-new_val) < DBL_EPSILON ? 0 : 1; // zero if same
51}
