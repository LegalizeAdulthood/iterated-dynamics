96void GDIDriver::get_max_size(int *width, int *height, bool *center_x, bool *center_y) const
97{
98 *width = m_win_text.get_max_width();
99 *height = m_win_text.get_max_height();
100 if (g_video_table[g_adapter].x_dots > *width)
101 {
102 *width = g_video_table[g_adapter].x_dots;
103 *center_x = false;
104 }
105 if (g_video_table[g_adapter].y_dots > *height)
106 {
107 *height = g_video_table[g_adapter].y_dots;
108 *center_y = false;
109 }
110}

112void GDIDriver::center_windows(const bool center_x, const bool center_y)
113{
114 POINT text_pos{};
115 POINT plot_pos{};
116
117 if (center_x)
118 {
119 plot_pos.x = (g_frame.get_width() - m_plot.get_width())/2;
120 }
121 else
122 {
123 text_pos.x = (g_frame.get_width() - m_win_text.get_max_width())/2;
124 }
125
126 if (center_y)
127 {
128 plot_pos.y = (g_frame.get_height() - m_plot.get_height())/2;
129 }
130 else
131 {
132 text_pos.y = (g_frame.get_height() - m_win_text.get_max_height())/2;
133 }
134
135 bool status = SetWindowPos(m_plot.get_window(), nullptr,
136 plot_pos.x, plot_pos.y, 0, 0, SWP_NOZORDER | SWP_NOSIZE) == TRUE;
137 _ASSERTE(status);
138 status = SetWindowPos(m_win_text.get_window(), nullptr,
139 text_pos.x, text_pos.y, 0, 0, SWP_NOZORDER | SWP_NOSIZE) == TRUE;
140 _ASSERTE(status);
141}

156void GDIDriver::get_max_screen(int &width, int &height)
157{
158 RECT desktop;
159 GetClientRect(GetDesktopWindow(), &desktop);
160 desktop.right -= GetSystemMetrics(SM_CXFRAME) * 2;
161 desktop.bottom -= GetSystemMetrics(SM_CYFRAME) * 2 + GetSystemMetrics(SM_CYCAPTION) - 1;
162
163 width = desktop.right;
164 height = desktop.bottom;
165}

167bool GDIDriver::init(int *argc, char **argv)
168{
169 if (!Win32BaseDriver::init(argc, argv))
170 {
171 return false;
172 }
173
174 m_plot.init(g_instance, "Plot");
175
176 // add default list of video modes
177 {
178 int width;
179 int height;
180 get_max_screen(width, height);
181
182 for (VideoInfo &mode : s_modes)
183 {
184 if (mode.x_dots <= width && mode.y_dots <= height)
185 {
186 add_video_mode(this, &mode);
187 }
188 }
189 }
190
191 return true;
192}

194/* resize
195 *
196 * Check if we need resizing. If no, return false.
197 * If yes, resize internal buffers and return true.
198 */
199bool GDIDriver::resize()
200{
201 int width;
202 int height;
203 bool center_graphics_x = true;
204 bool center_graphics_y = true;
205
206 get_max_size(&width, &height, &center_graphics_x, &center_graphics_y);
207 if (g_video_table[g_adapter].x_dots == m_plot.get_width() //
208 && g_video_table[g_adapter].y_dots == m_plot.get_height() //
209 && width == g_frame.get_width() //
210 && height == g_frame.get_height())
211 {
212 return false;
213 }
214
215 g_frame.resize(width, height);
216 m_plot.resize();
217 center_windows(center_graphics_x, center_graphics_y);
218 return true;
219}

306void GDIDriver::set_video_mode(const VideoInfo &mode)
307{
308 // initially, set the virtual line to be the scan line length
309 g_is_true_color = false; // assume not truecolor
310 g_vesa_x_res = 0; // reset indicators used for
311 g_vesa_y_res = 0; // virtual screen limits estimation
312 g_good_mode = true;
313 g_and_color = g_colors - 1;
314 g_box_count = 0;
315 g_dac_count = g_cycle_limit;
316 g_got_real_dac = true; // we are "VGA"
317 read_palette();
318
319 resize();
320 m_plot.clear();
321 if (g_disk_flag)
322 {
323 end_disk();
324 }
325 set_normal_dot();
326 set_normal_span();
327 set_for_graphics();
328 set_clear();

331bool GDIDriver::validate_mode(const VideoInfo &mode)
332{
333 int width;
334 int height;
335 get_max_screen(width, height);
336
337 // allow modes <= size of screen with 256 colors
338 return mode.x_dots <= width
339 && mode.y_dots <= height
340 && mode.colors == 256;
341}
